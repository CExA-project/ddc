97  if constexpr (in_tags_v<QueryDDim, detail::TypeSeq<ODDims...>>) {
98  return std::pair<std::size_t, std::size_t>(
99  front<QueryDDim>(c) - front<QueryDDim>(this->m_domain),
100  back<QueryDDim>(c) + 1 - front<QueryDDim>(this->m_domain));

167  constexpr ChunkSpan(allocation_mdspan_type allocation_mdspan, mdomain_type const& domain)
168  {
169  namespace stdex = std::experimental;
170  extents_type extents_s((front<DDims>(domain) + ::extents<DDims>(domain)).uid()...);
171  std::array<std::size_t, sizeof...(DDims)> strides_s {allocation_mdspan.mapping().stride(
172  type_seq_rank_v<DDims, detail::TypeSeq<DDims...>>)...};
173  stdex::layout_stride::mapping<extents_type> mapping_s(extents_s, strides_s);
174  this->m_internal_mdspan = internal_mdspan_type(
175  allocation_mdspan.data() - mapping_s(front<DDims>(domain).uid()...),
176  mapping_s);
177  this->m_domain = domain;
178  }

194  template <class... QueryDDims>
195  constexpr auto operator[](DiscreteCoordinate<QueryDDims...> const& slice_spec) const
196  {
197  auto subview = std::experimental::
198  submdspan(allocation_mdspan(), get_slicer_for<DDims>(slice_spec)...);
199  using detail::TypeSeq;
200  using selected_meshes = type_seq_remove_t<TypeSeq<DDims...>, TypeSeq<QueryDDims...>>;
201  return ::ChunkSpan(subview, select_by_type_seq<selected_meshes>(this->m_domain));
202  }

211  return ::ChunkSpan(subview, this->m_domain.restrict(odomain));

218  template <class... ODDims>
219  constexpr reference operator()(DiscreteCoordinate<ODDims> const&... mcoords) const noexcept
220  {
221  static_assert(sizeof...(ODDims) == sizeof...(DDims), "Invalid number of dimensions");
222  assert(((mcoords >= front<ODDims>(this->m_domain)) && ...));
223  assert(((mcoords <= back<ODDims>(this->m_domain)) && ...));
224  return this->m_internal_mdspan(uid(take<DDims>(mcoords...))...);
225  }

231  template <class... ODDims, class = std::enable_if_t<sizeof...(ODDims) != 1>>
232  constexpr reference operator()(DiscreteCoordinate<ODDims...> const& mcoord) const noexcept
233  {
234  static_assert(sizeof...(ODDims) == sizeof...(DDims), "Invalid number of dimensions");
235  assert(((select<ODDims>(mcoord) >= front<ODDims>(this->m_domain)) && ...));
236  assert(((select<ODDims>(mcoord) <= back<ODDims>(this->m_domain)) && ...));
237  return this->m_internal_mdspan(uid<DDims>(mcoord)...);
238  }
