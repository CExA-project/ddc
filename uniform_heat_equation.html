46template <class ChunkType>
47void display(double time, ChunkType temp)
48{
49 double const mean_temp = ddc::transform_reduce(
50 temp.domain(),
51 0.,
52 ddc::reducer::sum<double>(),
53 temp)
54 / temp.domain().size();
55 std::cout << std::fixed << std::setprecision(3);
56 std::cout << "At t = " << time << ",\n";
57 std::cout << " * mean temperature = " << mean_temp << "\n";
58 ddc::ChunkSpan temp_slice
59 = temp[ddc::get_domain<DDimY>(temp).front()
60 + ddc::get_domain<DDimY>(temp).size() / 2];
61 std::cout << " * temperature[y:"
62 << ddc::get_domain<DDimY>(temp).size() / 2 << "] = {";
63 ddc::for_each(
64 ddc::get_domain<DDimX>(temp),
65 [=](ddc::DiscreteElement<DDimX> const ix) {
66 std::cout << std::setw(6) << temp_slice(ix);
67 });
68 std::cout << " }" << std::endl;
69
70#ifdef DDC_BUILD_PDI_WRAPPER
71 ddc::PdiEvent("display")
72 .with("temp", temp)
73 .and_with("mean_temp", mean_temp)
74 .and_with("temp_slice", temp_slice);
75#endif
76}

82int main(int argc, char** argv)
83{
84#ifdef DDC_BUILD_PDI_WRAPPER
85 auto pdi_conf = PC_parse_string("");
86 PDI_init(pdi_conf);
87#endif
88 Kokkos::ScopeGuard const kokkos_scope(argc, argv);
89 ddc::ScopeGuard const ddc_scope(argc, argv);
90
91
93 double const x_start = -1.;
94 double const x_end = 1.;
95 std::size_t const nb_x_points = 10;
96 double const kx = .01;
99 double const y_start = -1.;
100 double const y_end = 1.;
101 std::size_t const nb_y_points = 100;
102 double const ky = .002;
105 double const start_time = 0.;
106 double const end_time = 10.;
108 std::ptrdiff_t const t_output_period = 10;
111
113 ddc::DiscreteVector<DDimX> const gwx(1);
115
117 auto const [x_domain, ghosted_x_domain, x_pre_ghost, x_post_ghost]
118 = ddc::init_discrete_space<DDimX>(DDimX::init_ghosted<DDimX>(
119 ddc::Coordinate<X>(x_start),
120 ddc::Coordinate<X>(x_end),
121 ddc::DiscreteVector<DDimX>(nb_x_points),
122 gwx));
124
126 ddc::DiscreteDomain<DDimX> const
127 x_domain_begin(x_domain.front(), x_post_ghost.extents());
128 ddc::DiscreteDomain<DDimX> const x_domain_end(
129 x_domain.back() - x_pre_ghost.extents() + 1,
130 x_pre_ghost.extents());
132
134 ddc::DiscreteVector<DDimY> const gwy(1);
135
136 auto const [y_domain, ghosted_y_domain, y_pre_ghost, y_post_ghost]
137 = ddc::init_discrete_space<DDimY>(DDimY::init_ghosted<DDimY>(
138 ddc::Coordinate<Y>(y_start),
139 ddc::Coordinate<Y>(y_end),
140 ddc::DiscreteVector<DDimY>(nb_y_points),
141 gwy));
142
143 ddc::DiscreteDomain<DDimY> const
144 y_domain_begin(y_domain.front(), y_post_ghost.extents());
145
146 ddc::DiscreteDomain<DDimY> const y_domain_end(
147 y_domain.back() - y_pre_ghost.extents() + 1,
148 y_pre_ghost.extents());
150
152 double const dx = ddc::step<DDimX>();
153 double const dy = ddc::step<DDimY>();
154 double const invdx2 = 1. / (dx * dx);
155 double const invdy2 = 1. / (dy * dy);
156
157 ddc::Coordinate<T> const dt(.5 / (kx * invdx2 + ky * invdy2));
159
161 ddc::DiscreteVector<DDimT> const nb_time_steps(
162 std::ceil((end_time - start_time) / dt) + .2);
163
164 ddc::DiscreteDomain<DDimT> const time_domain
165 = ddc::init_discrete_space<DDimT>(DDimT::init<DDimT>(
166 ddc::Coordinate<T>(start_time),
167 ddc::Coordinate<T>(end_time),
168 nb_time_steps + 1));
170
172 ddc::Chunk ghosted_last_temp(
173 "ghosted_last_temp",
174 ddc::DiscreteDomain<
175 DDimX,
176 DDimY>(ghosted_x_domain, ghosted_y_domain),
177 ddc::DeviceAllocator<double>());
178
179 ddc::Chunk ghosted_next_temp(
180 "ghosted_next_temp",
181 ddc::DiscreteDomain<
182 DDimX,
183 DDimY>(ghosted_x_domain, ghosted_y_domain),
184 ddc::DeviceAllocator<double>());
186
188 ddc::ChunkSpan const ghosted_initial_temp
189 = ghosted_last_temp.span_view();
191
193 ddc::parallel_for_each(
194 ddc::DiscreteDomain<DDimX, DDimY>(x_domain, y_domain),
195 KOKKOS_LAMBDA(ddc::DiscreteElement<DDimX, DDimY> const ixy) {
196 double const x = ddc::coordinate(
197 ddc::DiscreteElement<DDimX>(ixy));
198 double const y = ddc::coordinate(
199 ddc::DiscreteElement<DDimY>(ixy));
200 ghosted_initial_temp(ixy)
201 = 9.999 * ((x * x + y * y) < 0.25);
202 });
204
206 ddc::Chunk ghosted_temp(
207 "ghost_temp",
208 ddc::DiscreteDomain<
209 DDimX,
210 DDimY>(ghosted_x_domain, ghosted_y_domain),
211 ddc::HostAllocator<double>());
213
215 ddc::parallel_deepcopy(ghosted_temp, ghosted_last_temp);
217
219 display(ddc::coordinate(time_domain.front()),
220 ghosted_temp[x_domain][y_domain]);
222
224 ddc::DiscreteElement<DDimT> last_output_iter = time_domain.front();
226
228 for (ddc::DiscreteElement<DDimT> const iter :
229 time_domain.remove_first(ddc::DiscreteVector<DDimT>(1))) {
231
233 ddc::parallel_deepcopy(
234 ghosted_last_temp[ddc::DiscreteDomain<
235 DDimX,
236 DDimY>(x_pre_ghost, y_domain)],
237 ghosted_last_temp[ddc::DiscreteDomain<
238 DDimX,
239 DDimY>(y_domain, x_domain_end)]);
240 ddc::parallel_deepcopy(
241 ghosted_last_temp[ddc::DiscreteDomain<
242 DDimX,
243 DDimY>(y_domain, x_post_ghost)],
244 ghosted_last_temp[ddc::DiscreteDomain<
245 DDimX,
246 DDimY>(y_domain, x_domain_begin)]);
247 ddc::parallel_deepcopy(
248 ghosted_last_temp[ddc::DiscreteDomain<
249 DDimX,
250 DDimY>(x_domain, y_pre_ghost)],
251 ghosted_last_temp[ddc::DiscreteDomain<
252 DDimX,
253 DDimY>(x_domain, y_domain_end)]);
254 ddc::parallel_deepcopy(
255 ghosted_last_temp[ddc::DiscreteDomain<
256 DDimX,
257 DDimY>(x_domain, y_post_ghost)],
258 ghosted_last_temp[ddc::DiscreteDomain<
259 DDimX,
260 DDimY>(x_domain, y_domain_begin)]);
262
264 ddc::ChunkSpan const next_temp(
265 ghosted_next_temp[ddc::DiscreteDomain<
266 DDimX,
267 DDimY>(x_domain, y_domain)]);
268 ddc::ChunkSpan const last_temp(ghosted_last_temp.span_cview());
270
272 ddc::parallel_for_each(
273 next_temp.domain(),
274 KOKKOS_LAMBDA(
275 ddc::DiscreteElement<DDimX, DDimY> const ixy) {
276 ddc::DiscreteElement<DDimX> const ix(ixy);
277 ddc::DiscreteElement<DDimY> const iy(ixy);
278 double const dt = ddc::step<DDimT>();
279
280 next_temp(ix, iy) = last_temp(ix, iy);
281 next_temp(ix, iy) += kx * dt
282 * (last_temp(ix + 1, iy)
283 - 2.0 * last_temp(ix, iy)
284 + last_temp(ix - 1, iy))
285 * invdx2;
286
287 next_temp(ix, iy) += ky * dt
288 * (last_temp(ix, iy + 1)
289 - 2.0 * last_temp(ix, iy)
290 + last_temp(ix, iy - 1))
291 * invdy2;
292 });
294
296 if (iter - last_output_iter >= t_output_period) {
297 last_output_iter = iter;
298 ddc::parallel_deepcopy(ghosted_temp, ghosted_last_temp);
299 display(ddc::coordinate(iter),
300 ghosted_temp[ddc::DiscreteDomain<
301 DDimX,
302 DDimY>(x_domain, y_domain)]);
303 }
305
307 std::swap(ghosted_last_temp, ghosted_next_temp);
309 }
310
311
313 if (last_output_iter < time_domain.back()) {
314 ddc::parallel_deepcopy(ghosted_temp, ghosted_last_temp);
315 display(ddc::coordinate(time_domain.back()),
316 ghosted_temp[ddc::DiscreteDomain<
317 DDimX,
318 DDimY>(x_domain, y_domain)]);
319 }
321
322#ifdef DDC_BUILD_PDI_WRAPPER
323 PDI_finalize();
324 PC_tree_destroy(&pdi_conf);
325#endif
326}
