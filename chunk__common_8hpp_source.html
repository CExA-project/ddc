250  template <
251  class Mapping = mapping_type,
252  std::enable_if_t<std::is_constructible_v<Mapping, extents_type>, int> = 0>
253  constexpr ChunkCommon(ElementType* ptr, mdomain_type const& domain)
254  {
255  namespace stdex = std::experimental;
256  // Handle the case where an allocation of size 0 returns a nullptr.
257  assert((domain.size() == 0) || ((ptr != nullptr) && (domain.size() != 0)));
258 
259  extents_type extents_r(::ddc::extents<DDims>(domain).value()...);
260  mapping_type mapping_r(extents_r);
261 
262  extents_type extents_s((front<DDims>(domain) + ddc::extents<DDims>(domain)).uid()...);
263  std::array<std::size_t, sizeof...(DDims)> strides_s {
264  mapping_r.stride(type_seq_rank_v<DDims, detail::TypeSeq<DDims...>>)...};
265  stdex::layout_stride::mapping<extents_type> mapping_s(extents_s, strides_s);
266 
267  // Pointer offset to handle non-zero indexing
268  ptr -= mapping_s(front<DDims>(domain).uid()...);
269  m_internal_mdspan = internal_mdspan_type(ptr, mapping_s);
270  m_domain = domain;
271  }
