31 template <typename ExecSpace>
32 static std::unique_ptr<SplinesLinearProblem<ExecSpace>> make_new_dense(int const n)

48 template <typename ExecSpace>
49 static std::unique_ptr<SplinesLinearProblem<ExecSpace>> make_new_band(
50 int const n,
51 int const kl,
52 int const ku,
53 bool const pds)
54 {
55 if (2 * kl + 1 + ku >= n) {
56 return std::make_unique<SplinesLinearProblemDense<ExecSpace>>(n);
57 } else if (kl == ku && kl == 1 && pds) {
58 return std::make_unique<SplinesLinearProblemPDSTridiag<ExecSpace>>(n);
59 } else if (kl == ku && pds) {
60 return std::make_unique<SplinesLinearProblemPDSBand<ExecSpace>>(n, kl);
61 } else {
62 return std::make_unique<SplinesLinearProblemBand<ExecSpace>>(n, kl, ku);
63 }
64 }
65
82 template <typename ExecSpace>
83 static std::unique_ptr<SplinesLinearProblem<ExecSpace>> make_new_sparse(
84 int const n,
85 std::optional<std::size_t> cols_per_chunk = std::nullopt,
86 std::optional<unsigned int> preconditionner_max_block_size = std::nullopt)
87 {
88 return std::make_unique<SplinesLinearProblemSparse<
89 ExecSpace>>(n, cols_per_chunk, preconditionner_max_block_size);
90 }
