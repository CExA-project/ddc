18namespace ddc::detail {
19/**
20 * @brief A static factory for SplinesLinearProblem classes.
21 */
22class SplinesLinearProblemMaker
23{
24public:
25 /**
26 * @brief Construct a dense matrix
27 *
28 * @tparam the Kokkos::ExecutionSpace on which matrix-related operation will be performed.
29 * @param n The size of one of the dimensions of the square matrix.
30 *
31 * @return The SplinesLinearProblem instance.
32 */
33 template <typename ExecSpace>
34 static std::unique_ptr<SplinesLinearProblem<ExecSpace>> make_new_dense(int const n)
35 {
36 return std::make_unique<SplinesLinearProblemDense<ExecSpace>>(n);
37 }
38
39 /**
40 * @brief Construct a band matrix
41 *
42 * @tparam the Kokkos::ExecutionSpace on which matrix-related operation will be performed.
43 * @param n The size of one of the dimensions of the square matrix.
44 * @param kl The number of subdiagonals.
45 * @param ku The number of superdiagonals.
46 * @param pds A boolean indicating if the matrix is positive-definite symetric or not.
47 *
48 * @return The SplinesLinearProblem instance.
49 */
50 template <typename ExecSpace>
51 static std::unique_ptr<SplinesLinearProblem<ExecSpace>> make_new_band(
52 int const n,
53 int const kl,
54 int const ku,
55 bool const pds)
56 {
57 if (kl == ku && kl == 1 && pds) {
58 return std::make_unique<SplinesLinearProblemPDSTridiag<ExecSpace>>(n);
59 } else if (kl == ku && pds) {
60 return std::make_unique<SplinesLinearProblemPDSBand<ExecSpace>>(n, kl);
61 } else if (2 * kl + ku + 1 >= n) {
62 return std::make_unique<SplinesLinearProblemDense<ExecSpace>>(n);
63 } else {
64 return std::make_unique<SplinesLinearProblemBand<ExecSpace>>(n, kl, ku);
65 }
66 }
67
68 /**
69 * @brief Construct a 2x2-blocks or 3x3-blocks linear problem with band "main" block (the one called
70 * Q in SplinesLinearProblem2x2Blocks and SplinesLinearProblem3x3Blocks).
71 *
72 * @tparam the Kokkos::ExecutionSpace on which matrix-related operation will be performed.
73 * @param n The size of one of the dimensions of the whole square matrix.
74 * @param kl The number of subdiagonals in the band block.
75 * @param ku The number of superdiagonals in the band block.
76 * @param pds A boolean indicating if the band block is positive-definite symetric or not.
77 * @param bottom_right_size The size of one of the dimensions of the bottom-right square block.
78 * @param top_left_size The size of one of the dimensions of the top-left square block.
79 *
80 * @return The SplinesLinearProblem instance.
81 */
82 template <typename ExecSpace>
83 static std::unique_ptr<SplinesLinearProblem<ExecSpace>>
84 make_new_block_matrix_with_band_main_block(
85 int const n,
86 int const kl,
87 int const ku,
88 bool const pds,
89 int const bottom_right_size,
90 int const top_left_size = 0)
91 {
92 int const main_size = n - top_left_size - bottom_right_size;
93 std::unique_ptr<SplinesLinearProblem<ExecSpace>> main_block
94 = make_new_band<ExecSpace>(main_size, kl, ku, pds);
95 if (top_left_size == 0) {
96 return std::make_unique<
97 SplinesLinearProblem2x2Blocks<ExecSpace>>(n, std::move(main_block));
98 }
99 return std::make_unique<
100 SplinesLinearProblem3x3Blocks<ExecSpace>>(n, top_left_size, std::move(main_block));
101 }
102
103 /**
104 * @brief Construct a 2x2-blocks linear problem with band "main" block (the one called
105 * Q in SplinesLinearProblem2x2Blocks) and other blocks containing the "periodic parts" of
106 * a periodic band matrix.
107 *
108 * It simply calls make_new_block_matrix_with_band_main_block with bottom_size being
109 * max(kl, ku) (except if the allocation would be higher than instantiating a SplinesLinearProblemDense).
110 *
111 * @tparam the Kokkos::ExecutionSpace on which matrix-related operation will be performed.
112 * @param n The size of one of the dimensions of the whole square matrix.
113 * @param kl The number of subdiagonals in the band block.
114 * @param ku The number of superdiagonals in the band block.
115 * @param pds A boolean indicating if the band block is positive-definite symetric or not.
116 *
117 * @return The SplinesLinearProblem instance.
118 */
119 template <typename ExecSpace>
120 static std::unique_ptr<SplinesLinearProblem<ExecSpace>> make_new_periodic_band_matrix(
121 int const n,
122 int const kl,
123 int const ku,
124 bool const pds)
125 {
126 assert(kl < n);
127 assert(ku < n);
128 int const bottom_size = std::max(kl, ku);
129 int const top_size = n - bottom_size;
130
131 if (bottom_size * (n + top_size) + (2 * kl + ku + 1) * top_size >= n * n) {
132 return std::make_unique<SplinesLinearProblemDense<ExecSpace>>(n);
133 }
134
135 return make_new_block_matrix_with_band_main_block<ExecSpace>(n, kl, ku, pds, bottom_size);
136 }
137
138 /**
139 * @brief Construct a sparse matrix
140 *
141 * @tparam the Kokkos::ExecutionSpace on which matrix-related operation will be performed.
142 * @param n The size of one of the dimensions of the square matrix.
143 * @param cols_per_chunk A parameter used by the slicer (internal to the solver) to define the size
144 * of a chunk of right-hand sides of the linear problem to be computed in parallel (chunks are treated
145 * by the linear solver one-after-the-other).
146 * This value is optional. If no value is provided then the default value is chosen by the requested solver.
147 *
148 * @param preconditioner_max_block_size A parameter used by the slicer (internal to the solver) to
149 * define the size of a block used by the Block-Jacobi preconditioner.
150 * This value is optional. If no value is provided then the default value is chosen by the requested solver.
151 *
152 * @return The SplinesLinearProblem instance.
153 */
154 template <typename ExecSpace>
155 static std::unique_ptr<SplinesLinearProblem<ExecSpace>> make_new_sparse(
156 int const n,
157 std::optional<std::size_t> cols_per_chunk = std::nullopt,
158 std::optional<unsigned int> preconditioner_max_block_size = std::nullopt)
159 {
160 return std::make_unique<SplinesLinearProblemSparse<
161 ExecSpace>>(n, cols_per_chunk, preconditioner_max_block_size);
162 }
163};
