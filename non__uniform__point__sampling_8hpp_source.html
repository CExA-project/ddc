121template <class DDim>
122struct is_non_uniform_sampling : public std::is_base_of<NonUniformPointSamplingBase, DDim>
123{
124};
125
126template <class DDim>
127constexpr bool is_non_uniform_sampling_v = is_non_uniform_sampling<DDim>::value;
128
129template <
130 class DDimImpl,
131 std::enable_if_t<
132 is_non_uniform_sampling_v<typename DDimImpl::discrete_dimension_type>,
133 int> = 0>
134std::ostream& operator<<(std::ostream& out, DDimImpl const& mesh)
135{
136 return out << "NonUniformPointSampling(" << mesh.size() << ")";
137}

139template <class DDim, std::enable_if_t<is_non_uniform_sampling_v<DDim>, int> = 0>
140KOKKOS_FUNCTION Coordinate<typename DDim::continuous_dimension_type> coordinate(
141 DiscreteElement<DDim> const& c)
142{
143 return discrete_space<DDim>().coordinate(c);
144}
145
146template <class DDim, std::enable_if_t<is_non_uniform_sampling_v<DDim>, int> = 0>
147KOKKOS_FUNCTION Coordinate<typename DDim::continuous_dimension_type> distance_at_left(
148 DiscreteElement<DDim> i)
149{
150 return coordinate(i) - coordinate(i - 1);
151}
152
153template <class DDim, std::enable_if_t<is_non_uniform_sampling_v<DDim>, int> = 0>
154KOKKOS_FUNCTION Coordinate<typename DDim::continuous_dimension_type> distance_at_right(
155 DiscreteElement<DDim> i)
156{
157 return coordinate(i + 1) - coordinate(i);
158}
159
160template <class DDim, std::enable_if_t<is_non_uniform_sampling_v<DDim>, int> = 0>
161KOKKOS_FUNCTION Coordinate<typename DDim::continuous_dimension_type> rmin(
162 DiscreteDomain<DDim> const& d)
163{
164 return coordinate(d.front());
165}
166
167template <class DDim, std::enable_if_t<is_non_uniform_sampling_v<DDim>, int> = 0>
168KOKKOS_FUNCTION Coordinate<typename DDim::continuous_dimension_type> rmax(
169 DiscreteDomain<DDim> const& d)
170{
171 return coordinate(d.back());
172}
173
174template <class DDim, std::enable_if_t<is_non_uniform_sampling_v<DDim>, int> = 0>
175KOKKOS_FUNCTION Coordinate<typename DDim::continuous_dimension_type> rlength(
176 DiscreteDomain<DDim> const& d)
177{
178 return rmax(d) - rmin(d);
179}
