34template <class ExecSpace>
35class SplinesLinearProblemPDSTridiag : public SplinesLinearProblem<ExecSpace>
36{
37public:
38 using typename SplinesLinearProblem<ExecSpace>::MultiRHS;
39 using SplinesLinearProblem<ExecSpace>::size;
40
41protected:
42 Kokkos::View<double**, Kokkos::HostSpace> m_q; // pds band matrix representation
43
44public:
50 explicit SplinesLinearProblemPDSTridiag(std::size_t const mat_size)
51 : SplinesLinearProblem<ExecSpace>(mat_size)
52 , m_q("q", 2, mat_size)
53 {
54 Kokkos::deep_copy(m_q, 0.);
55 }
56
57 double get_element(std::size_t i, std::size_t j) const override
58 {
59 assert(i < size());
60 assert(j < size());
61
62 // Indices are swapped for an element on subdiagonal
63 if (i > j) {
64 std::swap(i, j);
65 }
66 if (j - i < 2) {
67 return m_q(j - i, i);
68 } else {
69 return 0.0;
70 }
71 }
72
73 void set_element(std::size_t i, std::size_t j, double const aij) override
74 {
75 assert(i < size());
76 assert(j < size());
77
78 // Indices are swapped for an element on subdiagonal
79 if (i > j) {
80 std::swap(i, j);
81 }
82 if (j - i < 2) {
83 m_q(j - i, i) = aij;
84 } else {
85 assert(std::fabs(aij) < 1e-20);
86 }
87 }
88
94 void setup_solver() override
95 {
96 int const info = LAPACKE_dpttrf(size(), m_q.data(), m_q.data() + m_q.stride(0));
97 if (info != 0) {
98 throw std::runtime_error(
99 "LAPACKE_dpttrf failed with error code " + std::to_string(info));
100 }
101 }
102
111 void solve(MultiRHS b, bool const) const override
112 {
113 assert(b.extent(0) == size());
114
115 auto b_host = create_mirror_view(Kokkos::DefaultHostExecutionSpace(), b);
116 Kokkos::deep_copy(b_host, b);
117 int const info = LAPACKE_dpttrs(
118 LAPACK_ROW_MAJOR,
119 b_host.extent(0),
120 b_host.extent(1),
121 m_q.data(),
122 m_q.data() + m_q.stride(0),
123 b_host.data(),
124 b_host.stride(0));
125 if (info != 0) {
126 throw std::runtime_error(
127 "LAPACKE_dpttrs failed with error code " + std::to_string(info));
128 }
129 Kokkos::deep_copy(b, b_host);
130 }
131};
