343 ExecSpace,

350 BcXmax1,

353 Solver,
354 IDimX...>::
355operator()(
356 ddc::ChunkSpan<double, spline_domain_type, Layout, memory_space> spline,
357 ddc::ChunkSpan<double const, vals_domain_type, Layout, memory_space> vals,
358 std::optional<ddc::ChunkSpan<double const, derivs_domain_type1, Layout, memory_space>> const
359 derivs_min1,
360 std::optional<ddc::ChunkSpan<double const, derivs_domain_type1, Layout, memory_space>> const
361 derivs_max1,
362 std::optional<ddc::ChunkSpan<double const, derivs_domain_type2, Layout, memory_space>> const
363 derivs_min2,
364 std::optional<ddc::ChunkSpan<double const, derivs_domain_type2, Layout, memory_space>> const
365 derivs_max2,
366 std::optional<ddc::ChunkSpan<double const, derivs_domain_type, Layout, memory_space>> const
367 mixed_derivs_min1_min2,
368 std::optional<ddc::ChunkSpan<double const, derivs_domain_type, Layout, memory_space>> const
369 mixed_derivs_max1_min2,
370 std::optional<ddc::ChunkSpan<double const, derivs_domain_type, Layout, memory_space>> const
371 mixed_derivs_min1_max2,
372 std::optional<ddc::ChunkSpan<double const, derivs_domain_type, Layout, memory_space>> const
373 mixed_derivs_max1_max2) const
374{
375 // TODO: perform computations along dimension 1 on different streams ?
376 // Spline1-transform derivs_min2 (to spline1_deriv_min)
377 ddc::Chunk spline1_deriv_min_alloc(
378 m_spline_builder_deriv1.spline_domain(),
379 ddc::KokkosAllocator<double, MemorySpace>());
380 auto spline1_deriv_min = spline1_deriv_min_alloc.span_view();
381 auto spline1_deriv_min_opt = std::optional(spline1_deriv_min.span_cview());
382 if constexpr (BcXmin1 == ddc::BoundCond::HERMITE) {
383 m_spline_builder_deriv1(
384 spline1_deriv_min,
385 *derivs_min2,
386 mixed_derivs_min1_min2,
387 mixed_derivs_max1_min2);
388 } else {
389 spline1_deriv_min_opt = std::nullopt;
390 }
391
392 // Spline1-transform vals (to spline1)
393 ddc::Chunk spline1_alloc(
394 m_spline_builder1.spline_domain(),
395 ddc::KokkosAllocator<double, MemorySpace>());
396 ddc::ChunkSpan spline1 = spline1_alloc.span_view();
397
398 m_spline_builder1(spline1, vals, derivs_min1, derivs_max1);
399
400 // Spline1-transform derivs_max2 (to spline1_deriv_max)
401 ddc::Chunk spline1_deriv_max_alloc(
402 m_spline_builder_deriv1.spline_domain(),
403 ddc::KokkosAllocator<double, MemorySpace>());
404 auto spline1_deriv_max = spline1_deriv_max_alloc.span_view();
405 auto spline1_deriv_max_opt = std::optional(spline1_deriv_max.span_cview());
406 if constexpr (BcXmax1 == ddc::BoundCond::HERMITE) {
407 m_spline_builder_deriv1(
408 spline1_deriv_max,
409 *derivs_max2,
410 mixed_derivs_min1_max2,
411 mixed_derivs_max1_max2);
412 } else {
413 spline1_deriv_max_opt = std::nullopt;
414 }
415
416 // Spline2-transform spline1
417 m_spline_builder2(spline, spline1.span_cview(), spline1_deriv_min_opt, spline1_deriv_max_opt);
418}
