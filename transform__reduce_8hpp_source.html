21 template <
22  class... DDims,
23  class T,
24  class BinaryReductionOp,
25  class UnaryTransformOp,
26  class... DCoords>
27 inline T transform_reduce_serial(
28  DiscreteDomain<DDims...> const& domain,
29  T init,
30  BinaryReductionOp const& reduce,
31  UnaryTransformOp const& transform,
32  DCoords const&... dcoords) noexcept
33 {
34  if constexpr (sizeof...(DCoords) == sizeof...(DDims)) {
35  return reduce(init, transform(DiscreteCoordinate<DDims...>(dcoords...)));
36  } else {
37  using CurrentDDim = type_seq_element_t<sizeof...(DCoords), detail::TypeSeq<DDims...>>;
38  for (DiscreteCoordinate<CurrentDDim> const ii : select<CurrentDDim>(domain)) {
39  init = transform_reduce_serial(domain, init, reduce, transform, dcoords..., ii);
40  }
41  return init;
42  }
43 }
