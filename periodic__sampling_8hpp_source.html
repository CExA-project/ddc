140 template <class DDim>
141 static std::tuple<typename DDim::template Impl<DDim, Kokkos::HostSpace>, DiscreteDomain<DDim>>
142 init(Coordinate<CDim> a,
143 Coordinate<CDim> b,
144 DiscreteVector<DDim> n,
145 DiscreteVector<DDim> n_period)
146 {
147 assert(a < b);
148 assert(n > 1);
149 assert(n_period > 1);
150 typename DDim::template Impl<DDim, Kokkos::HostSpace>
151 disc(a, Coordinate<CDim> {(b - a) / (n - 1)}, n_period);
152 DiscreteDomain<DDim> domain {disc.front(), n};
153 return std::make_tuple(std::move(disc), std::move(domain));
154 }

166 template <class DDim>
167 std::tuple<
168 Impl<DDim, Kokkos::HostSpace>,
169 DiscreteDomain<DDim>,
170 DiscreteDomain<DDim>,
171 DiscreteDomain<DDim>,
172 DiscreteDomain<DDim>>
173 init_ghosted(
174 continuous_element_type a,
175 continuous_element_type b,
176 DiscreteVector<DDim> n,
177 DiscreteVector<DDim> n_period,
178 DiscreteVector<DDim> n_ghosts_before,
179 DiscreteVector<DDim> n_ghosts_after)
180 {
181 using discrete_domain_type = DiscreteDomain<DDim>;
182 assert(a < b);
183 assert(n > 1);
184 assert(n_period > 1);
185 Real discretization_step {(b - a) / (n - 1)};
186 Impl<DDim, Kokkos::HostSpace>
187 disc(a - n_ghosts_before.value() * discretization_step,
188 discretization_step,
189 n_period);
190 discrete_domain_type ghosted_domain
191 = discrete_domain_type(disc.front(), n + n_ghosts_before + n_ghosts_after);
192 discrete_domain_type pre_ghost
193 = discrete_domain_type(ghosted_domain.front(), n_ghosts_before);
194 discrete_domain_type main_domain
195 = discrete_domain_type(ghosted_domain.front() + n_ghosts_before, n);
196 discrete_domain_type post_ghost
197 = discrete_domain_type(main_domain.back() + 1, n_ghosts_after);
198 return std::make_tuple(
199 std::move(disc),
200 std::move(main_domain),
201 std::move(ghosted_domain),
202 std::move(pre_ghost),
203 std::move(post_ghost));
204 }

215 template <class DDim>
216 std::tuple<
217 Impl<DDim, Kokkos::HostSpace>,
218 DiscreteDomain<DDim>,
219 DiscreteDomain<DDim>,

221 DiscreteDomain<DDim>>
222 init_ghosted(
223 continuous_element_type a,
224 continuous_element_type b,
225 DiscreteVector<DDim> n,
226 DiscreteVector<DDim> n_period,
227 DiscreteVector<DDim> n_ghosts)
228 {
229 return init_ghosted(a, b, n, n_period, n_ghosts, n_ghosts);
230 }
