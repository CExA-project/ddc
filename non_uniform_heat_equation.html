23std::vector<double> generate_random_vector(
24 int n,
25 double lower_bound,
26 double higher_bound)
27{
28 assert(n > 1);
29 assert(lower_bound < higher_bound);
30
31 std::random_device rd;
32 std::mt19937 gen(rd());
33 // p represents the fraction of displacement
34 // it should be less than 0.5 to avoid reordering of nodes
35 double const p = 0.1;
36 std::uniform_real_distribution<double> dis(-p, +p);
37
38 double const dx = (higher_bound - lower_bound) / (n - 1);
39
40 std::vector<double> vec(n);
41
42 // Generate a uniform mesh
43 for (int i = 0; i < n; ++i) {
44 vec[i] = lower_bound + i * dx;
45 }
46 // Add a random perturbation
47 for (int i = 1; i < n - 1; ++i) {
48 vec[i] += dis(gen) * dx;
49 }
50
51 assert(std::is_sorted(vec.begin(), vec.end()));
52
53 return vec;
54}

88template <class ChunkType>
89void display(double time, ChunkType temp)
90{
91 double const mean_temp = ddc::transform_reduce(
92 temp.domain(),
93 0.,
94 ddc::reducer::sum<double>(),
95 temp)
96 / temp.domain().size();
97 std::cout << std::fixed << std::setprecision(3);
98 std::cout << "At t = " << time << ",\n";
99 std::cout << " * mean temperature = " << mean_temp << "\n";
100 ddc::ChunkSpan const temp_slice
101 = temp[ddc::get_domain<DDimY>(temp).front()
102 + ddc::get_domain<DDimY>(temp).size() / 2];
103 std::cout << " * temperature[y:"
104 << ddc::get_domain<DDimY>(temp).size() / 2 << "] = {";
105 ddc::for_each(
106 ddc::get_domain<DDimX>(temp),
107 [=](ddc::DiscreteElement<DDimX> const ix) {
108 std::cout << std::setw(6) << temp_slice(ix);
109 });
110 std::cout << " }" << std::endl;
111
112#ifdef DDC_BUILD_PDI_WRAPPER
113 ddc::PdiEvent("display")
114 .with("temp", temp)
115 .and_with("mean_temp", mean_temp)
116 .and_with("temp_slice", temp_slice);
117#endif
118}

121int main(int argc, char** argv)
122{
123#ifdef DDC_BUILD_PDI_WRAPPER
124 PC_tree_t pdi_conf = PC_parse_string("");
125 PDI_init(pdi_conf);
126#endif
127 Kokkos::ScopeGuard const kokkos_scope(argc, argv);
128 ddc::ScopeGuard const ddc_scope(argc, argv);
129
132 double const x_start = -1.;
133 double const x_end = 1.;
134 std::size_t const nb_x_points = 10;
135 double const kx = .01;
138 double const y_start = -1.;
139 double const y_end = 1.;
140 std::size_t const nb_y_points = 100;
141 double const ky = .002;
144 double const start_time = 0.;
145 double const end_time = 10.;
147 std::ptrdiff_t const t_output_period = 10;
149
151 std::vector<double> x_domain_vect
152 = generate_random_vector(nb_x_points, x_start, x_end);
154
155 std::size_t const size_x = x_domain_vect.size();
156
158 std::vector<double> const x_pre_ghost_vect {
159 x_domain_vect.front()
160 - (x_domain_vect.back() - x_domain_vect[size_x - 2])};
161
162 std::vector<double> const x_post_ghost_vect {
163 x_domain_vect.back()
164 + (x_domain_vect[1] - x_domain_vect.front())};
166
168 auto const [x_domain, ghosted_x_domain, x_pre_ghost, x_post_ghost]
169 = ddc::init_discrete_space<DDimX>(DDimX::init_ghosted<DDimX>(
170 x_domain_vect,
171 x_pre_ghost_vect,
172 x_post_ghost_vect));
174
175 ddc::DiscreteDomain<DDimX> const
176 x_domain_begin(x_domain.front(), x_post_ghost.extents());
177 ddc::DiscreteDomain<DDimX> const x_domain_end(
178 x_domain.back() - x_pre_ghost.extents() + 1,
179 x_pre_ghost.extents());
180
182 std::vector<double> y_domain_vect
183 = generate_random_vector(nb_y_points, y_start, y_end);
184
185 std::size_t const size_y = y_domain_vect.size();
186
188 std::vector<double> const y_pre_ghost_vect {
189 y_domain_vect.front()
190 - (y_domain_vect.back() - y_domain_vect[size_y - 2])};
191 std::vector<double> const y_post_ghost_vect {
192 y_domain_vect.back()
193 + (y_domain_vect[1] - y_domain_vect.front())};
196
198 auto const [y_domain, ghosted_y_domain, y_pre_ghost, y_post_ghost]
199 = ddc::init_discrete_space<DDimY>(DDimY::init_ghosted<DDimY>(
200 y_domain_vect,
201 y_pre_ghost_vect,
202 y_post_ghost_vect));
204
205 ddc::DiscreteDomain<DDimY> const
206 y_domain_begin(y_domain.front(), y_post_ghost.extents());
207
208 ddc::DiscreteDomain<DDimY> const y_domain_end(
209 y_domain.back() - y_pre_ghost.extents() + 1,
210 y_pre_ghost.extents());
211
213 double const invdx2_max = ddc::transform_reduce(
214 x_domain,
215 0.,
216 ddc::reducer::max<double>(),
217 [](ddc::DiscreteElement<DDimX> ix) {
218 return 1.
219 / (ddc::distance_at_left(ix)
220 * ddc::distance_at_right(ix));
221 });
222
223 double const invdy2_max = ddc::transform_reduce(
224 y_domain,
225 0.,
226 ddc::reducer::max<double>(),
227 [](ddc::DiscreteElement<DDimY> iy) {
228 return 1.
229 / (ddc::distance_at_left(iy)
230 * ddc::distance_at_right(iy));
231 });
232
233 ddc::Coordinate<T> const max_dt {
234 .5 / (kx * invdx2_max + ky * invdy2_max)};
236
238 ddc::DiscreteVector<DDimT> const nb_time_steps(
239 std::ceil((end_time - start_time) / max_dt) + .2);
240
241 ddc::DiscreteDomain<DDimT> const time_domain
242 = ddc::init_discrete_space<DDimT>(DDimT::init<DDimT>(
243 ddc::Coordinate<T>(start_time),
244 ddc::Coordinate<T>(end_time),
245 nb_time_steps + 1));
247
249 ddc::Chunk ghosted_last_temp(
250 "ghosted_last_temp",
251 ddc::DiscreteDomain<
252 DDimX,
253 DDimY>(ghosted_x_domain, ghosted_y_domain),
254 ddc::DeviceAllocator<double>());
255
256 ddc::Chunk ghosted_next_temp(
257 "ghosted_next_temp",
258 ddc::DiscreteDomain<
259 DDimX,
260 DDimY>(ghosted_x_domain, ghosted_y_domain),
261 ddc::DeviceAllocator<double>());
263
265 ddc::ChunkSpan const ghosted_initial_temp
266 = ghosted_last_temp.span_view();
268
270 ddc::parallel_for_each(
271 ddc::DiscreteDomain<DDimX, DDimY>(x_domain, y_domain),
272 KOKKOS_LAMBDA(ddc::DiscreteElement<DDimX, DDimY> const ixy) {
273 double const x = ddc::coordinate(
274 ddc::DiscreteElement<DDimX>(ixy));
275 double const y = ddc::coordinate(
276 ddc::DiscreteElement<DDimY>(ixy));
277 ghosted_initial_temp(ixy)
278 = 9.999 * ((x * x + y * y) < 0.25);
279 });
281
283 ddc::Chunk ghosted_temp
284 = ddc::create_mirror(ghosted_last_temp.span_cview());
286
288 ddc::parallel_deepcopy(ghosted_temp, ghosted_last_temp);
290
292 display(ddc::coordinate(time_domain.front()),
293 ghosted_temp[x_domain][y_domain]);
295
297 ddc::DiscreteElement<DDimT> last_output_iter = time_domain.front();
299
301 for (ddc::DiscreteElement<DDimT> const iter :
302 time_domain.remove_first(ddc::DiscreteVector<DDimT>(1))) {
304
306 ddc::parallel_deepcopy(
307 ghosted_last_temp[ddc::DiscreteDomain<
308 DDimX,
309 DDimY>(x_pre_ghost, y_domain)],
310 ghosted_last_temp[ddc::DiscreteDomain<
311 DDimX,
312 DDimY>(y_domain, x_domain_end)]);
313 ddc::parallel_deepcopy(
314 ghosted_last_temp[ddc::DiscreteDomain<
315 DDimX,
316 DDimY>(y_domain, x_post_ghost)],
317 ghosted_last_temp[ddc::DiscreteDomain<
318 DDimX,
319 DDimY>(y_domain, x_domain_begin)]);
320 ddc::parallel_deepcopy(
321 ghosted_last_temp[ddc::DiscreteDomain<
322 DDimX,
323 DDimY>(x_domain, y_pre_ghost)],
324 ghosted_last_temp[ddc::DiscreteDomain<
325 DDimX,
326 DDimY>(x_domain, y_domain_end)]);
327 ddc::parallel_deepcopy(
328 ghosted_last_temp[ddc::DiscreteDomain<
329 DDimX,
330 DDimY>(x_domain, y_post_ghost)],
331 ghosted_last_temp[ddc::DiscreteDomain<
332 DDimX,
333 DDimY>(x_domain, y_domain_begin)]);
335
337 ddc::ChunkSpan const next_temp(
338 ghosted_next_temp[ddc::DiscreteDomain<
339 DDimX,
340 DDimY>(x_domain, y_domain)]);
341 ddc::ChunkSpan const last_temp(ghosted_last_temp.span_cview());
343
345 ddc::parallel_for_each(
346 next_temp.domain(),
347 KOKKOS_LAMBDA(
348 ddc::DiscreteElement<DDimX, DDimY> const ixy) {
349 ddc::DiscreteElement<DDimX> const ix
350 = ddc::select<DDimX>(ixy);
351 ddc::DiscreteElement<DDimY> const iy
352 = ddc::select<DDimY>(ixy);
353 double const dx_l = ddc::distance_at_left(ix);
354 double const dx_r = ddc::distance_at_right(ix);
355 double const dx_m = 0.5 * (dx_l + dx_r);
356 double const dy_l = ddc::distance_at_left(iy);
357 double const dy_r = ddc::distance_at_right(iy);
358 double const dy_m = 0.5 * (dy_l + dy_r);
359 next_temp(ix, iy) = last_temp(ix, iy);
360 next_temp(ix, iy)
361 += kx * ddc::step<DDimT>()
362 * (dx_l * last_temp(ix + 1, iy)
363 - 2.0 * dx_m * last_temp(ix, iy)
364 + dx_r * last_temp(ix - 1, iy))
365 / (dx_l * dx_m * dx_r);
366 next_temp(ix, iy)
367 += ky * ddc::step<DDimT>()
368 * (dy_l * last_temp(ix, iy + 1)
369 - 2.0 * dy_m * last_temp(ix, iy)
370 + dy_r * last_temp(ix, iy - 1))
371 / (dy_l * dy_m * dy_r);
372 });
374
376 if (iter - last_output_iter >= t_output_period) {
377 last_output_iter = iter;
378 ddc::parallel_deepcopy(ghosted_temp, ghosted_next_temp);
379 display(ddc::coordinate(iter),
380 ghosted_temp[ddc::DiscreteDomain<
381 DDimX,
382 DDimY>(x_domain, y_domain)]);
383 }
385
387 std::swap(ghosted_last_temp, ghosted_next_temp);
389 }
390
392 if (last_output_iter < time_domain.back()) {
393 ddc::parallel_deepcopy(ghosted_temp, ghosted_last_temp);
394 display(ddc::coordinate(time_domain.back()),
395 ghosted_temp[ddc::DiscreteDomain<
396 DDimX,
397 DDimY>(x_domain, y_domain)]);
398 }
400
401#ifdef DDC_BUILD_PDI_WRAPPER
402 PDI_finalize();
403 PC_tree_destroy(&pdi_conf);
404#endif
405}
