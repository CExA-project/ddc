45template <class ChunkType>
46void display(double time, ChunkType temp)
47{
48 double const mean_temp = ddc::transform_reduce(
49 temp.domain(),
50 0.,
51 ddc::reducer::sum<double>(),
52 temp)
53 / temp.domain().size();
54 std::cout << std::fixed << std::setprecision(3);
55 std::cout << "At t = " << time << ",\n";
56 std::cout << " * mean temperature = " << mean_temp << "\n";
57 ddc::ChunkSpan temp_slice
58 = temp[ddc::get_domain<DDimY>(temp).front()
59 + ddc::get_domain<DDimY>(temp).size() / 2];
60 std::cout << " * temperature[y:"
61 << ddc::get_domain<DDimY>(temp).size() / 2 << "] = {";
62 ddc::for_each(
63 ddc::get_domain<DDimX>(temp),
64 [=](ddc::DiscreteElement<DDimX> const ix) {
65 std::cout << std::setw(6) << temp_slice(ix);
66 });
67 std::cout << " }" << std::endl;
68
69#ifdef DDC_BUILD_PDI_WRAPPER
70 ddc::PdiEvent("display")
71 .with("temp", temp)
72 .and_with("mean_temp", mean_temp)
73 .and_with("temp_slice", temp_slice);
74#endif
75}

78int main(int argc, char** argv)
79{
80#ifdef DDC_BUILD_PDI_WRAPPER
81 PC_tree_t pdi_conf = PC_parse_string("");
82 PDI_init(pdi_conf);
83#endif
84 Kokkos::ScopeGuard const kokkos_scope(argc, argv);
85 ddc::ScopeGuard const ddc_scope(argc, argv);
86
89 double const x_start = -1.;
90 double const x_end = 1.;
91 std::size_t const nb_x_points = 10;
92 double const kx = .01;
95 double const y_start = -1.;
96 double const y_end = 1.;
97 std::size_t const nb_y_points = 100;
98 double const ky = .002;
101 double const start_time = 0.;
102 double const end_time = 10.;
104 std::ptrdiff_t const t_output_period = 10;
106
108 ddc::DiscreteVector<DDimX> const gwx(1);
110
112 auto const [x_domain, ghosted_x_domain, x_pre_ghost, x_post_ghost]
113 = ddc::init_discrete_space<DDimX>(DDimX::init_ghosted<DDimX>(
114 ddc::Coordinate<X>(x_start),
115 ddc::Coordinate<X>(x_end),
116 ddc::DiscreteVector<DDimX>(nb_x_points),
117 gwx));
119
121 ddc::DiscreteDomain<DDimX> const
122 x_domain_begin(x_domain.front(), x_post_ghost.extents());
123 ddc::DiscreteDomain<DDimX> const x_domain_end(
124 x_domain.back() - x_pre_ghost.extents() + 1,
125 x_pre_ghost.extents());
127
129 ddc::DiscreteVector<DDimY> const gwy(1);
130
131 auto const [y_domain, ghosted_y_domain, y_pre_ghost, y_post_ghost]
132 = ddc::init_discrete_space<DDimY>(DDimY::init_ghosted<DDimY>(
133 ddc::Coordinate<Y>(y_start),
134 ddc::Coordinate<Y>(y_end),
135 ddc::DiscreteVector<DDimY>(nb_y_points),
136 gwy));
137
138 ddc::DiscreteDomain<DDimY> const
139 y_domain_begin(y_domain.front(), y_post_ghost.extents());
140
141 ddc::DiscreteDomain<DDimY> const y_domain_end(
142 y_domain.back() - y_pre_ghost.extents() + 1,
143 y_pre_ghost.extents());
145
147 double const dx = ddc::step<DDimX>();
148 double const dy = ddc::step<DDimY>();
149 double const invdx2 = 1. / (dx * dx);
150 double const invdy2 = 1. / (dy * dy);
151
152 ddc::Coordinate<T> const dt(.5 / (kx * invdx2 + ky * invdy2));
154
156 ddc::DiscreteVector<DDimT> const nb_time_steps(
157 std::ceil((end_time - start_time) / dt) + .2);
158
159 ddc::DiscreteDomain<DDimT> const time_domain
160 = ddc::init_discrete_space<DDimT>(DDimT::init<DDimT>(
161 ddc::Coordinate<T>(start_time),
162 ddc::Coordinate<T>(end_time),
163 nb_time_steps + 1));
165
167 ddc::Chunk ghosted_last_temp(
168 "ghosted_last_temp",
169 ddc::DiscreteDomain<
170 DDimX,
171 DDimY>(ghosted_x_domain, ghosted_y_domain),
172 ddc::DeviceAllocator<double>());
173
174 ddc::Chunk ghosted_next_temp(
175 "ghosted_next_temp",
176 ddc::DiscreteDomain<
177 DDimX,
178 DDimY>(ghosted_x_domain, ghosted_y_domain),
179 ddc::DeviceAllocator<double>());
181
183 ddc::ChunkSpan const ghosted_initial_temp
184 = ghosted_last_temp.span_view();
186
188 ddc::parallel_for_each(
189 ddc::DiscreteDomain<DDimX, DDimY>(x_domain, y_domain),
190 KOKKOS_LAMBDA(ddc::DiscreteElement<DDimX, DDimY> const ixy) {
191 double const x = ddc::coordinate(
192 ddc::DiscreteElement<DDimX>(ixy));
193 double const y = ddc::coordinate(
194 ddc::DiscreteElement<DDimY>(ixy));
195 ghosted_initial_temp(ixy)
196 = 9.999 * ((x * x + y * y) < 0.25);
197 });
199
201 ddc::Chunk ghosted_temp
202 = ddc::create_mirror(ghosted_last_temp.span_cview());
204
206 ddc::parallel_deepcopy(ghosted_temp, ghosted_last_temp);
208
210 display(ddc::coordinate(time_domain.front()),
211 ghosted_temp[x_domain][y_domain]);
213
215 ddc::DiscreteElement<DDimT> last_output_iter = time_domain.front();
217
219 for (ddc::DiscreteElement<DDimT> const iter :
220 time_domain.remove_first(ddc::DiscreteVector<DDimT>(1))) {
222
224 ddc::parallel_deepcopy(
225 ghosted_last_temp[ddc::DiscreteDomain<
226 DDimX,
227 DDimY>(x_pre_ghost, y_domain)],
228 ghosted_last_temp[ddc::DiscreteDomain<
229 DDimX,
230 DDimY>(y_domain, x_domain_end)]);
231 ddc::parallel_deepcopy(
232 ghosted_last_temp[ddc::DiscreteDomain<
233 DDimX,
234 DDimY>(y_domain, x_post_ghost)],
235 ghosted_last_temp[ddc::DiscreteDomain<
236 DDimX,
237 DDimY>(y_domain, x_domain_begin)]);
238 ddc::parallel_deepcopy(
239 ghosted_last_temp[ddc::DiscreteDomain<
240 DDimX,
241 DDimY>(x_domain, y_pre_ghost)],
242 ghosted_last_temp[ddc::DiscreteDomain<
243 DDimX,
244 DDimY>(x_domain, y_domain_end)]);
245 ddc::parallel_deepcopy(
246 ghosted_last_temp[ddc::DiscreteDomain<
247 DDimX,
248 DDimY>(x_domain, y_post_ghost)],
249 ghosted_last_temp[ddc::DiscreteDomain<
250 DDimX,
251 DDimY>(x_domain, y_domain_begin)]);
253
255 ddc::ChunkSpan const next_temp(
256 ghosted_next_temp[ddc::DiscreteDomain<
257 DDimX,
258 DDimY>(x_domain, y_domain)]);
259 ddc::ChunkSpan const last_temp(ghosted_last_temp.span_cview());
261
263 ddc::parallel_for_each(
264 next_temp.domain(),
265 KOKKOS_LAMBDA(
266 ddc::DiscreteElement<DDimX, DDimY> const ixy) {
267 ddc::DiscreteElement<DDimX> const ix(ixy);
268 ddc::DiscreteElement<DDimY> const iy(ixy);
269 double const dt = ddc::step<DDimT>();
270
271 next_temp(ix, iy) = last_temp(ix, iy);
272 next_temp(ix, iy) += kx * dt
273 * (last_temp(ix + 1, iy)
274 - 2.0 * last_temp(ix, iy)
275 + last_temp(ix - 1, iy))
276 * invdx2;
277
278 next_temp(ix, iy) += ky * dt
279 * (last_temp(ix, iy + 1)
280 - 2.0 * last_temp(ix, iy)
281 + last_temp(ix, iy - 1))
282 * invdy2;
283 });
285
287 if (iter - last_output_iter >= t_output_period) {
288 last_output_iter = iter;
289 ddc::parallel_deepcopy(ghosted_temp, ghosted_last_temp);
290 display(ddc::coordinate(iter),
291 ghosted_temp[ddc::DiscreteDomain<
292 DDimX,
293 DDimY>(x_domain, y_domain)]);
294 }
296
298 std::swap(ghosted_last_temp, ghosted_next_temp);
300 }
301
303 if (last_output_iter < time_domain.back()) {
304 ddc::parallel_deepcopy(ghosted_temp, ghosted_last_temp);
305 display(ddc::coordinate(time_domain.back()),
306 ghosted_temp[ddc::DiscreteDomain<
307 DDimX,
308 DDimY>(x_domain, y_domain)]);
309 }
311
312#ifdef DDC_BUILD_PDI_WRAPPER
313 PDI_finalize();
314 PC_tree_destroy(&pdi_conf);
315#endif
316}
