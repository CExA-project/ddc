53template <class ChunkType>
54void display(double time, ChunkType temp)
55{
56 double const mean_temp = ddc::transform_reduce(
57 temp.domain(),
58 0.,
59 ddc::reducer::sum<double>(),
60 temp)
61 / temp.domain().size();
62 std::cout << std::fixed << std::setprecision(3);
63 std::cout << "At t = " << time << ",\n";
64 std::cout << " * mean temperature = " << mean_temp << "\n";
65 ddc::ChunkSpan const temp_slice
66 = temp[ddc::get_domain<DDimY>(temp).front()
67 + ddc::get_domain<DDimY>(temp).size() / 2];
68 std::cout << " * temperature[y:"
69 << ddc::get_domain<DDimY>(temp).size() / 2 << "] = {";
70 ddc::for_each(
71 ddc::get_domain<DDimX>(temp),
72 [=](ddc::DiscreteElement<DDimX> const ix) {
73 std::cout << std::setw(6) << temp_slice(ix);
74 });
75 std::cout << " }" << std::endl;
76
77#ifdef DDC_BUILD_PDI_WRAPPER
78 ddc::PdiEvent("display")
79 .with("temp", temp)
80 .and_with("mean_temp", mean_temp)
81 .and_with("temp_slice", temp_slice);
82#endif
83}

86int main(int argc, char** argv)
87{
88#ifdef DDC_BUILD_PDI_WRAPPER
89 PC_tree_t pdi_conf = PC_parse_string("");
90 PDI_init(pdi_conf);
91#endif
92 Kokkos::ScopeGuard const kokkos_scope(argc, argv);
93 ddc::ScopeGuard const ddc_scope(argc, argv);
94
97 double const x_start = -1.;
98 double const x_end = 1.;
99 std::size_t const nb_x_points = 10;
100 double const kx = .01;
103 double const y_start = -1.;
104 double const y_end = 1.;
105 std::size_t const nb_y_points = 100;
106 double const ky = .002;
109 double const start_time = 0.;
110 double const end_time = 10.;
112 std::ptrdiff_t const t_output_period = 10;
114
116 ddc::DiscreteVector<DDimX> const gwx(1);
118
120 auto const [x_domain, ghosted_x_domain, x_pre_ghost, x_post_ghost]
121 = ddc::init_discrete_space<DDimX>(DDimX::init_ghosted<DDimX>(
122 ddc::Coordinate<X>(x_start),
123 ddc::Coordinate<X>(x_end),
124 ddc::DiscreteVector<DDimX>(nb_x_points),
125 gwx));
127
129 ddc::DiscreteDomain<DDimX> const
130 x_domain_begin(x_domain.front(), x_post_ghost.extents());
131 ddc::DiscreteDomain<DDimX> const x_domain_end(
132 x_domain.back() - x_pre_ghost.extents() + 1,
133 x_pre_ghost.extents());
135
137 ddc::DiscreteVector<DDimY> const gwy(1);
138
139 auto const [y_domain, ghosted_y_domain, y_pre_ghost, y_post_ghost]
140 = ddc::init_discrete_space<DDimY>(DDimY::init_ghosted<DDimY>(
141 ddc::Coordinate<Y>(y_start),
142 ddc::Coordinate<Y>(y_end),
143 ddc::DiscreteVector<DDimY>(nb_y_points),
144 gwy));
145
146 ddc::DiscreteDomain<DDimY> const
147 y_domain_begin(y_domain.front(), y_post_ghost.extents());
148
149 ddc::DiscreteDomain<DDimY> const y_domain_end(
150 y_domain.back() - y_pre_ghost.extents() + 1,
151 y_pre_ghost.extents());
153
155 double const dx = ddc::step<DDimX>();
156 double const dy = ddc::step<DDimY>();
157 double const invdx2 = 1. / (dx * dx);
158 double const invdy2 = 1. / (dy * dy);
159
160 ddc::Coordinate<T> const dt(.5 / (kx * invdx2 + ky * invdy2));
162
164 ddc::DiscreteVector<DDimT> const nb_time_steps(
165 std::ceil((end_time - start_time) / dt) + .2);
166
167 ddc::DiscreteDomain<DDimT> const time_domain
168 = ddc::init_discrete_space<DDimT>(DDimT::init<DDimT>(
169 ddc::Coordinate<T>(start_time),
170 ddc::Coordinate<T>(end_time),
171 nb_time_steps + 1));
173
175 ddc::Chunk ghosted_last_temp(
176 "ghosted_last_temp",
177 ddc::DiscreteDomain<
178 DDimX,
179 DDimY>(ghosted_x_domain, ghosted_y_domain),
180 ddc::DeviceAllocator<double>());
181
182 ddc::Chunk ghosted_next_temp(
183 "ghosted_next_temp",
184 ddc::DiscreteDomain<
185 DDimX,
186 DDimY>(ghosted_x_domain, ghosted_y_domain),
187 ddc::DeviceAllocator<double>());
189
191 ddc::ChunkSpan const ghosted_initial_temp
192 = ghosted_last_temp.span_view();
194
196 ddc::parallel_for_each(
197 ddc::DiscreteDomain<DDimX, DDimY>(x_domain, y_domain),
198 KOKKOS_LAMBDA(ddc::DiscreteElement<DDimX, DDimY> const ixy) {
199 double const x = ddc::coordinate(
200 ddc::DiscreteElement<DDimX>(ixy));
201 double const y = ddc::coordinate(
202 ddc::DiscreteElement<DDimY>(ixy));
203 ghosted_initial_temp(ixy)
204 = 9.999 * ((x * x + y * y) < 0.25);
205 });
207
209 ddc::Chunk ghosted_temp
210 = ddc::create_mirror(ghosted_last_temp.span_cview());
212
214 ddc::parallel_deepcopy(ghosted_temp, ghosted_last_temp);
216
218 display(ddc::coordinate(time_domain.front()),
219 ghosted_temp[x_domain][y_domain]);
221
223 ddc::DiscreteElement<DDimT> last_output_iter = time_domain.front();
225
227 for (ddc::DiscreteElement<DDimT> const iter :
228 time_domain.remove_first(ddc::DiscreteVector<DDimT>(1))) {
230
232 ddc::parallel_deepcopy(
233 ghosted_last_temp[ddc::DiscreteDomain<
234 DDimX,
235 DDimY>(x_pre_ghost, y_domain)],
236 ghosted_last_temp[ddc::DiscreteDomain<
237 DDimX,
238 DDimY>(y_domain, x_domain_end)]);
239 ddc::parallel_deepcopy(
240 ghosted_last_temp[ddc::DiscreteDomain<
241 DDimX,
242 DDimY>(y_domain, x_post_ghost)],
243 ghosted_last_temp[ddc::DiscreteDomain<
244 DDimX,
245 DDimY>(y_domain, x_domain_begin)]);
246 ddc::parallel_deepcopy(
247 ghosted_last_temp[ddc::DiscreteDomain<
248 DDimX,
249 DDimY>(x_domain, y_pre_ghost)],
250 ghosted_last_temp[ddc::DiscreteDomain<
251 DDimX,
252 DDimY>(x_domain, y_domain_end)]);
253 ddc::parallel_deepcopy(
254 ghosted_last_temp[ddc::DiscreteDomain<
255 DDimX,
256 DDimY>(x_domain, y_post_ghost)],
257 ghosted_last_temp[ddc::DiscreteDomain<
258 DDimX,
259 DDimY>(x_domain, y_domain_begin)]);
261
263 ddc::ChunkSpan const next_temp(
264 ghosted_next_temp[ddc::DiscreteDomain<
265 DDimX,
266 DDimY>(x_domain, y_domain)]);
267 ddc::ChunkSpan const last_temp(ghosted_last_temp.span_cview());
269
271 ddc::parallel_for_each(
272 next_temp.domain(),
273 KOKKOS_LAMBDA(
274 ddc::DiscreteElement<DDimX, DDimY> const ixy) {
275 ddc::DiscreteElement<DDimX> const ix(ixy);
276 ddc::DiscreteElement<DDimY> const iy(ixy);
277 double const dt = ddc::step<DDimT>();
278
279 next_temp(ix, iy) = last_temp(ix, iy);
280 next_temp(ix, iy) += kx * dt
281 * (last_temp(ix + 1, iy)
282 - 2.0 * last_temp(ix, iy)
283 + last_temp(ix - 1, iy))
284 * invdx2;
285
286 next_temp(ix, iy) += ky * dt
287 * (last_temp(ix, iy + 1)
288 - 2.0 * last_temp(ix, iy)
289 + last_temp(ix, iy - 1))
290 * invdy2;
291 });
293
295 if (iter - last_output_iter >= t_output_period) {
296 last_output_iter = iter;
297 ddc::parallel_deepcopy(ghosted_temp, ghosted_next_temp);
298 display(ddc::coordinate(iter),
299 ghosted_temp[ddc::DiscreteDomain<
300 DDimX,
301 DDimY>(x_domain, y_domain)]);
302 }
304
306 std::swap(ghosted_last_temp, ghosted_next_temp);
308 }
309
311 if (last_output_iter < time_domain.back()) {
312 ddc::parallel_deepcopy(ghosted_temp, ghosted_last_temp);
313 display(ddc::coordinate(time_domain.back()),
314 ghosted_temp[ddc::DiscreteDomain<
315 DDimX,
316 DDimY>(x_domain, y_domain)]);
317 }
319
320#ifdef DDC_BUILD_PDI_WRAPPER
321 PDI_finalize();
322 PC_tree_destroy(&pdi_conf);
323#endif
324}
