18constexpr bool is_spline_interpolation_mesh_uniform(
19 bool const is_uniform,
20 ddc::BoundCond const BcXmin,
21 ddc::BoundCond const BcXmax,
22 int degree)
23{
24 int N_BE_MIN = n_boundary_equations(BcXmin, degree);
25 int N_BE_MAX = n_boundary_equations(BcXmax, degree);
26 bool is_periodic = (BcXmin == ddc::BoundCond::PERIODIC) && (BcXmax == ddc::BoundCond::PERIODIC);
27 return is_uniform && ((N_BE_MIN != 0 && N_BE_MAX != 0) || is_periodic);
28}

330int SplineBuilder<
331 ExecSpace,
332 MemorySpace,
333 BSplines,
334 InterpolationMesh,
335 BcXmin,
336 BcXmax,
337 Solver,
338 IDimX...>::compute_offset(interpolation_domain_type const& interpolation_domain)
339{
340 int offset;
341 if constexpr (bsplines_type::is_periodic()) {
342 // Calculate offset so that the matrix is diagonally dominant
343 std::array<double, bsplines_type::degree() + 1> values_ptr;
344 std::experimental::mdspan<
345 double,
346 std::experimental::extents<std::size_t, bsplines_type::degree() + 1>> const
347 values(values_ptr.data());
348 ddc::DiscreteElement<interpolation_mesh_type> start(interpolation_domain.front());
349 auto jmin = ddc::discrete_space<BSplines>()
350 .eval_basis(values, ddc::coordinate(start + BSplines::degree()));
351 if constexpr (bsplines_type::degree() % 2 == 0) {
352 offset = jmin.uid() - start.uid() + bsplines_type::degree() / 2 - BSplines::degree();
353 } else {
354 int const mid = bsplines_type::degree() / 2;
355 offset = jmin.uid() - start.uid() + (values(mid) > values(mid + 1) ? mid : mid + 1)
356 - BSplines::degree();
357 }
358 } else {
359 offset = 0;
360 }
361 return offset;
362}

377 InterpolationMesh,
378 BcXmin,
379 BcXmax,
380 Solver,
381 IDimX...>::compute_block_sizes_uniform(int& lower_block_size, int& upper_block_size) const
382{
383 switch (BcXmin) {
384 case ddc::BoundCond::PERIODIC:
385 upper_block_size = (bsplines_type::degree()) / 2;
386 break;
387 case ddc::BoundCond::HERMITE:
388 upper_block_size = s_nbc_xmin;
389 break;
390 case ddc::BoundCond::GREVILLE:
391 upper_block_size = bsplines_type::degree() - 1;
392 break;
393 default:
394 throw std::runtime_error("ddc::BoundCond not handled");
395 }
396 switch (BcXmax) {
397 case ddc::BoundCond::PERIODIC:
398 lower_block_size = (bsplines_type::degree()) / 2;
399 break;
400 case ddc::BoundCond::HERMITE:
401 lower_block_size = s_nbc_xmax;
402 break;
403 case ddc::BoundCond::GREVILLE:
404 lower_block_size = bsplines_type::degree() - 1;
405 break;
406 default:
407 throw std::runtime_error("ddc::BoundCond not handled");
408 }
409}

501template <
502 class ExecSpace,
503 class MemorySpace,
504 class BSplines,
505 class InterpolationMesh,
506 ddc::BoundCond BcXmin,
507 ddc::BoundCond BcXmax,
508 SplineSolver Solver,
509 class... IDimX>
510void SplineBuilder<
511 ExecSpace,
512 MemorySpace,
513 BSplines,
514 InterpolationMesh,
515 BcXmin,
516 BcXmax,
517 Solver,
518 IDimX...>::build_matrix_system()
519{
520 // Hermite boundary conditions at xmin, if any
521 if constexpr (BcXmin == ddc::BoundCond::HERMITE) {
522 std::array<double, (bsplines_type::degree() / 2 + 1) * (bsplines_type::degree() + 1)>
523 derivs_ptr;
524 ddc::DSpan2D
525 derivs(derivs_ptr.data(),
526 bsplines_type::degree() + 1,
527 bsplines_type::degree() / 2 + 1);
528 ddc::discrete_space<BSplines>().eval_basis_and_n_derivs(
529 derivs,
530 ddc::discrete_space<BSplines>().rmin(),
531 s_nbc_xmin);
532
533 // In order to improve the condition number of the matrix, we normalize
534 // all derivatives by multiplying the i-th derivative by dx^i
535 for (std::size_t i = 0; i < bsplines_type::degree() + 1; ++i) {
536 for (std::size_t j = 1; j < bsplines_type::degree() / 2 + 1; ++j) {
537 derivs(i, j) *= ddc::detail::ipow(m_dx, j);
538 }
539 }
540
541 // iterate only to deg as last bspline is 0
542 for (std::size_t i = 0; i < s_nbc_xmin; ++i) {
543 for (std::size_t j = 0; j < bsplines_type::degree(); ++j) {
544 matrix->set_element(i, j, derivs(j, s_nbc_xmin - i - 1 + s_odd));
545 }
546 }
547 }
548
549 // Interpolation points
550 std::array<double, bsplines_type::degree() + 1> values_ptr;
551 std::experimental::mdspan<
552 double,
553 std::experimental::extents<std::size_t, bsplines_type::degree() + 1>> const
554 values(values_ptr.data());
555
556 int start = interpolation_domain().front().uid();
557 ddc::for_each(interpolation_domain(), [&](auto ix) {
558 auto jmin = ddc::discrete_space<BSplines>().eval_basis(
559 values,
560 ddc::coordinate(ddc::DiscreteElement<interpolation_mesh_type>(ix)));
561 for (std::size_t s = 0; s < bsplines_type::degree() + 1; ++s) {
562 int const j = ddc::detail::
563 modulo(int(jmin.uid() - m_offset + s),
564 (int)ddc::discrete_space<BSplines>().nbasis());
565 matrix->set_element(ix.uid() - start + s_nbc_xmin, j, values(s));
566 }
567 });
568
569 // Hermite boundary conditions at xmax, if any
570 if constexpr (BcXmax == ddc::BoundCond::HERMITE) {
571 std::array<double, (bsplines_type::degree() / 2 + 1) * (bsplines_type::degree() + 1)>
572 derivs_ptr;
573 std::experimental::mdspan<
574 double,
575 std::experimental::extents<
576 std::size_t,
577 bsplines_type::degree() + 1,
578 bsplines_type::degree() / 2 + 1>> const derivs(derivs_ptr.data());
579
580 ddc::discrete_space<BSplines>().eval_basis_and_n_derivs(
581 derivs,
582 ddc::discrete_space<BSplines>().rmax(),
583 s_nbc_xmax);
584
585 // In order to improve the condition number of the matrix, we normalize
586 // all derivatives by multiplying the i-th derivative by dx^i
587 for (std::size_t i = 0; i < bsplines_type::degree() + 1; ++i) {
588 for (std::size_t j = 1; j < bsplines_type::degree() / 2 + 1; ++j) {
589 derivs(i, j) *= ddc::detail::ipow(m_dx, j);
590 }
591 }
592
593 int const i0 = ddc::discrete_space<BSplines>().nbasis() - s_nbc_xmax;
594 int const j0 = ddc::discrete_space<BSplines>().nbasis() - bsplines_type::degree();
595 for (std::size_t j = 0; j < bsplines_type::degree(); ++j) {
596 for (std::size_t i = 0; i < s_nbc_xmax; ++i) {
597 matrix->set_element(i0 + i, j0 + j, derivs(j + 1, i + s_odd));
598 }
599 }
600 }
601}

604 class ExecSpace,
605 class MemorySpace,

609 ddc::BoundCond BcXmax,

612template <class Layout>
613void SplineBuilder<
614 ExecSpace,
615 MemorySpace,
616 BSplines,
617 InterpolationMesh,
618 BcXmin,
619 BcXmax,
620 Solver,
621 IDimX...>::
622operator()(
623 ddc::ChunkSpan<double, batched_spline_domain_type, Layout, memory_space> spline,
624 ddc::ChunkSpan<double const, batched_interpolation_domain_type, Layout, memory_space> vals,
625 std::optional<ddc::ChunkSpan<
626 double const,
627 batched_derivs_domain_type,
628 Layout,
629 memory_space>> const derivs_xmin,
630 std::optional<ddc::ChunkSpan<
631 double const,
632 batched_derivs_domain_type,
633 Layout,
634 memory_space>> const derivs_xmax) const
635{
636 assert(vals.template extent<interpolation_mesh_type>()
637 == ddc::discrete_space<bsplines_type>().nbasis() - s_nbc_xmin - s_nbc_xmax);
638
639 assert((BcXmin == ddc::BoundCond::HERMITE)
640 != (!derivs_xmin.has_value() || derivs_xmin->template extent<deriv_type>() == 0));
641 assert((BcXmax == ddc::BoundCond::HERMITE)
642 != (!derivs_xmax.has_value() || derivs_xmax->template extent<deriv_type>() == 0));
643 if constexpr (BcXmin == BoundCond::HERMITE) {
644 assert(ddc::DiscreteElement<deriv_type>(derivs_xmin->domain().front()).uid() == 1);
645 }
646 if constexpr (BcXmax == BoundCond::HERMITE) {
647 assert(ddc::DiscreteElement<deriv_type>(derivs_xmax->domain().front()).uid() == 1);
648 }
649
650 // Hermite boundary conditions at xmin, if any
651 // NOTE: For consistency with the linear system, the i-th derivative
652 // provided by the user must be multiplied by dx^i
653 if constexpr (BcXmin == BoundCond::HERMITE) {
654 assert(derivs_xmin->template extent<deriv_type>() == s_nbc_xmin);
655 auto derivs_xmin_values = *derivs_xmin;
656 auto const dx_proxy = m_dx;
657 ddc::parallel_for_each(
658 exec_space(),
659 batch_domain(),
660 KOKKOS_LAMBDA(typename batch_domain_type::discrete_element_type j) {
661 for (int i = s_nbc_xmin; i > 0; --i) {
662 spline(ddc::DiscreteElement<bsplines_type>(s_nbc_xmin - i), j)
663 = derivs_xmin_values(ddc::DiscreteElement<deriv_type>(i), j)
664 * ddc::detail::ipow(dx_proxy, i + s_odd - 1);
665 }
666 });
667 }
668
669 // TODO : Consider optimizing
670 // Fill spline with vals (to work in spline afterward and preserve vals)
671 auto const& offset_proxy = m_offset;
672 auto const& interp_size_proxy = interpolation_domain().extents();
673 auto const& nbasis_proxy = ddc::discrete_space<bsplines_type>().nbasis();
674 ddc::parallel_for_each(
675 exec_space(),
676 batch_domain(),
677 KOKKOS_LAMBDA(typename batch_domain_type::discrete_element_type j) {
678 for (int i = s_nbc_xmin; i < s_nbc_xmin + offset_proxy; ++i) {
679 spline(ddc::DiscreteElement<bsplines_type>(i), j) = 0.0;
680 }
681 for (int i = 0; i < interp_size_proxy; ++i) {
682 spline(ddc::DiscreteElement<bsplines_type>(s_nbc_xmin + i + offset_proxy), j)
683 = vals(ddc::DiscreteElement<interpolation_mesh_type>(i), j);
684 }
685 });
686
687 // Hermite boundary conditions at xmax, if any
688 // NOTE: For consistency with the linear system, the i-th derivative
689 // provided by the user must be multiplied by dx^i
690 if constexpr (BcXmax == BoundCond::HERMITE) {
691 assert(derivs_xmax->template extent<deriv_type>() == s_nbc_xmax);
692 auto derivs_xmax_values = *derivs_xmax;
693 auto const dx_proxy = m_dx;
694 ddc::parallel_for_each(
695 exec_space(),
696 batch_domain(),
697 KOKKOS_LAMBDA(typename batch_domain_type::discrete_element_type j) {
698 for (int i = 0; i < s_nbc_xmax; ++i) {
699 spline(ddc::DiscreteElement<bsplines_type>(nbasis_proxy - s_nbc_xmax - i),
700 j)
701 = derivs_xmax_values(ddc::DiscreteElement<deriv_type>(i + 1), j)
702 * ddc::detail::ipow(dx_proxy, i + s_odd);
703 }
704 });
705 }
706
707 // TODO : Consider optimizing
708 // Allocate and fill a transposed version of spline in order to get dimension of interest as last dimension (optimal for GPU, necessary for Ginkgo)
709 ddc::Chunk spline_tr_alloc(
710 batched_spline_tr_domain(),
711 ddc::KokkosAllocator<double, memory_space>());
712 ddc::ChunkSpan spline_tr = spline_tr_alloc.span_view();
713 ddc::parallel_for_each(
714 exec_space(),
715 batch_domain(),
716 KOKKOS_LAMBDA(typename batch_domain_type::discrete_element_type const j) {
717 for (std::size_t i = 0; i < nbasis_proxy; i++) {
718 spline_tr(ddc::DiscreteElement<bsplines_type>(i), j)
719 = spline(ddc::DiscreteElement<bsplines_type>(i + offset_proxy), j);
720 }
721 });
722 // Create a 2D Kokkos::View to manage spline_tr as a matrix
723 Kokkos::View<double**, Kokkos::LayoutRight, exec_space> bcoef_section(
724 spline_tr.data_handle(),
