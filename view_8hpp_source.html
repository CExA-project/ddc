36/// Note: We use the comma operator to fill the input parameters
37///
38/// If Is=[1, 2], `subspan(s, i0, (Is, all)...)` will be expanded as
39/// `subspan(s, i0, (1, all), (2, all))` which is equivalent to
40/// `subspan(s, i0, all, all)`
41template <
42 class ElementType,
43 class Extents,
44 class Layout,
45 class Accessor,
46 std::size_t I0,
47 std::size_t... Is>
48std::ostream& stream_impl(
49 std::ostream& os,
50 std::experimental::mdspan<ElementType, Extents, Layout, Accessor> const& s,
51 std::index_sequence<I0, Is...>)
52{
53 if constexpr (sizeof...(Is) > 0) {
54 os << '[';
55 for (std::size_t i0 = 0; i0 < s.extent(I0); ++i0) {
56 stream_impl(
57 os,
58 std::experimental::
59 submdspan(s, i0, ((void)Is, std::experimental::full_extent)...),
60 std::make_index_sequence<sizeof...(Is)>());
61 }
62 os << ']';
63 } else {
64 os << '[';
65 for (std::size_t i0 = 0; i0 < s.extent(I0) - 1; ++i0) {
66 os << s(i0) << ',';
67 }
68 os << s(s.extent(I0) - 1) << ']';
69 }
70 return os;
71}
