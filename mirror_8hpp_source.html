14/// Returns a new `Chunk` with the same layout as `src` allocated on the memory space `Space::memory_space`.
15template <class Space, class ElementType, class Support, class Layout, class MemorySpace>
16auto create_mirror(
17 Space const& space,
18 ChunkSpan<ElementType, Support, Layout, MemorySpace> const& src)
19{
20 return Chunk<
21 std::remove_const_t<ElementType>,
22 Support,
23 KokkosAllocator<std::remove_const_t<ElementType>, typename Space::memory_space>>(
24 src.domain());
25}

27/// Returns a new host `Chunk` with the same layout as `src`.
28template <class ElementType, class Support, class Layout, class MemorySpace>
29auto create_mirror(ChunkSpan<ElementType, Support, Layout, MemorySpace> const& src)
30{
31 return create_mirror(Kokkos::DefaultHostExecutionSpace(), src);
32}

34/// Returns a new `Chunk` with the same layout as `src` allocated on the memory space `Space::memory_space` and operates a deep copy between the two.
35template <class Space, class ElementType, class Support, class Layout, class MemorySpace>
36auto create_mirror_and_copy(
37 Space const& space,
38 ChunkSpan<ElementType, Support, Layout, MemorySpace> const& src)
39{
40 Chunk<std::remove_const_t<ElementType>,
41 Support,
42 KokkosAllocator<std::remove_const_t<ElementType>, typename Space::memory_space>>
43 chunk(src.domain());
44 deepcopy(chunk, src);
45 return chunk;
46}

55/// If `src` is accessible from `space` then returns a copy of `src`,
56/// otherwise returns a new `Chunk` with the same layout as `src` allocated on the memory space `Space::memory_space`.
57template <class Space, class ElementType, class Support, class Layout, class MemorySpace>
58auto create_mirror_view(
59 Space const& space,
60 ChunkSpan<ElementType, Support, Layout, MemorySpace> const& src)
61{
62 if constexpr (Kokkos::SpaceAccessibility<Space, MemorySpace>::accessible) {
63 return src;
64 } else {
65 return create_mirror(space, src);
66 }
67}
