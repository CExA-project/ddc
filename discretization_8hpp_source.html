23 template <class Tuple, std::size_t... Ids>
24 auto extract_after(Tuple&& t, std::index_sequence<Ids...>)
25 {
26  return std::make_tuple(std::move(std::get<Ids + 1>(t))...);
27 }

31 template <class D, class Arg>
32 Arg init_discretization(std::tuple<D, Arg>&& a)
33 {
34  if (detail::Discretization<std::remove_cv_t<std::remove_reference_t<D>>>::s_disc) {
35  throw std::runtime_error("Discretization function already initialized.");
36  }
37  detail::Discretization<std::remove_cv_t<std::remove_reference_t<D>>>::s_disc.emplace(
38  std::move(std::get<0>(a)));
39  return std::get<1>(a);
40 }
41 
42 template <class D, class... Args>
43 std::enable_if_t<2 <= sizeof...(Args), std::tuple<Args...>> init_discretization(
44  std::tuple<D, Args...>&& a)
45 {
46  if (detail::Discretization<std::remove_cv_t<std::remove_reference_t<D>>>::s_disc) {
47  throw std::runtime_error("Discretization function already initialized.");
48  }
49  detail::Discretization<std::remove_cv_t<std::remove_reference_t<D>>>::s_disc.emplace(
50  std::move(std::get<0>(a)));
51  return detail::extract_after(std::move(a), std::index_sequence_for<Args...>());
52 }
53 
54 template <class D, class... Args>
55 void init_discretization(Args&&... a)
56 {
57  if (detail::Discretization<std::remove_cv_t<std::remove_reference_t<D>>>::s_disc) {
58  throw std::runtime_error("Discretization function already initialized.");
59  }
60  detail::Discretization<std::remove_cv_t<std::remove_reference_t<D>>>::s_disc.emplace(
61  std::forward<Args>(a)...);
62 }
