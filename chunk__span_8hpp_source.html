93 template <class QueryDDim, class... ODDims>
94 auto get_slicer_for(DiscreteElement<ODDims...> const& c) const
95 {
96 DDC_IF_NVCC_THEN_PUSH_AND_SUPPRESS(implicit_return_from_non_void_function)
97 if constexpr (in_tags_v<QueryDDim, detail::TypeSeq<ODDims...>>) {
98 return (uid<QueryDDim>(c) - front<QueryDDim>(this->m_domain).uid());
99 } else {
100 return std::experimental::full_extent;
101 }
102 DDC_IF_NVCC_THEN_POP
103 }
104
105 template <class QueryDDim, class... ODDims>
106 auto get_slicer_for(DiscreteDomain<ODDims...> const& c) const
107 {
108 DDC_IF_NVCC_THEN_PUSH_AND_SUPPRESS(implicit_return_from_non_void_function)
109 if constexpr (in_tags_v<QueryDDim, detail::TypeSeq<ODDims...>>) {
110 return std::pair<std::size_t, std::size_t>(
111 front<QueryDDim>(c) - front<QueryDDim>(this->m_domain),
112 back<QueryDDim>(c) + 1 - front<QueryDDim>(this->m_domain));
113 } else {
114 return std::experimental::full_extent;
115 }
116 DDC_IF_NVCC_THEN_POP
117 }

177 /** Constructs a new ChunkSpan from scratch
178 * @param allocation_mdspan the allocation mdspan to the data
179 * @param domain the domain that sustains the view
180 */
181 constexpr ChunkSpan(allocation_mdspan_type allocation_mdspan, mdomain_type const& domain)
182 {
183 namespace stdex = std::experimental;
184 extents_type extents_s((front<DDims>(domain) + extents<DDims>(domain)).uid()...);
185 std::array<std::size_t, sizeof...(DDims)> strides_s {allocation_mdspan.mapping().stride(
186 type_seq_rank_v<DDims, detail::TypeSeq<DDims...>>)...};
187 stdex::layout_stride::mapping<extents_type> mapping_s(extents_s, strides_s);
188 this->m_internal_mdspan = internal_mdspan_type(
189 allocation_mdspan.data() - mapping_s(front<DDims>(domain).uid()...),
190 mapping_s);
191 this->m_domain = domain;
192 }

218 /** Slice out some dimensions
219 */
220 template <class... QueryDDims>
221 constexpr auto operator[](DiscreteElement<QueryDDims...> const& slice_spec) const
222 {
223 auto subview = std::experimental::
224 submdspan(allocation_mdspan(), get_slicer_for<DDims>(slice_spec)...);
225 using detail::TypeSeq;
226 using selected_meshes = type_seq_remove_t<TypeSeq<DDims...>, TypeSeq<QueryDDims...>>;
227 return ChunkSpan<
228 ElementType,
229 decltype(select_by_type_seq<selected_meshes>(this->m_domain)),
230 typename decltype(subview)::layout_type,
231 memory_space>(subview, select_by_type_seq<selected_meshes>(this->m_domain));
232 }
233
234 /** Slice out some dimensions
235 */
236 template <class... QueryDDims>
237 constexpr auto operator[](DiscreteDomain<QueryDDims...> const& odomain) const
238 {
239 auto subview = std::experimental::
240 submdspan(allocation_mdspan(), get_slicer_for<DDims>(odomain)...);
241 return ChunkSpan<
242 ElementType,
243 decltype(this->m_domain.restrict(odomain)),
244 typename decltype(subview)::layout_type,
245 memory_space>(subview, this->m_domain.restrict(odomain));
246 }

248 /** Element access using a list of DiscreteElement
249 * @param delems 1D discrete elements
250 * @return reference to this element
251 */
252 template <class... ODDims>
253 constexpr reference operator()(DiscreteElement<ODDims> const&... delems) const noexcept
254 {
255 static_assert(sizeof...(ODDims) == sizeof...(DDims), "Invalid number of dimensions");
256 assert(((delems >= front<ODDims>(this->m_domain)) && ...));
257 assert(((delems <= back<ODDims>(this->m_domain)) && ...));
258 return this->m_internal_mdspan(uid(take<DDims>(delems...))...);
259 }
260
261 /** Element access using a multi-dimensional DiscreteElement
262 * @param delems discrete elements
263 * @return reference to this element
264 */
265 template <class... ODDims, class = std::enable_if_t<sizeof...(ODDims) != 1>>
266 constexpr reference operator()(DiscreteElement<ODDims...> const& delems) const noexcept
267 {
268 static_assert(sizeof...(ODDims) == sizeof...(DDims), "Invalid number of dimensions");
269 assert(((select<ODDims>(delems) >= front<ODDims>(this->m_domain)) && ...));
270 assert(((select<ODDims>(delems) <= back<ODDims>(this->m_domain)) && ...));
271 return this->m_internal_mdspan(uid<DDims>(delems)...);
272 }

290 /** Provide a mdspan on the memory allocation
291 * @return allocation mdspan
292 */
293 constexpr auto allocation_kokkos_view() const
294 {
295 auto s = this->allocation_mdspan();
296 auto kokkos_layout = detail::build_kokkos_layout(
297 s.extents(),
298 s.mapping(),
299 std::make_index_sequence<sizeof...(DDims)> {});
300 return Kokkos::View<
301 detail::mdspan_to_kokkos_element_t<ElementType, sizeof...(DDims)>,
302 decltype(kokkos_layout),
303 MemorySpace>(s.data(), kokkos_layout);
304 }
