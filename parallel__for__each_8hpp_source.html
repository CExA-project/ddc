25template <class F, class... DDims>
26class ForEachKokkosLambdaAdapter
27{
28 template <class T>
29 using index_type = std::size_t;
30
31 F m_f;
32
33public:
34 explicit ForEachKokkosLambdaAdapter(F const& f) : m_f(f) {}
35
36 template <std::size_t N = sizeof...(DDims), std::enable_if_t<(N == 0), bool> = true>
37 void operator()([[maybe_unused]] index_type<void> unused_id) const
38 {
39 m_f(DiscreteElement<>());
40 }
41
42 template <std::size_t N = sizeof...(DDims), std::enable_if_t<(N == 0), bool> = true>
43 KOKKOS_FUNCTION void operator()(
44 use_annotated_operator,
45 [[maybe_unused]] index_type<void> unused_id) const
46 {
47 m_f(DiscreteElement<>());
48 }
49
50 template <std::size_t N = sizeof...(DDims), std::enable_if_t<(N > 0), bool> = true>
51 void operator()(index_type<DDims>... ids) const
52 {
53 m_f(DiscreteElement<DDims...>(ids...));
54 }
55
56 template <std::size_t N = sizeof...(DDims), std::enable_if_t<(N > 0), bool> = true>
57 KOKKOS_FUNCTION void operator()(use_annotated_operator, index_type<DDims>... ids) const
58 {
59 m_f(DiscreteElement<DDims...>(ids...));
60 }
61};

63template <class ExecSpace, class Functor>
64void for_each_kokkos(
65 std::string const& label,
66 ExecSpace const& execution_space,
67 [[maybe_unused]] DiscreteDomain<> const& domain,
68 Functor const& f) noexcept
69{
70 if constexpr (need_annotated_operator<ExecSpace>()) {
71 Kokkos::parallel_for(
72 label,
73 Kokkos::RangePolicy<ExecSpace, use_annotated_operator>(execution_space, 0, 1),
74 ForEachKokkosLambdaAdapter<Functor>(f));
75 } else {
76 Kokkos::parallel_for(
77 label,
78 Kokkos::RangePolicy<ExecSpace>(execution_space, 0, 1),
79 ForEachKokkosLambdaAdapter<Functor>(f));
80 }
81}

83template <class ExecSpace, class Functor, class DDim0>
84void for_each_kokkos(
85 std::string const& label,
86 ExecSpace const& execution_space,
87 DiscreteDomain<DDim0> const& domain,
88 Functor const& f) noexcept
89{
90 DiscreteElement<DDim0> const ddc_begin = domain.front();
91 DiscreteElement<DDim0> const ddc_end = domain.front() + domain.extents();
92 std::size_t const begin = ddc::uid<DDim0>(ddc_begin);
93 std::size_t const end = ddc::uid<DDim0>(ddc_end);
94 if constexpr (need_annotated_operator<ExecSpace>()) {
95 Kokkos::parallel_for(
96 label,
97 Kokkos::RangePolicy<ExecSpace, use_annotated_operator>(execution_space, begin, end),
98 ForEachKokkosLambdaAdapter<Functor, DDim0>(f));
99 } else {
100 Kokkos::parallel_for(
101 label,
102 Kokkos::RangePolicy<ExecSpace>(execution_space, begin, end),
103 ForEachKokkosLambdaAdapter<Functor, DDim0>(f));
104 }
105}

107template <class ExecSpace, class Functor, class DDim0, class DDim1, class... DDims>
108void for_each_kokkos(
109 std::string const& label,
110 ExecSpace const& execution_space,
111 DiscreteDomain<DDim0, DDim1, DDims...> const& domain,
112 Functor const& f) noexcept
113{
114 DiscreteElement<DDim0, DDim1, DDims...> const ddc_begin = domain.front();
115 DiscreteElement<DDim0, DDim1, DDims...> const ddc_end = domain.front() + domain.extents();
116 Kokkos::Array<std::size_t, 2 + sizeof...(DDims)> const
117 begin {ddc::uid<DDim0>(ddc_begin),
118 ddc::uid<DDim1>(ddc_begin),
119 ddc::uid<DDims>(ddc_begin)...};
120 Kokkos::Array<std::size_t, 2 + sizeof...(DDims)> const
121 end {ddc::uid<DDim0>(ddc_end), ddc::uid<DDim1>(ddc_end), ddc::uid<DDims>(ddc_end)...};
122 if constexpr (need_annotated_operator<ExecSpace>()) {
123 Kokkos::parallel_for(
124 label,
125 Kokkos::MDRangePolicy<
126 ExecSpace,
127 Kokkos::Rank<
128 2 + sizeof...(DDims),
129 Kokkos::Iterate::Right,
130 Kokkos::Iterate::Right>,
131 use_annotated_operator>(execution_space, begin, end),
132 ForEachKokkosLambdaAdapter<Functor, DDim0, DDim1, DDims...>(f));
133 } else {
134 Kokkos::parallel_for(
135 label,
136 Kokkos::MDRangePolicy<
137 ExecSpace,
138 Kokkos::Rank<
139 2 + sizeof...(DDims),
140 Kokkos::Iterate::Right,
141 Kokkos::Iterate::Right>>(execution_space, begin, end),
142 ForEachKokkosLambdaAdapter<Functor, DDim0, DDim1, DDims...>(f));
143 }
144}
