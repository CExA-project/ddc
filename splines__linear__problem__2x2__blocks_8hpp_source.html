18/**
19 * @brief A 2x2-blocks linear problem dedicated to the computation of a spline approximation,
20 * with all blocks except top-left one being stored in dense format.
21 *
22 * A = | Q | gamma |
23 * | lambda | delta |
24 *
25 * The storage format is dense row-major for top-left, top-right and bottom-left blocks, and determined by
26 * its type for the top-left block.
27 *
28 * This class implements a Schur complement method to perform a block-LU factorization and solve,
29 * calling top-left block and bottom-right block setup_solver() and solve() methods for internal operations.
30 *
31 * @tparam ExecSpace The Kokkos::ExecutionSpace on which operations related to the matrix are supposed to be performed.
32 */
33template <class ExecSpace>
34class SplinesLinearProblem2x2Blocks : public SplinesLinearProblem<ExecSpace>
35{
36public:
37 using typename SplinesLinearProblem<ExecSpace>::MultiRHS;
38 using SplinesLinearProblem<ExecSpace>::size;
39
40 /**
41 * @brief COO storage.
42 *
43 * [SHOULD BE PRIVATE (GPU programming limitation)]
44 */
45 struct Coo
46 {
47 std::size_t m_nrows;
48 std::size_t m_ncols;
49 Kokkos::View<int*, Kokkos::LayoutRight, typename ExecSpace::memory_space> m_rows_idx;
50 Kokkos::View<int*, Kokkos::LayoutRight, typename ExecSpace::memory_space> m_cols_idx;
51 Kokkos::View<double*, Kokkos::LayoutRight, typename ExecSpace::memory_space> m_values;
52
53 Coo() : m_nrows(0), m_ncols(0) {}
54
55 Coo(std::size_t const nrows_,
56 std::size_t const ncols_,
57 Kokkos::View<int*, Kokkos::LayoutRight, typename ExecSpace::memory_space> rows_idx_,
58 Kokkos::View<int*, Kokkos::LayoutRight, typename ExecSpace::memory_space> cols_idx_,
59 Kokkos::View<double*, Kokkos::LayoutRight, typename ExecSpace::memory_space> values_)
60 : m_nrows(nrows_)
61 , m_ncols(ncols_)
62 , m_rows_idx(std::move(rows_idx_))
63 , m_cols_idx(std::move(cols_idx_))
64 , m_values(std::move(values_))
65 {
66 assert(m_rows_idx.extent(0) == m_cols_idx.extent(0));
67 assert(m_rows_idx.extent(0) == m_values.extent(0));
68 }

104protected:
105 std::unique_ptr<SplinesLinearProblem<ExecSpace>> m_top_left_block;
106 Kokkos::DualView<double**, Kokkos::LayoutRight, typename ExecSpace::memory_space>
107 m_top_right_block;
108 Coo m_top_right_block_coo;
109 Kokkos::DualView<double**, Kokkos::LayoutRight, typename ExecSpace::memory_space>
110 m_bottom_left_block;
111 Coo m_bottom_left_block_coo;
112 std::unique_ptr<SplinesLinearProblem<ExecSpace>> m_bottom_right_block;
113
114public:
115 /**
116 * @brief SplinesLinearProblem2x2Blocks constructor.
117 *
118 * @param mat_size The size of one of the dimensions of the square matrix.
119 * @param top_left_block A pointer toward the top-left SplinesLinearProblem. `setup_solver` must not have been called on it.
120 */
121 explicit SplinesLinearProblem2x2Blocks(
122 std::size_t const mat_size,
123 std::unique_ptr<SplinesLinearProblem<ExecSpace>> top_left_block)
124 : SplinesLinearProblem<ExecSpace>(mat_size)
125 , m_top_left_block(std::move(top_left_block))
126 , m_top_right_block(
127 "top_right_block",
128 m_top_left_block->size(),
129 mat_size - m_top_left_block->size())
130 , m_bottom_left_block(
131 "bottom_left_block",
132 mat_size - m_top_left_block->size(),
133 m_top_left_block->size())
134 , m_bottom_right_block(
135 new SplinesLinearProblemDense<ExecSpace>(mat_size - m_top_left_block->size()))
136 {
137 assert(m_top_left_block->size() <= mat_size);
138
139 Kokkos::deep_copy(m_top_right_block.h_view, 0.);
140 Kokkos::deep_copy(m_bottom_left_block.h_view, 0.);
141 }
142
143 double get_element(std::size_t const i, std::size_t const j) const override
144 {
145 assert(i < size());
146 assert(j < size());
147
148 std::size_t const nq = m_top_left_block->size();
149 if (i < nq && j < nq) {
150 return m_top_left_block->get_element(i, j);
151 } else if (i >= nq && j >= nq) {
152 return m_bottom_right_block->get_element(i - nq, j - nq);
153 } else if (j >= nq) {
154 return m_top_right_block.h_view(i, j - nq);
155 } else {
156 return m_bottom_left_block.h_view(i - nq, j);
157 }
158 }
159
160 void set_element(std::size_t const i, std::size_t const j, double const aij) override
161 {
162 assert(i < size());
163 assert(j < size());
164
165 std::size_t const nq = m_top_left_block->size();
166 if (i < nq && j < nq) {
167 m_top_left_block->set_element(i, j, aij);
168 } else if (i >= nq && j >= nq) {
169 m_bottom_right_block->set_element(i - nq, j - nq, aij);
170 } else if (j >= nq) {
171 m_top_right_block.h_view(i, j - nq) = aij;
172 } else {
173 m_bottom_left_block.h_view(i - nq, j) = aij;
174 }
175 }

177 /**
178 * @brief Fill a COO version of a Dense matrix (remove zeros).
179 *
180 * [SHOULD BE PRIVATE (GPU programming limitation)]
181 *
182 * Runs on a single thread to garantee ordering.
183 *
184 * @param[in] dense_matrix The dense storage matrix whose non-zeros are extracted to fill the COO matrix.
185 * @param[in] tol The tolerancy applied to filter the non-zeros.
186 *
187 * @return The COO storage matrix filled with the non-zeros from dense_matrix.
188 */
189 Coo dense2coo(
190 Kokkos::View<const double**, Kokkos::LayoutRight, typename ExecSpace::memory_space>
191 dense_matrix,
192 double const tol = 1e-14)
193 {
194 Kokkos::View<int*, Kokkos::LayoutRight, typename ExecSpace::memory_space> rows_idx(
195 "ddc_splines_coo_rows_idx",
196 dense_matrix.extent(0) * dense_matrix.extent(1));
197 Kokkos::View<int*, Kokkos::LayoutRight, typename ExecSpace::memory_space> cols_idx(
198 "ddc_splines_coo_cols_idx",
199 dense_matrix.extent(0) * dense_matrix.extent(1));
200 Kokkos::View<double*, Kokkos::LayoutRight, typename ExecSpace::memory_space>
201 values("ddc_splines_coo_values", dense_matrix.extent(0) * dense_matrix.extent(1));
202
203 Kokkos::DualView<std::size_t, Kokkos::LayoutRight, typename ExecSpace::memory_space>
204 n_nonzeros("ddc_splines_n_nonzeros");
205 n_nonzeros.h_view() = 0;
206 n_nonzeros.modify_host();
207 n_nonzeros.sync_device();
208
209 Kokkos::parallel_for(
210 "dense2coo",
211 Kokkos::RangePolicy(ExecSpace(), 0, 1),
212 KOKKOS_LAMBDA(const int) {
213 for (int i = 0; i < dense_matrix.extent(0); i++) {
214 for (int j = 0; j < dense_matrix.extent(1); j++) {
215 double const aij = dense_matrix(i, j);
216 if (Kokkos::abs(aij) >= tol) {
217 rows_idx(n_nonzeros.d_view()) = i;
218 cols_idx(n_nonzeros.d_view()) = j;
219 values(n_nonzeros.d_view()) = aij;
220 n_nonzeros.d_view()++;
221 }
222 }
223 }
224 });
225 n_nonzeros.modify_device();
226 n_nonzeros.sync_host();
227 Kokkos::resize(rows_idx, n_nonzeros.h_view());
228 Kokkos::resize(cols_idx, n_nonzeros.h_view());
229 Kokkos::resize(values, n_nonzeros.h_view());
230
231 return Coo(dense_matrix.extent(0), dense_matrix.extent(1), rows_idx, cols_idx, values);
232 }
233
234private:
235 /// @brief Compute the Schur complement delta - lambda*Q^-1*gamma.
236 void compute_schur_complement()
237 {
238 Kokkos::parallel_for(
239 "compute_schur_complement",
240 Kokkos::MDRangePolicy<Kokkos::DefaultHostExecutionSpace, Kokkos::Rank<2>>(
241 {0, 0},
242 {m_bottom_right_block->size(), m_bottom_right_block->size()}),
243 [&](const int i, const int j) {
244 double val = 0.0;
245 for (int l = 0; l < m_top_left_block->size(); ++l) {
246 val += m_bottom_left_block.h_view(i, l) * m_top_right_block.h_view(l, j);
247 }
248 m_bottom_right_block
249 ->set_element(i, j, m_bottom_right_block->get_element(i, j) - val);
250 });
251 }
252
253public:
254 /**
255 * @brief Perform a pre-process operation on the solver. Must be called after filling the matrix.
256 *
257 * Block-LU factorize the matrix A according to the Schur complement method. The block-LU factorization is:
258 *
259 * A = | Q | 0 | | I | Q^-1*gamma |
260 * | lambda | delta - lambda*Q^-1*gamma | | 0 | I |
261 *
262 * So we perform the factorization inplace to store only the relevant blocks in the matrix (while factorizing
263 * the blocks themselves if necessary):
264 *
265 * | Q | Q^-1*gamma |
266 * | lambda | delta - lambda*Q^-1*gamma |
267 */
268 void setup_solver() override
269 {
270 // Setup the top-left solver
271 m_top_left_block->setup_solver();
272
273 // Compute Q^-1*gamma in top-right block
274 m_top_right_block.modify_host();
275 m_top_right_block.sync_device();
276 m_top_left_block->solve(m_top_right_block.d_view);
277 m_top_right_block_coo = dense2coo(m_top_right_block.d_view);
278 m_top_right_block.modify_device();
279 m_top_right_block.sync_host();
280
281 // Push lambda on device in bottom-left block
282 m_bottom_left_block.modify_host();
283 m_bottom_left_block.sync_device();
284 m_bottom_left_block_coo = dense2coo(m_bottom_left_block.d_view);
285
286 // Compute delta - lambda*Q^-1*gamma in bottom-right block & setup the bottom-right solver
287 compute_schur_complement();
288 m_bottom_right_block->setup_solver();
289 }

291 /**
292 * @brief Compute y <- y - LinOp*x or y <- y - LinOp^t*x with a sparse LinOp.
293 *
294 * [SHOULD BE PRIVATE (GPU programming limitation)]
295 *
296 * Perform a spdm operation (sparse-dense matrix multiplication) with parameters alpha=-1 and beta=1 between
297 * a sparse matrix stored in COO format and a dense matrix x.
298 *
299 * @param[in] LinOp The sparse matrix, left side of the matrix multiplication.
300 * @param[in] x The dense matrix, right side of the matrix multiplication.
301 * @param[inout] y The dense matrix to be altered by the operation.
302 * @param transpose A flag to indicate if the direct or transposed version of the operation is performed.
303 */
304 void spdm_minus1_1(Coo LinOp, MultiRHS const x, MultiRHS const y, bool const transpose = false)
305 const
306 {
307 assert((!transpose && LinOp.nrows() == y.extent(0))
308 || (transpose && LinOp.ncols() == y.extent(0)));
309 assert((!transpose && LinOp.ncols() == x.extent(0))
310 || (transpose && LinOp.nrows() == x.extent(0)));
311 assert(x.extent(1) == y.extent(1));
312
313 if (!transpose) {
314 Kokkos::parallel_for(
315 "ddc_splines_spdm_minus1_1",
316 Kokkos::RangePolicy(ExecSpace(), 0, y.extent(1)),
317 KOKKOS_LAMBDA(const int j) {
318 for (int nz_idx = 0; nz_idx < LinOp.nnz(); ++nz_idx) {
319 const int i = LinOp.rows_idx()(nz_idx);
320 const int k = LinOp.cols_idx()(nz_idx);
321 y(i, j) -= LinOp.values()(nz_idx) * x(k, j);
322 }
323 });
324 } else {
325 Kokkos::parallel_for(
326 "ddc_splines_spdm_minus1_1_tr",
327 Kokkos::RangePolicy(ExecSpace(), 0, y.extent(1)),
328 KOKKOS_LAMBDA(const int j) {
329 for (int nz_idx = 0; nz_idx < LinOp.nnz(); ++nz_idx) {
330 const int i = LinOp.rows_idx()(nz_idx);
331 const int k = LinOp.cols_idx()(nz_idx);
332 y(k, j) -= LinOp.values()(nz_idx) * x(i, j);
333 }
334 });
335 }
336 }

338 /**
339 * @brief Solve the multiple right-hand sides linear problem Ax=b or its transposed version A^tx=b inplace.
340 *
341 * The solver method is the one known as Schur complement method. It can be summarized as follow,
342 * starting with the pre-computed elements of the matrix:
343 *
344 * | Q | Q^-1*gamma |
345 * | lambda | delta - lambda*Q^-1*gamma |
346 *
347 * For the non-transposed case:
348 * - Solve inplace Q * x'1 = b1 (using the solver internal to Q).
349 * - Compute inplace b'2 = b2 - lambda*x'1.
350 * - Solve inplace (delta - lambda*Q^-1*gamma) * x2 = b'2.
351 * - Compute inplace x1 = x'1 - (delta - lambda*Q^-1*gamma)*x2.
352 *
353 * @param[in, out] b A 2D Kokkos::View storing the multiple right-hand sides of the problem and receiving the corresponding solution.
354 * @param transpose Choose between the direct or transposed version of the linear problem.
355 */
356 void solve(MultiRHS const b, bool const transpose) const override
357 {
358 assert(b.extent(0) == size());
359
360 MultiRHS const b1 = Kokkos::
361 subview(b,
362 std::pair<std::size_t, std::size_t>(0, m_top_left_block->size()),
363 Kokkos::ALL);
364 MultiRHS const b2 = Kokkos::
365 subview(b,
366 std::pair<std::size_t, std::size_t>(m_top_left_block->size(), b.extent(0)),
367 Kokkos::ALL);
368 if (!transpose) {
369 m_top_left_block->solve(b1);
370 spdm_minus1_1(m_bottom_left_block_coo, b1, b2);
371 m_bottom_right_block->solve(b2);
372 spdm_minus1_1(m_top_right_block_coo, b2, b1);
373 } else {
374 spdm_minus1_1(m_top_right_block_coo, b1, b2, true);
375 m_bottom_right_block->solve(b2, true);
376 spdm_minus1_1(m_bottom_left_block_coo, b2, b1, true);
377 m_top_left_block->solve(b1, true);
378 }
379 }
