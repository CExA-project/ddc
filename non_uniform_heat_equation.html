18std::vector<double> generate_random_vector(
19 int n,
20 double lower_bound,
21 double higher_bound)
22{
23 std::random_device rd;
24 std::mt19937 gen(rd());
25 std::uniform_real_distribution<double>
26 dis(lower_bound, higher_bound);
27
28 std::vector<double> vec(n);
29 vec[0] = lower_bound;
30 vec[n - 1] = higher_bound;
31
32 for (int i = 1; i < vec.size() - 1; ++i) {
33 vec[i] = dis(gen);
34 }
35
36 std::sort(vec.begin(), vec.end());
37 return vec;
38}

72template <class ChunkType>
73void display(double time, ChunkType temp)
74{
75 double const mean_temp = ddc::transform_reduce(
76 temp.domain(),
77 0.,
78 ddc::reducer::sum<double>(),
79 temp)
80 / temp.domain().size();
81 std::cout << std::fixed << std::setprecision(3);
82 std::cout << "At t = " << time << ",\n";
83 std::cout << " * mean temperature = " << mean_temp << "\n";
84 ddc::ChunkSpan temp_slice
85 = temp[ddc::get_domain<DDimY>(temp).front()
86 + ddc::get_domain<DDimY>(temp).size() / 2];
87 std::cout << " * temperature[y:"
88 << ddc::get_domain<DDimY>(temp).size() / 2 << "] = {";
89 ddc::for_each(
90 ddc::get_domain<DDimX>(temp),
91 [=](ddc::DiscreteElement<DDimX> const ix) {
92 std::cout << std::setw(6) << temp_slice(ix);
93 });
94 std::cout << " }" << std::endl;
95
96#ifdef DDC_BUILD_PDI_WRAPPER
97 ddc::PdiEvent("display")
98 .with("temp", temp)
99 .and_with("mean_temp", mean_temp)
100 .and_with("temp_slice", temp_slice);
101#endif
102}

108int main(int argc, char** argv)
109{
110#ifdef DDC_BUILD_PDI_WRAPPER
111 auto pdi_conf = PC_parse_string("");
112 PDI_init(pdi_conf);
113#endif
114 Kokkos::ScopeGuard const kokkos_scope(argc, argv);
115 ddc::ScopeGuard const ddc_scope(argc, argv);
116
117
119 double const x_start = -1.;
120 double const x_end = 1.;
121 std::size_t const nb_x_points = 10;
122 double const kx = .01;
125 double const y_start = -1.;
126 double const y_end = 1.;
127 std::size_t const nb_y_points = 100;
128 double const ky = .002;
131 double const start_time = 0.;
132 double const end_time = 10.;
134 std::ptrdiff_t const t_output_period = 10;
136
138 std::vector<double> x_domain_vect
139 = generate_random_vector(nb_x_points, x_start, x_end);
141
142 std::size_t size_x = x_domain_vect.size();
143
145 std::vector<double> x_pre_ghost_vect {
146 x_domain_vect.front()
147 - (x_domain_vect.back() - x_domain_vect[size_x - 2])};
148
149 std::vector<double> x_post_ghost_vect {
150 x_domain_vect.back()
151 + (x_domain_vect[1] - x_domain_vect.front())};
153
155 auto const [x_domain, ghosted_x_domain, x_pre_ghost, x_post_ghost]
156 = ddc::init_discrete_space<DDimX>(DDimX::init_ghosted<DDimX>(
157 x_domain_vect,
158 x_pre_ghost_vect,
159 x_post_ghost_vect));
161
162 ddc::DiscreteDomain<DDimX> const
163 x_domain_begin(x_domain.front(), x_post_ghost.extents());
164 ddc::DiscreteDomain<DDimX> const x_domain_end(
165 x_domain.back() - x_pre_ghost.extents() + 1,
166 x_pre_ghost.extents());
167
169 std::vector<double> y_domain_vect
170 = generate_random_vector(nb_y_points, y_start, y_end);
171
172 std::size_t size_y = y_domain_vect.size();
173
175 std::vector<double> y_pre_ghost_vect {
176 y_domain_vect.front()
177 - (y_domain_vect.back() - y_domain_vect[size_y - 2])};
178 std::vector<double> y_post_ghost_vect {
179 y_domain_vect.back()
180 + (y_domain_vect[1] - y_domain_vect.front())};
182
184
186 auto const [y_domain, ghosted_y_domain, y_pre_ghost, y_post_ghost]
187 = ddc::init_discrete_space<DDimY>(DDimY::init_ghosted<DDimY>(
188 y_domain_vect,
189 y_pre_ghost_vect,
190 y_post_ghost_vect));
192
193 ddc::DiscreteDomain<DDimY> const
194 y_domain_begin(y_domain.front(), y_post_ghost.extents());
195
196 ddc::DiscreteDomain<DDimY> const y_domain_end(
197 y_domain.back() - y_pre_ghost.extents() + 1,
198 y_pre_ghost.extents());
199
201
202 double const invdx2_max = ddc::transform_reduce(
203 x_domain,
204 0.,
205 ddc::reducer::max<double>(),
206 [](ddc::DiscreteElement<DDimX> ix) {
207 return 1.
208 / (ddc::distance_at_left(ix)
209 * ddc::distance_at_right(ix));
210 });
211
212 double const invdy2_max = ddc::transform_reduce(
213 y_domain,
214 0.,
215 ddc::reducer::max<double>(),
216 [](ddc::DiscreteElement<DDimY> iy) {
217 return 1.
218 / (ddc::distance_at_left(iy)
219 * ddc::distance_at_right(iy));
220 });
221
222 ddc::Coordinate<T> const max_dt {
223 .5 / (kx * invdx2_max + ky * invdy2_max)};
224
226
228 ddc::DiscreteVector<DDimT> const nb_time_steps(
229 std::ceil((end_time - start_time) / max_dt) + .2);
230
231 ddc::DiscreteDomain<DDimT> const time_domain
232 = ddc::init_discrete_space<DDimT>(DDimT::init<DDimT>(
233 ddc::Coordinate<T>(start_time),
234 ddc::Coordinate<T>(end_time),
235 nb_time_steps + 1));
236
238
240 ddc::Chunk ghosted_last_temp(
241 "ghosted_last_temp",
242 ddc::DiscreteDomain<
243 DDimX,
244 DDimY>(ghosted_x_domain, ghosted_y_domain),
245 ddc::DeviceAllocator<double>());
246
247 ddc::Chunk ghosted_next_temp(
248 "ghosted_next_temp",
249 ddc::DiscreteDomain<
250 DDimX,
251 DDimY>(ghosted_x_domain, ghosted_y_domain),
252 ddc::DeviceAllocator<double>());
254
256 ddc::ChunkSpan const ghosted_initial_temp
257 = ghosted_last_temp.span_view();
259
261 ddc::parallel_for_each(
262 ddc::DiscreteDomain<DDimX, DDimY>(x_domain, y_domain),
263 KOKKOS_LAMBDA(ddc::DiscreteElement<DDimX, DDimY> const ixy) {
264 double const x = ddc::coordinate(
265 ddc::DiscreteElement<DDimX>(ixy));
266 double const y = ddc::coordinate(
267 ddc::DiscreteElement<DDimY>(ixy));
268 ghosted_initial_temp(ixy)
269 = 9.999 * ((x * x + y * y) < 0.25);
270 });
272
274 ddc::Chunk ghosted_temp(
275 "ghost_temp",
276 ddc::DiscreteDomain<
277 DDimX,
278 DDimY>(ghosted_x_domain, ghosted_y_domain),
279 ddc::HostAllocator<double>());
281
283 ddc::parallel_deepcopy(ghosted_temp, ghosted_last_temp);
285
287 display(ddc::coordinate(time_domain.front()),
288 ghosted_temp[x_domain][y_domain]);
290
292 ddc::DiscreteElement<DDimT> last_output_iter = time_domain.front();
294
296 for (ddc::DiscreteElement<DDimT> const iter :
297 time_domain.remove_first(ddc::DiscreteVector<DDimT>(1))) {
299
301 ddc::parallel_deepcopy(
302 ghosted_last_temp[ddc::DiscreteDomain<
303 DDimX,
304 DDimY>(x_pre_ghost, y_domain)],
305 ghosted_last_temp[ddc::DiscreteDomain<
306 DDimX,
307 DDimY>(y_domain, x_domain_end)]);
308 ddc::parallel_deepcopy(
309 ghosted_last_temp[ddc::DiscreteDomain<
310 DDimX,
311 DDimY>(y_domain, x_post_ghost)],
312 ghosted_last_temp[ddc::DiscreteDomain<
313 DDimX,
314 DDimY>(y_domain, x_domain_begin)]);
315 ddc::parallel_deepcopy(
316 ghosted_last_temp[ddc::DiscreteDomain<
317 DDimX,
318 DDimY>(x_domain, y_pre_ghost)],
319 ghosted_last_temp[ddc::DiscreteDomain<
320 DDimX,
321 DDimY>(x_domain, y_domain_end)]);
322 ddc::parallel_deepcopy(
323 ghosted_last_temp[ddc::DiscreteDomain<
324 DDimX,
325 DDimY>(x_domain, y_post_ghost)],
326 ghosted_last_temp[ddc::DiscreteDomain<
327 DDimX,
328 DDimY>(x_domain, y_domain_begin)]);
330
332 ddc::ChunkSpan const next_temp(
333 ghosted_next_temp[ddc::DiscreteDomain<
334 DDimX,
335 DDimY>(x_domain, y_domain)]);
336 ddc::ChunkSpan const last_temp(ghosted_last_temp.span_cview());
338
340 ddc::parallel_for_each(
341 next_temp.domain(),
342 KOKKOS_LAMBDA(
343 ddc::DiscreteElement<DDimX, DDimY> const ixy) {
344 ddc::DiscreteElement<DDimX> const ix
345 = ddc::select<DDimX>(ixy);
346 ddc::DiscreteElement<DDimY> const iy
347 = ddc::select<DDimY>(ixy);
348 double const dx_l = ddc::distance_at_left(ix);
349 double const dx_r = ddc::distance_at_right(ix);
350 double const dx_m = 0.5 * (dx_l + dx_r);
351 double const dy_l = ddc::distance_at_left(iy);
352 double const dy_r = ddc::distance_at_right(iy);
353 double const dy_m = 0.5 * (dy_l + dy_r);
354 next_temp(ix, iy) = last_temp(ix, iy);
355 next_temp(ix, iy)
356 += kx * ddc::step<DDimT>()
357 * (dx_l * last_temp(ix + 1, iy)
358 - 2.0 * dx_m * last_temp(ix, iy)
359 + dx_r * last_temp(ix - 1, iy))
360 / (dx_l * dx_m * dx_r);
361 next_temp(ix, iy)
362 += ky * ddc::step<DDimT>()
363 * (dy_l * last_temp(ix, iy + 1)
364 - 2.0 * dy_m * last_temp(ix, iy)
365 + dy_r * last_temp(ix, iy - 1))
366 / (dy_l * dy_m * dy_r);
367 });
369
371 if (iter - last_output_iter >= t_output_period) {
372 last_output_iter = iter;
373 ddc::parallel_deepcopy(ghosted_temp, ghosted_last_temp);
374 display(ddc::coordinate(iter),
375 ghosted_temp[ddc::DiscreteDomain<
376 DDimX,
377 DDimY>(x_domain, y_domain)]);
378 }
380
382 std::swap(ghosted_last_temp, ghosted_next_temp);
384 }
385
386
388 if (last_output_iter < time_domain.back()) {
389 ddc::parallel_deepcopy(ghosted_temp, ghosted_last_temp);
390 display(ddc::coordinate(time_domain.back()),
391 ghosted_temp[ddc::DiscreteDomain<
392 DDimX,
393 DDimY>(x_domain, y_domain)]);
394 }
396
397#ifdef DDC_BUILD_PDI_WRAPPER
398 PDI_finalize();
399 PC_tree_destroy(&pdi_conf);
400#endif
401}
