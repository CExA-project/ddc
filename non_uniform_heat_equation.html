29std::vector<double> generate_random_vector(
30 int n,
31 double lower_bound,
32 double higher_bound)
33{
34 assert(n > 1);
35 assert(lower_bound < higher_bound);
36
37 std::random_device rd;
38 std::mt19937 gen(rd());
39 // p represents the fraction of displacement
40 // it should be less than 0.5 to avoid reordering of nodes
41 double const p = 0.1;
42 std::uniform_real_distribution<double> dis(-p, +p);
43
44 double const dx = (higher_bound - lower_bound) / (n - 1);
45
46 std::vector<double> vec(n);
47
48 // Generate a uniform mesh
49 for (int i = 0; i < n; ++i) {
50 vec[i] = lower_bound + i * dx;
51 }
52 // Add a random perturbation
53 for (int i = 1; i < n - 1; ++i) {
54 vec[i] += dis(gen) * dx;
55 }
56
57 assert(std::is_sorted(vec.begin(), vec.end()));
58
59 return vec;
60}

94template <class ChunkType>
95void display(double time, ChunkType temp)
96{
97 double const mean_temp = ddc::transform_reduce(
98 temp.domain(),
99 0.,
100 ddc::reducer::sum<double>(),
101 temp)
102 / temp.domain().size();
103 std::cout << std::fixed << std::setprecision(3);
104 std::cout << "At t = " << time << ",\n";
105 std::cout << " * mean temperature = " << mean_temp << "\n";
106 ddc::ChunkSpan const temp_slice
107 = temp[ddc::get_domain<DDimY>(temp).front()
108 + ddc::get_domain<DDimY>(temp).size() / 2];
109 std::cout << " * temperature[y:"
110 << ddc::get_domain<DDimY>(temp).size() / 2 << "] = {";
111 ddc::for_each(
112 ddc::get_domain<DDimX>(temp),
113 [=](ddc::DiscreteElement<DDimX> const ix) {
114 std::cout << std::setw(6) << temp_slice(ix);
115 });
116 std::cout << " }\n" << std::flush;
117
118#ifdef DDC_BUILD_PDI_WRAPPER
119 ddc::PdiEvent("display")
120 .with("temp", temp)
121 .and_with("mean_temp", mean_temp)
122 .and_with("temp_slice", temp_slice);
123#endif
124}

127int main(int argc, char** argv)
128{
129#ifdef DDC_BUILD_PDI_WRAPPER
130 PC_tree_t pdi_conf = PC_parse_string("");
131 PDI_init(pdi_conf);
132#endif
133 Kokkos::ScopeGuard const kokkos_scope(argc, argv);
134 ddc::ScopeGuard const ddc_scope(argc, argv);
135
138 double const x_start = -1.;
139 double const x_end = 1.;
140 std::size_t const nb_x_points = 10;
141 double const kx = .01;
144 double const y_start = -1.;
145 double const y_end = 1.;
146 std::size_t const nb_y_points = 100;
147 double const ky = .002;
150 double const start_time = 0.;
151 double const end_time = 10.;
153 std::ptrdiff_t const t_output_period = 10;
155
157 std::vector<double> x_domain_vect
158 = generate_random_vector(nb_x_points, x_start, x_end);
160
161 std::size_t const size_x = x_domain_vect.size();
162
164 std::vector<double> const x_pre_ghost_vect {
165 x_domain_vect.front()
166 - (x_domain_vect.back() - x_domain_vect[size_x - 2])};
167
168 std::vector<double> const x_post_ghost_vect {
169 x_domain_vect.back()
170 + (x_domain_vect[1] - x_domain_vect.front())};
172
174 auto const [x_domain, ghosted_x_domain, x_pre_ghost, x_post_ghost]
175 = ddc::init_discrete_space<DDimX>(DDimX::init_ghosted<DDimX>(
176 x_domain_vect,
177 x_pre_ghost_vect,
178 x_post_ghost_vect));
180
181 ddc::DiscreteDomain<DDimX> const
182 x_domain_begin(x_domain.front(), x_post_ghost.extents());
183 ddc::DiscreteDomain<DDimX> const x_domain_end(
184 x_domain.back() - x_pre_ghost.extents() + 1,
185 x_pre_ghost.extents());
186
188 std::vector<double> y_domain_vect
189 = generate_random_vector(nb_y_points, y_start, y_end);
190
191 std::size_t const size_y = y_domain_vect.size();
192
194 std::vector<double> const y_pre_ghost_vect {
195 y_domain_vect.front()
196 - (y_domain_vect.back() - y_domain_vect[size_y - 2])};
197 std::vector<double> const y_post_ghost_vect {
198 y_domain_vect.back()
199 + (y_domain_vect[1] - y_domain_vect.front())};
202
204 auto const [y_domain, ghosted_y_domain, y_pre_ghost, y_post_ghost]
205 = ddc::init_discrete_space<DDimY>(DDimY::init_ghosted<DDimY>(
206 y_domain_vect,
207 y_pre_ghost_vect,
208 y_post_ghost_vect));
210
211 ddc::DiscreteDomain<DDimY> const
212 y_domain_begin(y_domain.front(), y_post_ghost.extents());
213
214 ddc::DiscreteDomain<DDimY> const y_domain_end(
215 y_domain.back() - y_pre_ghost.extents() + 1,
216 y_pre_ghost.extents());
217
219 double const invdx2_max = ddc::transform_reduce(
220 x_domain,
221 0.,
222 ddc::reducer::max<double>(),
223 [](ddc::DiscreteElement<DDimX> ix) {
224 return 1.
225 / (ddc::distance_at_left(ix)
226 * ddc::distance_at_right(ix));
227 });
228
229 double const invdy2_max = ddc::transform_reduce(
230 y_domain,
231 0.,
232 ddc::reducer::max<double>(),
233 [](ddc::DiscreteElement<DDimY> iy) {
234 return 1.
235 / (ddc::distance_at_left(iy)
236 * ddc::distance_at_right(iy));
237 });
238
239 ddc::Coordinate<T> const max_dt {
240 .5 / (kx * invdx2_max + ky * invdy2_max)};
242
244 ddc::DiscreteVector<DDimT> const nb_time_steps(
245 std::ceil((end_time - start_time) / max_dt) + .2);
246
247 ddc::DiscreteDomain<DDimT> const time_domain
248 = ddc::init_discrete_space<DDimT>(DDimT::init<DDimT>(
249 ddc::Coordinate<T>(start_time),
250 ddc::Coordinate<T>(end_time),
251 nb_time_steps + 1));
253
255 ddc::Chunk ghosted_last_temp(
256 "ghosted_last_temp",
257 ddc::DiscreteDomain<
258 DDimX,
259 DDimY>(ghosted_x_domain, ghosted_y_domain),
260 ddc::DeviceAllocator<double>());
261
262 ddc::Chunk ghosted_next_temp(
263 "ghosted_next_temp",
264 ddc::DiscreteDomain<
265 DDimX,
266 DDimY>(ghosted_x_domain, ghosted_y_domain),
267 ddc::DeviceAllocator<double>());
269
271 ddc::ChunkSpan const ghosted_initial_temp
272 = ghosted_last_temp.span_view();
274
276 ddc::parallel_for_each(
277 ddc::DiscreteDomain<DDimX, DDimY>(x_domain, y_domain),
278 KOKKOS_LAMBDA(ddc::DiscreteElement<DDimX, DDimY> const ixy) {
279 double const x = ddc::coordinate(
280 ddc::DiscreteElement<DDimX>(ixy));
281 double const y = ddc::coordinate(
282 ddc::DiscreteElement<DDimY>(ixy));
283 ghosted_initial_temp(ixy)
284 = 9.999 * ((x * x + y * y) < 0.25);
285 });
287
289 ddc::Chunk ghosted_temp
290 = ddc::create_mirror(ghosted_last_temp.span_cview());
292
294 ddc::parallel_deepcopy(ghosted_temp, ghosted_last_temp);
296
298 display(ddc::coordinate(time_domain.front()),
299 ghosted_temp[x_domain][y_domain]);
301
303 ddc::DiscreteElement<DDimT> last_output_iter = time_domain.front();
305
307 for (ddc::DiscreteElement<DDimT> const iter :
308 time_domain.remove_first(ddc::DiscreteVector<DDimT>(1))) {
310
312 ddc::parallel_deepcopy(
313 ghosted_last_temp[ddc::DiscreteDomain<
314 DDimX,
315 DDimY>(x_pre_ghost, y_domain)],
316 ghosted_last_temp[ddc::DiscreteDomain<
317 DDimX,
318 DDimY>(y_domain, x_domain_end)]);
319 ddc::parallel_deepcopy(
320 ghosted_last_temp[ddc::DiscreteDomain<
321 DDimX,
322 DDimY>(y_domain, x_post_ghost)],
323 ghosted_last_temp[ddc::DiscreteDomain<
324 DDimX,
325 DDimY>(y_domain, x_domain_begin)]);
326 ddc::parallel_deepcopy(
327 ghosted_last_temp[ddc::DiscreteDomain<
328 DDimX,
329 DDimY>(x_domain, y_pre_ghost)],
330 ghosted_last_temp[ddc::DiscreteDomain<
331 DDimX,
332 DDimY>(x_domain, y_domain_end)]);
333 ddc::parallel_deepcopy(
334 ghosted_last_temp[ddc::DiscreteDomain<
335 DDimX,
336 DDimY>(x_domain, y_post_ghost)],
337 ghosted_last_temp[ddc::DiscreteDomain<
338 DDimX,
339 DDimY>(x_domain, y_domain_begin)]);
341
343 ddc::ChunkSpan const next_temp(
344 ghosted_next_temp[ddc::DiscreteDomain<
345 DDimX,
346 DDimY>(x_domain, y_domain)]);
347 ddc::ChunkSpan const last_temp(ghosted_last_temp.span_cview());
349
351 ddc::parallel_for_each(
352 next_temp.domain(),
353 KOKKOS_LAMBDA(
354 ddc::DiscreteElement<DDimX, DDimY> const ixy) {
355 ddc::DiscreteElement<DDimX> const ix
356 = ddc::select<DDimX>(ixy);
357 ddc::DiscreteElement<DDimY> const iy
358 = ddc::select<DDimY>(ixy);
359 double const dx_l = ddc::distance_at_left(ix);
360 double const dx_r = ddc::distance_at_right(ix);
361 double const dx_m = 0.5 * (dx_l + dx_r);
362 double const dy_l = ddc::distance_at_left(iy);
363 double const dy_r = ddc::distance_at_right(iy);
364 double const dy_m = 0.5 * (dy_l + dy_r);
365 next_temp(ix, iy) = last_temp(ix, iy);
366 next_temp(ix, iy)
367 += kx * ddc::step<DDimT>()
368 * (dx_l * last_temp(ix + 1, iy)
369 - 2.0 * dx_m * last_temp(ix, iy)
370 + dx_r * last_temp(ix - 1, iy))
371 / (dx_l * dx_m * dx_r);
372 next_temp(ix, iy)
373 += ky * ddc::step<DDimT>()
374 * (dy_l * last_temp(ix, iy + 1)
375 - 2.0 * dy_m * last_temp(ix, iy)
376 + dy_r * last_temp(ix, iy - 1))
377 / (dy_l * dy_m * dy_r);
378 });
380
382 if (iter - last_output_iter >= t_output_period) {
383 last_output_iter = iter;
384 ddc::parallel_deepcopy(ghosted_temp, ghosted_next_temp);
385 display(ddc::coordinate(iter),
386 ghosted_temp[ddc::DiscreteDomain<
387 DDimX,
388 DDimY>(x_domain, y_domain)]);
389 }
391
393 std::swap(ghosted_last_temp, ghosted_next_temp);
395 }
396
398 if (last_output_iter < time_domain.back()) {
399 ddc::parallel_deepcopy(ghosted_temp, ghosted_last_temp);
400 display(ddc::coordinate(time_domain.back()),
401 ghosted_temp[ddc::DiscreteDomain<
402 DDimX,
403 DDimY>(x_domain, y_domain)]);
404 }
406
407#ifdef DDC_BUILD_PDI_WRAPPER
408 PDI_finalize();
409 PC_tree_destroy(&pdi_conf);
410#endif
411}
