172  static std::tuple<Impl<Kokkos::HostSpace>, discrete_domain_type> init(
173  continuous_element_type a,
174  continuous_element_type b,
175  discrete_vector_type n,
176  discrete_vector_type n_period)
177  {
178  assert(a < b);
179  assert(n > 1);
180  assert(n_period > 1);
181  Impl<Kokkos::HostSpace> disc(a, continuous_element_type {(b - a) / (n - 1)}, n_period);
182  discrete_domain_type domain {disc.front(), n};
183  return std::make_tuple(std::move(disc), std::move(domain));
184  }

196  static std::tuple<
197  Impl<Kokkos::HostSpace>,
198  discrete_domain_type,
199  discrete_domain_type,
200  discrete_domain_type,
201  discrete_domain_type>
202  init_ghosted(
203  continuous_element_type a,
204  continuous_element_type b,
205  discrete_vector_type n,
206  discrete_vector_type n_period,
207  discrete_vector_type n_ghosts_before,
208  discrete_vector_type n_ghosts_after)
209  {
210  using discrete_domain_type = discrete_domain_type;
211  assert(a < b);
212  assert(n > 1);
213  assert(n_period > 1);
214  double discretization_step {(b - a) / (n - 1)};
215  Impl<Kokkos::HostSpace>
216  disc(a - n_ghosts_before.value() * discretization_step,
217  discretization_step,
218  n_period);
219  discrete_domain_type ghosted_domain
220  = discrete_domain_type(disc.front(), n + n_ghosts_before + n_ghosts_after);
221  discrete_domain_type pre_ghost
222  = discrete_domain_type(ghosted_domain.front(), n_ghosts_before);
223  discrete_domain_type main_domain
224  = discrete_domain_type(ghosted_domain.front() + n_ghosts_before, n);
225  discrete_domain_type post_ghost
226  = discrete_domain_type(main_domain.back() + 1, n_ghosts_after);
227  return std::make_tuple(
228  std::move(disc),
229  std::move(main_domain),
230  std::move(ghosted_domain),
231  std::move(pre_ghost),
232  std::move(post_ghost));
233  }
