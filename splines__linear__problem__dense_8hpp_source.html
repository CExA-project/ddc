16/**
17 * @brief A dense linear problem dedicated to the computation of a spline approximation.
18 *
19 * The storage format is dense row-major. Lapack is used to perform every matrix and linear solver-related operations.
20 *
21 * @tparam ExecSpace The Kokkos::ExecutionSpace on which operations related to the matrix are supposed to be performed.
22 */
23template <class ExecSpace>
24class SplinesLinearProblemDense : public SplinesLinearProblem<ExecSpace>
25{
26public:
27 using typename SplinesLinearProblem<ExecSpace>::MultiRHS;
28 using SplinesLinearProblem<ExecSpace>::size;
29
30protected:
31 Kokkos::DualView<double**, Kokkos::LayoutRight, typename ExecSpace::memory_space> m_a;
32 Kokkos::DualView<int*, typename ExecSpace::memory_space> m_ipiv;
33
34public:
35 /**
36 * @brief SplinesLinearProblemDense constructor.
37 *
38 * @param mat_size The size of one of the dimensions of the square matrix.
39 */
40 explicit SplinesLinearProblemDense(std::size_t mat_size);
41
42 SplinesLinearProblemDense(SplinesLinearProblemDense const& rhs) = delete;
43
44 SplinesLinearProblemDense(SplinesLinearProblemDense&& rhs) = delete;
45
46 ~SplinesLinearProblemDense() override;
47
48 SplinesLinearProblemDense& operator=(SplinesLinearProblemDense const& rhs) = delete;
49
50 SplinesLinearProblemDense& operator=(SplinesLinearProblemDense&& rhs) = delete;
51
52 double get_element(std::size_t i, std::size_t j) const override;
53
54 void set_element(std::size_t i, std::size_t j, double aij) override;
55
56 /**
57 * @brief Perform a pre-process operation on the solver. Must be called after filling the matrix.
58 *
59 * LU-factorize the matrix A and store the pivots using the LAPACK dgetrf() implementation.
60 */
61 void setup_solver() override;
62
63 /**
64 * @brief Solve the multiple right-hand sides linear problem Ax=b or its transposed version A^tx=b inplace.
65 *
66 * The solver method is gaussian elimination with partial pivoting using the LU-factorized matrix A. The implementation is LAPACK method dgetrs.
67 *
68 * @param[in, out] b A 2D Kokkos::View storing the multiple right-hand sides of the problem and receiving the corresponding solution.
69 * @param transpose Choose between the direct or transposed version of the linear problem.
70 */
71 void solve(MultiRHS b, bool transpose) const override;
72};
