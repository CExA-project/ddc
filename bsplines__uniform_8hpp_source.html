31/**
32 * The type of a uniform 1D spline basis (B-spline).
33 *
34 * Knots for uniform B-splines are uniformly distributed (the associated discrete dimension
35 * is a UniformPointSampling).
36 *
37 * @tparam CDim The tag identifying the continuous dimension on which the support of the B-spline functions are defined.
38 * @tparam D The degree of the B-splines.
39 */
40template <class CDim, std::size_t D>
41class UniformBSplines : detail::UniformBSplinesBase
42{
43 static_assert(D > 0, "Parameter `D` must be positive");
44
45public:
46 /// @brief The tag identifying the continuous dimension on which the support of the B-splines are defined.
47 using continuous_dimension_type = CDim;
48
49 /// @brief The discrete dimension representing B-splines.
50 using discrete_dimension_type = UniformBSplines;
51
52 /** @brief The degree of B-splines.
53 *
54 * @return The degree.
55 */
56 static constexpr std::size_t degree() noexcept
57 {
58 return D;
59 }

61 /** @brief Indicates if the B-splines are periodic or not.
62 *
63 * @return A boolean indicating if the B-splines are periodic or not.
64 */
65 static constexpr bool is_periodic() noexcept
66 {
67 return CDim::PERIODIC;

70 /** @brief Indicates if the B-splines are uniform or not (this is the case here).
71 *
72 * @return A boolean indicating if the B-splines are uniform or not.

74 static constexpr bool is_uniform() noexcept

76 return true;

114 /** Constructs a spline basis (B-splines) with n equidistant knots over \f$[a, b]\f$.

176 /** @brief Evaluates non-zero B-splines at a given coordinate.
177 *
178 * The values are computed for every B-spline with support at the given coordinate x. There are only (degree+1)
179 * B-splines which are non-zero at any given point. It is these B-splines which are evaluated.
180 * This can be useful to calculate a spline approximation of a function. A spline approximation at coordinate x
181 * is a linear combination of these B-spline evaluations weighted with the spline coefficients of the spline-transformed
182 * initial discrete function.
183 *
184 * @param[out] values The values of the B-splines evaluated at coordinate x. It has to be a 1D mdspan with (degree+1) elements.
185 * @param[in] x The coordinate where B-splines are evaluated. It has to be in the range of break points coordinates.
186 * @return The index of the first B-spline which is evaluated.
187 */
188 KOKKOS_INLINE_FUNCTION discrete_element_type
189 eval_basis(DSpan1D values, ddc::Coordinate<CDim> const& x) const
190 {
191 assert(values.size() == degree() + 1);
192 return eval_basis(values, x, degree());
193 }
194
195 /** @brief Evaluates non-zero B-spline derivatives at a given coordinate
196 *
197 * The derivatives are computed for every B-spline with support at the given coordinate x. There are only (degree+1)
198 * B-splines which are non-zero at any given point. It is these B-splines which are derivated.
199 * A spline approximation of a derivative at coordinate x is a linear
200 * combination of those B-spline derivatives weighted with the spline coefficients of the spline-transformed
201 * initial discrete function.
202 *
203 * @param[out] derivs The derivatives of the B-splines evaluated at coordinate x. It has to be a 1D mdspan with (degree+1) elements.
204 * @param[in] x The coordinate where B-spline derivatives are evaluated. It has to be in the range of break points coordinates.
205 * @return The index of the first B-spline which is evaluated.
206 */
207 KOKKOS_INLINE_FUNCTION discrete_element_type
208 eval_deriv(DSpan1D derivs, ddc::Coordinate<CDim> const& x) const;
209
210 /** @brief Evaluates non-zero B-spline values and \f$n\f$ derivatives at a given coordinate
211 *
212 * The values and derivatives are computed for every B-spline with support at the given coordinate x. There are only (degree+1)
213 * B-splines which are non-zero at any given point. It is these B-splines which are evaluated and derivated.
214 * A spline approximation of a derivative at coordinate x is a linear
215 * combination of those B-spline derivatives weighted with spline coefficients of the spline-transformed
216 * initial discrete function.
217 *
218 * @param[out] derivs The values and \f$n\f$ derivatives of the B-splines evaluated at coordinate x. It has to be a 2D mdspan of sizes (degree+1, n+1).
219 * @param[in] x The coordinate where B-spline derivatives are evaluated. It has to be in the range of break points coordinates.
220 * @param[in] n The number of derivatives to evaluate (in addition to the B-spline values themselves).
221 * @return The index of the first B-spline which is evaluated.
222 */
223 KOKKOS_INLINE_FUNCTION discrete_element_type eval_basis_and_n_derivs(
224 ddc::DSpan2D derivs,
225 ddc::Coordinate<CDim> const& x,
226 std::size_t n) const;
227
228 /** @brief Compute the integrals of the B-splines.
229 *
230 * The integral of each of the B-splines over their support within the domain on which this basis was defined.
231 *
232 * @param[out] int_vals The values of the integrals. It has to be a 1D Chunkspan of size (nbasis).
233 * @return The values of the integrals.
234 */
235 template <class Layout, class MemorySpace2>
236 KOKKOS_INLINE_FUNCTION ddc::
237 ChunkSpan<double, ddc::DiscreteDomain<DDim>, Layout, MemorySpace2>
238 integrals(ddc::ChunkSpan<double, discrete_domain_type, Layout, MemorySpace2>
239 int_vals) const;

241 /** @brief Returns the coordinate of the first support knot associated to a DiscreteElement identifying a B-spline.
242 *
243 * Each B-spline has a support defined over (degree+2) knots. For a B-spline identified by the
244 * provided DiscreteElement, this function returns the first knot in the support of the B-spline.
245 * In other words it returns the lower bound of the support.
246 *
247 * @param[in] ix DiscreteElement identifying the B-spline.
248 * @return DiscreteElement of the lower bound of the support of the B-spline.
249 */
250 KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<knot_discrete_dimension_type>
251 get_first_support_knot(discrete_element_type const& ix) const
252 {
253 return ddc::DiscreteElement<knot_discrete_dimension_type>(
254 (ix - discrete_element_type(0)).value());
255 }
256
257 /** @brief Returns the coordinate of the last support knot associated to a DiscreteElement identifying a B-spline.
258 *
259 * Each B-spline has a support defined over (degree+2) knots. For a B-spline identified by the
260 * provided DiscreteElement, this function returns the last knot in the support of the B-spline.
261 * In other words it returns the upper bound of the support.
262 *
263 * @param[in] ix DiscreteElement identifying the B-spline.
264 * @return DiscreteElement of the upper bound of the support of the B-spline.
265 */
266 KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<knot_discrete_dimension_type>
267 get_last_support_knot(discrete_element_type const& ix) const
268 {
269 return get_first_support_knot(ix)
270 + ddc::DiscreteVector<knot_discrete_dimension_type>(degree() + 1);
271 }

300 /** @brief Returns the number of elements necessary to construct a spline representation of a function.
301 *
302 * For a non-periodic domain the number of elements necessary to construct a spline representation of a function
303 * is equal to the number of basis functions. However in the periodic case it additionally includes degree additional elements
304 * which allow the first B-splines to be evaluated close to rmax (where they also appear due to the periodicity).
305 *
306 * @return The number of elements necessary to construct a spline representation of a function.
307 */

309 {
310 return degree() + ncells();
311 }

319 return discrete_domain_type(discrete_element_type(0), discrete_vector_type(size()));

345 * The number of cells over which the B-splines and any spline representation are defined.

347 *
348 * @return The number of cells over which the B-splines are defined.

384template <class CDim, std::size_t D>
385template <class DDim, class MemorySpace>
386KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDim> UniformBSplines<CDim, D>::
387 Impl<DDim, MemorySpace>::eval_basis(
388 DSpan1D values,
389 ddc::Coordinate<CDim> const& x,
390 [[maybe_unused]] std::size_t const deg) const
391{
392 assert(values.size() == deg + 1);
393
394 double offset;
395 int jmin;
396 // 1. Compute cell index 'icell' and x_offset
397 // 2. Compute index range of B-splines with support over cell 'icell'
398 get_icell_and_offset(jmin, offset, x);
399
400 // 3. Compute values of aforementioned B-splines
401 double xx, temp, saved;
402 values(0) = 1.0;
403 for (std::size_t j = 1; j < values.size(); ++j) {
404 xx = -offset;
405 saved = 0.0;
406 for (std::size_t r = 0; r < j; ++r) {
407 xx += 1;
408 temp = values(r) / j;
409 values(r) = saved + xx * temp;
410 saved = (j - xx) * temp;
411 }
412 values(j) = saved;
413 }
414
415 return discrete_element_type(jmin);
416}

418template <class CDim, std::size_t D>
419template <class DDim, class MemorySpace>
420KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDim> UniformBSplines<CDim, D>::
421 Impl<DDim, MemorySpace>::eval_deriv(DSpan1D derivs, ddc::Coordinate<CDim> const& x) const
422{
423 assert(derivs.size() == degree() + 1);
424
425 double offset;
426 int jmin;
427 // 1. Compute cell index 'icell' and x_offset
428 // 2. Compute index range of B-splines with support over cell 'icell'
429 get_icell_and_offset(jmin, offset, x);
430
431 // 3. Compute derivatives of aforementioned B-splines
432 // Derivatives are normalized, hence they should be divided by dx
433 double xx, temp, saved;
434 derivs(0) = 1.0 / ddc::step<knot_discrete_dimension_type>();
435 for (std::size_t j = 1; j < degree(); ++j) {
436 xx = -offset;
437 saved = 0.0;
438 for (std::size_t r = 0; r < j; ++r) {
439 xx += 1.0;
440 temp = derivs(r) / j;
441 derivs(r) = saved + xx * temp;
442 saved = (j - xx) * temp;
443 }
444 derivs(j) = saved;
445 }
446
447 // Compute derivatives
448 double bjm1 = derivs[0];
449 double bj = bjm1;
450 derivs(0) = -bjm1;
451 for (std::size_t j = 1; j < degree(); ++j) {
452 bj = derivs(j);
453 derivs(j) = bjm1 - bj;
454 bjm1 = bj;
455 }
456 derivs(degree()) = bj;
457
458 return discrete_element_type(jmin);
459}

461template <class CDim, std::size_t D>
462template <class DDim, class MemorySpace>
463KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDim> UniformBSplines<CDim, D>::
464 Impl<DDim, MemorySpace>::eval_basis_and_n_derivs(
465 ddc::DSpan2D const derivs,
466 ddc::Coordinate<CDim> const& x,
467 std::size_t const n) const
468{
469 std::array<double, (degree() + 1) * (degree() + 1)> ndu_ptr;
470 std::experimental::mdspan<
471 double,
472 std::experimental::extents<std::size_t, degree() + 1, degree() + 1>> const
473 ndu(ndu_ptr.data());
474 std::array<double, 2 * (degree() + 1)> a_ptr;
475 std::experimental::
476 mdspan<double, std::experimental::extents<std::size_t, degree() + 1, 2>> const a(
477 a_ptr.data());
478 double offset;
479 int jmin;
480
481 assert(x - rmin() >= -length() * 1e-14);
482 assert(rmax() - x >= -length() * 1e-14);
483 // assert(n >= 0); as long as n is unsigned
484 assert(n <= degree());
485 assert(derivs.extent(0) == 1 + degree());
486 assert(derivs.extent(1) == 1 + n);
487
488 // 1. Compute cell index 'icell' and x_offset
489 // 2. Compute index range of B-splines with support over cell 'icell'
490 get_icell_and_offset(jmin, offset, x);
491
492 // 3. Recursively evaluate B-splines (eval_basis)
493 // up to self%degree, and store them all in the upper-right triangle of
494 // ndu
495 double xx, temp, saved;
496 ndu(0, 0) = 1.0;
497 for (std::size_t j = 1; j < degree() + 1; ++j) {
498 xx = -offset;
499 saved = 0.0;
500 for (std::size_t r = 0; r < j; ++r) {
501 xx += 1.0;
502 temp = ndu(j - 1, r) / j;
503 ndu(j, r) = saved + xx * temp;
504 saved = (j - xx) * temp;
505 }
506 ndu(j, j) = saved;
507 }
508 for (std::size_t i = 0; i < ndu.extent(1); ++i) {
509 derivs(i, 0) = ndu(degree(), i);
510 }
511
512 for (int r = 0; r < int(degree() + 1); ++r) {
513 int s1 = 0;
514 int s2 = 1;
515 a(0, 0) = 1.0;
516 for (int k = 1; k < int(n + 1); ++k) {
517 double d = 0.0;
518 int const rk = r - k;
519 int const pk = degree() - k;
520 if (r >= k) {
521 a(0, s2) = a(0, s1) / (pk + 1);
522 d = a(0, s2) * ndu(pk, rk);
523 }
524 int const j1 = rk > -1 ? 1 : (-rk);
525 int const j2 = (r - 1) <= pk ? k : (degree() - r + 1);
526 for (int j = j1; j < j2; ++j) {
527 a(j, s2) = (a(j, s1) - a(j - 1, s1)) / (pk + 1);
528 d += a(j, s2) * ndu(pk, rk + j);
529 }
530 if (r <= pk) {
531 a(k, s2) = -a(k - 1, s1) / (pk + 1);
532 d += a(k, s2) * ndu(pk, r);
533 }
534 derivs(r, k) = d;
535 Kokkos::kokkos_swap(s1, s2);
536 }
537 }
538
539 // Multiply result by correct factors:
540 // degree!/(degree-n)! = degree*(degree-1)*...*(degree-n+1)
541 // k-th derivatives are normalized, hence they should be divided by dx^k
542 double const inv_dx = inv_step();
543 double d = degree() * inv_dx;
544 for (int k = 1; k < int(n + 1); ++k) {
545 for (std::size_t i = 0; i < derivs.extent(0); ++i) {
546 derivs(i, k) *= d;
547 }
548 d *= (degree() - k) * inv_dx;
549 }
550
551 return discrete_element_type(jmin);
552}

554template <class CDim, std::size_t D>
555template <class DDim, class MemorySpace>
556KOKKOS_INLINE_FUNCTION void UniformBSplines<CDim, D>::Impl<DDim, MemorySpace>::get_icell_and_offset(
557 int& icell,
558 double& offset,
559 ddc::Coordinate<CDim> const& x) const
560{
561 assert(x - rmin() >= -length() * 1e-14);
562 assert(rmax() - x >= -length() * 1e-14);
563
564 double const inv_dx = inv_step();
565 if (x == rmin()) {
566 icell = 0;
567 offset = 0.0;
568 } else if (x == rmax()) {
569 icell = ncells() - 1;
570 offset = 1.0;
571 } else {
572 offset = (x - rmin()) * inv_dx;
573 icell = static_cast<int>(offset);
574 offset = offset - icell;
575
576 // When x is very close to xmax, round-off may cause the wrong answer
577 // icell=ncells and x_offset=0, which we convert to the case x=xmax:
578 if (icell == int(ncells()) && offset == 0.0) {
579 icell = ncells() - 1;
580 offset = 1.0;
581 }
582 }
583}

585template <class CDim, std::size_t D>
586template <class DDim, class MemorySpace>
587template <class Layout, class MemorySpace2>
588KOKKOS_INLINE_FUNCTION ddc::ChunkSpan<double, ddc::DiscreteDomain<DDim>, Layout, MemorySpace2>
589UniformBSplines<CDim, D>::Impl<DDim, MemorySpace>::integrals(
590 ddc::ChunkSpan<double, discrete_domain_type, Layout, MemorySpace2> int_vals) const
591{
592 if constexpr (is_periodic()) {
593 assert(int_vals.size() == nbasis() || int_vals.size() == size());
594 } else {
595 assert(int_vals.size() == nbasis());
596 }
597 discrete_domain_type const full_dom_splines(full_domain());
598
599 if constexpr (is_periodic()) {
600 discrete_domain_type const dom_bsplines(
601 full_dom_splines.take_first(discrete_vector_type {nbasis()}));
602 for (auto ix : dom_bsplines) {
603 int_vals(ix) = ddc::step<knot_discrete_dimension_type>();
604 }
605 if (int_vals.size() == size()) {
606 discrete_domain_type const dom_bsplines_repeated(
607 full_dom_splines.take_last(discrete_vector_type {degree()}));
608 for (auto ix : dom_bsplines_repeated) {
609 int_vals(ix) = 0;
610 }
611 }
612 } else {
613 discrete_domain_type const dom_bspline_entirely_in_domain
614 = full_dom_splines
615 .remove(discrete_vector_type(degree()), discrete_vector_type(degree()));
616 for (auto ix : dom_bspline_entirely_in_domain) {
617 int_vals(ix) = ddc::step<knot_discrete_dimension_type>();
618 }
619
620 std::array<double, degree() + 2> edge_vals_ptr;
621 std::experimental::
622 mdspan<double, std::experimental::extents<std::size_t, degree() + 2>> const
623 edge_vals(edge_vals_ptr.data());
624
625 eval_basis(edge_vals, rmin(), degree() + 1);
626
627 double const d_eval = ddc::detail::sum(edge_vals);
628
629 for (std::size_t i = 0; i < degree(); ++i) {
630 double const c_eval = ddc::detail::sum(edge_vals, 0, degree() - i);
631
