403template <class Tag, std::size_t D>
404template <class DDim, class MemorySpace>
405KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDim> UniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::
406 eval_basis(
407 DSpan1D values,
408 ddc::Coordinate<Tag> const& x,
409 [[maybe_unused]] std::size_t const deg) const
410{
411 assert(values.size() == deg + 1);
412
413 double offset;
414 int jmin;
415 // 1. Compute cell index 'icell' and x_offset
416 // 2. Compute index range of B-splines with support over cell 'icell'
417 get_icell_and_offset(jmin, offset, x);
418
419 // 3. Compute values of aforementioned B-splines
420 double xx, temp, saved;
421 values(0) = 1.0;
422 for (std::size_t j = 1; j < values.size(); ++j) {
423 xx = -offset;
424 saved = 0.0;
425 for (std::size_t r = 0; r < j; ++r) {
426 xx += 1;
427 temp = values(r) / j;
428 values(r) = saved + xx * temp;
429 saved = (j - xx) * temp;
430 }
431 values(j) = saved;
432 }
433
434 return discrete_element_type(jmin);
435}

438template <class DDim, class MemorySpace>
439KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDim> UniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::
440 eval_deriv(DSpan1D derivs, ddc::Coordinate<Tag> const& x) const
441{
442 assert(derivs.size() == degree() + 1);
443
444 double offset;
445 int jmin;
446 // 1. Compute cell index 'icell' and x_offset
447 // 2. Compute index range of B-splines with support over cell 'icell'
448 get_icell_and_offset(jmin, offset, x);
449
450 // 3. Compute derivatives of aforementioned B-splines
451 // Derivatives are normalized, hence they should be divided by dx
452 double xx, temp, saved;
453 derivs(0) = 1.0 / ddc::step<knot_mesh_type>();
454 for (std::size_t j = 1; j < degree(); ++j) {
455 xx = -offset;
456 saved = 0.0;
457 for (std::size_t r = 0; r < j; ++r) {
458 xx += 1.0;
459 temp = derivs(r) / j;
460 derivs(r) = saved + xx * temp;
461 saved = (j - xx) * temp;
462 }
463 derivs(j) = saved;
464 }
465
466 // Compute derivatives
467 double bjm1 = derivs[0];
468 double bj = bjm1;
469 derivs(0) = -bjm1;
470 for (std::size_t j = 1; j < degree(); ++j) {
471 bj = derivs(j);
472 derivs(j) = bjm1 - bj;
473 bjm1 = bj;
474 }
475 derivs(degree()) = bj;
476
477 return discrete_element_type(jmin);
478}

481template <class DDim, class MemorySpace>
482KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDim> UniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::
483 eval_basis_and_n_derivs(
484 ddc::DSpan2D const derivs,
485 ddc::Coordinate<Tag> const& x,
486 std::size_t const n) const
487{
488 std::array<double, (degree() + 1) * (degree() + 1)> ndu_ptr;
489 std::experimental::mdspan<
490 double,
491 std::experimental::extents<std::size_t, degree() + 1, degree() + 1>> const
492 ndu(ndu_ptr.data());
493 std::array<double, 2 * (degree() + 1)> a_ptr;
494 std::experimental::
495 mdspan<double, std::experimental::extents<std::size_t, degree() + 1, 2>> const a(
496 a_ptr.data());
497 double offset;
498 int jmin;
499
500 assert(x >= rmin());
501 assert(x <= rmax());
502 // assert(n >= 0); as long as n is unsigned
503 assert(n <= degree());
504 assert(derivs.extent(0) == 1 + degree());
505 assert(derivs.extent(1) == 1 + n);
506
507 // 1. Compute cell index 'icell' and x_offset
508 // 2. Compute index range of B-splines with support over cell 'icell'
509 get_icell_and_offset(jmin, offset, x);
510
511 // 3. Recursively evaluate B-splines (eval_basis)
512 // up to self%degree, and store them all in the upper-right triangle of
513 // ndu
514 double xx, temp, saved;
515 ndu(0, 0) = 1.0;
516 for (std::size_t j = 1; j < degree() + 1; ++j) {
517 xx = -offset;
518 saved = 0.0;
519 for (std::size_t r = 0; r < j; ++r) {
520 xx += 1.0;
521 temp = ndu(j - 1, r) / j;
522 ndu(j, r) = saved + xx * temp;
523 saved = (j - xx) * temp;
524 }
525 ndu(j, j) = saved;
526 }
527 for (std::size_t i = 0; i < ndu.extent(1); ++i) {
528 derivs(i, 0) = ndu(degree(), i);
529 }
530
531 for (int r = 0; r < int(degree() + 1); ++r) {
532 int s1 = 0;
533 int s2 = 1;
534 a(0, 0) = 1.0;
535 for (int k = 1; k < int(n + 1); ++k) {
536 double d = 0.0;
537 int const rk = r - k;
538 int const pk = degree() - k;
539 if (r >= k) {
540 a(0, s2) = a(0, s1) / (pk + 1);
541 d = a(0, s2) * ndu(pk, rk);
542 }
543 int const j1 = rk > -1 ? 1 : (-rk);
544 int const j2 = (r - 1) <= pk ? k : (degree() - r + 1);
545 for (int j = j1; j < j2; ++j) {
546 a(j, s2) = (a(j, s1) - a(j - 1, s1)) / (pk + 1);
547 d += a(j, s2) * ndu(pk, rk + j);
548 }
549 if (r <= pk) {
550 a(k, s2) = -a(k - 1, s1) / (pk + 1);
551 d += a(k, s2) * ndu(pk, r);
552 }
553 derivs(r, k) = d;
554 Kokkos::kokkos_swap(s1, s2);
555 }
556 }
557
558 // Multiply result by correct factors:
559 // degree!/(degree-n)! = degree*(degree-1)*...*(degree-n+1)
560 // k-th derivatives are normalized, hence they should be divided by dx^k
561 double const inv_dx = inv_step();
562 double d = degree() * inv_dx;
563 for (int k = 1; k < int(n + 1); ++k) {
564 for (std::size_t i = 0; i < derivs.extent(0); ++i) {
565 derivs(i, k) *= d;
566 }
567 d *= (degree() - k) * inv_dx;
568 }
569
570 return discrete_element_type(jmin);
571}

573template <class Tag, std::size_t D>
574template <class DDim, class MemorySpace>
575KOKKOS_INLINE_FUNCTION void UniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::get_icell_and_offset(
576 int& icell,
577 double& offset,
578 ddc::Coordinate<Tag> const& x) const
579{
580 assert(x >= rmin());
581 assert(x <= rmax());
582
583 double const inv_dx = inv_step();
584 if (x == rmin()) {
585 icell = 0;
586 offset = 0.0;
587 } else if (x == rmax()) {
588 icell = ncells() - 1;
589 offset = 1.0;
590 } else {
591 offset = (x - rmin()) * inv_dx;
592 icell = static_cast<int>(offset);
593 offset = offset - icell;
594
595 // When x is very close to xmax, round-off may cause the wrong answer
596 // icell=ncells and x_offset=0, which we convert to the case x=xmax:
597 if (icell == int(ncells()) && offset == 0.0) {
598 icell = ncells() - 1;
599 offset = 1.0;
600 }
601 }
602}

604template <class Tag, std::size_t D>
605template <class DDim, class MemorySpace>
606template <class Layout, class MemorySpace2>
607KOKKOS_INLINE_FUNCTION ddc::ChunkSpan<double, ddc::DiscreteDomain<DDim>, Layout, MemorySpace2>
608UniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::integrals(
609 ddc::ChunkSpan<double, discrete_domain_type, Layout, MemorySpace2> int_vals) const
610{
611 if constexpr (is_periodic()) {
612 assert(int_vals.size() == nbasis() || int_vals.size() == size());
613 } else {
614 assert(int_vals.size() == nbasis());
615 }
616 discrete_domain_type const full_dom_splines(full_domain());
617
618 if constexpr (is_periodic()) {
619 discrete_domain_type const dom_bsplines(
620 full_dom_splines.take_first(discrete_vector_type {nbasis()}));
621 for (auto ix : dom_bsplines) {
622 int_vals(ix) = ddc::step<knot_mesh_type>();
623 }
624 if (int_vals.size() == size()) {
625 discrete_domain_type const dom_bsplines_repeated(
626 full_dom_splines.take_last(discrete_vector_type {degree()}));
627 for (auto ix : dom_bsplines_repeated) {
628 int_vals(ix) = 0;
629 }
630 }
631 } else {
632 discrete_domain_type const dom_bspline_entirely_in_domain
633 = full_dom_splines
634 .remove(discrete_vector_type(degree()), discrete_vector_type(degree()));
635 for (auto ix : dom_bspline_entirely_in_domain) {
636 int_vals(ix) = ddc::step<knot_mesh_type>();
637 }
638
639 std::array<double, degree() + 2> edge_vals_ptr;
640 std::experimental::
641 mdspan<double, std::experimental::extents<std::size_t, degree() + 2>> const
642 edge_vals(edge_vals_ptr.data());
643
644 eval_basis(edge_vals, rmin(), degree() + 1);
645
646 double const d_eval = ddc::detail::sum(edge_vals);
647
648 for (std::size_t i = 0; i < degree(); ++i) {
649 double const c_eval = ddc::detail::sum(edge_vals, 0, degree() - i);
650
651 double const edge_value = ddc::step<knot_mesh_type>() * (d_eval - c_eval);
652
653 int_vals(discrete_element_type(i)) = edge_value;
654 int_vals(discrete_element_type(nbasis() - 1 - i)) = edge_value;
655 }
656 }
657 return int_vals;
658}
