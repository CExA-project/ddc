106 static KOKKOS_FUNCTION internal_mdspan_type build_internal_mdspan(
107 allocation_mdspan_type const& allocation_mdspan,
108 discrete_domain_type const& domain)
109 {
110 if (!domain.empty()) {
111 extents_type const extents_s((front<DDims>(domain) + extents<DDims>(domain)).uid()...);
112 std::array<std::size_t, sizeof...(DDims)> const strides_s {
113 allocation_mdspan.mapping().stride(
114 type_seq_rank_v<DDims, detail::TypeSeq<DDims...>>)...};
115 Kokkos::layout_stride::mapping<extents_type> const mapping_s(extents_s, strides_s);
116 return internal_mdspan_type(
117 allocation_mdspan.data_handle() - mapping_s(front<DDims>(domain).uid()...),
118 mapping_s);
119 }
120
121 return internal_mdspan_type(allocation_mdspan);
122 }

124 template <class QueryDDim, class... ODDims>
125 KOKKOS_FUNCTION constexpr auto get_slicer_for(DiscreteElement<ODDims...> const& c) const
126 {
127 DDC_IF_NVCC_THEN_PUSH_AND_SUPPRESS(implicit_return_from_non_void_function)
128 if constexpr (in_tags_v<QueryDDim, detail::TypeSeq<ODDims...>>) {
129 return (uid<QueryDDim>(c) - front<QueryDDim>(this->m_domain).uid());
130 } else {
131 return Kokkos::full_extent;
132 }
133 DDC_IF_NVCC_THEN_POP
134 }
135
136 template <class QueryDDim, class... ODDims>
137 KOKKOS_FUNCTION constexpr auto get_slicer_for(DiscreteDomain<ODDims...> const& c) const
138 {
139 DDC_IF_NVCC_THEN_PUSH_AND_SUPPRESS(implicit_return_from_non_void_function)
140 if constexpr (in_tags_v<QueryDDim, detail::TypeSeq<ODDims...>>) {
141 return std::pair<std::size_t, std::size_t>(
142 front<QueryDDim>(c) - front<QueryDDim>(this->m_domain),
143 back<QueryDDim>(c) + 1 - front<QueryDDim>(this->m_domain));
144 } else {
145 return Kokkos::full_extent;
146 }
147 DDC_IF_NVCC_THEN_POP
148 }

266 /** Slice out some dimensions
267 */
268 template <class... QueryDDims>
269 KOKKOS_FUNCTION constexpr auto operator[](
270 DiscreteElement<QueryDDims...> const& slice_spec) const
271 {
272 auto subview = Kokkos::submdspan(allocation_mdspan(), get_slicer_for<DDims>(slice_spec)...);
273 using layout_type = typename decltype(subview)::layout_type;
274 using extents_type = typename decltype(subview)::extents_type;
275 using detail::TypeSeq;
276 using OutTypeSeqDDims = type_seq_remove_t<TypeSeq<DDims...>, TypeSeq<QueryDDims...>>;
277 using OutDDom = detail::convert_type_seq_to_discrete_domain_t<OutTypeSeqDDims>;
278 if constexpr (
279 std::is_same_v<layout_type, Kokkos::Experimental::layout_left_padded<>>
280 || std::is_same_v<layout_type, Kokkos::Experimental::layout_right_padded<>>) {
281 Kokkos::layout_stride::mapping<extents_type> const mapping_stride(subview.mapping());
282 Kokkos::mdspan<ElementType, extents_type, Kokkos::layout_stride> const
283 a(subview.data_handle(), mapping_stride);
284 return ChunkSpan<
285 ElementType,
286 OutDDom,
287 Kokkos::layout_stride,
288 memory_space>(a, OutDDom(this->m_domain));
289 } else {
290 return ChunkSpan<
291 ElementType,
292 OutDDom,
293 layout_type,
294 memory_space>(subview, OutDDom(this->m_domain));
295 }
296 }

298 /** Restrict to a subdomain
299 */
300 template <class... QueryDDims>
301 KOKKOS_FUNCTION constexpr auto operator[](DiscreteDomain<QueryDDims...> const& odomain) const
302 {
303 auto subview = Kokkos::submdspan(allocation_mdspan(), get_slicer_for<DDims>(odomain)...);
304 using layout_type = typename decltype(subview)::layout_type;
305 using extents_type = typename decltype(subview)::extents_type;
306 if constexpr (
307 std::is_same_v<layout_type, Kokkos::Experimental::layout_left_padded<>>
308 || std::is_same_v<layout_type, Kokkos::Experimental::layout_right_padded<>>) {
309 Kokkos::layout_stride::mapping<extents_type> const mapping_stride(subview.mapping());
310 Kokkos::mdspan<ElementType, extents_type, Kokkos::layout_stride> const
311 a(subview.data_handle(), mapping_stride);
312 return ChunkSpan<
313 ElementType,
314 decltype(this->m_domain.restrict(odomain)),
315 Kokkos::layout_stride,
316 memory_space>(a, this->m_domain.restrict(odomain));
317 } else {
318 return ChunkSpan<
319 ElementType,
320 decltype(this->m_domain.restrict(odomain)),
321 layout_type,
322 memory_space>(subview, this->m_domain.restrict(odomain));
323 }
324 }

326 /** Element access using a list of DiscreteElement
327 * @param delems discrete elements
328 * @return reference to this element
329 */
330 template <class... DElems>
331 KOKKOS_FUNCTION constexpr reference operator()(DElems const&... delems) const noexcept
332 {
333 static_assert(
334 sizeof...(DDims) == (0 + ... + DElems::size()),
335 "Invalid number of dimensions");
336 static_assert((is_discrete_element_v<DElems> && ...), "Expected DiscreteElements");
337 assert(((select<DDims>(take<DDims>(delems...)) >= front<DDims>(this->m_domain)) && ...));
338 assert(((select<DDims>(take<DDims>(delems...)) <= back<DDims>(this->m_domain)) && ...));
339 return DDC_MDSPAN_ACCESS_OP(this->m_internal_mdspan, uid<DDims>(take<DDims>(delems...))...);
340 }
