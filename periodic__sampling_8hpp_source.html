136 static std::tuple<Impl<Kokkos::HostSpace>, discrete_domain_type> init(
137 continuous_element_type a,
138 continuous_element_type b,
139 discrete_vector_type n,
140 discrete_vector_type n_period)
141 {
142 assert(a < b);
143 assert(n > 1);
144 assert(n_period > 1);
145 Impl<Kokkos::HostSpace> disc(a, continuous_element_type {(b - a) / (n - 1)}, n_period);
146 discrete_domain_type domain {disc.front(), n};
147 return std::make_tuple(std::move(disc), std::move(domain));
148 }

160 static std::tuple<
161 Impl<Kokkos::HostSpace>,
162 discrete_domain_type,
163 discrete_domain_type,
164 discrete_domain_type,
165 discrete_domain_type>
166 init_ghosted(
167 continuous_element_type a,
168 continuous_element_type b,
169 discrete_vector_type n,
170 discrete_vector_type n_period,
171 discrete_vector_type n_ghosts_before,
172 discrete_vector_type n_ghosts_after)
173 {
174 using discrete_domain_type = discrete_domain_type;
175 assert(a < b);
176 assert(n > 1);
177 assert(n_period > 1);
178 Real discretization_step {(b - a) / (n - 1)};
179 Impl<Kokkos::HostSpace>
180 disc(a - n_ghosts_before.value() * discretization_step,
181 discretization_step,
182 n_period);
183 discrete_domain_type ghosted_domain
184 = discrete_domain_type(disc.front(), n + n_ghosts_before + n_ghosts_after);
185 discrete_domain_type pre_ghost
186 = discrete_domain_type(ghosted_domain.front(), n_ghosts_before);
187 discrete_domain_type main_domain
188 = discrete_domain_type(ghosted_domain.front() + n_ghosts_before, n);
189 discrete_domain_type post_ghost
190 = discrete_domain_type(main_domain.back() + 1, n_ghosts_after);
191 return std::make_tuple(
192 std::move(disc),
193 std::move(main_domain),
194 std::move(ghosted_domain),
195 std::move(pre_ghost),
196 std::move(post_ghost));
197 }
198
208 static std::tuple<
209 Impl<Kokkos::HostSpace>,
210 discrete_domain_type,

214 init_ghosted(

216 continuous_element_type b,
217 discrete_vector_type n,
218 discrete_vector_type n_period,
219 discrete_vector_type n_ghosts)
220 {
221 return init_ghosted(a, b, n, n_period, n_ghosts, n_ghosts);
222 }
