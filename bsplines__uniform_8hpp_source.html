229template <std::size_t Size>
230KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<UniformBSplines<Tag, D>> UniformBSplines<Tag, D>::
231 Impl<MemorySpace>::eval_basis(
232 std::array<double, Size>& values,
233 ddc::Coordinate<Tag> const& x,
234 [[maybe_unused]] std::size_t const deg) const
235{
236 assert(values.size() == deg + 1);
237
238 double offset;
239 int jmin;
240 // 1. Compute cell index 'icell' and x_offset
241 // 2. Compute index range of B-splines with support over cell 'icell'
242 get_icell_and_offset(jmin, offset, x);
243
244 // 3. Compute values of aforementioned B-splines
245 double xx, temp, saved;
246 values[0] = 1.0;
247 for (std::size_t j = 1; j < Size; ++j) {
248 xx = -offset;
249 saved = 0.0;
250 for (std::size_t r = 0; r < j; ++r) {
251 xx += 1;
252 temp = values[r] / j;
253 values[r] = saved + xx * temp;
254 saved = (j - xx) * temp;
255 }
256 values[j] = saved;
257 }
258
259 return discrete_element_type(jmin);
260}

263template <class MemorySpace>
264KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<UniformBSplines<Tag, D>> UniformBSplines<Tag, D>::Impl<
265 MemorySpace>::eval_deriv(std::array<double, D + 1>& derivs, ddc::Coordinate<Tag> const& x)
266 const
267{
268 assert(derivs.size() == degree() + 1);
269
270 double offset;
271 int jmin;
272 // 1. Compute cell index 'icell' and x_offset
273 // 2. Compute index range of B-splines with support over cell 'icell'
274 get_icell_and_offset(jmin, offset, x);
275
276 // 3. Compute derivatives of aforementioned B-splines
277 // Derivatives are normalized, hence they should be divided by dx
278 double xx, temp, saved;
279 derivs[0] = 1.0 / ddc::step<mesh_type>();
280 for (std::size_t j = 1; j < degree(); ++j) {
281 xx = -offset;
282 saved = 0.0;
283 for (std::size_t r = 0; r < j; ++r) {
284 xx += 1.0;
285 temp = derivs[r] / j;
286 derivs[r] = saved + xx * temp;
287 saved = (j - xx) * temp;
288 }
289 derivs[j] = saved;
290 }
291
292 // Compute derivatives
293 double bjm1 = derivs[0];
294 double bj = bjm1;
295 derivs[0] = -bjm1;
296 for (std::size_t j = 1; j < degree(); ++j) {
297 bj = derivs[j];
298 derivs[j] = bjm1 - bj;
299 bjm1 = bj;
300 }
301 derivs[degree()] = bj;
302
303 return discrete_element_type(jmin);
304}

307template <class MemorySpace>
308KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<UniformBSplines<Tag, D>> UniformBSplines<Tag, D>::
309 Impl<MemorySpace>::eval_basis_and_n_derivs(
310 ddc::DSpan2D const derivs,
311 ddc::Coordinate<Tag> const& x,
312 std::size_t const n) const
313{
314 std::array<double, (degree() + 1) * (degree() + 1)> ndu_ptr;
315 std::experimental::mdspan<
316 double,
317 std::experimental::extents<std::size_t, degree() + 1, degree() + 1>> const
318 ndu(ndu_ptr.data());
319 std::array<double, 2 * (degree() + 1)> a_ptr;
320 std::experimental::
321 mdspan<double, std::experimental::extents<std::size_t, degree() + 1, 2>> const a(
322 a_ptr.data());
323 double offset;
324 int jmin;
325
326 assert(x >= rmin());
327 assert(x <= rmax());
328 // assert(n >= 0); as long as n is unsigned
329 assert(n <= degree());
330 assert(derivs.extent(0) == 1 + degree());
331 assert(derivs.extent(1) == 1 + n);
332
333 // 1. Compute cell index 'icell' and x_offset
334 // 2. Compute index range of B-splines with support over cell 'icell'
335 get_icell_and_offset(jmin, offset, x);
336
337 // 3. Recursively evaluate B-splines (see
338 // "sll_s_uniform_BSplines_eval_basis")
339 // up to self%degree, and store them all in the upper-right triangle of
340 // ndu
341 double xx, temp, saved;
342 ndu(0, 0) = 1.0;
343 for (std::size_t j = 1; j < degree() + 1; ++j) {
344 xx = -offset;
345 saved = 0.0;
346 for (std::size_t r = 0; r < j; ++r) {
347 xx += 1.0;
348 temp = ndu(j - 1, r) / j;
349 ndu(j, r) = saved + xx * temp;
350 saved = (j - xx) * temp;
351 }
352 ndu(j, j) = saved;
353 }
354 for (std::size_t i = 0; i < ndu.extent(1); ++i) {
355 derivs(i, 0) = ndu(degree(), i);
356 }
357
358 for (int r = 0; r < int(degree() + 1); ++r) {
359 int s1 = 0;
360 int s2 = 1;
361 a(0, 0) = 1.0;
362 for (int k = 1; k < int(n + 1); ++k) {
363 double d = 0.0;
364 int const rk = r - k;
365 int const pk = degree() - k;
366 if (r >= k) {
367 a(0, s2) = a(0, s1) / (pk + 1);
368 d = a(0, s2) * ndu(pk, rk);
369 }
370 int const j1 = rk > -1 ? 1 : (-rk);
371 int const j2 = (r - 1) <= pk ? k : (degree() - r + 1);
372 for (int j = j1; j < j2; ++j) {
373 a(j, s2) = (a(j, s1) - a(j - 1, s1)) / (pk + 1);
374 d += a(j, s2) * ndu(pk, rk + j);
375 }
376 if (r <= pk) {
377 a(k, s2) = -a(k - 1, s1) / (pk + 1);
378 d += a(k, s2) * ndu(pk, r);
379 }
380 derivs(r, k) = d;
381 // swap s1 <-> s2
382 auto tmp = s1;
383 s1 = s2;
384 s2 = tmp;
385 }
386 }
387
388 // Multiply result by correct factors:
389 // degree!/(degree-n)! = degree*(degree-1)*...*(degree-n+1)
390 // k-th derivatives are normalized, hence they should be divided by dx^k
391 double const inv_dx = inv_step();
392 double d = degree() * inv_dx;
393 for (int k = 1; k < int(n + 1); ++k) {
394 for (std::size_t i = 0; i < derivs.extent(0); ++i) {
395 derivs(i, k) *= d;
396 }
397 d *= (degree() - k) * inv_dx;
398 }
399
400 return discrete_element_type(jmin);
401}

403template <class Tag, std::size_t D>
404template <class MemorySpace>
405KOKKOS_INLINE_FUNCTION void UniformBSplines<Tag, D>::Impl<MemorySpace>::get_icell_and_offset(
406 int& icell,
407 double& offset,
408 ddc::Coordinate<Tag> const& x) const
409{
410 assert(x >= rmin());
411 assert(x <= rmax());
412
413 double const inv_dx = inv_step();
414 if (x == rmin()) {
415 icell = 0;
416 offset = 0.0;
417 } else if (x == rmax()) {
418 icell = ncells() - 1;
419 offset = 1.0;
420 } else {
421 offset = (x - rmin()) * inv_dx;
422 icell = static_cast<int>(offset);
423 offset = offset - icell;
424
425 // When x is very close to xmax, round-off may cause the wrong answer
426 // icell=ncells and x_offset=0, which we convert to the case x=xmax:
427 if (icell == int(ncells()) && offset == 0.0) {
428 icell = ncells() - 1;
429 offset = 1.0;
430 }
431 }
432}

434template <class Tag, std::size_t D>
435template <class MemorySpace>
436template <class Layout, class MemorySpace2>
437KOKKOS_INLINE_FUNCTION ddc::ChunkSpan<
438 double,
439 ddc::DiscreteDomain<UniformBSplines<Tag, D>>,
440 Layout,
441 MemorySpace2>
442UniformBSplines<Tag, D>::Impl<MemorySpace>::integrals(
443 ddc::ChunkSpan<double, ddc::DiscreteDomain<UniformBSplines<Tag, D>>, Layout, MemorySpace2>
444 int_vals) const
445{
446 if constexpr (is_periodic()) {
447 assert(int_vals.size() == nbasis() || int_vals.size() == size());
448 } else {
449 assert(int_vals.size() == nbasis());
450 }
451 discrete_domain_type const full_dom_splines(full_domain());
452
453 if constexpr (is_periodic()) {
454 discrete_domain_type const dom_bsplines(
455 full_dom_splines.take_first(discrete_vector_type {nbasis()}));
456 for (auto ix : dom_bsplines) {
457 int_vals(ix) = ddc::step<mesh_type>();
458 }
459 if (int_vals.size() == size()) {
460 discrete_domain_type const dom_bsplines_repeated(
461 full_dom_splines.take_last(discrete_vector_type {degree()}));
462 for (auto ix : dom_bsplines_repeated) {
463 int_vals(ix) = 0;
464 }
465 }
466 } else {
467 discrete_domain_type const dom_bspline_entirely_in_domain
468 = full_dom_splines
469 .remove(discrete_vector_type(degree()), discrete_vector_type(degree()));
470 for (auto ix : dom_bspline_entirely_in_domain) {
471 int_vals(ix) = ddc::step<mesh_type>();
472 }
473
474 std::array<double, degree() + 2> edge_vals_ptr;
475 std::experimental::
476 mdspan<double, std::experimental::extents<std::size_t, degree() + 2>> const
477 edge_vals(edge_vals_ptr.data());
478
479 eval_basis(edge_vals_ptr, rmin(), degree() + 1);
480
481 double const d_eval = ddc::detail::sum(edge_vals);
482
483 for (std::size_t i = 0; i < degree(); ++i) {
484 double const c_eval = ddc::detail::sum(edge_vals, 0, degree() - i);
485
486 double const edge_value = ddc::step<mesh_type>() * (d_eval - c_eval);
487
488 int_vals(discrete_element_type(i)) = edge_value;
489 int_vals(discrete_element_type(nbasis() - 1 - i)) = edge_value;
490 }
491 }
492 return int_vals;
493}
