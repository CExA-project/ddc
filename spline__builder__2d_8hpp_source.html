377 ddc::ChunkSpan<double const, batched_interpolation_domain_type, Layout, memory_space> vals,
378 std::optional<ddc::ChunkSpan<
379 double const,
380 batched_derivs_domain_type1,
381 Layout,
382 memory_space>> const derivs_min1,
383 std::optional<ddc::ChunkSpan<
384 double const,
385 batched_derivs_domain_type1,
386 Layout,
387 memory_space>> const derivs_max1,
388 std::optional<ddc::ChunkSpan<
389 double const,
390 batched_derivs_domain_type2,
391 Layout,
392 memory_space>> const derivs_min2,
393 std::optional<ddc::ChunkSpan<
394 double const,
395 batched_derivs_domain_type2,
396 Layout,
397 memory_space>> const derivs_max2,
398 std::optional<ddc::ChunkSpan<
399 double const,
400 batched_derivs_domain_type,
401 Layout,
402 memory_space>> const mixed_derivs_min1_min2,
403 std::optional<ddc::ChunkSpan<
404 double const,
405 batched_derivs_domain_type,
406 Layout,
407 memory_space>> const mixed_derivs_max1_min2,
408 std::optional<ddc::ChunkSpan<
409 double const,
410 batched_derivs_domain_type,
411 Layout,
412 memory_space>> const mixed_derivs_min1_max2,
413 std::optional<ddc::ChunkSpan<
414 double const,
415 batched_derivs_domain_type,
416 Layout,
417 memory_space>> const mixed_derivs_max1_max2) const
418{
419 // TODO: perform computations along dimension 1 on different streams ?
420 // Spline1-transform derivs_min2 (to spline1_deriv_min)
421 ddc::Chunk spline1_deriv_min_alloc(
422 m_spline_builder_deriv1.batched_spline_domain(),
423 ddc::KokkosAllocator<double, MemorySpace>());
424 auto spline1_deriv_min = spline1_deriv_min_alloc.span_view();
425 auto spline1_deriv_min_opt = std::optional(spline1_deriv_min.span_cview());
426 if constexpr (BcXmin2 == ddc::BoundCond::HERMITE) {
427 m_spline_builder_deriv1(
428 spline1_deriv_min,
429 *derivs_min2,
430 mixed_derivs_min1_min2,
431 mixed_derivs_max1_min2);
432 } else {
433 spline1_deriv_min_opt = std::nullopt;
434 }
435
436 // Spline1-transform vals (to spline1)
437 ddc::Chunk spline1_alloc(
438 m_spline_builder1.batched_spline_domain(),
439 ddc::KokkosAllocator<double, MemorySpace>());
440 ddc::ChunkSpan spline1 = spline1_alloc.span_view();
441
442 m_spline_builder1(spline1, vals, derivs_min1, derivs_max1);
443
444 // Spline1-transform derivs_max2 (to spline1_deriv_max)
445 ddc::Chunk spline1_deriv_max_alloc(
446 m_spline_builder_deriv1.batched_spline_domain(),
447 ddc::KokkosAllocator<double, MemorySpace>());
448 auto spline1_deriv_max = spline1_deriv_max_alloc.span_view();
449 auto spline1_deriv_max_opt = std::optional(spline1_deriv_max.span_cview());
450 if constexpr (BcXmax2 == ddc::BoundCond::HERMITE) {
451 m_spline_builder_deriv1(
452 spline1_deriv_max,
453 *derivs_max2,
454 mixed_derivs_min1_max2,
455 mixed_derivs_max1_max2);
456 } else {
457 spline1_deriv_max_opt = std::nullopt;
458 }
459
460 // Spline2-transform spline1
461 m_spline_builder2(spline, spline1.span_cview(), spline1_deriv_min_opt, spline1_deriv_max_opt);
462}
