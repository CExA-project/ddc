16/**
17 * @brief A functor for describing a spline boundary value by a constant extrapolation for 1D evaluator.
18 *
19 * To define the value of a function on B-splines out of the domain, we here use a constant
20 * extrapolation on the edge.
21 */
22template <class DimI>
23struct ConstantExtrapolationRule<DimI>
24{
25private:
26 ddc::Coordinate<DimI> m_eval_pos;
27
28public:
29 /**
30 * @brief Instantiate a ConstantExtrapolationRule.
31 *
32 * The boundary value will be the same as at the coordinate eval_pos given.
33 *
34 * @param[in] eval_pos
35 * Coordinate inside the domain where we will evaluate each points outside the domain.
36 */
37 explicit ConstantExtrapolationRule(ddc::Coordinate<DimI> eval_pos) : m_eval_pos(eval_pos) {}
38
39 /**
40 * @brief Get the value of the function on B-splines at a coordinate outside the domain.
41 *
42 * @param[in] pos
43 * The coordinate where we want to evaluate the function on B-splines.
44 * @param[in] spline_coef
45 * The coefficients of the function on B-splines.
46 *
47 * @return A double with the value of the function on B-splines evaluated at the coordinate.
48 */
49 template <class CoordType, class BSplines, class Layout, class MemorySpace>
50 KOKKOS_FUNCTION double operator()(
51 CoordType,
52 ddc::ChunkSpan<double const, ddc::DiscreteDomain<BSplines>, Layout, MemorySpace> const
53 spline_coef) const
54 {
55 static_assert(in_tags_v<DimI, to_type_seq_t<CoordType>>);
56
57 std::array<double, BSplines::degree() + 1> vals;
58
59 ddc::DiscreteElement<BSplines> idx
60 = ddc::discrete_space<BSplines>().eval_basis(vals, m_eval_pos);
61
62 double y = 0.0;
63 for (std::size_t i = 0; i < BSplines::degree() + 1; ++i) {
64 y += spline_coef(idx + i) * vals[i];
65 }
66 return y;
67 }
68};

70/**
71 * @brief A functor for describing a spline boundary value by a constant extrapolation for 2D evaluator.
72 *
73 * To define the value of a function on B-splines out of the domain, we here use a constant
74 * extrapolation on the edge.
75 */
76template <class DimI, class DimNI>
77struct ConstantExtrapolationRule<DimI, DimNI>
78{
79private:
80 ddc::Coordinate<DimI> m_eval_pos;
81 ddc::Coordinate<DimNI> m_eval_pos_not_interest_min;
82 ddc::Coordinate<DimNI> m_eval_pos_not_interest_max;
83
84public:
85 /**
86 * @brief Instantiate a ConstantExtrapolationRule.
87 *
88 * The boundary value will be the same as at the coordinate given in a dimension given.
89 * The dimension of the input defines the dimension of the boundary condition.
90 * The second and the third parameters are needed in case of non-periodic splines on the
91 * dimension off-interest (the complementary dimension of the boundary condition),
92 * because the evaluator can receive coordinates outside the domain in both dimension.
93 *
94 * @param[in] eval_pos_bc
95 * Coordinate in the dimension given inside the domain where we will evaluate
96 * each points outside the domain.
97 * @param[in] eval_pos_not_interest_min
98 * The minimum coordinate inside the domain on the complementary dimension of the boundary condition.
99 * @param[in] eval_pos_not_interest_max
100 * The maximum coordinate inside the domain on the complementary dimension of the boundary condition.
101 */
102 explicit ConstantExtrapolationRule(
103 ddc::Coordinate<DimI> eval_pos,
104 ddc::Coordinate<DimNI> eval_pos_not_interest_min,
105 ddc::Coordinate<DimNI> eval_pos_not_interest_max)
106 : m_eval_pos(eval_pos)
107 , m_eval_pos_not_interest_min(eval_pos_not_interest_min)
108 , m_eval_pos_not_interest_max(eval_pos_not_interest_max)
109 {
110 }
111
112 /**
113 * @brief Instantiate a ConstantExtrapolationRule.
114 *
115 * The boundary value will be the same as at the coordinate given in a dimension given.
116 * The dimension of the input defines the dimension of the boundary condition.
117 * No second and third parameters are needed in case of periodic splines on the
118 * dimension off-interest (the complementary dimension of the boundary condition).
119 *
120 * @param[in] eval_pos_bc
121 * Coordinate in the dimension given inside the domain where we will evaluate
122 * each points outside the domain.
123 */
124 template <class DimNI_sfinae = DimNI, std::enable_if_t<DimNI_sfinae::PERIODIC, int> = 0>
125 explicit ConstantExtrapolationRule(ddc::Coordinate<DimI> eval_pos)
126 : m_eval_pos(eval_pos)
127 , m_eval_pos_not_interest_min(0.)
128 , m_eval_pos_not_interest_max(0.)
129 {
130 }
131
132 /**
133 * @brief Get the value of the function on B-splines at a coordinate outside the domain.
134 *
135 * In the dimension defined in the constructor Dim1 (or Dim2), it sets the coordinate pos_1 (or pos_2)
136 * given at the m_eval_pos coordinate if it is outside the domain.
137 * If the coordinate on the complementary dimension of the boundary condition dimension ddc::select<DimNI>(coord_extrap) is
138 * outside the domain, then it also sets the coordinate at eval_pos_not_interest_min
139 * (if ddc::select<DimNI>(coord_extrap) @f$ < @f$ eval_pos_not_interest_min) or
140 * at eval_pos_not_interest_max (if ddc::select<DimNI>(coord_extrap) @f$ > @f$ eval_pos_not_interest_max).
141 *
142 * @param[in] coord_extrap
143 * The coordinates where we want to evaluate the function on B-splines
144 * @param[in] spline_coef
145 * The coefficients of the function on B-splines.
146 *
147 *@return A double with the value of the function on B-splines evaluated at the coordinate.
148 */
149 template <class CoordType, class BSplines1, class BSplines2, class Layout, class MemorySpace>
150 KOKKOS_FUNCTION double operator()(
151 CoordType coord_extrap,
152 ddc::ChunkSpan<
153 double const,
154 ddc::DiscreteDomain<BSplines1, BSplines2>,
155 Layout,
156 MemorySpace> const spline_coef) const
157 {
158 static_assert(
159 in_tags_v<
160 DimI,
161 to_type_seq_t<CoordType>> && in_tags_v<DimNI, to_type_seq_t<CoordType>>);
162
163 ddc::Coordinate<DimI, DimNI> eval_pos;
164 if constexpr (DimNI::PERIODIC) {
165 eval_pos = ddc::Coordinate<DimI, DimNI>(m_eval_pos, ddc::select<DimNI>(coord_extrap));
166 } else {
167 eval_pos = ddc::Coordinate<DimI, DimNI>(
168 m_eval_pos,
169 Kokkos::
170 clamp(ddc::select<DimNI>(coord_extrap),
171 m_eval_pos_not_interest_min,
172 m_eval_pos_not_interest_max));
173 }
174
175 std::array<double, BSplines1::degree() + 1> vals1;
176 std::array<double, BSplines2::degree() + 1> vals2;
177
178 ddc::DiscreteElement<BSplines1> idx1
179 = ddc::discrete_space<BSplines1>()
180 .eval_basis(vals1, ddc::select<typename BSplines1::tag_type>(eval_pos));
181 ddc::DiscreteElement<BSplines2> idx2
182 = ddc::discrete_space<BSplines2>()
183 .eval_basis(vals2, ddc::select<typename BSplines2::tag_type>(eval_pos));
184
185 double y = 0.0;
186 for (std::size_t i = 0; i < BSplines1::degree() + 1; ++i) {
187 for (std::size_t j = 0; j < BSplines2::degree() + 1; ++j) {
188 y += spline_coef(idx1 + i, idx2 + j) * vals1[i] * vals2[j];
189 }
190 }
191
192 return y;
193 }
194};
195} // namespace ddc
