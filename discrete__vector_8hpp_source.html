73 DiscreteVector<Tags...> const& lhs,
74 DiscreteVector<OTags...> const& rhs)
75{
76 using detail::TypeSeq;
77 if constexpr (sizeof...(Tags) >= sizeof...(OTags)) {
78 static_assert(((type_seq_contains_v<TypeSeq<OTags>, TypeSeq<Tags...>>)&&...));
79 DiscreteVector<Tags...> result(lhs);

81 return result;

84 DiscreteVector<OTags...> result(rhs);

86 return result;

164template <class QueryTag, class HeadTag, class... TailTags>
165constexpr DiscreteVector<QueryTag> const& take(
166 DiscreteVector<HeadTag> const& head,
167 DiscreteVector<TailTags> const&... tags)
168{
169 DDC_IF_NVCC_THEN_PUSH_AND_SUPPRESS(implicit_return_from_non_void_function)
170 static_assert(
171 !type_seq_contains_v<detail::TypeSeq<HeadTag>, detail::TypeSeq<TailTags...>>,
172 "ERROR: tag redundant");
173 if constexpr (std::is_same_v<QueryTag, HeadTag>) {
174 return head;

176 static_assert(sizeof...(TailTags) > 0, "ERROR: tag not found");
177 return take<QueryTag>(tags...);
178 }

180}
