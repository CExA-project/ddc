95 template <class QueryDDim, class... ODDims>
96 KOKKOS_FUNCTION constexpr auto get_slicer_for(DiscreteElement<ODDims...> const& c) const
97 {
98 DDC_IF_NVCC_THEN_PUSH_AND_SUPPRESS(implicit_return_from_non_void_function)
99 if constexpr (in_tags_v<QueryDDim, detail::TypeSeq<ODDims...>>) {
100 return (uid<QueryDDim>(c) - front<QueryDDim>(this->m_domain).uid());
101 } else {
102 return std::experimental::full_extent;
103 }
104 DDC_IF_NVCC_THEN_POP
105 }
106
107 template <class QueryDDim, class... ODDims>
108 KOKKOS_FUNCTION constexpr auto get_slicer_for(DiscreteDomain<ODDims...> const& c) const
109 {
110 DDC_IF_NVCC_THEN_PUSH_AND_SUPPRESS(implicit_return_from_non_void_function)
111 if constexpr (in_tags_v<QueryDDim, detail::TypeSeq<ODDims...>>) {
112 return std::pair<std::size_t, std::size_t>(
113 front<QueryDDim>(c) - front<QueryDDim>(this->m_domain),
114 back<QueryDDim>(c) + 1 - front<QueryDDim>(this->m_domain));
115 } else {
116 return std::experimental::full_extent;
117 }
118 DDC_IF_NVCC_THEN_POP
119 }

190 KOKKOS_FUNCTION constexpr ChunkSpan(
191 allocation_mdspan_type allocation_mdspan,
192 mdomain_type const& domain)
193 {
194 namespace stdex = std::experimental;
195 extents_type extents_s((front<DDims>(domain) + extents<DDims>(domain)).uid()...);
196 std::array<std::size_t, sizeof...(DDims)> strides_s {allocation_mdspan.mapping().stride(
197 type_seq_rank_v<DDims, detail::TypeSeq<DDims...>>)...};
198 stdex::layout_stride::mapping<extents_type> mapping_s(extents_s, strides_s);
199 this->m_internal_mdspan = internal_mdspan_type(
200 allocation_mdspan.data_handle() - mapping_s(front<DDims>(domain).uid()...),
201 mapping_s);
202 this->m_domain = domain;
203 }

233 template <class... QueryDDims>
234 KOKKOS_FUNCTION constexpr auto operator[](
235 DiscreteElement<QueryDDims...> const& slice_spec) const
236 {
237 auto subview = std::experimental::
238 submdspan(allocation_mdspan(), get_slicer_for<DDims>(slice_spec)...);
239 using detail::TypeSeq;
240 using selected_meshes = type_seq_remove_t<TypeSeq<DDims...>, TypeSeq<QueryDDims...>>;
241 return ChunkSpan<
242 ElementType,
243 decltype(select_by_type_seq<selected_meshes>(this->m_domain)),
244 typename decltype(subview)::layout_type,
245 memory_space>(subview, select_by_type_seq<selected_meshes>(this->m_domain));
246 }
247
250 template <class... QueryDDims>
251 KOKKOS_FUNCTION constexpr auto operator[](DiscreteDomain<QueryDDims...> const& odomain) const
252 {
253 auto subview = std::experimental::
254 submdspan(allocation_mdspan(), get_slicer_for<DDims>(odomain)...);
255 return ChunkSpan<
256 ElementType,
257 decltype(this->m_domain.restrict(odomain)),
258 typename decltype(subview)::layout_type,
259 memory_space>(subview, this->m_domain.restrict(odomain));
260 }
261
266 template <class... DElems>
267 KOKKOS_FUNCTION constexpr reference operator()(DElems const&... delems) const noexcept
268 {
269 static_assert(
270 sizeof...(DDims) == (0 + ... + DElems::size()),
271 "Invalid number of dimensions");
272 static_assert((is_discrete_element_v<DElems> && ...), "Expected DiscreteElements");
273 assert(((select<DDims>(take<DDims>(delems...)) >= front<DDims>(this->m_domain)) && ...));
274 assert(((select<DDims>(take<DDims>(delems...)) <= back<DDims>(this->m_domain)) && ...));
275 return this->m_internal_mdspan(uid<DDims>(take<DDims>(delems...))...);
276 }

308 }
