124 template <class DDim>
125 static std::tuple<typename DDim::template Impl<DDim, Kokkos::HostSpace>, DiscreteDomain<DDim>>
126 init(Coordinate<CDim> a, Coordinate<CDim> b, DiscreteVector<DDim> n)
127 {
128 assert(a < b);
129 assert(n > 1);
130 typename DDim::template Impl<DDim, Kokkos::HostSpace>
131 disc(a, Coordinate<CDim> {(b - a) / (n - 1)});
132 DiscreteDomain<DDim> domain {disc.front(), n};
133 return std::make_tuple(std::move(disc), std::move(domain));
134 }

145 template <class DDim>
146 static std::tuple<
147 typename DDim::template Impl<DDim, Kokkos::HostSpace>,
148 DiscreteDomain<DDim>,
149 DiscreteDomain<DDim>,
150 DiscreteDomain<DDim>,
151 DiscreteDomain<DDim>>
152 init_ghosted(
153 Coordinate<CDim> a,
154 Coordinate<CDim> b,
155 DiscreteVector<DDim> n,
156 DiscreteVector<DDim> n_ghosts_before,
157 DiscreteVector<DDim> n_ghosts_after)
158 {
159 using discrete_domain_type = DiscreteDomain<DDim>;
160 assert(a < b);
161 assert(n > 1);
162 Real discretization_step {(b - a) / (n - 1)};
163 typename DDim::template Impl<DDim, Kokkos::HostSpace>
164 disc(a - n_ghosts_before.value() * discretization_step, discretization_step);
165 discrete_domain_type ghosted_domain
166 = discrete_domain_type(disc.front(), n + n_ghosts_before + n_ghosts_after);
167 discrete_domain_type pre_ghost
168 = discrete_domain_type(ghosted_domain.front(), n_ghosts_before);
169 discrete_domain_type main_domain
170 = discrete_domain_type(ghosted_domain.front() + n_ghosts_before, n);
171 discrete_domain_type post_ghost
172 = discrete_domain_type(main_domain.back() + 1, n_ghosts_after);
173 return std::make_tuple(
174 std::move(disc),
175 std::move(main_domain),
176 std::move(ghosted_domain),
177 std::move(pre_ghost),
178 std::move(post_ghost));
179 }

198 Coordinate<CDim> b,
