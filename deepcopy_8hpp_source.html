13/** Copy the content of a borrowed chunk into another
14 * @param[out] dst the borrowed chunk in which to copy
15 * @param[in] src the borrowed chunk from which to copy
16 * @return dst as a ChunkSpan
17*/
18template <class ChunkDst, class ChunkSrc>
19auto deepcopy(ChunkDst&& dst, ChunkSrc&& src)
20{
21 static_assert(is_borrowed_chunk_v<ChunkDst>);
22 static_assert(is_borrowed_chunk_v<ChunkSrc>);
23 static_assert(
24 std::is_assignable_v<chunk_reference_t<ChunkDst>, chunk_reference_t<ChunkSrc>>,
25 "Not assignable");
26 assert(dst.domain().extents() == src.domain().extents());
27 Kokkos::deep_copy(dst.allocation_kokkos_view(), src.allocation_kokkos_view());
28 return dst.span_view();
29}
