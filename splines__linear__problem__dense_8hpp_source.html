29class SplinesLinearProblemDense : public SplinesLinearProblem<ExecSpace>

35protected:
36 Kokkos::View<double**, Kokkos::LayoutRight, Kokkos::HostSpace> m_a;

39public:
45 explicit SplinesLinearProblemDense(const std::size_t mat_size)
46 : SplinesLinearProblem<ExecSpace>(mat_size)
47 , m_a("a", mat_size, mat_size)
48 , m_ipiv("ipiv", mat_size)
49 {
50 Kokkos::deep_copy(m_a, 0.);
51 }
52
53 virtual double get_element(std::size_t const i, std::size_t const j) const override
54 {
55 assert(i < size());
56 assert(j < size());
57 return m_a(i, j);
58 }
59
60 virtual void set_element(std::size_t const i, std::size_t const j, double const aij) override
61 {
62 assert(i < size());
63 assert(j < size());
64 m_a(i, j) = aij;
65 }
66
72 void setup_solver() override
73 {
74 int const info = LAPACKE_dgetrf(
75 LAPACK_ROW_MAJOR,
76 size(),
77 size(),
78 m_a.data(),
79 size(),
80 m_ipiv.data());
81 if (info != 0) {
82 throw std::runtime_error("LAPACK failed with error code " + std::to_string(info));
83 }
84 }
85
94 void solve(MultiRHS b, bool const transpose) const override
95 {
96 assert(b.extent(0) == size());
97
98 auto b_host = create_mirror_view(Kokkos::DefaultHostExecutionSpace(), b);
99 Kokkos::deep_copy(b_host, b);
100 int const info = LAPACKE_dgetrs(
101 LAPACK_ROW_MAJOR,
102 transpose ? 'T' : 'N',
103 b_host.extent(0),
104 b_host.extent(1),
105 m_a.data(),
106 b_host.extent(0),
107 m_ipiv.data(),
108 b_host.data(),
109 b_host.stride(0));
110 if (info != 0) {
111 throw std::runtime_error("LAPACK failed with error code " + std::to_string(info));
112 }
113 Kokkos::deep_copy(b, b_host);
114 }
115};
