411template <class Tag, std::size_t D>
412template <class DDim, class MemorySpace>
413template <class RandomIt>
414NonUniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::Impl(
415 RandomIt const break_begin,
416 RandomIt const break_end)
417 : m_knot_domain(
418 ddc::DiscreteElement<knot_mesh_type>(0),
419 ddc::DiscreteVector<knot_mesh_type>(
420 (break_end - break_begin)
421 + 2 * degree())) // Create a mesh of knots including the eventual periodic point
422 , m_break_point_domain(
423 ddc::DiscreteElement<knot_mesh_type>(degree()),
424 ddc::DiscreteVector<knot_mesh_type>(
425 (break_end - break_begin))) // Create a mesh of break points
426{
427 std::vector<ddc::Coordinate<Tag>> knots((break_end - break_begin) + 2 * degree());
428 // Fill the provided knots
429 int ii = 0;
430 for (RandomIt it = break_begin; it < break_end; ++it) {
431 knots[degree() + ii] = *it;
432 ++ii;
433 }
434 ddc::Coordinate<Tag> const rmin = knots[degree()];
435 ddc::Coordinate<Tag> const rmax = knots[(break_end - break_begin) + degree() - 1];
436 assert(rmin < rmax);
437
438 // Fill out the extra knots
439 if constexpr (is_periodic()) {
440 double const period = rmax - rmin;
441 for (std::size_t i = 1; i < degree() + 1; ++i) {
442 knots[degree() + -i] = knots[degree() + ncells() - i] - period;
443 knots[degree() + ncells() + i] = knots[degree() + i] + period;
444 }
445 } else // open
446 {
447 for (std::size_t i = 1; i < degree() + 1; ++i) {
448 knots[degree() + -i] = rmin;
449 knots[degree() + npoints() - 1 + i] = rmax;
450 }
451 }
452 ddc::init_discrete_space<knot_mesh_type>(knots);
453}

456template <class DDim, class MemorySpace>
457KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDim> NonUniformBSplines<Tag, D>::
458 Impl<DDim, MemorySpace>::eval_basis(DSpan1D values, ddc::Coordinate<Tag> const& x) const
459{
460 assert(values.size() == D + 1);
461
462 std::array<double, degree()> left;
463 std::array<double, degree()> right;
464
465 assert(x >= rmin());
466 assert(x <= rmax());
467 assert(values.size() == degree() + 1);
468
469 // 1. Compute cell index 'icell'
470 ddc::DiscreteElement<knot_mesh_type> const icell = find_cell_start(x);
471
472 assert(icell >= m_break_point_domain.front());
473 assert(icell <= m_break_point_domain.back());
474 assert(ddc::coordinate(icell) <= x);
475 assert(ddc::coordinate(icell + 1) >= x);
476
477 // 2. Compute values of B-splines with support over cell 'icell'
478 double temp;
479 values[0] = 1.0;
480 for (std::size_t j = 0; j < degree(); ++j) {
481 left[j] = x - ddc::coordinate(icell - j);
482 right[j] = ddc::coordinate(icell + j + 1) - x;
483 double saved = 0.0;
484 for (std::size_t r = 0; r < j + 1; ++r) {
485 temp = values[r] / (right[r] + left[j - r]);
486 values[r] = saved + right[r] * temp;
487 saved = left[j - r] * temp;
488 }
489 values[j + 1] = saved;
490 }
491
492 return get_first_bspline_in_cell(icell);
493}

495template <class Tag, std::size_t D>
496template <class DDim, class MemorySpace>
497KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDim> NonUniformBSplines<Tag, D>::
498 Impl<DDim, MemorySpace>::eval_deriv(DSpan1D derivs, ddc::Coordinate<Tag> const& x) const
499{
500 std::array<double, degree()> left;
501 std::array<double, degree()> right;
502
503 assert(x >= rmin());
504 assert(x <= rmax());
505 assert(derivs.size() == degree() + 1);
506
507 // 1. Compute cell index 'icell'
508 ddc::DiscreteElement<knot_mesh_type> const icell = find_cell_start(x);
509
510 assert(icell >= m_break_point_domain.front());
511 assert(icell <= m_break_point_domain.back());
512 assert(ddc::coordinate(icell) <= x);
513 assert(ddc::coordinate(icell + 1) >= x);
514
515 // 2. Compute values of derivatives of B-splines with support over cell 'icell'
516
517 /*
518 * Compute nonzero basis functions and knot differences
519 * for splines up to degree degree-1 which are needed to compute derivative
520 * First part of Algorithm A3.2 of NURBS book
521 */
522 double saved, temp;
523 derivs[0] = 1.0;
524 for (std::size_t j = 0; j < degree() - 1; ++j) {
525 left[j] = x - ddc::coordinate(icell - j);
526 right[j] = ddc::coordinate(icell + j + 1) - x;
527 saved = 0.0;
528 for (std::size_t r = 0; r < j + 1; ++r) {
529 temp = derivs[r] / (right[r] + left[j - r]);
530 derivs[r] = saved + right[r] * temp;
531 saved = left[j - r] * temp;
532 }
533 derivs[j + 1] = saved;
534 }
535
536 /*
537 * Compute derivatives at x using values stored in bsdx and formula
538 * for spline derivative based on difference of splines of degree degree-1
539 */
540 saved = degree() * derivs[0]
541 / (ddc::coordinate(icell + 1) - ddc::coordinate(icell + 1 - degree()));
542 derivs[0] = -saved;
543 for (std::size_t j = 1; j < degree(); ++j) {
544 temp = saved;
545 saved = degree() * derivs[j]
546 / (ddc::coordinate(icell + j + 1) - ddc::coordinate(icell + j + 1 - degree()));
547 derivs[j] = temp - saved;
548 }
549 derivs[degree()] = saved;
550
551 return get_first_bspline_in_cell(icell);
552}

554template <class Tag, std::size_t D>
555template <class DDim, class MemorySpace>
556KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDim> NonUniformBSplines<Tag, D>::
557 Impl<DDim, MemorySpace>::eval_basis_and_n_derivs(
558 ddc::DSpan2D const derivs,
559 ddc::Coordinate<Tag> const& x,
560 std::size_t const n) const
561{
562 std::array<double, degree()> left;
563 std::array<double, degree()> right;
564
565 std::array<double, 2 * (degree() + 1)> a_ptr;
566 std::experimental::
567 mdspan<double, std::experimental::extents<std::size_t, degree() + 1, 2>> const a(
568 a_ptr.data());
569
570 std::array<double, (degree() + 1) * (degree() + 1)> ndu_ptr;
571 std::experimental::mdspan<
572 double,
573 std::experimental::extents<std::size_t, degree() + 1, degree() + 1>> const
574 ndu(ndu_ptr.data());
575
576 assert(x >= rmin());
577 assert(x <= rmax());
578 // assert(n >= 0); as long as n is unsigned
579 assert(n <= degree());
580 assert(derivs.extent(0) == 1 + degree());
581 assert(derivs.extent(1) == 1 + n);
582
583 // 1. Compute cell index 'icell' and x_offset
584 ddc::DiscreteElement<knot_mesh_type> const icell = find_cell_start(x);
585
586 assert(icell >= m_break_point_domain.front());
587 assert(icell <= m_break_point_domain.back());
588 assert(ddc::coordinate(icell) <= x);
589 assert(ddc::coordinate(icell + 1) >= x);
590
591 // 2. Compute nonzero basis functions and knot differences for splines
592 // up to degree (degree-1) which are needed to compute derivative
593 // Algorithm A2.3 of NURBS book
594 //
595 // 21.08.2017: save inverse of knot differences to avoid unnecessary
596 // divisions
597 // [Yaman Gl, Edoardo Zoni]
598
599 double saved, temp;
600 ndu(0, 0) = 1.0;
601 for (std::size_t j = 0; j < degree(); ++j) {
602 left[j] = x - ddc::coordinate(icell - j);
603 right[j] = ddc::coordinate(icell + j + 1) - x;
604 saved = 0.0;
605 for (std::size_t r = 0; r < j + 1; ++r) {
606 // compute inverse of knot differences and save them into lower
607 // triangular part of ndu
608 ndu(r, j + 1) = 1.0 / (right[r] + left[j - r]);
609 // compute basis functions and save them into upper triangular part
610 // of ndu
611 temp = ndu(j, r) * ndu(r, j + 1);
612 ndu(j + 1, r) = saved + right[r] * temp;
613 saved = left[j - r] * temp;
614 }
615 ndu(j + 1, j + 1) = saved;
616 }
617 // Save 0-th derivative
618 for (std::size_t j = 0; j < degree() + 1; ++j) {
619 derivs(j, 0) = ndu(degree(), j);
620 }
621
622 for (int r = 0; r < int(degree() + 1); ++r) {
623 int s1 = 0;
624 int s2 = 1;
625 a(0, 0) = 1.0;
626 for (int k = 1; k < int(n + 1); ++k) {
627 double d = 0.0;
628 int const rk = r - k;
629 int const pk = degree() - k;
630 if (r >= k) {
631 a(0, s2) = a(0, s1) * ndu(rk, pk + 1);
632 d = a(0, s2) * ndu(pk, rk);
633 }
634 int const j1 = rk > -1 ? 1 : (-rk);
635 int const j2 = (r - 1) <= pk ? k : (degree() - r + 1);
636 for (int j = j1; j < j2; ++j) {
637 a(j, s2) = (a(j, s1) - a(j - 1, s1)) * ndu(rk + j, pk + 1);
638 d += a(j, s2) * ndu(pk, rk + j);
639 }
640 if (r <= pk) {
641 a(k, s2) = -a(k - 1, s1) * ndu(r, pk + 1);
642 d += a(k, s2) * ndu(pk, r);
643 }
644 derivs(r, k) = d;
645 Kokkos::kokkos_swap(s1, s2);
646 }
647 }
648
649 int r = degree();
650 for (int k = 1; k < int(n + 1); ++k) {
651 for (std::size_t i = 0; i < derivs.extent(0); i++) {
652 derivs(i, k) *= r;
653 }
654 r *= degree() - k;
655 }
656
657 return get_first_bspline_in_cell(icell);
658}

660template <class Tag, std::size_t D>
661template <class DDim, class MemorySpace>
662KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<NonUniformBsplinesKnots<DDim>> NonUniformBSplines<
663 Tag,
664 D>::Impl<DDim, MemorySpace>::find_cell_start(ddc::Coordinate<Tag> const& x) const
665{
666 assert(x <= rmax());
667 assert(x >= rmin());
668
669 if (x == rmin())
670 return m_break_point_domain.front();
671 if (x == rmax())
672 return m_break_point_domain.back() - 1;
673
674 // Binary search
675 ddc::DiscreteElement<knot_mesh_type> low = m_break_point_domain.front();
676 ddc::DiscreteElement<knot_mesh_type> high = m_break_point_domain.back();
677 ddc::DiscreteElement<knot_mesh_type> icell = low + (high - low) / 2;
678 while (x < ddc::coordinate(icell) || x >= ddc::coordinate(icell + 1)) {
679 if (x < ddc::coordinate(icell)) {
680 high = icell;
681 } else {
682 low = icell;
683 }
684 icell = low + (high - low) / 2;
685 }
686 return icell;
687}

689template <class Tag, std::size_t D>
690template <class DDim, class MemorySpace>
691template <class Layout, class MemorySpace2>
692KOKKOS_INLINE_FUNCTION ddc::ChunkSpan<double, ddc::DiscreteDomain<DDim>, Layout, MemorySpace2>
693NonUniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::integrals(
694 ddc::ChunkSpan<double, discrete_domain_type, Layout, MemorySpace2> int_vals) const
695{
696 if constexpr (is_periodic()) {
697 assert(int_vals.size() == nbasis() || int_vals.size() == size());
698 } else {
699 assert(int_vals.size() == nbasis());
700 }
701
702 double const inv_deg = 1.0 / (degree() + 1);
703
704 discrete_domain_type const dom_bsplines(
705 full_domain().take_first(discrete_vector_type {nbasis()}));
706 for (auto ix : dom_bsplines) {
707 int_vals(ix) = double(ddc::coordinate(get_last_support_knot(ix))
708 - ddc::coordinate(get_first_support_knot(ix)))
709 * inv_deg;
710 }
711
712 if constexpr (is_periodic()) {
713 if (int_vals.size() == size()) {
714 discrete_domain_type const dom_bsplines_wrap(
715 full_domain().take_last(discrete_vector_type {degree()}));
716 for (auto ix : dom_bsplines_wrap) {
717 int_vals(ix) = 0;
718 }
719 }
720 }
721 return int_vals;
722}
