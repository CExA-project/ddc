122  static std::tuple<UniformDiscretization, ddom_type> init(
123  rcoord_type a,
124  rcoord_type b,
125  dvect_type n)
126  {
127  assert(a < b);
128  assert(n > 1);
129  UniformDiscretization disc(a, rcoord_type {(b - a) / (n - 1)});
130  ddom_type domain {disc.front(), n};
131  return std::make_tuple(std::move(disc), std::move(domain));

143  static std::tuple<UniformDiscretization, ddom_type, ddom_type, ddom_type, ddom_type>
144  init_ghosted(
145  rcoord_type a,
146  rcoord_type b,
147  dvect_type n,
148  dvect_type n_ghosts_before,
149  dvect_type n_ghosts_after)
150  {
151  using rcoord_type = rcoord_type;
152  using ddom_type = ddom_type;
153  assert(a < b);
154  assert(n > 1);
155  rcoord_type discretization_step {(b - a) / (n - 1)};
156  UniformDiscretization
157  disc(a - n_ghosts_before[0] * discretization_step, discretization_step);
158  ddom_type ghosted_domain = ddom_type(disc.front(), n + n_ghosts_before + n_ghosts_after);
159  ddom_type pre_ghost = ddom_type(ghosted_domain.front(), n_ghosts_before);
160  ddom_type main_domain = ddom_type(ghosted_domain.front() + n_ghosts_before, n);
161  ddom_type post_ghost = ddom_type(main_domain.back() + 1, n_ghosts_after);
162  return std::make_tuple(
163  std::move(disc),
164  std::move(main_domain),
165  std::move(ghosted_domain),
166  std::move(pre_ghost),
167  std::move(post_ghost));
168  }
169

181  return init_ghosted(a, b, n, n_ghosts, n_ghosts);
