20 template <class DDim, class T, class BinaryReductionOp, class UnaryTransformOp>
21 inline T transform_reduce(
22  serial_policy policy,
23  DiscreteDomain<DDim> const& domain,
24  T init,
25  BinaryReductionOp&& reduce,
26  UnaryTransformOp&& transform) noexcept
27 {
28  DiscreteDomainIterator<DDim> const it_b = domain.begin();
29  DiscreteDomainIterator<DDim> const it_e = domain.end();
30  for (DiscreteDomainIterator<DDim> it = it_b; it != it_e; ++it) {
