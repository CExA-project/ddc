326 /**
327 * @brief Evaluate 2D spline function (described by its spline coefficients) at a given coordinate.
328 *
329 * The spline coefficients represent a 2D spline function defined on a B-splines (basis splines). They can be obtained via various methods, such as using a SplineBuilder2D.
330 *
331 * Remark: calling SplineBuilder2D then SplineEvaluator2D corresponds to a 2D spline interpolation.
332 *
333 * @param coord_eval The coordinate where the spline is evaluated. Note that only the components along the dimensions of interest are used.
334 * @param spline_coef A ChunkSpan storing the 2D spline coefficients.
335 *
336 * @return The value of the spline function at the desired coordinate.
337 */
338 template <class Layout, class... CoordsDims>
339 KOKKOS_FUNCTION double operator()(
340 ddc::Coordinate<CoordsDims...> const& coord_eval,
341 ddc::ChunkSpan<double const, spline_domain_type, Layout, memory_space> const
342 spline_coef) const
343 {
344 return eval(coord_eval, spline_coef);
345 }

347 /**
348 * @brief Evaluate 2D spline function (described by its spline coefficients) on a mesh.
349 *
350 * The spline coefficients represent a 2D spline function defined on a cartesian product of batch_domain and B-splines
351 * (basis splines). They can be obtained via various methods, such as using a SplineBuilder2D.
352 *
353 * This is not a nD evaluation. This is a batched 2D evaluation. This means that for each slice of coordinates
354 * identified by a batch_domain_type::discrete_element_type, the evaluation is performed with the 2D set of
355 * spline coefficients identified by the same batch_domain_type::discrete_element_type.
356 *
357 * Remark: calling SplineBuilder2D then SplineEvaluator2D corresponds to a 2D spline interpolation.
358 *
359 * @param[out] spline_eval The values of the 2D spline function at the desired coordinates. For practical reasons those are
360 * stored in a ChunkSpan defined on a batched_evaluation_domain_type.
361 * @param[in] coords_eval The coordinates where the spline is evaluated. Those are
362 * stored in a ChunkSpan defined on a batched_evaluation_domain_type. Note that the coordinates of the
363 * points represented by this domain are unused and irrelevant (but the points themselves (DiscreteElement) are used to select
364 * the set of 2D spline coefficients retained to perform the evaluation).
365 * @param[in] spline_coef A ChunkSpan storing the 2D spline coefficients.
366 */
367 template <class Layout1, class Layout2, class Layout3, class... CoordsDims>
368 void operator()(
369 ddc::ChunkSpan<double, batched_evaluation_domain_type, Layout1, memory_space> const
370 spline_eval,
371 ddc::ChunkSpan<
372 ddc::Coordinate<CoordsDims...> const,
373 batched_evaluation_domain_type,
374 Layout2,
375 memory_space> const coords_eval,
376 ddc::ChunkSpan<double const, batched_spline_domain_type, Layout3, memory_space> const
377 spline_coef) const
378 {
379 batch_domain_type batch_domain(coords_eval.domain());
380 evaluation_domain_type1 const evaluation_domain1(spline_eval.domain());
381 evaluation_domain_type2 const evaluation_domain2(spline_eval.domain());
382 ddc::parallel_for_each(
383 "ddc_splines_evaluate_2d",
384 exec_space(),
385 batch_domain,
386 KOKKOS_CLASS_LAMBDA(typename batch_domain_type::discrete_element_type const j) {
387 const auto spline_eval_2D = spline_eval[j];
388 const auto coords_eval_2D = coords_eval[j];
389 const auto spline_coef_2D = spline_coef[j];
390 for (auto const i1 : evaluation_domain1) {
391 for (auto const i2 : evaluation_domain2) {
392 spline_eval_2D(i1, i2) = eval(coords_eval_2D(i1, i2), spline_coef_2D);
393 }
394 }
395 });
396 }

458 /**
459 * @brief Differentiate 2D spline function (described by its spline coefficients) at a given coordinate along a specified dimension of interest.
460 *
461 * The spline coefficients represent a 2D spline function defined on a B-splines (basis splines). They can be
462 * obtained via various methods, such as using a SplineBuilder2D.
463 *
464 * @tparam InterestDim Dimension along which differentiation is performed.
465 *
466 * @param coord_eval The coordinate where the spline is differentiated. Note that only the components along the dimensions of interest are used.
467 * @param spline_coef A ChunkSpan storing the 2D spline coefficients.
468 *
469 * @return The derivative of the spline function at the desired coordinate.
470 */
471 template <class InterestDim, class Layout, class... CoordsDims>
472 KOKKOS_FUNCTION double deriv(
473 ddc::Coordinate<CoordsDims...> const& coord_eval,
474 ddc::ChunkSpan<double const, spline_domain_type, Layout, memory_space> const
475 spline_coef) const
476 {
477 static_assert(
478 std::is_same_v<
479 InterestDim,
480 typename evaluation_mesh_type1::
481 continuous_dimension_type> || std::is_same_v<InterestDim, typename evaluation_mesh_type2::continuous_dimension_type>);
482 if constexpr (std::is_same_v<
483 InterestDim,
484 typename evaluation_mesh_type1::continuous_dimension_type>) {
485 return deriv_dim_1(coord_eval, spline_coef);
486 } else if constexpr (std::is_same_v<
487 InterestDim,
488 typename evaluation_mesh_type2::continuous_dimension_type>) {
489 return deriv_dim_2(coord_eval, spline_coef);
490 }
491 }

496 * The spline coefficients represent a 2D spline function defined on a B-splines (basis splines). They can be
497 * obtained via various methods, such as using a SplineBuilder2D.
498 *
499 * Note: double-differentiation other than cross-differentiation is not supported atm. See #440
500 *
501 * @tparam InterestDim1 First dimension along which differentiation is performed.
502 * @tparam InterestDim2 Second dimension along which differentiation is performed.
503 *
504 * @param coord_eval The coordinate where the spline is double-differentiated. Note that only the components along the dimensions of interest are used.
505 * @param spline_coef A ChunkSpan storing the 2D spline coefficients.
506 *
507 * @return The derivative of the spline function at the desired coordinate.
508 */
509 template <class InterestDim1, class InterestDim2, class Layout, class... CoordsDims>
510 KOKKOS_FUNCTION double deriv2(
511 ddc::Coordinate<CoordsDims...> const& coord_eval,
512 ddc::ChunkSpan<double const, spline_domain_type, Layout, memory_space> const
513 spline_coef) const
514 {
515 static_assert(
516 (std::is_same_v<
517 InterestDim1,
518 typename evaluation_mesh_type1::
519 continuous_dimension_type> && std::is_same_v<InterestDim2, typename evaluation_mesh_type2::continuous_dimension_type>)
520 || (std::is_same_v<
521 InterestDim2,
522 typename evaluation_mesh_type1::
523 continuous_dimension_type> && std::is_same_v<InterestDim1, typename evaluation_mesh_type2::continuous_dimension_type>));
524 return deriv_1_and_2(coord_eval, spline_coef);
525 }

545 * @param[in] spline_coef A ChunkSpan storing the 2D spline coefficients.

550 spline_eval,

552 ddc::Coordinate<CoordsDims...> const,

782 /** @brief Perform batched 2D integrations of a spline function (described by its spline coefficients) along the dimensions of interest and store results on a subdomain of batch_domain.
783 *
784 * The spline coefficients represent a 2D spline function defined on a B-splines (basis splines). They can be obtained via various methods, such as using a SplineBuilder2D.
785 *
786 * This is not a nD integration. This is a batched 2D integration.
787 * This means that for each element of integrals, the integration is performed with the 2D set of
788 * spline coefficients identified by the same DiscreteElement.
789 *
790 * @param[out] integrals The integrals of the 2D spline function on the subdomain of batch_domain. For practical reasons those are
791 * stored in a ChunkSpan defined on a batch_domain_type. Note that the coordinates of the
792 * points represented by this domain are unused and irrelevant.
793 * @param[in] spline_coef A ChunkSpan storing the 2D spline coefficients.
794 */
795 template <class Layout1, class Layout2>
796 void integrate(
797 ddc::ChunkSpan<double, batch_domain_type, Layout1, memory_space> const integrals,
798 ddc::ChunkSpan<double const, batched_spline_domain_type, Layout2, memory_space> const
799 spline_coef) const
800 {
801 batch_domain_type batch_domain(integrals.domain());
802 ddc::Chunk values1_alloc(
803 ddc::DiscreteDomain<bsplines_type1>(spline_coef.domain()),
804 ddc::KokkosAllocator<double, memory_space>());
805 ddc::ChunkSpan values1 = values1_alloc.span_view();
806 ddc::Chunk values2_alloc(
807 ddc::DiscreteDomain<bsplines_type2>(spline_coef.domain()),
808 ddc::KokkosAllocator<double, memory_space>());
809 ddc::ChunkSpan values2 = values2_alloc.span_view();
810 Kokkos::parallel_for(
811 "ddc_splines_integrate_bsplines_2d",
812 Kokkos::RangePolicy<exec_space>(0, 1),
813 KOKKOS_LAMBDA(int) {
814 ddc::discrete_space<bsplines_type1>().integrals(values1);
815 ddc::discrete_space<bsplines_type2>().integrals(values2);
816 });
817
818 ddc::parallel_for_each(
819 "ddc_splines_integrate_bsplines",
820 exec_space(),
821 batch_domain,
822 KOKKOS_LAMBDA(typename batch_domain_type::discrete_element_type const j) {
823 integrals(j) = 0;
824 for (typename spline_domain_type1::discrete_element_type const i1 :
825 values1.domain()) {
826 for (typename spline_domain_type2::discrete_element_type const i2 :
827 values2.domain()) {
828 integrals(j) += spline_coef(i1, i2, j) * values1(i1) * values2(i2);
829 }
830 }
831 });
832 }

835 /**
836 * @brief Evaluate the function on B-splines at the coordinate given.
837 *
838 * This function firstly deals with the boundary conditions and calls the SplineEvaluator2D::eval_no_bc function
839 * to evaluate.
840 *
841 * @param[in] coord_eval
842 * The 2D coordinate where we want to evaluate.
843 * @param[in] spline_coef
844 * The B-splines coefficients of the function we want to evaluate.
845 * @param[out] vals1
846 * A ChunkSpan with the not-null values of each function of the spline in the first dimension.
847 * @param[out] vals2
848 * A ChunkSpan with the not-null values of each function of the spline in the second dimension.
849 *
850 * @return A double with the value of the function at the coordinate given.
851 *
852 * @see SplineBoundaryValue
853 */
854 template <class Layout, class... CoordsDims>
855 KOKKOS_INLINE_FUNCTION double eval(
856 ddc::Coordinate<CoordsDims...> coord_eval,
857 ddc::ChunkSpan<double const, spline_domain_type, Layout, memory_space> const
858 spline_coef) const
859 {
860 using Dim1 = typename evaluation_mesh_type1::continuous_dimension_type;
861 using Dim2 = typename evaluation_mesh_type2::continuous_dimension_type;
862 if constexpr (bsplines_type1::is_periodic()) {
863 if (ddc::get<Dim1>(coord_eval) < ddc::discrete_space<bsplines_type1>().rmin()
864 || ddc::get<Dim1>(coord_eval) > ddc::discrete_space<bsplines_type1>().rmax()) {
865 ddc::get<Dim1>(coord_eval)
866 -= Kokkos::floor(
867 (ddc::get<Dim1>(coord_eval)
868 - ddc::discrete_space<bsplines_type1>().rmin())
869 / ddc::discrete_space<bsplines_type1>().length())
870 * ddc::discrete_space<bsplines_type1>().length();
871 }
872 }
873 if constexpr (bsplines_type2::is_periodic()) {
874 if (ddc::get<Dim2>(coord_eval) < ddc::discrete_space<bsplines_type2>().rmin()
875 || ddc::get<Dim2>(coord_eval) > ddc::discrete_space<bsplines_type2>().rmax()) {
876 ddc::get<Dim2>(coord_eval)
877 -= Kokkos::floor(
878 (ddc::get<Dim2>(coord_eval)
879 - ddc::discrete_space<bsplines_type2>().rmin())
880 / ddc::discrete_space<bsplines_type2>().length())
881 * ddc::discrete_space<bsplines_type2>().length();
882 }
883 }
884 if constexpr (!bsplines_type1::is_periodic()) {
885 if (ddc::get<Dim1>(coord_eval) < ddc::discrete_space<bsplines_type1>().rmin()) {
886 return m_lower_extrap_rule_1(coord_eval, spline_coef);
887 }
888 if (ddc::get<Dim1>(coord_eval) > ddc::discrete_space<bsplines_type1>().rmax()) {
889 return m_upper_extrap_rule_1(coord_eval, spline_coef);
890 }
891 }
892 if constexpr (!bsplines_type2::is_periodic()) {
893 if (ddc::get<Dim2>(coord_eval) < ddc::discrete_space<bsplines_type2>().rmin()) {
894 return m_lower_extrap_rule_2(coord_eval, spline_coef);
895 }
896 if (ddc::get<Dim2>(coord_eval) > ddc::discrete_space<bsplines_type2>().rmax()) {
897 return m_upper_extrap_rule_2(coord_eval, spline_coef);
898 }
899 }
900 return eval_no_bc<eval_type, eval_type>(
901 ddc::Coordinate<
902 typename evaluation_mesh_type1::continuous_dimension_type,
903 typename evaluation_mesh_type2::continuous_dimension_type>(
904 ddc::get<Dim1>(coord_eval),
905 ddc::get<Dim2>(coord_eval)),
906 spline_coef);
907 }

909 /**
910 * @brief Evaluate the function or its derivative at the coordinate given.
911 *
912 * @param[in] coord_eval
913 * The coordinate where we want to evaluate.
914 * @param[in] splne_coef
915 * The B-splines coefficients of the function we want to evaluate.
916 * @tparam EvalType1
917 * A flag indicating if we evaluate the function or its derivative in the first dimension.
918 * The type of this object is either `eval_type` or `eval_deriv_type`.
919 * @tparam EvalType2
920 * A flag indicating if we evaluate the function or its derivative in the second dimension.
921 * The type of this object is either `eval_type` or `eval_deriv_type`.
922 */
923 template <class EvalType1, class EvalType2, class Layout, class... CoordsDims>
924 KOKKOS_INLINE_FUNCTION double eval_no_bc(
925 ddc::Coordinate<CoordsDims...> const& coord_eval,
926 ddc::ChunkSpan<double const, spline_domain_type, Layout, memory_space> const
927 spline_coef) const
928 {
929 static_assert(
930 std::is_same_v<EvalType1, eval_type> || std::is_same_v<EvalType1, eval_deriv_type>);
931 static_assert(
932 std::is_same_v<EvalType2, eval_type> || std::is_same_v<EvalType2, eval_deriv_type>);
933 ddc::DiscreteElement<bsplines_type1> jmin1;
934 ddc::DiscreteElement<bsplines_type2> jmin2;
935
936 std::array<double, bsplines_type1::degree() + 1> vals1_ptr;
937 std::experimental::mdspan<
938 double,
939 std::experimental::extents<std::size_t, bsplines_type1::degree() + 1>> const
940 vals1(vals1_ptr.data());
941 std::array<double, bsplines_type2::degree() + 1> vals2_ptr;
942 std::experimental::mdspan<
943 double,
944 std::experimental::extents<std::size_t, bsplines_type2::degree() + 1>> const
945 vals2(vals2_ptr.data());
946 ddc::Coordinate<typename evaluation_mesh_type1::continuous_dimension_type>
947 coord_eval_interest1
948 = ddc::select<typename evaluation_mesh_type1::continuous_dimension_type>(
949 coord_eval);
950 ddc::Coordinate<typename evaluation_mesh_type2::continuous_dimension_type>
951 coord_eval_interest2
952 = ddc::select<typename evaluation_mesh_type2::continuous_dimension_type>(
953 coord_eval);
954
955 if constexpr (std::is_same_v<EvalType1, eval_type>) {
956 jmin1 = ddc::discrete_space<bsplines_type1>().eval_basis(vals1, coord_eval_interest1);
957 } else if constexpr (std::is_same_v<EvalType1, eval_deriv_type>) {
958 jmin1 = ddc::discrete_space<bsplines_type1>().eval_deriv(vals1, coord_eval_interest1);
959 }
960 if constexpr (std::is_same_v<EvalType2, eval_type>) {
961 jmin2 = ddc::discrete_space<bsplines_type2>().eval_basis(vals2, coord_eval_interest2);
962 } else if constexpr (std::is_same_v<EvalType2, eval_deriv_type>) {
963 jmin2 = ddc::discrete_space<bsplines_type2>().eval_deriv(vals2, coord_eval_interest2);
964 }
965
966 double y = 0.0;
967 for (std::size_t i = 0; i < bsplines_type1::degree() + 1; ++i) {
968 for (std::size_t j = 0; j < bsplines_type2::degree() + 1; ++j) {
969 y += spline_coef(ddc::DiscreteElement<
970 bsplines_type1,
971 bsplines_type2>(jmin1 + i, jmin2 + j))
972 * vals1[i] * vals2[j];
973 }
974 }
975 return y;
976 }
