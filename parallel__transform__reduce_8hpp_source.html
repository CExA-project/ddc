91class TransformReducerKokkosLambdaAdapter

93 template <class T>
94 using index_type = std::size_t;
95
96 Reducer reducer;
97
98 Functor functor;
99
100public:
101 TransformReducerKokkosLambdaAdapter(Reducer const& r, Functor const& f) : reducer(r), functor(f)
102 {
103 }

139/** A parallel reduction over a nD domain using the default Kokkos execution space
140 * @param[in] execution_space a Kokkos execution space where the loop will be executed on
141 * @param[in] domain the range over which to apply the algorithm
142 * @param[in] neutral the neutral element of the reduction operation
143 * @param[in] reduce a binary FunctionObject that will be applied in unspecified order to the
144 * results of transform, the results of other reduce and neutral.
145 * @param[in] transform a unary FunctionObject that will be applied to each element of the input
146 * range. The return type must be acceptable as input to reduce
147 */
148template <class ExecSpace, class T, class BinaryReductionOp, class UnaryTransformOp>
149T transform_reduce_kokkos(
150 ExecSpace const& execution_space,
151 [[maybe_unused]] DiscreteDomain<> const& domain,
152 T neutral,
153 BinaryReductionOp const& reduce,
154 UnaryTransformOp const& transform) noexcept
155{
156 T result = neutral;
157 if constexpr (need_annotated_operator<ExecSpace>()) {
158 Kokkos::parallel_reduce(
159 Kokkos::RangePolicy<ExecSpace, use_annotated_operator>(execution_space, 0, 1),
160 TransformReducerKokkosLambdaAdapter<
161 BinaryReductionOp,
162 UnaryTransformOp>(reduce, transform),
163 ddc_to_kokkos_reducer_t<BinaryReductionOp>(result));
164 } else {
165 Kokkos::parallel_reduce(
166 Kokkos::RangePolicy<ExecSpace>(execution_space, 0, 1),
167 TransformReducerKokkosLambdaAdapter<
168 BinaryReductionOp,
169 UnaryTransformOp>(reduce, transform),
170 ddc_to_kokkos_reducer_t<BinaryReductionOp>(result));
171 }
172 return result;
173}

175/** A parallel reduction over a nD domain using the default Kokkos execution space
176 * @param[in] execution_space a Kokkos execution space where the loop will be executed on
177 * @param[in] domain the range over which to apply the algorithm
178 * @param[in] neutral the neutral element of the reduction operation
179 * @param[in] reduce a binary FunctionObject that will be applied in unspecified order to the
180 * results of transform, the results of other reduce and neutral.
181 * @param[in] transform a unary FunctionObject that will be applied to each element of the input
182 * range. The return type must be acceptable as input to reduce
183 */
184template <class ExecSpace, class DDim0, class T, class BinaryReductionOp, class UnaryTransformOp>
185T transform_reduce_kokkos(
186 ExecSpace const& execution_space,
187 DiscreteDomain<DDim0> const& domain,
188 T neutral,
189 BinaryReductionOp const& reduce,
190 UnaryTransformOp const& transform) noexcept
191{
192 T result = neutral;
193 if constexpr (need_annotated_operator<ExecSpace>()) {
194 Kokkos::parallel_reduce(
195 Kokkos::RangePolicy<ExecSpace, use_annotated_operator>(
196 execution_space,
197 domain.front().uid(),
198 domain.back().uid() + 1),
199 TransformReducerKokkosLambdaAdapter<
200 BinaryReductionOp,
201 UnaryTransformOp,
202 DDim0>(reduce, transform),
203 ddc_to_kokkos_reducer_t<BinaryReductionOp>(result));
204 } else {
205 Kokkos::parallel_reduce(
206 Kokkos::RangePolicy<
207 ExecSpace>(execution_space, domain.front().uid(), domain.back().uid() + 1),
208 TransformReducerKokkosLambdaAdapter<
209 BinaryReductionOp,
210 UnaryTransformOp,
211 DDim0>(reduce, transform),
212 ddc_to_kokkos_reducer_t<BinaryReductionOp>(result));
213 }
214 return result;
215}

217/** A parallel reduction over a nD domain using the default Kokkos execution space
218 * @param[in] execution_space a Kokkos execution space where the loop will be executed on
219 * @param[in] domain the range over which to apply the algorithm
220 * @param[in] neutral the neutral element of the reduction operation
221 * @param[in] reduce a binary FunctionObject that will be applied in unspecified order to the
222 * results of transform, the results of other reduce and neutral.
223 * @param[in] transform a unary FunctionObject that will be applied to each element of the input
224 * range. The return type must be acceptable as input to reduce
225 */
226template <
227 class ExecSpace,
228 class DDim0,
229 class DDim1,
230 class... DDims,
231 class T,
232 class BinaryReductionOp,
233 class UnaryTransformOp>
234T transform_reduce_kokkos(
235 ExecSpace const& execution_space,
236 DiscreteDomain<DDim0, DDim1, DDims...> const& domain,
237 T neutral,
238 BinaryReductionOp const& reduce,
239 UnaryTransformOp const& transform) noexcept
240{
241 T result = neutral;
242 Kokkos::Array<std::size_t, 2 + sizeof...(DDims)> const
243 begin {select<DDim0>(domain).front().uid(),
244 select<DDim1>(domain).front().uid(),
245 select<DDims>(domain).front().uid()...};
246 Kokkos::Array<std::size_t, 2 + sizeof...(DDims)> const
247 end {select<DDim0>(domain).back().uid() + 1,
248 select<DDim1>(domain).back().uid() + 1,
249 (select<DDims>(domain).back().uid() + 1)...};
250 if constexpr (need_annotated_operator<ExecSpace>()) {
251 Kokkos::parallel_reduce(
252 Kokkos::MDRangePolicy<
253 ExecSpace,
254 Kokkos::Rank<2 + sizeof...(DDims)>,
255 use_annotated_operator>(execution_space, begin, end),
256 TransformReducerKokkosLambdaAdapter<
257 BinaryReductionOp,
258 UnaryTransformOp,
259 DDim0,
260 DDim1,
261 DDims...>(reduce, transform),
262 ddc_to_kokkos_reducer_t<BinaryReductionOp>(result));
263 } else {
264 Kokkos::parallel_reduce(
265 Kokkos::MDRangePolicy<
266 ExecSpace,
267 Kokkos::Rank<2 + sizeof...(DDims)>>(execution_space, begin, end),
268 TransformReducerKokkosLambdaAdapter<
269 BinaryReductionOp,
270 UnaryTransformOp,
271 DDim0,
272 DDim1,
273 DDims...>(reduce, transform),
274 ddc_to_kokkos_reducer_t<BinaryReductionOp>(result));
275 }
276 return result;
277}

281/** A reduction over a nD domain using a given `Kokkos` execution space

290template <class ExecSpace, class... DDims, class T, class BinaryReductionOp, class UnaryTransformOp>

293 DiscreteDomain<DDims...> const& domain,
294 T neutral,
295 BinaryReductionOp&& reduce,
296 UnaryTransformOp&& transform) noexcept

298 return detail::transform_reduce_kokkos(

302 std::forward<BinaryReductionOp>(reduce),
