244  template <

246  std::enable_if_t<std::is_constructible_v<Mapping, extents_type>, int> = 0>
247  constexpr ChunkCommon(ElementType* ptr, mdomain_type const& domain)
248  {
249  namespace stdex = std::experimental;
250  assert(ptr != nullptr);
251 
252  extents_type extents_r(::extents<DDims>(domain)...);
253  mapping_type mapping_r(extents_r);
254 
255  extents_type extents_s((front<DDims>(domain) + ::extents<DDims>(domain))...);
256  std::array<std::size_t, sizeof...(DDims)> strides_s {
257  mapping_r.stride(type_seq_rank_v<DDims, detail::TypeSeq<DDims...>>)...};
258  stdex::layout_stride::mapping<extents_type> mapping_s(extents_s, strides_s);
259 
260  // Pointer offset to handle non-zero indexing
261  ptr -= mapping_s(front<DDims>(domain)...);
262  m_internal_mdspan = internal_mdspan_type(ptr, mapping_s);
263  m_domain = domain;
264  }
