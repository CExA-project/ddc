136 /** Construct a Impl<Kokkos::HostSpace> and associated discrete_domain_type from a segment
137 * \f$[a, b] \subset [a, +\infty[\f$ and a number of points `n`.
138 * Note that there is no guarantee that either the boundaries a or b will be exactly represented in the sampling.
139 * One should expect usual floating point rounding errors.
140 *
141 * @param a coordinate of the first point of the domain
142 * @param b coordinate of the last point of the domain
143 * @param n number of points to map on the segment \f$[a, b]\f$ including a & b
144 * @param n_period the number of steps in a period
145 */
146 template <class DDim>
147 static std::tuple<typename DDim::template Impl<DDim, Kokkos::HostSpace>, DiscreteDomain<DDim>>
148 init(Coordinate<CDim> a,
149 Coordinate<CDim> b,
150 DiscreteVector<DDim> n,
151 DiscreteVector<DDim> n_period)
152 {
153 assert(a < b);
154 assert(n > 1);
155 assert(n_period > 1);
156 typename DDim::template Impl<DDim, Kokkos::HostSpace>
157 disc(a, Coordinate<CDim> {(b - a) / (n - 1)}, n_period);
158 DiscreteDomain<DDim> domain {disc.front(), n};
159 return std::make_tuple(std::move(disc), std::move(domain));
160 }
161
162 /** Construct a periodic `DiscreteDomain` from a segment \f$[a, b] \subset [a, +\infty[\f$ and a
163 * number of points `n`.
164 * Note that there is no guarantee that either the boundaries a or b will be exactly represented in the sampling.
165 * One should expect usual floating point rounding errors.
166 *
167 * @param a coordinate of the first point of the domain
168 * @param b coordinate of the last point of the domain
169 * @param n the number of points to map the segment \f$[a, b]\f$ including a & b
170 * @param n_period the number of steps in a period
171 * @param n_ghosts_before number of additional "ghost" points before the segment
172 * @param n_ghosts_after number of additional "ghost" points after the segment
173 */
174 template <class DDim>
175 std::tuple<
176 Impl<DDim, Kokkos::HostSpace>,
177 DiscreteDomain<DDim>,
178 DiscreteDomain<DDim>,
179 DiscreteDomain<DDim>,
180 DiscreteDomain<DDim>>
181 init_ghosted(
182 continuous_element_type a,
183 continuous_element_type b,
184 DiscreteVector<DDim> n,
185 DiscreteVector<DDim> n_period,
186 DiscreteVector<DDim> n_ghosts_before,
187 DiscreteVector<DDim> n_ghosts_after)
188 {
189 using discrete_domain_type = DiscreteDomain<DDim>;
190 assert(a < b);
191 assert(n > 1);
192 assert(n_period > 1);
193 Real discretization_step {(b - a) / (n - 1)};
194 Impl<DDim, Kokkos::HostSpace>
195 disc(a - n_ghosts_before.value() * discretization_step,
196 discretization_step,
197 n_period);
198 discrete_domain_type ghosted_domain
199 = discrete_domain_type(disc.front(), n + n_ghosts_before + n_ghosts_after);
200 discrete_domain_type pre_ghost
201 = discrete_domain_type(ghosted_domain.front(), n_ghosts_before);
202 discrete_domain_type main_domain
203 = discrete_domain_type(ghosted_domain.front() + n_ghosts_before, n);
204 discrete_domain_type post_ghost
205 = discrete_domain_type(main_domain.back() + 1, n_ghosts_after);
206 return std::make_tuple(
207 std::move(disc),
208 std::move(main_domain),
209 std::move(ghosted_domain),
210 std::move(pre_ghost),
211 std::move(post_ghost));
212 }
213
214 /** Construct a periodic `DiscreteDomain` from a segment \f$[a, b] \subset [a, +\infty[\f$ and a
215 * number of points `n`.
216 * Note that there is no guarantee that either the boundaries a or b will be exactly represented in the sampling.
217 * One should expect usual floating point rounding errors.
218 *
219 * @param a coordinate of the first point of the domain
220 * @param b coordinate of the last point of the domain
221 * @param n the number of points to map the segment \f$[a, b]\f$ including a & b
222 * @param n_period the number of steps in a period
223 * @param n_ghosts number of additional "ghost" points before and after the segment
224 */
225 template <class DDim>
226 std::tuple<
227 Impl<DDim, Kokkos::HostSpace>,
228 DiscreteDomain<DDim>,
229 DiscreteDomain<DDim>,
230 DiscreteDomain<DDim>,
231 DiscreteDomain<DDim>>
232 init_ghosted(
233 continuous_element_type a,
234 continuous_element_type b,
235 DiscreteVector<DDim> n,
236 DiscreteVector<DDim> n_period,
237 DiscreteVector<DDim> n_ghosts)
238 {
239 return init_ghosted(a, b, n, n_period, n_ghosts, n_ghosts);
240 }
