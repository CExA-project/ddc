26template <class T>

29template <class T>
30inline constexpr bool is_borrowed_chunk_v

34template <class T>
35struct chunk_traits
36{
37 static_assert(is_chunk_v<T>);
38 using value_type
39 = std::remove_cv_t<std::remove_pointer_t<decltype(std::declval<T>().data_handle())>>;
40 using pointer_type = decltype(std::declval<T>().data_handle());
41 using reference_type = decltype(*std::declval<T>().data_handle());
42};
43
44template <class T>
45using chunk_value_t = typename chunk_traits<T>::value_type;
46
47template <class T>
48using chunk_pointer_t = typename chunk_traits<T>::pointer_type;
49
50template <class T>
51using chunk_reference_t = typename chunk_traits<T>::reference_type;
52
53template <class T>
54inline constexpr bool is_writable_chunk_v
55 = !std::is_const_v<std::remove_pointer_t<chunk_pointer_t<T>>>;

57/** Access the domain (or subdomain) of a view
58 * @param[in] chunk the view whose domain to access
59 * @return the domain of view in the queried dimensions
60 */
61template <class... QueryDDims, class ChunkType>
62auto get_domain(ChunkType const& chunk) noexcept
63{
64 static_assert(is_chunk_v<ChunkType>, "Not a chunk span type");
65 return chunk.template domain<QueryDDims...>();
66}

301 /** Provide a modifiable view of the data
302 * @return a modifiable view of the data
303 */
304 KOKKOS_FUNCTION constexpr allocation_mdspan_type allocation_mdspan() const
305 {
306 DDC_IF_NVCC_THEN_PUSH_AND_SUPPRESS(implicit_return_from_non_void_function)
307 extents_type extents_s(::ddc::extents<DDims>(m_domain).value()...);
308 if constexpr (std::is_same_v<LayoutStridedPolicy, std::experimental::layout_stride>) {
309 mapping_type map(extents_s, m_internal_mdspan.mapping().strides());
310 return allocation_mdspan_type(data_handle(), map);
311 } else {
312 mapping_type map(extents_s);
313 return allocation_mdspan_type(data_handle(), map);
314 }
315 DDC_IF_NVCC_THEN_POP
316 }
317
318private:
319 /** builds the mapping to use in the internal mdspan
320 * @param domain the domain that sustains the view
321 */
322 template <class Mapping = mapping_type>
323 KOKKOS_FUNCTION constexpr std::enable_if_t<
324 std::is_constructible_v<Mapping, extents_type>,
325 std::experimental::layout_stride::mapping<extents_type>>
326 make_mapping_for(mdomain_type const& domain)
327 {
328 extents_type extents_r(::ddc::extents<DDims>(domain).value()...);
329 mapping_type mapping_r(extents_r);
330
331 extents_type extents_s((front<DDims>(domain) + ddc::extents<DDims>(domain)).uid()...);
332 std::array<std::size_t, sizeof...(DDims)> strides_s {
333 mapping_r.stride(type_seq_rank_v<DDims, detail::TypeSeq<DDims...>>)...};
334 return std::experimental::layout_stride::mapping<extents_type>(extents_s, strides_s);
335 }
336
337 /** Constructs a new ChunkCommon from scratch
338 * @param ptr the allocation pointer to the data_handle()
339 * @param domain the domain that sustains the view
340 */
341 KOKKOS_FUNCTION constexpr ChunkCommon(
342 ElementType* ptr,
343 mdomain_type const& domain,
344 std::experimental::layout_stride::mapping<extents_type>&& s_domain)
345 : m_internal_mdspan {ptr - s_domain(front<DDims>(domain).uid()...), s_domain}
346 , m_domain {domain}
347 {
348 // Handle the case where an allocation of size 0 returns a nullptr.
349 assert((domain.size() == 0) || ((ptr != nullptr) && (domain.size() != 0)));
350 }
351};
