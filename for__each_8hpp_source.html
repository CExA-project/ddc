46 template <class ExecSpace, class Functor, class DDim0, class DDim1, class... DDims>
47 inline void for_each_kokkos(
48  DiscreteDomain<DDim0, DDim1, DDims...> const& domain,
49  Functor&& f) noexcept
50 {
51  Kokkos::Array<std::size_t, 2 + sizeof...(DDims)> const
52  begin {select<DDim0>(domain).front().uid(),
53  select<DDim1>(domain).front().uid(),
54  select<DDims>(domain).front().uid()...};
55  Kokkos::Array<std::size_t, 2 + sizeof...(DDims)> const
56  end {(select<DDim0>(domain).back().uid() + 1),
57  (select<DDim1>(domain).back().uid() + 1),
58  (select<DDims>(domain).back().uid() + 1)...};
59  Kokkos::parallel_for(
60  Kokkos::MDRangePolicy<
61  ExecSpace,
62  Kokkos::Rank<
63  2 + sizeof...(DDims),
64  Kokkos::Iterate::Right,
65  Kokkos::Iterate::Right>>(begin, end),
66  ForEachKokkosLambdaAdapter<Functor, DDim0, DDim1, DDims...>(f));
67 }

69 template <class RetType, class Element, std::size_t N, class Functor, class... Is>
70 inline void for_each_serial(
71  std::array<Element, N> const& start,
72  std::array<Element, N> const& end,
73  Functor const& f,
74  Is const&... is) noexcept
75 {
76  static constexpr std::size_t I = sizeof...(Is);
77  if constexpr (I == N) {
78  f(RetType(is...));
79  } else {
80  for (Element ii = start[I]; ii <= end[I]; ++ii) {
81  for_each_serial<RetType>(start, end, f, is..., ii);
82  }
83  }
84 }
