33template <class ExecSpace>
34class SplinesLinearProblem2x2Blocks : public SplinesLinearProblem<ExecSpace>
35{
36public:
37 using typename SplinesLinearProblem<ExecSpace>::MultiRHS;
38 using SplinesLinearProblem<ExecSpace>::size;
39
40protected:
41 std::unique_ptr<SplinesLinearProblem<ExecSpace>> m_top_left_block;
42 Kokkos::DualView<double**, Kokkos::LayoutRight, typename ExecSpace::memory_space>
43 m_top_right_block;
44 Kokkos::DualView<double**, Kokkos::LayoutRight, typename ExecSpace::memory_space>
45 m_bottom_left_block;
46 std::unique_ptr<SplinesLinearProblem<ExecSpace>> m_bottom_right_block;
47
48public:
55 explicit SplinesLinearProblem2x2Blocks(
56 std::size_t const mat_size,
57 std::unique_ptr<SplinesLinearProblem<ExecSpace>> top_left_block)
58 : SplinesLinearProblem<ExecSpace>(mat_size)
59 , m_top_left_block(std::move(top_left_block))
60 , m_top_right_block(
61 "top_right_block",
62 m_top_left_block->size(),
63 mat_size - m_top_left_block->size())
64 , m_bottom_left_block(
65 "bottom_left_block",
66 mat_size - m_top_left_block->size(),
67 m_top_left_block->size())
68 , m_bottom_right_block(
69 new SplinesLinearProblemDense<ExecSpace>(mat_size - m_top_left_block->size()))
70 {
71 assert(m_top_left_block->size() <= mat_size);
72
73 Kokkos::deep_copy(m_top_right_block.h_view, 0.);
74 Kokkos::deep_copy(m_bottom_left_block.h_view, 0.);
75 }
76
77 double get_element(std::size_t const i, std::size_t const j) const override
78 {
79 assert(i < size());
80 assert(j < size());
81
82 std::size_t const nq = m_top_left_block->size();
83 if (i < nq && j < nq) {
84 return m_top_left_block->get_element(i, j);
85 } else if (i >= nq && j >= nq) {
86 return m_bottom_right_block->get_element(i - nq, j - nq);
87 } else if (j >= nq) {
88 return m_top_right_block.h_view(i, j - nq);
89 } else {
90 return m_bottom_left_block.h_view(i - nq, j);
91 }
92 }
93
94 void set_element(std::size_t const i, std::size_t const j, double const aij) override
95 {
96 assert(i < size());
97 assert(j < size());
98
99 std::size_t const nq = m_top_left_block->size();
100 if (i < nq && j < nq) {
101 m_top_left_block->set_element(i, j, aij);
102 } else if (i >= nq && j >= nq) {
103 m_bottom_right_block->set_element(i - nq, j - nq, aij);
104 } else if (j >= nq) {
105 m_top_right_block.h_view(i, j - nq) = aij;
106 } else {
107 m_bottom_left_block.h_view(i - nq, j) = aij;
108 }
109 }
110
111private:
113 void compute_schur_complement()
114 {
115 Kokkos::parallel_for(
116 "compute_schur_complement",
117 Kokkos::MDRangePolicy<Kokkos::DefaultHostExecutionSpace, Kokkos::Rank<2>>(
118 {0, 0},
119 {m_bottom_right_block->size(), m_bottom_right_block->size()}),
120 [&](const int i, const int j) {
121 double val = 0.0;
122 for (int l = 0; l < m_top_left_block->size(); ++l) {
123 val += m_bottom_left_block.h_view(i, l) * m_top_right_block.h_view(l, j);
124 }
125 m_bottom_right_block
126 ->set_element(i, j, m_bottom_right_block->get_element(i, j) - val);
127 });
128 }
129
130public:
145 void setup_solver() override
146 {
147 // Setup the top-left solver
148 m_top_left_block->setup_solver();
149
150 // Compute Q^-1*gamma in top-right block
151 m_top_right_block.modify_host();
152 m_top_right_block.sync_device();
153 m_top_left_block->solve(m_top_right_block.d_view);
154 m_top_right_block.modify_device();
155 m_top_right_block.sync_host();
156
157 // Push lambda on device in bottom-left block
158 m_bottom_left_block.modify_host();
159 m_bottom_left_block.sync_device();
160
161 // Compute delta - lambda*Q^-1*gamma in bottom-right block & setup the bottom-right solver
162 compute_schur_complement();
163 m_bottom_right_block->setup_solver();
164 }

176 void gemv_minus1_1(
177 MultiRHS const x,
178 MultiRHS const y,
179 Kokkos::View<double**, Kokkos::LayoutRight, typename ExecSpace::memory_space> const
180 LinOp,
181 bool const transpose = false) const
182 {
183 assert((!transpose && LinOp.extent(0) == y.extent(0))
184 || (transpose && LinOp.extent(1) == y.extent(0)));
185 assert((!transpose && LinOp.extent(1) == x.extent(0))
186 || (transpose && LinOp.extent(0) == x.extent(0)));
187 assert(x.extent(1) == y.extent(1));
188
189 if (!transpose) {
190 Kokkos::parallel_for(
191 "gemv_minus1_1",
192 Kokkos::TeamPolicy<ExecSpace>(y.extent(0) * y.extent(1), Kokkos::AUTO),
193 KOKKOS_LAMBDA(
194 const typename Kokkos::TeamPolicy<ExecSpace>::member_type& teamMember) {
195 const int i = teamMember.league_rank() / y.extent(1);
196 const int j = teamMember.league_rank() % y.extent(1);
197
198 double LinOpTimesX = 0.;
199 Kokkos::parallel_reduce(
200 Kokkos::TeamThreadRange(teamMember, x.extent(0)),
201 [&](const int l, double& LinOpTimesX_tmp) {
202 LinOpTimesX_tmp += LinOp(i, l) * x(l, j);
203 },
204 LinOpTimesX);
205 Kokkos::single(Kokkos::PerTeam(teamMember), [&]() {
206 y(i, j) -= LinOpTimesX;
207 });
208 });
209 } else {
210 Kokkos::parallel_for(
211 "gemv_minus1_1_tr",
212 Kokkos::TeamPolicy<ExecSpace>(y.extent(0) * y.extent(1), Kokkos::AUTO),
213 KOKKOS_LAMBDA(
214 const typename Kokkos::TeamPolicy<ExecSpace>::member_type& teamMember) {
215 const int i = teamMember.league_rank() / y.extent(1);
216 const int j = teamMember.league_rank() % y.extent(1);
217
218 double LinOpTimesX = 0.;
219 Kokkos::parallel_reduce(
220 Kokkos::TeamThreadRange(teamMember, x.extent(0)),
221 [&](const int l, double& LinOpTimesX_tmp) {
222 LinOpTimesX_tmp += LinOp(l, i) * x(l, j);
223 },
224 LinOpTimesX);
225 Kokkos::single(Kokkos::PerTeam(teamMember), [&]() {
226 y(i, j) -= LinOpTimesX;
227 });
228 });
229 }
230 }

250 void solve(MultiRHS b, bool const transpose) const override
251 {
252 assert(b.extent(0) == size());
253
254 MultiRHS b1 = Kokkos::
255 subview(b,
256 std::pair<std::size_t, std::size_t>(0, m_top_left_block->size()),
257 Kokkos::ALL);
258 MultiRHS b2 = Kokkos::
259 subview(b,
260 std::pair<std::size_t, std::size_t>(m_top_left_block->size(), b.extent(0)),
261 Kokkos::ALL);
262 if (!transpose) {
263 m_top_left_block->solve(b1);
264 gemv_minus1_1(b1, b2, m_bottom_left_block.d_view);
265 m_bottom_right_block->solve(b2);
266 gemv_minus1_1(b2, b1, m_top_right_block.d_view);
267 } else {
268 gemv_minus1_1(b1, b2, m_top_right_block.d_view, true);
269 m_bottom_right_block->solve(b2, true);
270 gemv_minus1_1(b2, b1, m_bottom_left_block.d_view, true);
271 m_top_left_block->solve(b1, true);
272 }
273 }
274};
