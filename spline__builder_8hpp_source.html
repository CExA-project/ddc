17constexpr bool is_spline_interpolation_mesh_uniform(
18 bool const is_uniform,
19 ddc::BoundCond const BcXmin,
20 ddc::BoundCond const BcXmax,
21 int degree)
22{
23 int N_BE_MIN = n_boundary_equations(BcXmin, degree);
24 int N_BE_MAX = n_boundary_equations(BcXmax, degree);
25 bool is_periodic = (BcXmin == ddc::BoundCond::PERIODIC) && (BcXmax == ddc::BoundCond::PERIODIC);
26 return is_uniform && ((N_BE_MIN != 0 && N_BE_MAX != 0) || is_periodic);
27}

346int SplineBuilder<
347 ExecSpace,
348 MemorySpace,
349 BSplines,
350 InterpolationMesh,
351 BcXmin,
352 BcXmax,
353 Solver,
354 IDimX...>::compute_offset(interpolation_domain_type const& interpolation_domain)
355{
356 int offset;
357 if constexpr (bsplines_type::is_periodic()) {
358 // Calculate offset so that the matrix is diagonally dominant
359 std::array<double, bsplines_type::degree() + 1> values_ptr;
360 std::experimental::mdspan<
361 double,
362 std::experimental::extents<std::size_t, bsplines_type::degree() + 1>> const
363 values(values_ptr.data());
364 ddc::DiscreteElement<interpolation_mesh_type> start(interpolation_domain.front());
365 auto jmin = ddc::discrete_space<BSplines>()
366 .eval_basis(values, ddc::coordinate(start + BSplines::degree()));
367 if constexpr (bsplines_type::degree() % 2 == 0) {
368 offset = jmin.uid() - start.uid() + bsplines_type::degree() / 2 - BSplines::degree();
369 } else {
370 int const mid = bsplines_type::degree() / 2;
371 offset = jmin.uid() - start.uid() + (values(mid) > values(mid + 1) ? mid : mid + 1)
372 - BSplines::degree();
373 }
374 } else {
375 offset = 0;
376 }
377 return offset;
378}

394 BcXmin,
395 BcXmax,
396 Solver,
397 IDimX...>::compute_block_sizes_uniform(int& lower_block_size, int& upper_block_size) const
398{
399 switch (BcXmin) {
400 case ddc::BoundCond::PERIODIC:
401 upper_block_size = (bsplines_type::degree()) / 2;
402 break;
403 case ddc::BoundCond::NATURAL:
404 case ddc::BoundCond::HERMITE:
405 upper_block_size = s_nbc_xmin;
406 break;
407 case ddc::BoundCond::GREVILLE:
408 upper_block_size = bsplines_type::degree() - 1;
409 break;
410 default:
411 throw std::runtime_error("ddc::BoundCond not handled");
412 }
413 switch (BcXmax) {
414 case ddc::BoundCond::PERIODIC:
415 lower_block_size = (bsplines_type::degree()) / 2;
416 break;
417 case ddc::BoundCond::NATURAL:
418 case ddc::BoundCond::HERMITE:
419 lower_block_size = s_nbc_xmax;
420 break;
421 case ddc::BoundCond::GREVILLE:
422 lower_block_size = bsplines_type::degree() - 1;
423 break;
424 default:
425 throw std::runtime_error("ddc::BoundCond not handled");
426 }
427}

497 [[maybe_unused]] int upper_block_size,

501 // Special case: linear spline
502 // No need for matrix assembly
503 // (desactivated)
504 /*
505 if constexpr (bsplines_type::degree() == 1)
506 return;
507 */
508
509 matrix = ddc::detail::MatrixMaker::make_new_sparse<ExecSpace>(
510 ddc::discrete_space<BSplines>().nbasis(),

512 preconditionner_max_block_size);
513
514 build_matrix_system();
515
516 matrix->factorize();
517}

519template <
520 class ExecSpace,
521 class MemorySpace,
522 class BSplines,
523 class InterpolationMesh,
524 ddc::BoundCond BcXmin,
525 ddc::BoundCond BcXmax,
526 SplineSolver Solver,
527 class... IDimX>
528void SplineBuilder<
529 ExecSpace,
530 MemorySpace,
531 BSplines,
532 InterpolationMesh,
533 BcXmin,
534 BcXmax,
535 Solver,
536 IDimX...>::build_matrix_system()
537{
538 // Hermite boundary conditions at xmin, if any
539 if constexpr (BcXmin == ddc::BoundCond::HERMITE) {
540 std::array<double, (bsplines_type::degree() / 2 + 1) * (bsplines_type::degree() + 1)>
541 derivs_ptr;
542 ddc::DSpan2D
543 derivs(derivs_ptr.data(),
544 bsplines_type::degree() + 1,
545 bsplines_type::degree() / 2 + 1);
546 ddc::discrete_space<BSplines>().eval_basis_and_n_derivs(
547 derivs,
548 ddc::discrete_space<BSplines>().rmin(),
549 s_nbc_xmin);
550
551 // In order to improve the condition number of the matrix, we normalize
552 // all derivatives by multiplying the i-th derivative by dx^i
553 for (std::size_t i = 0; i < bsplines_type::degree() + 1; ++i) {
554 for (std::size_t j = 1; j < bsplines_type::degree() / 2 + 1; ++j) {
555 derivs(i, j) *= ddc::detail::ipow(m_dx, j);
556 }
557 }
558
559 // iterate only to deg as last bspline is 0
560 for (std::size_t i = 0; i < s_nbc_xmin; ++i) {
561 for (std::size_t j = 0; j < bsplines_type::degree(); ++j) {
562 matrix->set_element(i, j, derivs(j, s_nbc_xmin - i - 1 + s_odd));
563 }
564 }
565 }
566
567 // Interpolation points
568 std::array<double, bsplines_type::degree() + 1> values_ptr;
569 std::experimental::mdspan<
570 double,
571 std::experimental::extents<std::size_t, bsplines_type::degree() + 1>> const
572 values(values_ptr.data());
573
574 int start = interpolation_domain().front().uid();
575 ddc::for_each(interpolation_domain(), [&](auto ix) {
576 auto jmin = ddc::discrete_space<BSplines>().eval_basis(
577 values,
578 ddc::coordinate(ddc::DiscreteElement<interpolation_mesh_type>(ix)));
579 for (std::size_t s = 0; s < bsplines_type::degree() + 1; ++s) {
580 int const j = ddc::detail::
581 modulo(int(jmin.uid() - m_offset + s),
582 (int)ddc::discrete_space<BSplines>().nbasis());
583 matrix->set_element(ix.uid() - start + s_nbc_xmin, j, values(s));
584 }
585 });
586
587 // Hermite boundary conditions at xmax, if any
588 if constexpr (BcXmax == ddc::BoundCond::HERMITE) {
589 std::array<double, (bsplines_type::degree() / 2 + 1) * (bsplines_type::degree() + 1)>
590 derivs_ptr;
591 std::experimental::mdspan<
592 double,
593 std::experimental::extents<
594 std::size_t,
595 bsplines_type::degree() + 1,
596 bsplines_type::degree() / 2 + 1>> const derivs(derivs_ptr.data());
597
598 ddc::discrete_space<BSplines>().eval_basis_and_n_derivs(
599 derivs,
600 ddc::discrete_space<BSplines>().rmax(),
601 s_nbc_xmax);
602
603 // In order to improve the condition number of the matrix, we normalize
604 // all derivatives by multiplying the i-th derivative by dx^i
605 for (std::size_t i = 0; i < bsplines_type::degree() + 1; ++i) {
606 for (std::size_t j = 1; j < bsplines_type::degree() / 2 + 1; ++j) {
607 derivs(i, j) *= ddc::detail::ipow(m_dx, j);
608 }
609 }
610
611 int const i0 = ddc::discrete_space<BSplines>().nbasis() - s_nbc_xmax;
612 int const j0 = ddc::discrete_space<BSplines>().nbasis() - bsplines_type::degree();
613 for (std::size_t j = 0; j < bsplines_type::degree(); ++j) {
614 for (std::size_t i = 0; i < s_nbc_xmax; ++i) {
615 matrix->set_element(i0 + i, j0 + j, derivs(j + 1, i + s_odd));
616 }
617 }
618 }
619}

622 class ExecSpace,
623 class MemorySpace,
624 class BSplines,
625 class InterpolationMesh,
626 ddc::BoundCond BcXmin,
627 ddc::BoundCond BcXmax,
628 SplineSolver Solver,

630template <class Layout>
631void SplineBuilder<
632 ExecSpace,
633 MemorySpace,
634 BSplines,
635 InterpolationMesh,
636 BcXmin,
637 BcXmax,
638 Solver,
639 IDimX...>::
640operator()(
641 ddc::ChunkSpan<double, spline_domain_type, Layout, memory_space> spline,
642 ddc::ChunkSpan<double const, vals_domain_type, Layout, memory_space> vals,
643 std::optional<ddc::ChunkSpan<double const, derivs_domain_type, Layout, memory_space>> const
644 derivs_xmin,
645 std::optional<ddc::ChunkSpan<double const, derivs_domain_type, Layout, memory_space>> const
646 derivs_xmax) const
647{
648 assert(vals.template extent<interpolation_mesh_type>()
649 == ddc::discrete_space<bsplines_type>().nbasis() - s_nbe_xmin - s_nbe_xmax);
650
651 assert((BcXmin == ddc::BoundCond::HERMITE)
652 != (!derivs_xmin.has_value() || derivs_xmin->template extent<deriv_type>() == 0));
653 assert((BcXmax == ddc::BoundCond::HERMITE)
654 != (!derivs_xmax.has_value() || derivs_xmax->template extent<deriv_type>() == 0));
655 if constexpr (BcXmin == BoundCond::HERMITE) {
656 assert(ddc::DiscreteElement<deriv_type>(derivs_xmin->domain().front()).uid() == 1);
657 }
658 if constexpr (BcXmax == BoundCond::HERMITE) {
659 assert(ddc::DiscreteElement<deriv_type>(derivs_xmax->domain().front()).uid() == 1);
660 }
661
662 // Hermite boundary conditions at xmin, if any
663 // NOTE: For consistency with the linear system, the i-th derivative
664 // provided by the user must be multiplied by dx^i
665 if constexpr (BcXmin == BoundCond::HERMITE) {
666 assert(derivs_xmin->template extent<deriv_type>() == s_nbc_xmin);
667 auto derivs_xmin_values = *derivs_xmin;
668 auto const dx_proxy = m_dx;
669 ddc::parallel_for_each(
670 exec_space(),
671 batch_domain(),
672 KOKKOS_LAMBDA(typename batch_domain_type::discrete_element_type j) {
673 for (int i = s_nbc_xmin; i > 0; --i) {
674 spline(ddc::DiscreteElement<bsplines_type>(s_nbc_xmin - i), j)
675 = derivs_xmin_values(ddc::DiscreteElement<deriv_type>(i), j)
676 * ddc::detail::ipow(dx_proxy, i + s_odd - 1);
677 }
678 });
679 }
680
681 // TODO : Consider optimizing
682 // Fill spline with vals (to work in spline afterward and preserve vals)
683 auto const& offset_proxy = m_offset;
684 auto const& interp_size_proxy = interpolation_domain().extents();
685 auto const& nbasis_proxy = ddc::discrete_space<bsplines_type>().nbasis();
686 ddc::parallel_for_each(
687 exec_space(),
688 batch_domain(),
689 KOKKOS_LAMBDA(typename batch_domain_type::discrete_element_type j) {
690 for (int i = s_nbc_xmin; i < s_nbc_xmin + offset_proxy; ++i) {
691 spline(ddc::DiscreteElement<bsplines_type>(i), j) = 0.0;
692 }
693 for (int i = 0; i < interp_size_proxy; ++i) {
694 spline(ddc::DiscreteElement<bsplines_type>(s_nbc_xmin + i + offset_proxy), j)
695 = vals(ddc::DiscreteElement<interpolation_mesh_type>(i), j);
696 }
697 });
698
699 // Hermite boundary conditions at xmax, if any
700 // NOTE: For consistency with the linear system, the i-th derivative
701 // provided by the user must be multiplied by dx^i
702 if constexpr (BcXmax == BoundCond::HERMITE) {
703 assert(derivs_xmax->template extent<deriv_type>() == s_nbc_xmax);
704 auto derivs_xmax_values = *derivs_xmax;
705 auto const dx_proxy = m_dx;
706 ddc::parallel_for_each(
707 exec_space(),
708 batch_domain(),
709 KOKKOS_LAMBDA(typename batch_domain_type::discrete_element_type j) {
710 for (int i = 0; i < s_nbc_xmax; ++i) {
711 spline(ddc::DiscreteElement<bsplines_type>(nbasis_proxy - s_nbc_xmax - i),
712 j)
713 = derivs_xmax_values(ddc::DiscreteElement<deriv_type>(i + 1), j)
714 * ddc::detail::ipow(dx_proxy, i + s_odd);
715 }
716 });
717 }
718
719 // TODO : Consider optimizing
720 // Allocate and fill a transposed version of spline in order to get dimension of interest as last dimension (optimal for GPU, necessary for Ginkgo)
721 ddc::Chunk spline_tr_alloc(spline_tr_domain(), ddc::KokkosAllocator<double, memory_space>());
722 ddc::ChunkSpan spline_tr = spline_tr_alloc.span_view();
723 ddc::parallel_for_each(
724 exec_space(),
725 batch_domain(),
726 KOKKOS_LAMBDA(typename batch_domain_type::discrete_element_type const j) {
727 for (std::size_t i = 0; i < nbasis_proxy; i++) {
728 spline_tr(ddc::DiscreteElement<bsplines_type>(i), j)
729 = spline(ddc::DiscreteElement<bsplines_type>(i + offset_proxy), j);
730 }
731 });
732 // Create a 2D Kokkos::View to manage spline_tr as a matrix
733 Kokkos::View<double**, Kokkos::LayoutRight, exec_space> bcoef_section(
734 spline_tr.data_handle(),
735 ddc::discrete_space<bsplines_type>().nbasis(),
736 batch_domain().size());
737 // Compute spline coef

746 = spline_tr(ddc::DiscreteElement<bsplines_type>(i), j);
747 }
748 });
749
750 // Not sure yet of what this part do
751 if (bsplines_type::is_periodic()) {
752 ddc::parallel_for_each(
753 exec_space(),
754 batch_domain(),
755 KOKKOS_LAMBDA(typename batch_domain_type::discrete_element_type const j) {
756 if (offset_proxy != 0) {

758 spline(ddc::DiscreteElement<bsplines_type>(i), j) = spline(

761 }

764 = spline(ddc::DiscreteElement<bsplines_type>(i), j);
765 }
766 }
767 for (std::size_t i(0); i < bsplines_type::degree(); ++i) {
768 const ddc::DiscreteElement<bsplines_type> i_start(i);
769 const ddc::DiscreteElement<bsplines_type> i_end(nbasis_proxy + i);
770
771 spline(i_end, j) = spline(i_start, j);
772 }
773 });
774 }
