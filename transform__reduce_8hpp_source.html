95 template <
96  class... DDims,
97  class T,
98  class BinaryReductionOp,
99  class UnaryTransformOp,
100  class... DCoords>
101 inline T transform_reduce_serial(
102  DiscreteDomain<DDims...> const& domain,
103  [[maybe_unused]] T const neutral,
104  BinaryReductionOp const& reduce,
105  UnaryTransformOp const& transform,
106  DCoords const&... dcoords) noexcept
107 {
108  DDC_IF_NVCC_THEN_PUSH_AND_SUPPRESS(implicit_return_from_non_void_function)
109  if constexpr (sizeof...(DCoords) == sizeof...(DDims)) {
110  return transform(DiscreteElement<DDims...>(dcoords...));
111  } else {
112  using CurrentDDim = type_seq_element_t<sizeof...(DCoords), ddc_detail::TypeSeq<DDims...>>;
113  T result = neutral;
114  for (DiscreteElement<CurrentDDim> const ii : select<CurrentDDim>(domain)) {
115  result = reduce(
116  result,
117  transform_reduce_serial(domain, neutral, reduce, transform, dcoords..., ii));
118  }
119  return result;
120  }
121  DDC_IF_NVCC_THEN_POP
122 }

125 class TransformReducerKokkosLambdaAdapter
126 {

128  using index_type = std::size_t;
129 
130  Reducer reducer;
131 
132  Functor functor;
133 
134 public:
135  DDC_FORCEINLINE_FUNCTION
136  TransformReducerKokkosLambdaAdapter(Reducer const& r, Functor const& f) : reducer(r), functor(f)

156 template <class ExecSpace, class DDim0, class T, class BinaryReductionOp, class UnaryTransformOp>
157 inline T transform_reduce_kokkos(
158  DiscreteDomain<DDim0> const& domain,
159  T neutral,
160  BinaryReductionOp const& reduce,
161  UnaryTransformOp const& transform) noexcept
162 {
163  T result = neutral;
164  Kokkos::parallel_reduce(
165  Kokkos::RangePolicy<ExecSpace>(
166  select<DDim0>(domain).front().uid(),
167  select<DDim0>(domain).back().uid() + 1),
168  TransformReducerKokkosLambdaAdapter<
169  BinaryReductionOp,
170  UnaryTransformOp,
171  DDim0>(reduce, transform),
172  ddc_to_kokkos_reducer_t<BinaryReductionOp>(result));
173  return result;
174 }

184 template <
185  class ExecSpace,
186  class DDim0,
187  class DDim1,
188  class... DDims,
189  class T,
190  class BinaryReductionOp,
191  class UnaryTransformOp>
192 inline T transform_reduce_kokkos(
193  DiscreteDomain<DDim0, DDim1, DDims...> const& domain,
194  T neutral,
195  BinaryReductionOp const& reduce,
196  UnaryTransformOp const& transform) noexcept
197 {
198  T result = neutral;
199  Kokkos::Array<std::size_t, 2 + sizeof...(DDims)> const
200  begin {select<DDim0>(domain).front().uid(),
201  select<DDim1>(domain).front().uid(),
202  select<DDims>(domain).front().uid()...};
203  Kokkos::Array<std::size_t, 2 + sizeof...(DDims)> const
204  end {select<DDim0>(domain).back().uid() + 1,
205  select<DDim1>(domain).back().uid() + 1,
206  (select<DDims>(domain).back().uid() + 1)...};
207  Kokkos::parallel_reduce(
208  Kokkos::MDRangePolicy<ExecSpace, Kokkos::Rank<2 + sizeof...(DDims)>>(begin, end),
209  TransformReducerKokkosLambdaAdapter<
210  BinaryReductionOp,
211  UnaryTransformOp,
212  DDim0,
213  DDim1,
214  DDims...>(reduce, transform),
215  ddc_to_kokkos_reducer_t<BinaryReductionOp>(result));
216  return result;
217 }
