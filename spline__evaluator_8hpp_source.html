147 template <class Layout1, class Layout2, class Layout3, class... CoordsDims>
148 void operator()(
149 ddc::ChunkSpan<double, vals_domain_type, Layout1, memory_space> const spline_eval,
150 ddc::ChunkSpan<
151 ddc::Coordinate<CoordsDims...> const,
152 vals_domain_type,
153 Layout2,
154 memory_space> const coords_eval,
155 ddc::ChunkSpan<double const, spline_domain_type, Layout3, memory_space> const
156 spline_coef) const
157 {
158 interpolation_domain_type const interpolation_domain(spline_eval.domain());
159 batch_domain_type const batch_domain(spline_eval.domain());
160
161 ddc::parallel_for_each(
162 exec_space(),
163 batch_domain,
164 KOKKOS_CLASS_LAMBDA(typename batch_domain_type::discrete_element_type const j) {
165 const auto spline_eval_1D = spline_eval[j];
166 const auto coords_eval_1D = coords_eval[j];
167 const auto spline_coef_1D = spline_coef[j];
168 for (auto const i : interpolation_domain) {
169 spline_eval_1D(i) = eval(coords_eval_1D(i), spline_coef_1D);
170 }
171 });
172 }

211 template <class Layout1, class Layout2>
212 void integrate(
213 ddc::ChunkSpan<double, batch_domain_type, Layout1, memory_space> const integrals,
214 ddc::ChunkSpan<double const, spline_domain_type, Layout2, memory_space> const
215 spline_coef) const
216 {
217 batch_domain_type const batch_domain(integrals.domain());
218 ddc::Chunk values_alloc(
219 ddc::DiscreteDomain<bsplines_type>(spline_coef.domain()),
220 ddc::KokkosAllocator<double, memory_space>());
221 ddc::ChunkSpan values = values_alloc.span_view();
222 Kokkos::parallel_for(
223 Kokkos::RangePolicy<exec_space>(0, 1),
224 KOKKOS_LAMBDA(int) { ddc::discrete_space<bsplines_type>().integrals(values); });
225
226 ddc::parallel_for_each(
227 exec_space(),
228 batch_domain,
229 KOKKOS_LAMBDA(typename batch_domain_type::discrete_element_type const j) {
230 integrals(j) = 0;
231 for (typename bsplines_domain_type::discrete_element_type const i :
232 values.domain()) {
233 integrals(j) += spline_coef(i, j) * values(i);
234 }
235 });
236 }

239 template <class Layout, class... CoordsDims>
240 KOKKOS_INLINE_FUNCTION double eval(
241 ddc::Coordinate<CoordsDims...> const& coord_eval,
242 ddc::ChunkSpan<double const, bsplines_domain_type, Layout, memory_space> const
243 spline_coef) const
244 {
245 ddc::Coordinate<typename interpolation_mesh_type::continuous_dimension_type>
246 coord_eval_interpolation
247 = ddc::select<typename interpolation_mesh_type::continuous_dimension_type>(
248 coord_eval);
249 if constexpr (bsplines_type::is_periodic()) {
250 if (coord_eval_interpolation < ddc::discrete_space<bsplines_type>().rmin()
251 || coord_eval_interpolation > ddc::discrete_space<bsplines_type>().rmax()) {
252 coord_eval_interpolation -= Kokkos::floor(
253 (coord_eval_interpolation
254 - ddc::discrete_space<bsplines_type>().rmin())
255 / ddc::discrete_space<bsplines_type>().length())
256 * ddc::discrete_space<bsplines_type>().length();
257 }
258 } else {
259 if (coord_eval_interpolation < ddc::discrete_space<bsplines_type>().rmin()) {
260 return m_left_extrap_rule(coord_eval_interpolation, spline_coef);
261 }
262 if (coord_eval_interpolation > ddc::discrete_space<bsplines_type>().rmax()) {
263 return m_right_extrap_rule(coord_eval_interpolation, spline_coef);
264 }
265 }
266 return eval_no_bc<eval_type>(coord_eval_interpolation, spline_coef);
267 }

269 template <class EvalType, class Layout, class... CoordsDims>
270 KOKKOS_INLINE_FUNCTION double eval_no_bc(
271 ddc::Coordinate<CoordsDims...> const& coord_eval,
272 ddc::ChunkSpan<double const, bsplines_domain_type, Layout, memory_space> const
273 spline_coef) const
274 {
275 static_assert(
276 std::is_same_v<EvalType, eval_type> || std::is_same_v<EvalType, eval_deriv_type>);
277 ddc::DiscreteElement<bsplines_type> jmin;
278 std::array<double, bsplines_type::degree() + 1> vals;
279 ddc::Coordinate<typename interpolation_mesh_type::continuous_dimension_type>
280 coord_eval_interpolation
281 = ddc::select<typename interpolation_mesh_type::continuous_dimension_type>(
282 coord_eval);
283 if constexpr (std::is_same_v<EvalType, eval_type>) {
284 jmin = ddc::discrete_space<bsplines_type>().eval_basis(vals, coord_eval_interpolation);
285 } else if constexpr (std::is_same_v<EvalType, eval_deriv_type>) {
286 jmin = ddc::discrete_space<bsplines_type>().eval_deriv(vals, coord_eval_interpolation);
287 }
288 double y = 0.0;
289 for (std::size_t i = 0; i < bsplines_type::degree() + 1; ++i) {
290 y += spline_coef(ddc::DiscreteElement<bsplines_type>(jmin + i)) * vals[i];
291 }
292 return y;
293 }
