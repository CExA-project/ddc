276 template <class Layout1, class Layout2, class Layout3, class... CoordsDims>
277 void operator()(
278 ddc::ChunkSpan<double, batched_evaluation_domain_type, Layout1, memory_space> const
279 spline_eval,
280 ddc::ChunkSpan<
281 ddc::Coordinate<CoordsDims...> const,
282 batched_evaluation_domain_type,
283 Layout2,
284 memory_space> const coords_eval,
285 ddc::ChunkSpan<double const, batched_spline_domain_type, Layout3, memory_space> const
286 spline_coef) const
287 {
288 batch_domain_type batch_domain(coords_eval.domain());
289 evaluation_domain_type1 const evaluation_domain1(spline_eval.domain());
290 evaluation_domain_type2 const evaluation_domain2(spline_eval.domain());
291 ddc::parallel_for_each(
292 exec_space(),
293 batch_domain,
294 KOKKOS_CLASS_LAMBDA(typename batch_domain_type::discrete_element_type const j) {
295 const auto spline_eval_2D = spline_eval[j];
296 const auto coords_eval_2D = coords_eval[j];
297 const auto spline_coef_2D = spline_coef[j];
298 for (auto const i1 : evaluation_domain1) {
299 for (auto const i2 : evaluation_domain2) {
300 spline_eval_2D(i1, i2) = eval(coords_eval_2D(i1, i2), spline_coef_2D);
301 }
302 }
303 });
304 }

587 /**
588 * @brief Get the the integral of the function on B-splines on the domain.
589 * @param[out] integrals
590 * The integrals of the function
591 * @param[in] spline_coef
592 * The B-splines coefficients of the function we want to integrate.
593 */
594 template <class Layout1, class Layout2>
595 void integrate(
596 ddc::ChunkSpan<double, batch_domain_type, Layout1, memory_space> const integrals,
597 ddc::ChunkSpan<double const, batched_spline_domain_type, Layout2, memory_space> const
598 spline_coef) const
599 {
600 batch_domain_type batch_domain(integrals.domain());
601 ddc::Chunk values1_alloc(
602 ddc::DiscreteDomain<bsplines_type1>(spline_coef.domain()),
603 ddc::KokkosAllocator<double, memory_space>());
604 ddc::ChunkSpan values1 = values1_alloc.span_view();
605 ddc::Chunk values2_alloc(
606 ddc::DiscreteDomain<bsplines_type2>(spline_coef.domain()),
607 ddc::KokkosAllocator<double, memory_space>());
608 ddc::ChunkSpan values2 = values2_alloc.span_view();
609 Kokkos::parallel_for(
610 Kokkos::RangePolicy<exec_space>(0, 1),
611 KOKKOS_LAMBDA(int) {
612 ddc::discrete_space<bsplines_type1>().integrals(values1);
613 ddc::discrete_space<bsplines_type2>().integrals(values2);
614 });
615
616 ddc::parallel_for_each(
617 exec_space(),
618 batch_domain,
619 KOKKOS_LAMBDA(typename batch_domain_type::discrete_element_type const j) {

632 /**
633 * @brief Evaluate the function on B-splines at the coordinate given.
634 *
635 * This function firstly deals with the boundary conditions and calls the SplineEvaluator2D::eval_no_bc function
636 * to evaluate.
637 *
638 * @param[in] coord_eval
639 * The 2D coordinate where we want to evaluate.
640 * @param[in] spline_coef
641 * The B-splines coefficients of the function we want to evaluate.
642 * @param[out] vals1
643 * A ChunkSpan with the not-null values of each function of the spline in the first dimension.
644 * @param[out] vals2
645 * A ChunkSpan with the not-null values of each function of the spline in the second dimension.
646 *
647 * @return A double with the value of the function at the coordinate given.
648 *
649 * @see SplineBoundaryValue
650 */
651 template <class Layout, class... CoordsDims>
652 KOKKOS_INLINE_FUNCTION double eval(
653 ddc::Coordinate<CoordsDims...> coord_eval,
654 ddc::ChunkSpan<double const, spline_domain_type, Layout, memory_space> const
655 spline_coef) const
656 {
657 using Dim1 = typename evaluation_mesh_type1::continuous_dimension_type;
658 using Dim2 = typename evaluation_mesh_type2::continuous_dimension_type;
659 if constexpr (bsplines_type1::is_periodic()) {
660 if (ddc::get<Dim1>(coord_eval) < ddc::discrete_space<bsplines_type1>().rmin()
661 || ddc::get<Dim1>(coord_eval) > ddc::discrete_space<bsplines_type1>().rmax()) {
662 ddc::get<Dim1>(coord_eval)
663 -= Kokkos::floor(
664 (ddc::get<Dim1>(coord_eval)
665 - ddc::discrete_space<bsplines_type1>().rmin())
666 / ddc::discrete_space<bsplines_type1>().length())
667 * ddc::discrete_space<bsplines_type1>().length();
668 }
669 }
670 if constexpr (bsplines_type2::is_periodic()) {
671 if (ddc::get<Dim2>(coord_eval) < ddc::discrete_space<bsplines_type2>().rmin()
672 || ddc::get<Dim2>(coord_eval) > ddc::discrete_space<bsplines_type2>().rmax()) {
673 ddc::get<Dim2>(coord_eval)
674 -= Kokkos::floor(
675 (ddc::get<Dim2>(coord_eval)
676 - ddc::discrete_space<bsplines_type2>().rmin())
677 / ddc::discrete_space<bsplines_type2>().length())
678 * ddc::discrete_space<bsplines_type2>().length();
679 }
680 }
681 if constexpr (!bsplines_type1::is_periodic()) {
682 if (ddc::get<Dim1>(coord_eval) < ddc::discrete_space<bsplines_type1>().rmin()) {
683 return m_left_extrap_rule_1(coord_eval, spline_coef);
684 }
685 if (ddc::get<Dim1>(coord_eval) > ddc::discrete_space<bsplines_type1>().rmax()) {
686 return m_right_extrap_rule_1(coord_eval, spline_coef);
687 }
688 }
689 if constexpr (!bsplines_type2::is_periodic()) {
690 if (ddc::get<Dim2>(coord_eval) < ddc::discrete_space<bsplines_type2>().rmin()) {
691 return m_left_extrap_rule_2(coord_eval, spline_coef);
692 }
693 if (ddc::get<Dim2>(coord_eval) > ddc::discrete_space<bsplines_type2>().rmax()) {
694 return m_right_extrap_rule_2(coord_eval, spline_coef);
695 }
696 }
697 return eval_no_bc<eval_type, eval_type>(
698 ddc::Coordinate<
699 typename evaluation_mesh_type1::continuous_dimension_type,
700 typename evaluation_mesh_type2::continuous_dimension_type>(
701 ddc::get<Dim1>(coord_eval),
702 ddc::get<Dim2>(coord_eval)),
703 spline_coef);
704 }

706 /**
707 * @brief Evaluate the function or its derivative at the coordinate given.
708 *
709 * @param[in] coord_eval
710 * The coordinate where we want to evaluate.
711 * @param[in] splne_coef
712 * The B-splines coefficients of the function we want to evaluate.
713 * @tparam EvalType1
714 * A flag indicating if we evaluate the function or its derivative in the first dimension.
715 * The type of this object is either `eval_type` or `eval_deriv_type`.
716 * @tparam EvalType2
717 * A flag indicating if we evaluate the function or its derivative in the second dimension.
718 * The type of this object is either `eval_type` or `eval_deriv_type`.
719 */
720 template <class EvalType1, class EvalType2, class Layout, class... CoordsDims>
721 KOKKOS_INLINE_FUNCTION double eval_no_bc(
722 ddc::Coordinate<CoordsDims...> const& coord_eval,
723 ddc::ChunkSpan<double const, spline_domain_type, Layout, memory_space> const
724 spline_coef) const
725 {
726 static_assert(
727 std::is_same_v<EvalType1, eval_type> || std::is_same_v<EvalType1, eval_deriv_type>);
728 static_assert(
729 std::is_same_v<EvalType2, eval_type> || std::is_same_v<EvalType2, eval_deriv_type>);
730 ddc::DiscreteElement<bsplines_type1> jmin1;
731 ddc::DiscreteElement<bsplines_type2> jmin2;
732
733 std::array<double, bsplines_type1::degree() + 1> vals1_ptr;
734 std::experimental::mdspan<
735 double,
736 std::experimental::extents<std::size_t, bsplines_type1::degree() + 1>> const
737 vals1(vals1_ptr.data());
738 std::array<double, bsplines_type2::degree() + 1> vals2_ptr;
739 std::experimental::mdspan<
740 double,
741 std::experimental::extents<std::size_t, bsplines_type2::degree() + 1>> const
742 vals2(vals2_ptr.data());
743 ddc::Coordinate<typename evaluation_mesh_type1::continuous_dimension_type>
744 coord_eval_interest1
745 = ddc::select<typename evaluation_mesh_type1::continuous_dimension_type>(
746 coord_eval);
747 ddc::Coordinate<typename evaluation_mesh_type2::continuous_dimension_type>
748 coord_eval_interest2
749 = ddc::select<typename evaluation_mesh_type2::continuous_dimension_type>(
750 coord_eval);
751
752 if constexpr (std::is_same_v<EvalType1, eval_type>) {
753 jmin1 = ddc::discrete_space<bsplines_type1>().eval_basis(vals1, coord_eval_interest1);
754 } else if constexpr (std::is_same_v<EvalType1, eval_deriv_type>) {
755 jmin1 = ddc::discrete_space<bsplines_type1>().eval_deriv(vals1, coord_eval_interest1);
756 }
757 if constexpr (std::is_same_v<EvalType2, eval_type>) {
758 jmin2 = ddc::discrete_space<bsplines_type2>().eval_basis(vals2, coord_eval_interest2);
759 } else if constexpr (std::is_same_v<EvalType2, eval_deriv_type>) {
760 jmin2 = ddc::discrete_space<bsplines_type2>().eval_deriv(vals2, coord_eval_interest2);
761 }
762
763 double y = 0.0;
764 for (std::size_t i = 0; i < bsplines_type1::degree() + 1; ++i) {
765 for (std::size_t j = 0; j < bsplines_type2::degree() + 1; ++j) {
766 y += spline_coef(ddc::DiscreteElement<
767 bsplines_type1,
768 bsplines_type2>(jmin1 + i, jmin2 + j))
769 * vals1[i] * vals2[j];
770 }
771 }
772 return y;
773 }
