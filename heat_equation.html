84 void display(double time, ChunkType temp)
85 {
86  double const mean_temp
87  = transform_reduce(temp.domain(), 0., std::plus(), temp)
88  / temp.domain().size();
89  std::cout << std::fixed << std::setprecision(3);
90  std::cout << "At t = " << time << ",\n";
91  std::cout << " * mean temperature = " << mean_temp << "\n";
92  // take a slice in the middle of the box
93  ChunkSpan temp_slice
94  = temp[get_domain<DDimY>(temp).front()
95  + get_domain<DDimY>(temp).size() / 2];
96  std::cout << " * temperature[y:"
97  << get_domain<DDimY>(temp).size() / 2 << "] = {";
98  for_each(get_domain<DDimX>(temp), [&](auto&& ix) {
99  std::cout << std::setw(6) << temp_slice(ix);
100  });
101  std::cout << " }" << std::endl;

107 int main()
108 {
111  // Number of ghost points to use on each side in X
112  DiscreteVector<DDimX> static constexpr gwx {1};
114 
116  // Initialization of the global domain in X with gwx ghost points on
117  // each side
118  auto const [x_domain, ghosted_x_domain, x_pre_ghost, x_post_ghost]
119  = init_discretization<DDimX>(DDimX::init_ghosted(
120  Coordinate<X>(x_start),
121  Coordinate<X>(x_end),
122  DiscreteVector<DDimX>(nb_x_points),
123  gwx));
125 
127  // our zone at the start of the domain that will be mirrored to the
128  // ghost
129  DiscreteDomain const
130  x_domain_begin(x_domain.front(), x_post_ghost.extents());
131  // our zone at the end of the domain that will be mirrored to the
132  // ghost
133  DiscreteDomain const x_domain_end(
134  x_domain.back() - x_pre_ghost.extents() + 1,
135  x_pre_ghost.extents());
137 
139  // Number of ghost points to use on each side in Y
140  DiscreteVector<DDimY> static constexpr gwy {1};
141 
142  // Initialization of the global domain in Y with gwy ghost points on
143  // each side
144  auto const [y_domain, ghosted_y_domain, y_pre_ghost, y_post_ghost]
145  = init_discretization(DDimY::init_ghosted(
146  Coordinate<Y>(y_start),
147  Coordinate<Y>(y_end),
148  DiscreteVector<DDimY>(nb_y_points),
149  gwy));
150 
151  // our zone at the start of the domain that will be mirrored to the
152  // ghost
153  DiscreteDomain const
154  y_domain_begin(y_domain.front(), y_post_ghost.extents());
155  // our zone at the end of the domain that will be mirrored to the
156  // ghost
157  DiscreteDomain const y_domain_end(
158  y_domain.back() - y_pre_ghost.extents() + 1,
159  y_pre_ghost.extents());
161 
163  // max(1/dx^2)
164  double const invdx2_max = transform_reduce(
165  x_domain,
166  0.,
167  max,
168  [](DiscreteCoordinate<DDimX> ix) {
169  return 1.
170  / (distance_at_left(ix) * distance_at_right(ix));
171  });
172  // max(1/dy^2)
173  double const invdy2_max = transform_reduce(
174  y_domain,
175  0.,
176  max,
177  [](DiscreteCoordinate<DDimY> iy) {
178  return 1.
179  / (distance_at_left(iy) * distance_at_right(iy));
180  });
181  Coordinate<T> const max_dt {
182  .5 / (kx * invdx2_max + ky * invdy2_max)};
183 
184  // number of time intervals required to reach the end time
185  DiscreteVector<DDimT> const nb_time_steps {
186  std::ceil((end_time - start_time) / max_dt) + .2};
187  // Initialization of the global domain in time:
188  // - the number of discrete time-points is equal to the number of
189  // steps + 1
190  DiscreteDomain<DDimT> const time_domain
191  = init_discretization(DDimT::
192  init(Coordinate<T>(start_time),
193  Coordinate<T>(end_time),
194  nb_time_steps + 1));
196 
198  // Maps temperature into the full domain (including ghosts) twice:
199  // - once for the last fully computed time-step
200  Chunk<double, DiscreteDomain<DDimX, DDimY>> ghosted_last_temp(
201  DiscreteDomain<
202  DDimX,
203  DDimY>(ghosted_x_domain, ghosted_y_domain));
204  // - once for time-step being computed
205  Chunk<double, DiscreteDomain<DDimX, DDimY>> ghosted_next_temp(
206  DiscreteDomain<
207  DDimX,
208  DDimY>(ghosted_x_domain, ghosted_y_domain));
210 
212  // Initialize the temperature on the main domain
213  for_each(
214  DiscreteDomain<DDimX, DDimY>(x_domain, y_domain),
215  [&](DiscreteCoordinate<DDimX, DDimY> const ixy) {
216  double const x = to_real(select<DDimX>(ixy));
217  double const y = to_real(select<DDimY>(ixy));
218  ghosted_last_temp(ixy)
219  = 9.999 * ((x * x + y * y) < 0.25);
220  });
222 
224  // display the initial data
225  display(to_real(time_domain.front()),
226  ghosted_last_temp[x_domain][y_domain]);
227  // time of the iteration where the last output happened
228  DiscreteCoordinate<DDimT> last_output = time_domain.front();
230 
232  for (auto const iter :
233  DiscreteDomain {time_domain.front() + 1, nb_time_steps - 1}) {
235 
237  // Periodic boundary conditions
238  deepcopy(
239  ghosted_last_temp[x_pre_ghost][y_domain],
240  ghosted_last_temp[y_domain][x_domain_end]);
241  deepcopy(
242  ghosted_last_temp[y_domain][x_post_ghost],
243  ghosted_last_temp[y_domain][x_domain_begin]);
244  deepcopy(
245  ghosted_last_temp[x_domain][y_pre_ghost],
246  ghosted_last_temp[x_domain][y_domain_end]);
247  deepcopy(
248  ghosted_last_temp[x_domain][y_post_ghost],
249  ghosted_last_temp[x_domain][y_domain_begin]);
251 
253  // a span excluding ghosts of the temperature at the time-step we
254  // will build
255  ChunkSpan const next_temp {
256  ghosted_next_temp[x_domain][y_domain]};
257  // a read-only view of the temperature at the previous time-step
258  ChunkSpan const last_temp {ghosted_last_temp.span_view()};
260 
262  // Stencil computation on the main domain
263  for_each(
264  next_temp.domain(),
265  [&](DiscreteCoordinate<DDimX, DDimY> const ixy) {
266  DiscreteCoordinate<DDimX> const ix
267  = select<DDimX>(ixy);
268  DiscreteCoordinate<DDimY> const iy
269  = select<DDimY>(ixy);
270  double const dx_l = distance_at_left(ix);
271  double const dx_r = distance_at_right(ix);
272  double const dx_m = 0.5 * (dx_l + dx_r);
273  double const dy_l = distance_at_left(iy);
274  double const dy_r = distance_at_right(iy);
275  double const dy_m = 0.5 * (dy_l + dy_r);
276  next_temp(ix, iy) = last_temp(ix, iy);
277  next_temp(ix, iy)
278  += kx * max_dt
279  * (dx_l * last_temp(ix + 1, iy)
280  - 2.0 * dx_m * last_temp(ix, iy)
281  + dx_r * last_temp(ix - 1, iy))
282  / (dx_l * dx_m * dx_r);
283  next_temp(ix, iy)
284  += ky * max_dt
285  * (dy_l * last_temp(ix, iy + 1)
286  - 2.0 * dy_m * last_temp(ix, iy)
287  + dy_r * last_temp(ix, iy - 1))
288  / (dy_l * dy_m * dy_r);
289  });
291 
293  if (iter - last_output >= t_output_period) {
294  last_output = iter;
295  display(to_real(iter), next_temp);
296  }
298 
300  // Swap our two buffers
301  std::swap(ghosted_last_temp, ghosted_next_temp);
303  }
304 
306  if (last_output < time_domain.back()) {
307  display(to_real(time_domain.back()),
308  ghosted_last_temp[x_domain][y_domain]);
309  }
311 }
