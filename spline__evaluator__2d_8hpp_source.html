337 /**
338 * @brief Evaluate 2D spline function (described by its spline coefficients) at a given coordinate.
339 *
340 * The spline coefficients represent a 2D spline function defined on a B-splines (basis splines). They can be obtained via various methods, such as using a SplineBuilder2D.
341 *
342 * Remark: calling SplineBuilder2D then SplineEvaluator2D corresponds to a 2D spline interpolation.
343 *
344 * @param coord_eval The coordinate where the spline is evaluated. Note that only the components along the dimensions of interest are used.
345 * @param spline_coef A ChunkSpan storing the 2D spline coefficients.
346 *
347 * @return The value of the spline function at the desired coordinate.
348 */
349 template <class Layout, class... CoordsDims>
350 KOKKOS_FUNCTION double operator()(
351 ddc::Coordinate<CoordsDims...> const& coord_eval,
352 ddc::ChunkSpan<double const, spline_domain_type, Layout, memory_space> const
353 spline_coef) const
354 {
355 return eval(coord_eval, spline_coef);
356 }

358 /**
359 * @brief Evaluate 2D spline function (described by its spline coefficients) on a mesh.
360 *
361 * The spline coefficients represent a 2D spline function defined on a cartesian product of batch_domain and B-splines
362 * (basis splines). They can be obtained via various methods, such as using a SplineBuilder2D.
363 *
364 * This is not a nD evaluation. This is a batched 2D evaluation. This means that for each slice of coordinates
365 * identified by a batch_domain_type::discrete_element_type, the evaluation is performed with the 2D set of
366 * spline coefficients identified by the same batch_domain_type::discrete_element_type.
367 *
368 * Remark: calling SplineBuilder2D then SplineEvaluator2D corresponds to a 2D spline interpolation.
369 *
370 * @param[out] spline_eval The values of the 2D spline function at the desired coordinates. For practical reasons those are
371 * stored in a ChunkSpan defined on a batched_evaluation_domain_type.
372 * @param[in] coords_eval The coordinates where the spline is evaluated. Those are
373 * stored in a ChunkSpan defined on a batched_evaluation_domain_type. Note that the coordinates of the
374 * points represented by this domain are unused and irrelevant (but the points themselves (DiscreteElement) are used to select
375 * the set of 2D spline coefficients retained to perform the evaluation).
376 * @param[in] spline_coef A ChunkSpan storing the 2D spline coefficients.
377 */
378 template <class Layout1, class Layout2, class Layout3, class... CoordsDims>
379 void operator()(
380 ddc::ChunkSpan<double, batched_evaluation_domain_type, Layout1, memory_space> const
381 spline_eval,
382 ddc::ChunkSpan<
383 ddc::Coordinate<CoordsDims...> const,
384 batched_evaluation_domain_type,
385 Layout2,
386 memory_space> const coords_eval,
387 ddc::ChunkSpan<double const, batched_spline_domain_type, Layout3, memory_space> const
388 spline_coef) const
389 {
390 batch_domain_type const batch_domain(coords_eval.domain());
391 evaluation_domain_type1 const evaluation_domain1(spline_eval.domain());
392 evaluation_domain_type2 const evaluation_domain2(spline_eval.domain());
393 ddc::parallel_for_each(
394 "ddc_splines_evaluate_2d",
395 exec_space(),
396 batch_domain,
397 KOKKOS_CLASS_LAMBDA(typename batch_domain_type::discrete_element_type const j) {
398 const auto spline_eval_2D = spline_eval[j];
399 const auto coords_eval_2D = coords_eval[j];
400 const auto spline_coef_2D = spline_coef[j];
401 for (auto const i1 : evaluation_domain1) {
402 for (auto const i2 : evaluation_domain2) {
403 spline_eval_2D(i1, i2) = eval(coords_eval_2D(i1, i2), spline_coef_2D);
404 }
405 }
406 });
407 }

469 /**
470 * @brief Differentiate 2D spline function (described by its spline coefficients) at a given coordinate along a specified dimension of interest.
471 *
472 * The spline coefficients represent a 2D spline function defined on a B-splines (basis splines). They can be
473 * obtained via various methods, such as using a SplineBuilder2D.
474 *
475 * @tparam InterestDim Dimension along which differentiation is performed.
476 *
477 * @param coord_eval The coordinate where the spline is differentiated. Note that only the components along the dimensions of interest are used.
478 * @param spline_coef A ChunkSpan storing the 2D spline coefficients.
479 *
480 * @return The derivative of the spline function at the desired coordinate.
481 */
482 template <class InterestDim, class Layout, class... CoordsDims>
483 KOKKOS_FUNCTION double deriv(
484 ddc::Coordinate<CoordsDims...> const& coord_eval,
485 ddc::ChunkSpan<double const, spline_domain_type, Layout, memory_space> const
486 spline_coef) const
487 {
488 static_assert(
489 std::is_same_v<InterestDim, continuous_dimension_type1>
490 || std::is_same_v<InterestDim, continuous_dimension_type2>);
491 if constexpr (std::is_same_v<
492 InterestDim,
493 typename evaluation_discrete_dimension_type1::
494 continuous_dimension_type>) {
495 return deriv_dim_1(coord_eval, spline_coef);
496 } else if constexpr (std::is_same_v<
497 InterestDim,
498 typename evaluation_discrete_dimension_type2::
499 continuous_dimension_type>) {
500 return deriv_dim_2(coord_eval, spline_coef);
501 }
502 }

510 * Note: double-differentiation other than cross-differentiation is not supported atm. See #440
511 *
512 * @tparam InterestDim1 First dimension along which differentiation is performed.
513 * @tparam InterestDim2 Second dimension along which differentiation is performed.
514 *
515 * @param coord_eval The coordinate where the spline is double-differentiated. Note that only the components along the dimensions of interest are used.
516 * @param spline_coef A ChunkSpan storing the 2D spline coefficients.
517 *

520 template <class InterestDim1, class InterestDim2, class Layout, class... CoordsDims>

522 ddc::Coordinate<CoordsDims...> const& coord_eval,

527 (std::is_same_v<
528 InterestDim1,
529 typename evaluation_discrete_dimension_type1::continuous_dimension_type>

531 || (std::is_same_v<
532 InterestDim2,
533 typename evaluation_discrete_dimension_type1::continuous_dimension_type>

795 /** @brief Perform batched 2D integrations of a spline function (described by its spline coefficients) along the dimensions of interest and store results on a subdomain of batch_domain.
796 *
797 * The spline coefficients represent a 2D spline function defined on a B-splines (basis splines). They can be obtained via various methods, such as using a SplineBuilder2D.
798 *
799 * This is not a nD integration. This is a batched 2D integration.
800 * This means that for each element of integrals, the integration is performed with the 2D set of
801 * spline coefficients identified by the same DiscreteElement.
802 *
803 * @param[out] integrals The integrals of the 2D spline function on the subdomain of batch_domain. For practical reasons those are
804 * stored in a ChunkSpan defined on a batch_domain_type. Note that the coordinates of the
805 * points represented by this domain are unused and irrelevant.
806 * @param[in] spline_coef A ChunkSpan storing the 2D spline coefficients.
807 */
808 template <class Layout1, class Layout2>
809 void integrate(
810 ddc::ChunkSpan<double, batch_domain_type, Layout1, memory_space> const integrals,
811 ddc::ChunkSpan<double const, batched_spline_domain_type, Layout2, memory_space> const
812 spline_coef) const
813 {
814 batch_domain_type batch_domain(integrals.domain());
815 ddc::Chunk values1_alloc(
816 ddc::DiscreteDomain<bsplines_type1>(spline_coef.domain()),
817 ddc::KokkosAllocator<double, memory_space>());
818 ddc::ChunkSpan values1 = values1_alloc.span_view();
819 ddc::integrals(exec_space(), values1);
820 ddc::Chunk values2_alloc(
821 ddc::DiscreteDomain<bsplines_type2>(spline_coef.domain()),
822 ddc::KokkosAllocator<double, memory_space>());
823 ddc::ChunkSpan values2 = values2_alloc.span_view();
824 ddc::integrals(exec_space(), values2);
825
826 ddc::parallel_for_each(
827 "ddc_splines_integrate_bsplines",
828 exec_space(),
829 batch_domain,
830 KOKKOS_LAMBDA(typename batch_domain_type::discrete_element_type const j) {
831 integrals(j) = 0;
832 for (typename spline_domain_type1::discrete_element_type const i1 :
833 values1.domain()) {
834 for (typename spline_domain_type2::discrete_element_type const i2 :
835 values2.domain()) {
836 integrals(j) += spline_coef(i1, i2, j) * values1(i1) * values2(i2);
837 }
838 }
839 });
840 }

843 /**
844 * @brief Evaluate the function on B-splines at the coordinate given.
845 *
846 * This function firstly deals with the boundary conditions and calls the SplineEvaluator2D::eval_no_bc function
847 * to evaluate.
848 *
849 * @param[in] coord_eval
850 * The 2D coordinate where we want to evaluate.
851 * @param[in] spline_coef
852 * The B-splines coefficients of the function we want to evaluate.
853 * @param[out] vals1
854 * A ChunkSpan with the not-null values of each function of the spline in the first dimension.
855 * @param[out] vals2
856 * A ChunkSpan with the not-null values of each function of the spline in the second dimension.
857 *
858 * @return A double with the value of the function at the coordinate given.
859 *
860 * @see SplineBoundaryValue
861 */
862 template <class Layout, class... CoordsDims>
863 KOKKOS_INLINE_FUNCTION double eval(
864 ddc::Coordinate<CoordsDims...> coord_eval,
865 ddc::ChunkSpan<double const, spline_domain_type, Layout, memory_space> const
866 spline_coef) const
867 {
868 using Dim1 = continuous_dimension_type1;
869 using Dim2 = continuous_dimension_type2;
870 if constexpr (bsplines_type1::is_periodic()) {
871 if (ddc::get<Dim1>(coord_eval) < ddc::discrete_space<bsplines_type1>().rmin()
872 || ddc::get<Dim1>(coord_eval) > ddc::discrete_space<bsplines_type1>().rmax()) {
873 ddc::get<Dim1>(coord_eval)
874 -= Kokkos::floor(
875 (ddc::get<Dim1>(coord_eval)
876 - ddc::discrete_space<bsplines_type1>().rmin())
877 / ddc::discrete_space<bsplines_type1>().length())
878 * ddc::discrete_space<bsplines_type1>().length();
879 }
880 }
881 if constexpr (bsplines_type2::is_periodic()) {
882 if (ddc::get<Dim2>(coord_eval) < ddc::discrete_space<bsplines_type2>().rmin()
883 || ddc::get<Dim2>(coord_eval) > ddc::discrete_space<bsplines_type2>().rmax()) {
884 ddc::get<Dim2>(coord_eval)
885 -= Kokkos::floor(
886 (ddc::get<Dim2>(coord_eval)
887 - ddc::discrete_space<bsplines_type2>().rmin())
888 / ddc::discrete_space<bsplines_type2>().length())
889 * ddc::discrete_space<bsplines_type2>().length();
890 }
891 }
892 if constexpr (!bsplines_type1::is_periodic()) {
893 if (ddc::get<Dim1>(coord_eval) < ddc::discrete_space<bsplines_type1>().rmin()) {
894 return m_lower_extrap_rule_1(coord_eval, spline_coef);
895 }
896 if (ddc::get<Dim1>(coord_eval) > ddc::discrete_space<bsplines_type1>().rmax()) {
897 return m_upper_extrap_rule_1(coord_eval, spline_coef);
898 }
899 }
900 if constexpr (!bsplines_type2::is_periodic()) {
901 if (ddc::get<Dim2>(coord_eval) < ddc::discrete_space<bsplines_type2>().rmin()) {
902 return m_lower_extrap_rule_2(coord_eval, spline_coef);
903 }
904 if (ddc::get<Dim2>(coord_eval) > ddc::discrete_space<bsplines_type2>().rmax()) {
905 return m_upper_extrap_rule_2(coord_eval, spline_coef);
906 }
907 }
908 return eval_no_bc<eval_type, eval_type>(
909 ddc::Coordinate<continuous_dimension_type1, continuous_dimension_type2>(
910 ddc::get<Dim1>(coord_eval),
911 ddc::get<Dim2>(coord_eval)),
912 spline_coef);
913 }

915 /**
916 * @brief Evaluate the function or its derivative at the coordinate given.
917 *
918 * @param[in] coord_eval
919 * The coordinate where we want to evaluate.
920 * @param[in] splne_coef
921 * The B-splines coefficients of the function we want to evaluate.
922 * @tparam EvalType1
923 * A flag indicating if we evaluate the function or its derivative in the first dimension.
924 * The type of this object is either `eval_type` or `eval_deriv_type`.
925 * @tparam EvalType2
926 * A flag indicating if we evaluate the function or its derivative in the second dimension.
927 * The type of this object is either `eval_type` or `eval_deriv_type`.
928 */
929 template <class EvalType1, class EvalType2, class Layout, class... CoordsDims>
930 KOKKOS_INLINE_FUNCTION double eval_no_bc(
931 ddc::Coordinate<CoordsDims...> const& coord_eval,
932 ddc::ChunkSpan<double const, spline_domain_type, Layout, memory_space> const
933 spline_coef) const
934 {
935 static_assert(
936 std::is_same_v<EvalType1, eval_type> || std::is_same_v<EvalType1, eval_deriv_type>);
937 static_assert(
938 std::is_same_v<EvalType2, eval_type> || std::is_same_v<EvalType2, eval_deriv_type>);
939 ddc::DiscreteElement<bsplines_type1> jmin1;
940 ddc::DiscreteElement<bsplines_type2> jmin2;
941
942 std::array<double, bsplines_type1::degree() + 1> vals1_ptr;
943 std::experimental::mdspan<
944 double,
945 std::experimental::extents<std::size_t, bsplines_type1::degree() + 1>> const
946 vals1(vals1_ptr.data());
947 std::array<double, bsplines_type2::degree() + 1> vals2_ptr;
948 std::experimental::mdspan<
949 double,
950 std::experimental::extents<std::size_t, bsplines_type2::degree() + 1>> const
951 vals2(vals2_ptr.data());
952 ddc::Coordinate<continuous_dimension_type1> const coord_eval_interest1
953 = ddc::select<continuous_dimension_type1>(coord_eval);
954 ddc::Coordinate<continuous_dimension_type2> const coord_eval_interest2
955 = ddc::select<continuous_dimension_type2>(coord_eval);
956
957 if constexpr (std::is_same_v<EvalType1, eval_type>) {
958 jmin1 = ddc::discrete_space<bsplines_type1>().eval_basis(vals1, coord_eval_interest1);
959 } else if constexpr (std::is_same_v<EvalType1, eval_deriv_type>) {
960 jmin1 = ddc::discrete_space<bsplines_type1>().eval_deriv(vals1, coord_eval_interest1);
961 }
962 if constexpr (std::is_same_v<EvalType2, eval_type>) {
963 jmin2 = ddc::discrete_space<bsplines_type2>().eval_basis(vals2, coord_eval_interest2);
964 } else if constexpr (std::is_same_v<EvalType2, eval_deriv_type>) {
965 jmin2 = ddc::discrete_space<bsplines_type2>().eval_deriv(vals2, coord_eval_interest2);
966 }
967
968 double y = 0.0;
969 for (std::size_t i = 0; i < bsplines_type1::degree() + 1; ++i) {
970 for (std::size_t j = 0; j < bsplines_type2::degree() + 1; ++j) {
971 y += spline_coef(ddc::DiscreteElement<
972 bsplines_type1,
973 bsplines_type2>(jmin1 + i, jmin2 + j))
974 * vals1[i] * vals2[j];
975 }
976 }
977 return y;
978 }
