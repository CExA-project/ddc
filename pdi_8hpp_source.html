19template <class T>

21 = (std::is_lvalue_reference_v<T> && !std::is_const_v<std::remove_reference_t<T>>)

59 template <
60 PDI_inout_t access,
61 class BorrowedChunk,
62 std::enable_if_t<is_borrowed_chunk_v<BorrowedChunk>, int> = 0>
63 PdiEvent& with(std::string const& name, BorrowedChunk&& data)
64 {
65 static_assert(
66 !(access & PDI_IN) || (chunk_default_access_v<BorrowedChunk> & PDI_IN),
67 "Invalid access for constant data");
68 auto extents = detail::array(data.domain().extents());
69 std::size_t& rank = *std::pmr::polymorphic_allocator<std::size_t>(&m_metadata).allocate(1);
70 rank = extents.size();
71 PDI_share((name + "_rank").c_str(), &rank, PDI_OUT);
72 m_names.push_back(name + "_rank");
73 PDI_share(
74 (name + "_extents").c_str(),
75 std::pmr::vector<std::size_t>(extents.begin(), extents.end(), &m_metadata).data(),
76 PDI_OUT);
77 m_names.push_back(name + "_extents");
78 PDI_share(
79 name.c_str(),
80 const_cast<chunk_value_t<BorrowedChunk>*>(data.data_handle()),
81 access);
82 m_names.push_back(name);
83 return *this;
84 }
85
86 template <
87 PDI_inout_t access,
88 class Arithmetic,
89 std::enable_if_t<std::is_arithmetic_v<std::remove_reference_t<Arithmetic>>, int> = 0>
90 PdiEvent& with(std::string const& name, Arithmetic&& data)
91 {
92 static_assert(
93 !(access & PDI_IN) || (default_access_v<Arithmetic> & PDI_IN),
94 "Invalid access for constant data");
95 using value_type = std::remove_cv_t<std::remove_reference_t<Arithmetic>>;
96 value_type* data_ptr = const_cast<value_type*>(&data);
97 // for read-only data, we share a copy instead of the data itself in case we received a ref on a temporary,
98 if constexpr (!(access & PDI_IN)) {
99 data_ptr = std::pmr::polymorphic_allocator<value_type>(&m_metadata).allocate(1);
100 *data_ptr = data;
101 }
102 PDI_share(name.c_str(), data_ptr, access);
103 m_names.push_back(name);
104 return *this;
105 }
