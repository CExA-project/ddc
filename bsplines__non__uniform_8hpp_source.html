222template <class MemorySpace>
223template <class RandomIt>
224NonUniformBSplines<Tag, D>::Impl<MemorySpace>::Impl(
225 RandomIt const break_begin,
226 RandomIt const break_end)
227 : m_domain(
228 ddc::DiscreteElement<mesh_type>(0),
229 ddc::DiscreteVector<mesh_type>(
230 (break_end - break_begin)
231 + 2 * degree())) // Create a mesh including the eventual periodic point
232 , m_nknots((break_end - break_begin) + 2 * degree())
233{
234 std::vector<ddc::Coordinate<KnotDim>> knots((break_end - break_begin) + 2 * degree());
235 // Fill the provided knots
236 int ii = 0;
237 for (RandomIt it = break_begin; it < break_end; ++it) {
238 knots[degree() + ii] = knot_from_coord(*it);
239 ++ii;
240 }
241 ddc::Coordinate<KnotDim> const rmin = knots[degree()];
242 ddc::Coordinate<KnotDim> const rmax = knots[(break_end - break_begin) + degree() - 1];
243 assert(rmin < rmax);
244
245 // Fill out the extra knots
246 if constexpr (is_periodic()) {
247 double const period = rmax - rmin;
248 for (std::size_t i = 1; i < degree() + 1; ++i) {
249 knots[degree() + -i]
250 = ddc::Coordinate<KnotDim>(knots[degree() + ncells() - i] - period);
251 knots[degree() + ncells() + i] = ddc::Coordinate<KnotDim>(knots[degree() + i] + period);
252 }
253 } else // open
254 {
255 for (std::size_t i = 1; i < degree() + 1; ++i) {
256 knots[degree() + -i] = rmin;
257 knots[degree() + npoints() - 1 + i] = rmax;
258 }
259 }
260 ddc::init_discrete_space<mesh_type>(knots);
261}

263template <class Tag, std::size_t D>
264template <class MemorySpace>
265KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<NonUniformBSplines<Tag, D>> NonUniformBSplines<Tag, D>::
266 Impl<MemorySpace>::eval_basis(
267 std::array<double, D + 1>& values,
268 ddc::Coordinate<Tag> const& x) const
269{
270 std::array<double, degree()> left;
271 std::array<double, degree()> right;
272
273 assert(x >= rmin());
274 assert(x <= rmax());
275 assert(values.size() == degree() + 1);
276
277 // 1. Compute cell index 'icell'
278 int const icell = find_cell(x);
279
280 assert(icell >= 0);
281 assert(icell <= int(ncells() - 1));
282 assert(get_knot(icell) <= x);
283 assert(get_knot(icell + 1) >= x);
284
285 // 2. Compute values of B-splines with support over cell 'icell'
286 double temp;
287 values[0] = 1.0;
288 for (std::size_t j = 0; j < degree(); ++j) {
289 left[j] = x - get_knot(icell - j);
290 right[j] = get_knot(icell + j + 1) - x;
291 double saved = 0.0;
292 for (std::size_t r = 0; r < j + 1; ++r) {
293 temp = values[r] / (right[r] + left[j - r]);
294 values[r] = saved + right[r] * temp;
295 saved = left[j - r] * temp;
296 }
297 values[j + 1] = saved;
298 }
299
300 return discrete_element_type(icell);
301}

304template <class MemorySpace>
305KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<NonUniformBSplines<Tag, D>> NonUniformBSplines<Tag, D>::
306 Impl<MemorySpace>::eval_deriv(
307 std::array<double, D + 1>& derivs,
308 ddc::Coordinate<Tag> const& x) const
309{
310 std::array<double, degree()> left;
311 std::array<double, degree()> right;
312
313 assert(x >= rmin());
314 assert(x <= rmax());
315 assert(derivs.size() == degree() + 1);
316
317 // 1. Compute cell index 'icell'
318 int const icell = find_cell(x);
319
320 assert(icell >= 0);
321 assert(icell <= int(ncells() - 1));
322 assert(get_knot(icell) <= x);
323 assert(get_knot(icell + 1) >= x);
324
325 // 2. Compute values of derivatives of B-splines with support over cell 'icell'
326
327 /*
328 * Compute nonzero basis functions and knot differences
329 * for splines up to degree degree-1 which are needed to compute derivative
330 * First part of Algorithm A3.2 of NURBS book
331 */
332 double saved, temp;
333 derivs[0] = 1.0;
334 for (std::size_t j = 0; j < degree() - 1; ++j) {
335 left[j] = x - get_knot(icell - j);
336 right[j] = get_knot(icell + j + 1) - x;
337 saved = 0.0;
338 for (std::size_t r = 0; r < j + 1; ++r) {
339 temp = derivs[r] / (right[r] + left[j - r]);
340 derivs[r] = saved + right[r] * temp;
341 saved = left[j - r] * temp;
342 }
343 derivs[j + 1] = saved;
344 }
345
346 /*
347 * Compute derivatives at x using values stored in bsdx and formula
348 * for spline derivative based on difference of splines of degree degree-1
349 */
350 saved = degree() * derivs[0] / (get_knot(icell + 1) - get_knot(icell + 1 - degree()));
351 derivs[0] = -saved;
352 for (std::size_t j = 1; j < degree(); ++j) {
353 temp = saved;
354 saved = degree() * derivs[j]
355 / (get_knot(icell + j + 1) - get_knot(icell + j + 1 - degree()));
356 derivs[j] = temp - saved;
357 }
358 derivs[degree()] = saved;
359
360 return discrete_element_type(icell);
361}

364template <class MemorySpace>
365KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<NonUniformBSplines<Tag, D>> NonUniformBSplines<Tag, D>::
366 Impl<MemorySpace>::eval_basis_and_n_derivs(
367 ddc::DSpan2D const derivs,
368 ddc::Coordinate<Tag> const& x,
369 std::size_t const n) const
370{
371 std::array<double, degree()> left;
372 std::array<double, degree()> right;
373
374 std::array<double, 2 * (degree() + 1)> a_ptr;
375 std::experimental::
376 mdspan<double, std::experimental::extents<std::size_t, degree() + 1, 2>> const a(
377 a_ptr.data());
378
379 std::array<double, (degree() + 1) * (degree() + 1)> ndu_ptr;
380 std::experimental::mdspan<
381 double,
382 std::experimental::extents<std::size_t, degree() + 1, degree() + 1>> const
383 ndu(ndu_ptr.data());
384
385 assert(x >= rmin());
386 assert(x <= rmax());
387 // assert(n >= 0); as long as n is unsigned
388 assert(n <= degree());
389 assert(derivs.extent(0) == 1 + degree());
390 assert(derivs.extent(1) == 1 + n);
391
392 // 1. Compute cell index 'icell' and x_offset
393 int const icell = find_cell(x);
394
395 assert(icell >= 0);
396 assert(icell <= int(ncells() - 1));
397 assert(get_knot(icell) <= x);
398 assert(get_knot(icell + 1) >= x);
399
400 // 2. Compute nonzero basis functions and knot differences for splines
401 // up to degree (degree-1) which are needed to compute derivative
402 // Algorithm A2.3 of NURBS book
403 //
404 // 21.08.2017: save inverse of knot differences to avoid unnecessary
405 // divisions
406 // [Yaman Güçlü, Edoardo Zoni]
407
408 double saved, temp;
409 ndu(0, 0) = 1.0;
410 for (std::size_t j = 0; j < degree(); ++j) {
411 left[j] = x - get_knot(icell - j);
412 right[j] = get_knot(icell + j + 1) - x;
413 saved = 0.0;
414 for (std::size_t r = 0; r < j + 1; ++r) {
415 // compute inverse of knot differences and save them into lower
416 // triangular part of ndu
417 ndu(r, j + 1) = 1.0 / (right[r] + left[j - r]);
418 // compute basis functions and save them into upper triangular part
419 // of ndu
420 temp = ndu(j, r) * ndu(r, j + 1);
421 ndu(j + 1, r) = saved + right[r] * temp;
422 saved = left[j - r] * temp;
423 }
424 ndu(j + 1, j + 1) = saved;
425 }
426 // Save 0-th derivative
427 for (std::size_t j = 0; j < degree() + 1; ++j) {
428 derivs(j, 0) = ndu(degree(), j);
429 }
430
431 for (int r = 0; r < int(degree() + 1); ++r) {
432 int s1 = 0;
433 int s2 = 1;
434 a(0, 0) = 1.0;
435 for (int k = 1; k < int(n + 1); ++k) {
436 double d = 0.0;
437 int const rk = r - k;
438 int const pk = degree() - k;
439 if (r >= k) {
440 a(0, s2) = a(0, s1) * ndu(rk, pk + 1);
441 d = a(0, s2) * ndu(pk, rk);
442 }
443 int const j1 = rk > -1 ? 1 : (-rk);
444 int const j2 = (r - 1) <= pk ? k : (degree() - r + 1);
445 for (int j = j1; j < j2; ++j) {
446 a(j, s2) = (a(j, s1) - a(j - 1, s1)) * ndu(rk + j, pk + 1);
447 d += a(j, s2) * ndu(pk, rk + j);
448 }
449 if (r <= pk) {
450 a(k, s2) = -a(k - 1, s1) * ndu(r, pk + 1);
451 d += a(k, s2) * ndu(pk, r);
452 }
453 derivs(r, k) = d;
454 // swap s1 <-> s2;
455 auto tmp = s1;
456 s1 = s2;
457 s2 = tmp;
458 }
459 }
460
461 int r = degree();
462 for (int k = 1; k < int(n + 1); ++k) {
463 for (std::size_t i = 0; i < derivs.extent(0); i++) {
464 derivs(i, k) *= r;
465 }
466 r *= degree() - k;
467 }
468
469 return discrete_element_type(icell);
470}

472template <class Tag, std::size_t D>
473template <class MemorySpace>
474KOKKOS_INLINE_FUNCTION int NonUniformBSplines<Tag, D>::Impl<MemorySpace>::find_cell(
475 ddc::Coordinate<Tag> const& x) const
476{
477 if (x > rmax())
478 return -1;
479 if (x < rmin())
480 return -1;
481
482 if (x == rmin())
483 return 0;
484 if (x == rmax())
485 return ncells() - 1;
486
487 // Binary search
488 int low = 0, high = ncells();
489 int icell = (low + high) / 2;
490 while (x < get_knot(icell) || x >= get_knot(icell + 1)) {
491 if (x < get_knot(icell)) {
492 high = icell;
493 } else {
494 low = icell;
495 }
496 icell = (low + high) / 2;
497 }
498 return icell;
499}

501template <class Tag, std::size_t D>
502template <class MemorySpace>
503template <class Layout, class MemorySpace2>
504KOKKOS_INLINE_FUNCTION ddc::
505 ChunkSpan<double, ddc::DiscreteDomain<NonUniformBSplines<Tag, D>>, Layout, MemorySpace2>
506 NonUniformBSplines<Tag, D>::Impl<MemorySpace>::integrals(
507 ddc::ChunkSpan<
508 double,
509 ddc::DiscreteDomain<NonUniformBSplines<Tag, D>>,
510 Layout,
511 MemorySpace2> int_vals) const
512{
513 if constexpr (is_periodic()) {
514 assert(int_vals.size() == nbasis() || int_vals.size() == size());
515 } else {
516 assert(int_vals.size() == nbasis());
517 }
518
519 double const inv_deg = 1.0 / (degree() + 1);
520
521 discrete_domain_type const dom_bsplines(
522 full_domain().take_first(discrete_vector_type {nbasis()}));
523 for (auto ix : dom_bsplines) {
524 int_vals(ix) = (get_last_support_knot(ix) - get_first_support_knot(ix)) * inv_deg;
525 }
526
527 if constexpr (is_periodic()) {
528 if (int_vals.size() == size()) {
529 discrete_domain_type const dom_bsplines_wrap(
530 full_domain().take_last(discrete_vector_type {degree()}));
531 for (auto ix : dom_bsplines_wrap) {
532 int_vals(ix) = 0;
533 }
534 }
535 }
536 return int_vals;
537}
