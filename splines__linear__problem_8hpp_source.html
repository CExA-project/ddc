16/**
17 * @brief The parent class for linear problems dedicated to the computation of spline approximations.
18 *
19 * Store a square matrix and provide method to solve a multiple right-hand sides linear problem.
20 * Implementations may have different storage formats, filling methods and multiple right-hand sides linear solvers.
21 */
22template <class ExecSpace>
23class SplinesLinearProblem
24{
25public:
26 /// @brief The type of a Kokkos::View storing multiple right-hand sides.
27 using MultiRHS = Kokkos::View<double**, Kokkos::LayoutRight, ExecSpace>;
28
29private:
30 std::size_t m_size;
31
32protected:
33 explicit SplinesLinearProblem(const std::size_t size) : m_size(size) {}
34
35public:
36 /// @brief Destruct
37 virtual ~SplinesLinearProblem() = default;
38
39 /**
40 * @brief Get an element of the matrix at indexes i, j. It must not be called after `setup_solver`.
41 *
42 * @param i The row index of the desired element.
43 * @param j The column index of the desired element.
44 *
45 * @return The value of the element of the matrix.
46 */
47 virtual double get_element(std::size_t i, std::size_t j) const = 0;
48
49 /**
50 * @brief Set an element of the matrix at indexes i, j. It must not be called after `setup_solver`.
51 *
52 * @param i The row index of the setted element.
53 * @param j The column index of the setted element.
54 * @param aij The value to set in the element of the matrix.
55 */
56 virtual void set_element(std::size_t i, std::size_t j, double aij) = 0;
57
58 /**
59 * @brief Perform a pre-process operation on the solver. Must be called after filling the matrix.
60 */
61 virtual void setup_solver() = 0;
62
63 /**
64 * @brief Solve the multiple right-hand sides linear problem Ax=b or its transposed version A^tx=b inplace.
65 *
66 * @param[in, out] multi_rhs A 2D Kokkos::View storing the multiple right-hand sides of the problem and receiving the corresponding solution.
67 * @param transpose Choose between the direct or transposed version of the linear problem.
68 */
69 virtual void solve(MultiRHS b, bool transpose = false) const = 0;
70
71 /**
72 * @brief Get the size of the square matrix in one of its dimensions.
73 *
74 * @return The size of the matrix in one of its dimensions.
75 */
76 std::size_t size() const
77 {
78 return m_size;
79 }
80
81 /**
82 * @brief Get the required number of rows of the multi-rhs view passed to solve().
83 *
84 * Implementations may require a number of rows larger than what `size` returns for optimization purposes.
85 *
86 * @return The required number of rows of the multi-rhs view. It is guaranteed to be greater or equal to `size`.
87 */
88 std::size_t required_number_of_rhs_rows() const
89 {
90 std::size_t const nrows = impl_required_number_of_rhs_rows();
91 assert(nrows >= size());
92 return nrows;
93 }
94
95private:
96 virtual std::size_t impl_required_number_of_rhs_rows() const
97 {
98 return m_size;
99 }
100};
101
102/**
103 * @brief Prints the matrix of a SplinesLinearProblem in a std::ostream. It must not be called after `setup_solver`.
104 *
105 * @param[out] os The stream in which the matrix is printed.
106 * @param[in] linear_problem The SplinesLinearProblem of the matrix to print.
107 *
108 * @return The stream in which the matrix is printed.
109**/
110template <class ExecSpace>
111std::ostream& operator<<(std::ostream& os, SplinesLinearProblem<ExecSpace> const& linear_problem)
112{
113 std::size_t const n = linear_problem.size();
114 for (std::size_t i = 0; i < n; ++i) {
115 for (std::size_t j = 0; j < n; ++j) {
116 os << std::fixed << std::setprecision(3) << std::setw(10)
117 << linear_problem.get_element(i, j);
118 }
119 os << std::endl;
120 }
121 return os;
122}
