74 {
75  double const mean_temp = transform_reduce(
76  temp.domain(),
77  0.,
78  reducer::sum<double>(),

80  / temp.domain().size();

99 int main()
100 {
103  // Number of ghost points to use on each side in X
104  DiscreteVector<DDimX> static constexpr gwx {1};
106 
108  // Initialization of the global domain in X with gwx ghost points on
109  // each side
110  auto const [x_domain, ghosted_x_domain, x_pre_ghost, x_post_ghost]
111  = init_discretization<DDimX>(DDimX::init_ghosted(
112  Coordinate<X>(x_start),
113  Coordinate<X>(x_end),
114  DiscreteVector<DDimX>(nb_x_points),
115  gwx));
117 
119  // our zone at the start of the domain that will be mirrored to the
120  // ghost
121  DiscreteDomain const
122  x_domain_begin(x_domain.front(), x_post_ghost.extents());
123  // our zone at the end of the domain that will be mirrored to the
124  // ghost
125  DiscreteDomain const x_domain_end(
126  x_domain.back() - x_pre_ghost.extents() + 1,
127  x_pre_ghost.extents());
129 
131  // Number of ghost points to use on each side in Y
132  DiscreteVector<DDimY> static constexpr gwy {1};
133 
134  // Initialization of the global domain in Y with gwy ghost points on
135  // each side
136  auto const [y_domain, ghosted_y_domain, y_pre_ghost, y_post_ghost]
137  = init_discretization(DDimY::init_ghosted(
138  Coordinate<Y>(y_start),
139  Coordinate<Y>(y_end),
140  DiscreteVector<DDimY>(nb_y_points),
141  gwy));
142 
143  // our zone at the start of the domain that will be mirrored to the
144  // ghost
145  DiscreteDomain const
146  y_domain_begin(y_domain.front(), y_post_ghost.extents());
147  // our zone at the end of the domain that will be mirrored to the
148  // ghost
149  DiscreteDomain const y_domain_end(
150  y_domain.back() - y_pre_ghost.extents() + 1,
151  y_pre_ghost.extents());
153 
155  // max(1/dx^2)
156  double const invdx2_max = transform_reduce(
157  x_domain,
158  0.,
159  reducer::max<double>(),
160  [](DiscreteCoordinate<DDimX> ix) {
161  return 1.
162  / (distance_at_left(ix) * distance_at_right(ix));
163  });
164  // max(1/dy^2)
165  double const invdy2_max = transform_reduce(
166  y_domain,
167  0.,
168  reducer::max<double>(),
169  [](DiscreteCoordinate<DDimY> iy) {
170  return 1.
171  / (distance_at_left(iy) * distance_at_right(iy));
172  });
173  Coordinate<T> const max_dt {
174  .5 / (kx * invdx2_max + ky * invdy2_max)};
175 
176  // number of time intervals required to reach the end time
177  DiscreteVector<DDimT> const nb_time_steps {
178  std::ceil((end_time - start_time) / max_dt) + .2};
179  // Initialization of the global domain in time:
180  // - the number of discrete time-points is equal to the number of
181  // steps + 1
182  DiscreteDomain<DDimT> const time_domain
183  = init_discretization(DDimT::
184  init(Coordinate<T>(start_time),
185  Coordinate<T>(end_time),
186  nb_time_steps + 1));
188 
190  // Maps temperature into the full domain (including ghosts) twice:
191  // - once for the last fully computed time-step
192  Chunk<double, DiscreteDomain<DDimX, DDimY>> ghosted_last_temp(
193  DiscreteDomain<
194  DDimX,
195  DDimY>(ghosted_x_domain, ghosted_y_domain));
196  // - once for time-step being computed
197  Chunk<double, DiscreteDomain<DDimX, DDimY>> ghosted_next_temp(
198  DiscreteDomain<
199  DDimX,
200  DDimY>(ghosted_x_domain, ghosted_y_domain));
202 
204  // Initialize the temperature on the main domain
205  for_each(
206  DiscreteDomain<DDimX, DDimY>(x_domain, y_domain),
207  [&](DiscreteCoordinate<DDimX, DDimY> const ixy) {
208  double const x = to_real(select<DDimX>(ixy));
209  double const y = to_real(select<DDimY>(ixy));
210  ghosted_last_temp(ixy)
211  = 9.999 * ((x * x + y * y) < 0.25);
212  });
214 
216  // display the initial data
217  display(to_real(time_domain.front()),
218  ghosted_last_temp[x_domain][y_domain]);
219  // time of the iteration where the last output happened
220  DiscreteCoordinate<DDimT> last_output = time_domain.front();
222 
224  for (auto const iter :
225  DiscreteDomain {time_domain.front() + 1, nb_time_steps - 1}) {
227 
229  // Periodic boundary conditions
230  deepcopy(
231  ghosted_last_temp[x_pre_ghost][y_domain],
232  ghosted_last_temp[y_domain][x_domain_end]);
233  deepcopy(
234  ghosted_last_temp[y_domain][x_post_ghost],
235  ghosted_last_temp[y_domain][x_domain_begin]);
236  deepcopy(
237  ghosted_last_temp[x_domain][y_pre_ghost],
238  ghosted_last_temp[x_domain][y_domain_end]);
239  deepcopy(
240  ghosted_last_temp[x_domain][y_post_ghost],
241  ghosted_last_temp[x_domain][y_domain_begin]);
243 
245  // a span excluding ghosts of the temperature at the time-step we
246  // will build
247  ChunkSpan const next_temp {
248  ghosted_next_temp[x_domain][y_domain]};
249  // a read-only view of the temperature at the previous time-step
250  ChunkSpan const last_temp {ghosted_last_temp.span_view()};
252 
254  // Stencil computation on the main domain
255  for_each(
256  next_temp.domain(),
257  [&](DiscreteCoordinate<DDimX, DDimY> const ixy) {
258  DiscreteCoordinate<DDimX> const ix
259  = select<DDimX>(ixy);
260  DiscreteCoordinate<DDimY> const iy
261  = select<DDimY>(ixy);
262  double const dx_l = distance_at_left(ix);
263  double const dx_r = distance_at_right(ix);
264  double const dx_m = 0.5 * (dx_l + dx_r);
265  double const dy_l = distance_at_left(iy);
266  double const dy_r = distance_at_right(iy);
267  double const dy_m = 0.5 * (dy_l + dy_r);
268  next_temp(ix, iy) = last_temp(ix, iy);
269  next_temp(ix, iy)
270  += kx * max_dt
271  * (dx_l * last_temp(ix + 1, iy)
272  - 2.0 * dx_m * last_temp(ix, iy)
273  + dx_r * last_temp(ix - 1, iy))
274  / (dx_l * dx_m * dx_r);
275  next_temp(ix, iy)
276  += ky * max_dt
277  * (dy_l * last_temp(ix, iy + 1)
278  - 2.0 * dy_m * last_temp(ix, iy)
279  + dy_r * last_temp(ix, iy - 1))
280  / (dy_l * dy_m * dy_r);
281  });
283 
285  if (iter - last_output >= t_output_period) {
286  last_output = iter;
287  display(to_real(iter), next_temp);
288  }
290 
292  // Swap our two buffers
293  std::swap(ghosted_last_temp, ghosted_next_temp);
295  }
296 
298  if (last_output < time_domain.back()) {
299  display(to_real(time_domain.back()),
300  ghosted_last_temp[x_domain][y_domain]);
301  }
303 }
