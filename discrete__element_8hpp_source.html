98/// Returns a reference towards the DiscreteElement that contains the QueryTag
99template <
100 class QueryTag,
101 class HeadDElem,
102 class... TailDElems,
103 std::enable_if_t<
104 is_discrete_element_v<HeadDElem> && (is_discrete_element_v<TailDElems> && ...),
105 int> = 1>
106KOKKOS_FUNCTION constexpr auto const& take(HeadDElem const& head, TailDElems const&... tail)
107{
108 DDC_IF_NVCC_THEN_PUSH_AND_SUPPRESS(implicit_return_from_non_void_function)
109 if constexpr (type_seq_contains_v<detail::TypeSeq<QueryTag>, to_type_seq_t<HeadDElem>>) {
110 static_assert(
111 (!type_seq_contains_v<detail::TypeSeq<QueryTag>, to_type_seq_t<TailDElems>> && ...),
112 "ERROR: tag redundant");
113 return head;
114 } else {
115 static_assert(sizeof...(TailDElems) > 0, "ERROR: tag not found");
116 return take<QueryTag>(tail...);
117 }
118 DDC_IF_NVCC_THEN_POP
119}

141/** A DiscreteElement identifies an element of the discrete dimension
142 *
143 * Each one is tagged by its associated dimensions.
144 */
145template <class... Tags>
146class DiscreteElement
147{
148 using tags_seq = detail::TypeSeq<Tags...>;
149
150 friend KOKKOS_FUNCTION constexpr std::array<DiscreteElementType, sizeof...(Tags)>& detail::
151 array<Tags...>(DiscreteElement<Tags...>& v) noexcept;
152
153 friend KOKKOS_FUNCTION constexpr std::array<DiscreteElementType, sizeof...(Tags)> const&
154 detail::array<Tags...>(DiscreteElement<Tags...> const& v) noexcept;
155
156private:
157 std::array<DiscreteElementType, sizeof...(Tags)> m_values;
158
159public:
160 using value_type = DiscreteElementType;
161
162 static KOKKOS_FUNCTION constexpr std::size_t size() noexcept
163 {
164 return sizeof...(Tags);
165 }
166
167public:
168 KOKKOS_DEFAULTED_FUNCTION constexpr DiscreteElement() = default;
169
170 KOKKOS_DEFAULTED_FUNCTION constexpr DiscreteElement(DiscreteElement const&) = default;
171
172 KOKKOS_DEFAULTED_FUNCTION constexpr DiscreteElement(DiscreteElement&&) = default;
173
174 template <class... DElems, class = std::enable_if_t<(is_discrete_element_v<DElems> && ...)>>
175 explicit KOKKOS_FUNCTION constexpr DiscreteElement(DElems const&... delems) noexcept
176 : m_values {take<Tags>(delems...).template uid<Tags>()...}
177 {
178 }
179
180 template <
181 class... Params,
182 class = std::enable_if_t<(!is_discrete_element_v<Params> && ...)>,
183 class = std::enable_if_t<(std::is_integral_v<Params> && ...)>,
184 class = std::enable_if_t<sizeof...(Params) == sizeof...(Tags)>>
185 explicit KOKKOS_FUNCTION constexpr DiscreteElement(Params const&... params) noexcept
186 : m_values {static_cast<value_type>(params)...}
187 {
188 }

191

193

211 template <class QueryTag>
212 KOKKOS_FUNCTION constexpr value_type const& uid_or(value_type const& default_value) const&
213 {
214 DDC_IF_NVCC_THEN_PUSH_AND_SUPPRESS(implicit_return_from_non_void_function)
215 if constexpr (in_tags_v<QueryTag, tags_seq>) {
216 return m_values[type_seq_rank_v<QueryTag, tags_seq>];
217 } else {
218 return default_value;
219 }
220 DDC_IF_NVCC_THEN_POP
221 }

228 return m_values[type_seq_rank_v<QueryTag, tags_seq>];
229 }

283 {
284 static_assert(((type_seq_contains_v<detail::TypeSeq<OTags>, tags_seq>)&&...));

294 KOKKOS_FUNCTION constexpr DiscreteElement& operator+=(IntegralType const& rhs)
295 {
296 m_values[0] += rhs;
297 return *this;
298 }

302 {
303 static_assert(((type_seq_contains_v<detail::TypeSeq<OTags>, tags_seq>)&&...));
304 ((m_values[type_seq_rank_v<OTags, tags_seq>] -= rhs.template get<OTags>()), ...);
305 return *this;
306 }

308 template <
309 class IntegralType,
310 std::size_t N = sizeof...(Tags),
311 class = std::enable_if_t<N == 1>,
312 class = std::enable_if_t<std::is_integral_v<IntegralType>>>
313 KOKKOS_FUNCTION constexpr DiscreteElement& operator-=(IntegralType const& rhs)
314 {
315 m_values[0] -= rhs;
316 return *this;
317 }
