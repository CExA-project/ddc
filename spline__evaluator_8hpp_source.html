219 /**
220 * @brief Evaluate 1D spline function (described by its spline coefficients) at a given coordinate.
221 *
222 * The spline coefficients represent a 1D spline function defined on a B-splines (basis splines). They can be obtained via various methods, such as using a SplineBuilder.
223 *
224 * Remark: calling SplineBuilder then SplineEvaluator corresponds to a spline interpolation.
225 *
226 * @param coord_eval The coordinate where the spline is evaluated. Note that only the component along the dimension of interest is used.
227 * @param spline_coef A ChunkSpan storing the 1D spline coefficients.
228 *
229 * @return The value of the spline function at the desired coordinate.
230 */
231 template <class Layout, class... CoordsDims>
232 KOKKOS_FUNCTION double operator()(
233 ddc::Coordinate<CoordsDims...> const& coord_eval,
234 ddc::ChunkSpan<double const, spline_domain_type, Layout, memory_space> const
235 spline_coef) const
236 {
237 return eval(coord_eval, spline_coef);
238 }

240 /**
241 * @brief Evaluate spline function (described by its spline coefficients) on a mesh.
242 *
243 * The spline coefficients represent a spline function defined on a cartesian product of batch_domain and B-splines
244 * (basis splines). They can be obtained via various methods, such as using a SplineBuilder.
245 *
246 * This is not a multidimensional evaluation. This is a batched 1D evaluation. This means that for each slice of coordinates
247 * identified by a batch_domain_type::discrete_element_type, the evaluation is performed with the 1D set of
248 * spline coefficients identified by the same batch_domain_type::discrete_element_type.
249 *
250 * Remark: calling SplineBuilder then SplineEvaluator corresponds to a spline interpolation.
251 *
252 * @param[out] spline_eval The values of the spline function at the desired coordinates. For practical reasons those are
253 * stored in a ChunkSpan defined on a batched_evaluation_domain_type.
254 * @param[in] coords_eval The coordinates where the spline is evaluated. Those are
255 * stored in a ChunkSpan defined on a batched_evaluation_domain_type. Note that the coordinates of the
256 * points represented by this domain are unused and irrelevant (but the points themselves (DiscreteElement) are used to select
257 * the set of 1D spline coefficients retained to perform the evaluation).
258 * @param[in] spline_coef A ChunkSpan storing the spline coefficients.
259 */
260 template <class Layout1, class Layout2, class Layout3, class... CoordsDims>
261 void operator()(
262 ddc::ChunkSpan<double, batched_evaluation_domain_type, Layout1, memory_space> const
263 spline_eval,
264 ddc::ChunkSpan<
265 ddc::Coordinate<CoordsDims...> const,
266 batched_evaluation_domain_type,
267 Layout2,
268 memory_space> const coords_eval,
269 ddc::ChunkSpan<double const, batched_spline_domain_type, Layout3, memory_space> const
270 spline_coef) const
271 {
272 evaluation_domain_type const evaluation_domain(spline_eval.domain());
273 batch_domain_type const batch_domain(spline_eval.domain());
274
275 ddc::parallel_for_each(
276 "ddc_splines_evaluate",
277 exec_space(),
278 batch_domain,
279 KOKKOS_CLASS_LAMBDA(typename batch_domain_type::discrete_element_type const j) {
280 const auto spline_eval_1D = spline_eval[j];
281 const auto coords_eval_1D = coords_eval[j];
282 const auto spline_coef_1D = spline_coef[j];
283 for (auto const i : evaluation_domain) {
284 spline_eval_1D(i) = eval(coords_eval_1D(i), spline_coef_1D);
285 }
286 });
287 }

290 * @brief Differentiate 1D spline function (described by its spline coefficients) at a given coordinate.
291 *
292 * The spline coefficients represent a 1D spline function defined on a B-splines (basis splines). They can be
293 * obtained via various methods, such as using a SplineBuilder.
294 *
295 * @param coord_eval The coordinate where the spline is differentiated. Note that only the component along the dimension of interest is used.
296 * @param spline_coef A ChunkSpan storing the 1D spline coefficients.

300 template <class Layout, class... CoordsDims>
301 KOKKOS_FUNCTION double deriv(
302 ddc::Coordinate<CoordsDims...> const& coord_eval,
303 ddc::ChunkSpan<double const, spline_domain_type, Layout, memory_space> const
304 spline_coef) const
305 {
306 return eval_no_bc<eval_deriv_type>(coord_eval, spline_coef);
307 }

309 /**
310 * @brief Differentiate spline function (described by its spline coefficients) on a mesh.
311 *
312 * The spline coefficients represent a spline function defined on a cartesian product of batch_domain and B-splines
313 * (basis splines). They can be obtained via various methods, such as using a SplineBuilder.
314 *
315 * The derivation is not performed in a multidimensional way (in any sense). This is a batched 1D derivation.
316 * This means that for each slice of coordinates identified by a batch_domain_type::discrete_element_type,
317 * the derivation is performed with the 1D set of spline coefficients identified by the same batch_domain_type::discrete_element_type.
318 *
319 * @param[out] spline_eval The derivatives of the spline function at the desired coordinates. For practical reasons those are
320 * stored in a ChunkSpan defined on a batched_evaluation_domain_type.
321 * @param[in] coords_eval The coordinates where the spline is differentiated. Those are
322 * stored in a ChunkSpan defined on a batched_evaluation_domain_type. Note that the coordinates of the
323 * points represented by this domain are unused and irrelevant (but the points themselves (DiscreteElement) are used to select
324 * the set of 1D spline coefficients retained to perform the evaluation).
325 * @param[in] spline_coef A ChunkSpan storing the spline coefficients.
326 */
327 template <class Layout1, class Layout2, class Layout3, class... CoordsDims>
328 void deriv(
329 ddc::ChunkSpan<double, batched_evaluation_domain_type, Layout1, memory_space> const
330 spline_eval,
331 ddc::ChunkSpan<
332 ddc::Coordinate<CoordsDims...> const,
333 batched_evaluation_domain_type,
334 Layout2,
335 memory_space> const coords_eval,
336 ddc::ChunkSpan<double const, batched_spline_domain_type, Layout3, memory_space> const
337 spline_coef) const
338 {
339 evaluation_domain_type const evaluation_domain(spline_eval.domain());
340 batch_domain_type const batch_domain(spline_eval.domain());
341
342 ddc::parallel_for_each(
343 "ddc_splines_differentiate",
344 exec_space(),
345 batch_domain,
346 KOKKOS_CLASS_LAMBDA(typename batch_domain_type::discrete_element_type const j) {
347 const auto spline_eval_1D = spline_eval[j];
348 const auto coords_eval_1D = coords_eval[j];
349 const auto spline_coef_1D = spline_coef[j];
350 for (auto const i : evaluation_domain) {
351 spline_eval_1D(i)
352 = eval_no_bc<eval_deriv_type>(coords_eval_1D(i), spline_coef_1D);
353 }
354 });
355 }

357 /** @brief Perform batched 1D integrations of a spline function (described by its spline coefficients) along the dimension of interest and store results on a subdomain of batch_domain.
358 *
359 * The spline coefficients represent a spline function defined on a B-splines (basis splines). They can be obtained via the SplineBuilder.
360 *
361 * The integration is not performed in a multidimensional way (in any sense). This is a batched 1D integration.
362 * This means that for each element of integrals, the integration is performed with the 1D set of
363 * spline coefficients identified by the same DiscreteElement.
364 *
365 * @param[out] integrals The integrals of the spline function on the subdomain of batch_domain. For practical reasons those are
366 * stored in a ChunkSpan defined on a batch_domain_type. Note that the coordinates of the
367 * points represented by this domain are unused and irrelevant.
368 * @param[in] spline_coef A ChunkSpan storing the spline coefficients.
369 */
370 template <class Layout1, class Layout2>
371 void integrate(
372 ddc::ChunkSpan<double, batch_domain_type, Layout1, memory_space> const integrals,
373 ddc::ChunkSpan<double const, batched_spline_domain_type, Layout2, memory_space> const
374 spline_coef) const
375 {
376 batch_domain_type const batch_domain(integrals.domain());
377 ddc::Chunk values_alloc(
378 ddc::DiscreteDomain<bsplines_type>(spline_coef.domain()),
379 ddc::KokkosAllocator<double, memory_space>());
380 ddc::ChunkSpan values = values_alloc.span_view();
381 Kokkos::parallel_for(
382 "ddc_splines_integrate_bsplines",
383 Kokkos::RangePolicy<exec_space>(0, 1),
384 KOKKOS_LAMBDA(int) { ddc::discrete_space<bsplines_type>().integrals(values); });
385
386 ddc::parallel_for_each(
387 "ddc_splines_integrate",
388 exec_space(),
389 batch_domain,
390 KOKKOS_LAMBDA(typename batch_domain_type::discrete_element_type const j) {
391 integrals(j) = 0;
392 for (typename spline_domain_type::discrete_element_type const i :
393 values.domain()) {
394 integrals(j) += spline_coef(i, j) * values(i);
395 }
396 });
397 }

399private:
400 template <class Layout, class... CoordsDims>
401 KOKKOS_INLINE_FUNCTION double eval(
402 ddc::Coordinate<CoordsDims...> const& coord_eval,
403 ddc::ChunkSpan<double const, spline_domain_type, Layout, memory_space> const
404 spline_coef) const
405 {
406 ddc::Coordinate<continuous_dimension_type> coord_eval_interest
407 = ddc::select<continuous_dimension_type>(coord_eval);
408 if constexpr (bsplines_type::is_periodic()) {
409 if (coord_eval_interest < ddc::discrete_space<bsplines_type>().rmin()
410 || coord_eval_interest > ddc::discrete_space<bsplines_type>().rmax()) {
411 coord_eval_interest -= Kokkos::floor(
412 (coord_eval_interest
413 - ddc::discrete_space<bsplines_type>().rmin())
414 / ddc::discrete_space<bsplines_type>().length())
415 * ddc::discrete_space<bsplines_type>().length();
416 }
417 } else {
418 if (coord_eval_interest < ddc::discrete_space<bsplines_type>().rmin()) {
419 return m_lower_extrap_rule(coord_eval_interest, spline_coef);
420 }
421 if (coord_eval_interest > ddc::discrete_space<bsplines_type>().rmax()) {
422 return m_upper_extrap_rule(coord_eval_interest, spline_coef);
423 }
424 }
425 return eval_no_bc<eval_type>(coord_eval_interest, spline_coef);
426 }

428 template <class EvalType, class Layout, class... CoordsDims>
429 KOKKOS_INLINE_FUNCTION double eval_no_bc(
430 ddc::Coordinate<CoordsDims...> const& coord_eval,
431 ddc::ChunkSpan<double const, spline_domain_type, Layout, memory_space> const
432 spline_coef) const
433 {
434 static_assert(
435 std::is_same_v<EvalType, eval_type> || std::is_same_v<EvalType, eval_deriv_type>);
436 ddc::DiscreteElement<bsplines_type> jmin;
437 std::array<double, bsplines_type::degree() + 1> vals_ptr;
438 std::experimental::mdspan<
439 double,
440 std::experimental::extents<std::size_t, bsplines_type::degree() + 1>> const
441 vals(vals_ptr.data());
442 ddc::Coordinate<continuous_dimension_type> coord_eval_interest
443 = ddc::select<continuous_dimension_type>(coord_eval);
444 if constexpr (std::is_same_v<EvalType, eval_type>) {
445 jmin = ddc::discrete_space<bsplines_type>().eval_basis(vals, coord_eval_interest);
446 } else if constexpr (std::is_same_v<EvalType, eval_deriv_type>) {
447 jmin = ddc::discrete_space<bsplines_type>().eval_deriv(vals, coord_eval_interest);
448 }
449 double y = 0.0;
450 for (std::size_t i = 0; i < bsplines_type::degree() + 1; ++i) {
451 y += spline_coef(ddc::DiscreteElement<bsplines_type>(jmin + i)) * vals[i];
452 }
453 return y;
454 }
