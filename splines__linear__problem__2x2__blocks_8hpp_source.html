19/**
20 * @brief A 2x2-blocks linear problem dedicated to the computation of a spline approximation,
21 * with all blocks except top-left one being stored in dense format.
22 *
23 * A = | Q | gamma |
24 * | lambda | delta |
25 *
26 * The storage format is dense row-major for top-left, top-right and bottom-left blocks, and determined by
27 * its type for the top-left block.
28 *
29 * This class implements a Schur complement method to perform a block-LU factorization and solve,
30 * calling top-left block and bottom-right block setup_solver() and solve() methods for internal operations.
31 *
32 * @tparam ExecSpace The Kokkos::ExecutionSpace on which operations related to the matrix are supposed to be performed.
33 */
34template <class ExecSpace>
35class SplinesLinearProblem2x2Blocks : public SplinesLinearProblem<ExecSpace>
36{
37public:
38 using typename SplinesLinearProblem<ExecSpace>::MultiRHS;
39 using SplinesLinearProblem<ExecSpace>::size;
40
41protected:
42 std::unique_ptr<SplinesLinearProblem<ExecSpace>> m_top_left_block;
43 Kokkos::DualView<double**, Kokkos::LayoutRight, typename ExecSpace::memory_space>
44 m_top_right_block;
45 Kokkos::DualView<double**, Kokkos::LayoutRight, typename ExecSpace::memory_space>
46 m_bottom_left_block;
47 std::unique_ptr<SplinesLinearProblem<ExecSpace>> m_bottom_right_block;
48
49public:
50 /**
51 * @brief SplinesLinearProblem2x2Blocks constructor.
52 *
53 * @param mat_size The size of one of the dimensions of the square matrix.
54 * @param top_left_block A pointer toward the top-left SplinesLinearProblem. `setup_solver` must not have been called on it.
55 */
56 explicit SplinesLinearProblem2x2Blocks(
57 std::size_t const mat_size,
58 std::unique_ptr<SplinesLinearProblem<ExecSpace>> top_left_block)
59 : SplinesLinearProblem<ExecSpace>(mat_size)
60 , m_top_left_block(std::move(top_left_block))
61 , m_top_right_block(
62 "top_right_block",
63 m_top_left_block->size(),
64 mat_size - m_top_left_block->size())
65 , m_bottom_left_block(
66 "bottom_left_block",
67 mat_size - m_top_left_block->size(),
68 m_top_left_block->size())
69 , m_bottom_right_block(
70 new SplinesLinearProblemDense<ExecSpace>(mat_size - m_top_left_block->size()))
71 {
72 assert(m_top_left_block->size() <= mat_size);
73
74 Kokkos::deep_copy(m_top_right_block.h_view, 0.);
75 Kokkos::deep_copy(m_bottom_left_block.h_view, 0.);
76 }
77
78 double get_element(std::size_t const i, std::size_t const j) const override
79 {
80 assert(i < size());
81 assert(j < size());
82
83 std::size_t const nq = m_top_left_block->size();
84 if (i < nq && j < nq) {
85 return m_top_left_block->get_element(i, j);
86 } else if (i >= nq && j >= nq) {
87 return m_bottom_right_block->get_element(i - nq, j - nq);
88 } else if (j >= nq) {
89 return m_top_right_block.h_view(i, j - nq);
90 } else {
91 return m_bottom_left_block.h_view(i - nq, j);
92 }
93 }
94
95 void set_element(std::size_t const i, std::size_t const j, double const aij) override
96 {
97 assert(i < size());
98 assert(j < size());
99
100 std::size_t const nq = m_top_left_block->size();
101 if (i < nq && j < nq) {
102 m_top_left_block->set_element(i, j, aij);
103 } else if (i >= nq && j >= nq) {
104 m_bottom_right_block->set_element(i - nq, j - nq, aij);
105 } else if (j >= nq) {
106 m_top_right_block.h_view(i, j - nq) = aij;
107 } else {
108 m_bottom_left_block.h_view(i - nq, j) = aij;
109 }
110 }
111
112private:
113 /// @brief Compute the Schur complement delta - lambda*Q^-1*gamma.
114 void compute_schur_complement()
115 {
116 Kokkos::parallel_for(
117 "compute_schur_complement",
118 Kokkos::MDRangePolicy<Kokkos::DefaultHostExecutionSpace, Kokkos::Rank<2>>(
119 {0, 0},
120 {m_bottom_right_block->size(), m_bottom_right_block->size()}),
121 [&](const int i, const int j) {
122 double val = 0.0;
123 for (int l = 0; l < m_top_left_block->size(); ++l) {
124 val += m_bottom_left_block.h_view(i, l) * m_top_right_block.h_view(l, j);
125 }
126 m_bottom_right_block
127 ->set_element(i, j, m_bottom_right_block->get_element(i, j) - val);
128 });
129 }
130
131public:
132 /**
133 * @brief Perform a pre-process operation on the solver. Must be called after filling the matrix.
134 *
135 * Block-LU factorize the matrix A according to the Schur complement method. The block-LU factorization is:
136 *
137 * A = | Q | 0 | | I | Q^-1*gamma |
138 * | lambda | delta - lambda*Q^-1*gamma | | 0 | I |
139 *
140 * So we perform the factorization inplace to store only the relevant blocks in the matrix (while factorizing
141 * the blocks themselves if necessary):
142 *
143 * | Q | Q^-1*gamma |
144 * | lambda | delta - lambda*Q^-1*gamma |
145 */
146 void setup_solver() override
147 {
148 // Setup the top-left solver
149 m_top_left_block->setup_solver();
150
151 // Compute Q^-1*gamma in top-right block
152 m_top_right_block.modify_host();
153 m_top_right_block.sync_device();
154 m_top_left_block->solve(m_top_right_block.d_view);
155 m_top_right_block.modify_device();
156 m_top_right_block.sync_host();
157
158 // Push lambda on device in bottom-left block
159 m_bottom_left_block.modify_host();
160 m_bottom_left_block.sync_device();
161
162 // Compute delta - lambda*Q^-1*gamma in bottom-right block & setup the bottom-right solver
163 compute_schur_complement();
164 m_bottom_right_block->setup_solver();
165 }
166
167 /**
168 * @brief Solve the multiple right-hand sides linear problem Ax=b or its transposed version A^tx=b inplace.
169 *
170 * The solver method is the one known as Schur complement method. It can be summarized as follow,
171 * starting with the pre-computed elements of the matrix:
172 *
173 * | Q | Q^-1*gamma |
174 * | lambda | delta - lambda*Q^-1*gamma |
175 *
176 * For the non-transposed case:
177 * - Solve inplace Q * x'1 = b1 (using the solver internal to Q).
178 * - Compute inplace b'2 = b2 - lambda*x'1.
179 * - Solve inplace (delta - lambda*Q^-1*gamma) * x2 = b'2.
180 * - Compute inplace x1 = x'1 - (delta - lambda*Q^-1*gamma)*x2.
181 *
182 * @param[in, out] b A 2D Kokkos::View storing the multiple right-hand sides of the problem and receiving the corresponding solution.
183 * @param transpose Choose between the direct or transposed version of the linear problem.
184 */
185 void solve(MultiRHS const b, bool const transpose) const override
186 {
187 assert(b.extent(0) == size());
188
189 MultiRHS b1 = Kokkos::
190 subview(b,
191 std::pair<std::size_t, std::size_t>(0, m_top_left_block->size()),
192 Kokkos::ALL);
193 MultiRHS b2 = Kokkos::
194 subview(b,
195 std::pair<std::size_t, std::size_t>(m_top_left_block->size(), b.extent(0)),
196 Kokkos::ALL);
197 if (!transpose) {
198 m_top_left_block->solve(b1);
199 KokkosBlas::gemm(ExecSpace(), "N", "N", -1., m_bottom_left_block.d_view, b1, 1., b2);
200 m_bottom_right_block->solve(b2);
201 KokkosBlas::gemm(ExecSpace(), "N", "N", -1., m_top_right_block.d_view, b2, 1., b1);
202 } else {
203 KokkosBlas::gemm(ExecSpace(), "T", "N", -1., m_top_right_block.d_view, b1, 1., b2);
204 m_bottom_right_block->solve(b2, true);
205 KokkosBlas::gemm(ExecSpace(), "T", "N", -1., m_bottom_left_block.d_view, b2, 1., b1);
206 m_top_left_block->solve(b1, true);
207 }
208 }
209};
