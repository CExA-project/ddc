153 /** Construct a Impl<Kokkos::HostSpace> and associated discrete_domain_type from a segment
154 * \f$[a, b] \subset [a, +\infty[\f$ and a number of points `n`.
155 * Note that there is no guarantee that either the boundaries a or b will be exactly represented in the sampling.
156 * One should expect usual floating point rounding errors.
157 *
158 * @param a coordinate of the first point of the domain
159 * @param b coordinate of the last point of the domain
160 * @param n number of points to map on the segment \f$[a, b]\f$ including a & b
161 * @param n_period the number of steps in a period
162 */
163 template <class DDim>
164 static std::tuple<typename DDim::template Impl<DDim, Kokkos::HostSpace>, DiscreteDomain<DDim>>
165 init(Coordinate<CDim> a,
166 Coordinate<CDim> b,
167 DiscreteVector<DDim> n,
168 DiscreteVector<DDim> n_period)
169 {
170 assert(a < b);
171 assert(n > 1);
172 assert(n_period > 1);
173 typename DDim::template Impl<DDim, Kokkos::HostSpace>
174 disc(a, Coordinate<CDim>((b - a) / (n - 1)), n_period);
175 DiscreteDomain<DDim> domain(disc.front(), n);
176 return std::make_tuple(std::move(disc), std::move(domain));
177 }
178
179 /** Construct a periodic `DiscreteDomain` from a segment \f$[a, b] \subset [a, +\infty[\f$ and a
180 * number of points `n`.
181 * Note that there is no guarantee that either the boundaries a or b will be exactly represented in the sampling.
182 * One should expect usual floating point rounding errors.
183 *
184 * @param a coordinate of the first point of the domain
185 * @param b coordinate of the last point of the domain
186 * @param n the number of points to map the segment \f$[a, b]\f$ including a & b
187 * @param n_period the number of steps in a period
188 * @param n_ghosts_before number of additional "ghost" points before the segment
189 * @param n_ghosts_after number of additional "ghost" points after the segment
190 */
191 template <class DDim>
192 std::tuple<
193 Impl<DDim, Kokkos::HostSpace>,
194 DiscreteDomain<DDim>,
195 DiscreteDomain<DDim>,
196 DiscreteDomain<DDim>,
197 DiscreteDomain<DDim>>
198 init_ghosted(
199 Coordinate<CDim> a,
200 Coordinate<CDim> b,
201 DiscreteVector<DDim> n,
202 DiscreteVector<DDim> n_period,
203 DiscreteVector<DDim> n_ghosts_before,
204 DiscreteVector<DDim> n_ghosts_after)
205 {
206 assert(a < b);
207 assert(n > 1);
208 assert(n_period > 1);
209 Real const discretization_step = (b - a) / (n - 1);
210 Impl<DDim, Kokkos::HostSpace>
211 disc(a - n_ghosts_before.value() * discretization_step,
212 discretization_step,
213 n_period);
214 DiscreteDomain<DDim> ghosted_domain(disc.front(), n + n_ghosts_before + n_ghosts_after);
215 DiscreteDomain<DDim> pre_ghost = ghosted_domain.take_first(n_ghosts_before);
216 DiscreteDomain<DDim> main_domain = ghosted_domain.remove(n_ghosts_before, n_ghosts_after);
217 DiscreteDomain<DDim> post_ghost = ghosted_domain.take_last(n_ghosts_after);
218 return std::make_tuple(
219 std::move(disc),
220 std::move(main_domain),
221 std::move(ghosted_domain),
222 std::move(pre_ghost),
223 std::move(post_ghost));
224 }
225
226 /** Construct a periodic `DiscreteDomain` from a segment \f$[a, b] \subset [a, +\infty[\f$ and a
227 * number of points `n`.
228 * Note that there is no guarantee that either the boundaries a or b will be exactly represented in the sampling.
229 * One should expect usual floating point rounding errors.
230 *
231 * @param a coordinate of the first point of the domain
232 * @param b coordinate of the last point of the domain
233 * @param n the number of points to map the segment \f$[a, b]\f$ including a & b
234 * @param n_period the number of steps in a period
235 * @param n_ghosts number of additional "ghost" points before and after the segment
236 */
237 template <class DDim>
238 std::tuple<
239 Impl<DDim, Kokkos::HostSpace>,
240 DiscreteDomain<DDim>,
241 DiscreteDomain<DDim>,
242 DiscreteDomain<DDim>,
243 DiscreteDomain<DDim>>
244 init_ghosted(
245 Coordinate<CDim> a,
246 Coordinate<CDim> b,
247 DiscreteVector<DDim> n,
248 DiscreteVector<DDim> n_period,
249 DiscreteVector<DDim> n_ghosts)
250 {
251 return init_ghosted(a, b, n, n_period, n_ghosts, n_ghosts);
252 }
