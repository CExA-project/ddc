25/**
26 * @brief A band linear problem dedicated to the computation of a spline approximation.
27 *
28 * The storage format is dense row-major. Lapack is used to perform every matrix and linear solver-related operations.
29 *
30 * Given the linear system A*x=b, we assume that A is a square (n by n)
31 * with ku superdiagonals and kl subdiagonals.
32 * All non-zero elements of A are stored in the rectangular matrix q, using
33 * the format required by DGBTRF (LAPACK): diagonals of A are rows of q.
34 * q has 2*kl rows for the subdiagonals, 1 row for the diagonal, and ku rows
35 * for the superdiagonals. (The kl additional rows are needed for pivoting.)
36 *
37 * @tparam ExecSpace The Kokkos::ExecutionSpace on which operations related to the matrix are supposed to be performed.
38 */
39template <class ExecSpace>
40class SplinesLinearProblemBand : public SplinesLinearProblem<ExecSpace>
41{
42public:
43 using typename SplinesLinearProblem<ExecSpace>::MultiRHS;
44 using SplinesLinearProblem<ExecSpace>::size;
45
46protected:
47 std::size_t m_kl; // no. of subdiagonals
48 std::size_t m_ku; // no. of superdiagonals
49 Kokkos::DualView<double**, Kokkos::LayoutRight, typename ExecSpace::memory_space>
50 m_q; // band matrix representation
51 Kokkos::DualView<int*, typename ExecSpace::memory_space> m_ipiv; // pivot indices
52
53public:
54 /**
55 * @brief SplinesLinearProblemBand constructor.
56 *
57 * @param mat_size The size of one of the dimensions of the square matrix.
58 * @param kl The number of subdiagonals of the matrix.
59 * @param ku The number of superdiagonals of the matrix.
60 */
61 explicit SplinesLinearProblemBand(
62 std::size_t const mat_size,
63 std::size_t const kl,
64 std::size_t const ku)
65 : SplinesLinearProblem<ExecSpace>(mat_size)
66 , m_kl(kl)
67 , m_ku(ku)
68 /*
69 * The matrix itself stored in band format requires a (kl + ku + 1)*mat_size
70 * allocation, but the LU-factorization requires an additional kl*mat_size block
71 */
72 , m_q("q", 2 * kl + ku + 1, mat_size)
73 , m_ipiv("ipiv", mat_size)
74 {
75 assert(m_kl <= mat_size);
76 assert(m_ku <= mat_size);
77
78 Kokkos::deep_copy(m_q.h_view, 0.);
79 }
80
81private:
82 std::size_t band_storage_row_index(std::size_t const i, std::size_t const j) const
83 {
84 return m_kl + m_ku + i - j;
85 }
86
87public:
88 double get_element(std::size_t const i, std::size_t const j) const override
89 {
90 assert(i < size());
91 assert(j < size());
92 /*
93 * The "row index" of the band format storage identify the (sub/super)-diagonal
94 * while the column index is actually the column index of the matrix. Two layouts
95 * are supported by LAPACKE. The m_kl first lines are irrelevant for the storage of
96 * the matrix itself but required for the storage of its LU factorization.
97 */
98 if (i >= std::
99 max(static_cast<std::ptrdiff_t>(0),
100 static_cast<std::ptrdiff_t>(j) - static_cast<std::ptrdiff_t>(m_ku))
101 && i < std::min(size(), j + m_kl + 1)) {
102 return m_q.h_view(band_storage_row_index(i, j), j);
103 } else {
104 return 0.0;
105 }
106 }
107
108 void set_element(std::size_t const i, std::size_t const j, double const aij) override
109 {
110 assert(i < size());
111 assert(j < size());
112 /*
113 * The "row index" of the band format storage identify the (sub/super)-diagonal
114 * while the column index is actually the column index of the matrix. Two layouts
115 * are supported by LAPACKE. The m_kl first lines are irrelevant for the storage of
116 * the matrix itself but required for the storage of its LU factorization.
117 */
118 if (i >= std::
119 max(static_cast<std::ptrdiff_t>(0),
120 static_cast<std::ptrdiff_t>(j) - static_cast<std::ptrdiff_t>(m_ku))
121 && i < std::min(size(), j + m_kl + 1)) {
122 m_q.h_view(band_storage_row_index(i, j), j) = aij;
123 } else {
124 assert(std::fabs(aij) < 1e-20);
125 }
126 }
127
128 /**
129 * @brief Perform a pre-process operation on the solver. Must be called after filling the matrix.
130 *
131 * LU-factorize the matrix A and store the pivots using the LAPACK dgbtrf() implementation.
132 */
133 void setup_solver() override
134 {
135 int const info = LAPACKE_dgbtrf(
136 LAPACK_ROW_MAJOR,
137 size(),
138 size(),
139 m_kl,
140 m_ku,
141 m_q.h_view.data(),
142 m_q.h_view.stride(
143 0), // m_q.h_view.stride(0) if LAPACK_ROW_MAJOR, m_q.h_view.stride(1) if LAPACK_COL_MAJOR
144 m_ipiv.h_view.data());
145 if (info != 0) {
146 throw std::runtime_error(
147 "LAPACKE_dgbtrf failed with error code " + std::to_string(info));
148 }
149
150 // Convert 1-based index to 0-based index
151 for (int i = 0; i < size(); ++i) {
152 m_ipiv.h_view(i) -= 1;
153 }
154
155 // Push on device
156 m_q.modify_host();
157 m_q.sync_device();
158 m_ipiv.modify_host();
159 m_ipiv.sync_device();
160 }
161
162 /**
163 * @brief Solve the multiple right-hand sides linear problem Ax=b or its transposed version A^tx=b inplace.
164 *
165 * The solver method is band gaussian elimination with partial pivoting using the LU-factorized matrix A. The implementation is LAPACK method dgbtrs.
166 *
167 * @param[in, out] b A 2D Kokkos::View storing the multiple right-hand sides of the problem and receiving the corresponding solution.
168 * @param transpose Choose between the direct or transposed version of the linear problem.
169 */
170 void solve(MultiRHS const b, bool const transpose) const override
171 {
172 assert(b.extent(0) == size());
173
174 std::size_t kl_proxy = m_kl;
175 std::size_t ku_proxy = m_ku;
176 auto q_device = m_q.d_view;
177 auto ipiv_device = m_ipiv.d_view;
178 Kokkos::RangePolicy<ExecSpace> policy(0, b.extent(1));
179 if (transpose) {
180 Kokkos::parallel_for(
181 "gbtrs",
182 policy,
183 KOKKOS_LAMBDA(const int i) {
184 auto sub_b = Kokkos::subview(b, Kokkos::ALL, i);
185 KokkosBatched::SerialGbtrs<
186 KokkosBatched::Trans::Transpose,
187 KokkosBatched::Algo::Gbtrs::Unblocked>::
188 invoke(q_device, sub_b, ipiv_device, kl_proxy, ku_proxy);
189 });
190 } else {
191 Kokkos::parallel_for(
192 "gbtrs",
193 policy,
194 KOKKOS_LAMBDA(const int i) {
195 auto sub_b = Kokkos::subview(b, Kokkos::ALL, i);
196 KokkosBatched::SerialGbtrs<
197 KokkosBatched::Trans::NoTranspose,
198 KokkosBatched::Algo::Gbtrs::Unblocked>::
199 invoke(q_device, sub_b, ipiv_device, kl_proxy, ku_proxy);
200 });
201 }
202 }
203};
