145 template <class Layout1, class Layout2, class Layout3, class... CoordsDims>
146 void operator()(
147 ddc::ChunkSpan<double, batched_evaluation_domain_type, Layout1, memory_space> const
148 spline_eval,
149 ddc::ChunkSpan<

162 batch_domain,

164 const auto spline_eval_1D = spline_eval[j];
165 const auto coords_eval_1D = coords_eval[j];

167 for (auto const i : evaluation_domain) {
168 spline_eval_1D(i) = eval(coords_eval_1D(i), spline_coef_1D);

170 });

211 template <class Layout1, class Layout2>
212 void integrate(
213 ddc::ChunkSpan<double, batch_domain_type, Layout1, memory_space> const integrals,
214 ddc::ChunkSpan<double const, batched_spline_domain_type, Layout2, memory_space> const
215 spline_coef) const
216 {
217 batch_domain_type const batch_domain(integrals.domain());
218 ddc::Chunk values_alloc(
219 ddc::DiscreteDomain<bsplines_type>(spline_coef.domain()),
220 ddc::KokkosAllocator<double, memory_space>());
221 ddc::ChunkSpan values = values_alloc.span_view();
222 Kokkos::parallel_for(
223 Kokkos::RangePolicy<exec_space>(0, 1),
224 KOKKOS_LAMBDA(int) { ddc::discrete_space<bsplines_type>().integrals(values); });
225
226 ddc::parallel_for_each(
227 exec_space(),
228 batch_domain,
229 KOKKOS_LAMBDA(typename batch_domain_type::discrete_element_type const j) {
230 integrals(j) = 0;
231 for (typename spline_domain_type::discrete_element_type const i :
232 values.domain()) {
233 integrals(j) += spline_coef(i, j) * values(i);
234 }
235 });
236 }

239 template <class Layout, class... CoordsDims>
240 KOKKOS_INLINE_FUNCTION double eval(
241 ddc::Coordinate<CoordsDims...> const& coord_eval,
242 ddc::ChunkSpan<double const, spline_domain_type, Layout, memory_space> const
243 spline_coef) const
244 {
245 ddc::Coordinate<typename evaluation_mesh_type::continuous_dimension_type>
246 coord_eval_interest
247 = ddc::select<typename evaluation_mesh_type::continuous_dimension_type>(coord_eval);
248 if constexpr (bsplines_type::is_periodic()) {
249 if (coord_eval_interest < ddc::discrete_space<bsplines_type>().rmin()
250 || coord_eval_interest > ddc::discrete_space<bsplines_type>().rmax()) {
251 coord_eval_interest -= Kokkos::floor(
252 (coord_eval_interest
253 - ddc::discrete_space<bsplines_type>().rmin())
254 / ddc::discrete_space<bsplines_type>().length())
255 * ddc::discrete_space<bsplines_type>().length();
256 }
257 } else {
258 if (coord_eval_interest < ddc::discrete_space<bsplines_type>().rmin()) {
259 return m_left_extrap_rule(coord_eval_interest, spline_coef);
260 }
261 if (coord_eval_interest > ddc::discrete_space<bsplines_type>().rmax()) {
262 return m_right_extrap_rule(coord_eval_interest, spline_coef);
263 }
264 }
265 return eval_no_bc<eval_type>(coord_eval_interest, spline_coef);
266 }

268 template <class EvalType, class Layout, class... CoordsDims>
269 KOKKOS_INLINE_FUNCTION double eval_no_bc(
270 ddc::Coordinate<CoordsDims...> const& coord_eval,
271 ddc::ChunkSpan<double const, spline_domain_type, Layout, memory_space> const
272 spline_coef) const
273 {
274 static_assert(
275 std::is_same_v<EvalType, eval_type> || std::is_same_v<EvalType, eval_deriv_type>);
276 ddc::DiscreteElement<bsplines_type> jmin;
277 std::array<double, bsplines_type::degree() + 1> vals_ptr;
278 std::experimental::mdspan<
279 double,
280 std::experimental::extents<std::size_t, bsplines_type::degree() + 1>> const
281 vals(vals_ptr.data());
282 ddc::Coordinate<typename evaluation_mesh_type::continuous_dimension_type>
283 coord_eval_interest
284 = ddc::select<typename evaluation_mesh_type::continuous_dimension_type>(coord_eval);
285 if constexpr (std::is_same_v<EvalType, eval_type>) {
286 jmin = ddc::discrete_space<bsplines_type>().eval_basis(vals, coord_eval_interest);
287 } else if constexpr (std::is_same_v<EvalType, eval_deriv_type>) {
288 jmin = ddc::discrete_space<bsplines_type>().eval_deriv(vals, coord_eval_interest);
289 }
290 double y = 0.0;
291 for (std::size_t i = 0; i < bsplines_type::degree() + 1; ++i) {
292 y += spline_coef(ddc::DiscreteElement<bsplines_type>(jmin + i)) * vals[i];
293 }
294 return y;
295 }
