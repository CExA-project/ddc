83 template <class DDim, class... Args>
84 static inline void init_discrete_space(Args&&... args)
85 {
86  if (detail::g_discrete_space_host<DDim>) {
87  throw std::runtime_error("Discrete space function already initialized.");
88  }
89  detail::g_discrete_space_host<DDim>.emplace(std::forward<Args>(args)...);
90  detail::g_discretization_store->emplace(typeid(DDim).name(), []() {
91  detail::g_discrete_space_host<DDim>.reset();
92  });
93 #if defined(__CUDACC__)
94  using DDimImplDevice = typename DDim::template Impl<Kokkos::CudaSpace>;
95  DDimImplDevice* device_ptr = detail::g_discrete_space_host<DDim>->get_device_ptr();
96  cudaMemcpyToSymbol(
97  detail::g_discrete_space_device<DDimImplDevice>,
98  &device_ptr,
99  sizeof(DDimImplDevice*),
100  0,
101  cudaMemcpyHostToDevice);
102 #endif
103 #if defined(__HIPCC__)
104  using DDimImplDevice = typename DDim::template Impl<Kokkos::Experimental::HIPSpace>;
105  DDimImplDevice* device_ptr = detail::g_discrete_space_host<DDim>->get_device_ptr();
106  hipMemcpyToSymbol(
107  detail::g_discrete_space_device<DDimImplDevice>,
108  &device_ptr,
109  sizeof(DDimImplDevice*),
110  0,
111  hipMemcpyHostToDevice);
112 #endif
113 }

120 template <class DDimImpl, class Arg>
121 static inline Arg init_discrete_space(std::tuple<DDimImpl, Arg>&& a)
122 {
123  using DDim = typename DDimImpl::discrete_dimension_type;
124  init_discrete_space<DDim>(std::move(std::get<0>(a)));
125  return std::get<1>(a);
126 }

133 template <class DDimImpl, class... Args>
134 static inline std::enable_if_t<2 <= sizeof...(Args), std::tuple<Args...>> init_discrete_space(
135  std::tuple<DDimImpl, Args...>&& a)
136 {
137  using DDim = typename DDimImpl::discrete_dimension_type;
138  init_discrete_space<DDim>(std::move(std::get<0>(a)));
139  return detail::extract_after(std::move(a), std::index_sequence_for<Args...>());
140 }
