13/**
14 * @brief A class for creating a 2D spline approximation of a function.
15 *
16 * A class which contains an operator () which can be used to build a 2D spline approximation
17 * of a function. A 2D spline approximation uses a cross-product between two 1D SplineBuilder.
18 *
19 * @see SplineBuilder
20 */
21template <
22 class ExecSpace,
23 class MemorySpace,
24 class BSpline1,
25 class BSpline2,
26 class IDimI1,
27 class IDimI2,
28 ddc::BoundCond BcXmin1,
29 ddc::BoundCond BcXmax1,
30 ddc::BoundCond BcXmin2,
31 ddc::BoundCond BcXmax2,
32 ddc::SplineSolver Solver,
33 class... IDimX>
34class SplineBuilder2D
35{
36public:
37 /// @brief The type of the Kokkos execution space used by this class.
38 using exec_space = ExecSpace;
39
40 /// @brief The type of the Kokkos memory space used by this class.
41 using memory_space = MemorySpace;
42
43 /// @brief The type of the SplineBuilder used by this class to spline-approximate along first dimension.
44 using builder_type1 = ddc::SplineBuilder<
45 ExecSpace,
46 MemorySpace,
47 BSpline1,
48 IDimI1,
49 BcXmin1,
50 BcXmax1,
51 Solver,
52 IDimX...>;
53
54 /// @brief The type of the SplineBuilder used by this class to spline-approximate along second dimension.
55 using builder_type2 = ddc::SplineBuilder<
56 ExecSpace,
57 MemorySpace,
58 BSpline2,
59 IDimI2,
60 BcXmin2,
61 BcXmax2,
62 Solver,
63 std::conditional_t<std::is_same_v<IDimX, IDimI1>, BSpline1, IDimX>...>;
64
65 /// @brief The type of the SplineBuilder used by this class to spline-approximate the second-dimension-derivatives along first dimension.
66 using builder_deriv_type1 = ddc::SplineBuilder<
67 ExecSpace,
68 MemorySpace,
69 BSpline1,
70 IDimI1,
71 BcXmin1,
72 BcXmax1,

118 /**
119 * @brief The type of the batch domain (obtained by removing the dimensions of interest
120 * from the whole domain).
121 *
122 * Example: For batched_interpolation_domain_type = DiscreteDomain<X,Y,Z> and dimensions of interest X and Y,

131 * @brief The type of the whole spline domain (cartesian product of 2D spline domain

137 using batched_spline_domain_type

156 * Example: For batched_interpolation_domain_type = DiscreteDomain<X,Y,Z> and dimensions of interest X and Y,

159 using batched_derivs_domain_type2
160 = ddc::detail::convert_type_seq_to_discrete_domain<ddc::type_seq_replace_t<
161 ddc::detail::TypeSeq<IDimX...>,
162 ddc::detail::TypeSeq<interpolation_mesh_type2>,
163 ddc::detail::TypeSeq<deriv_type2>>>;
164
165 /**
166 * @brief The type of the whole Derivs domain (cartesian product of the 2D Deriv domain

169 * Example: For batched_interpolation_domain_type = DiscreteDomain<X,Y,Z> and dimensions of interest X and Y,
170 * this is DiscreteDomain<Deriv<X>, Deriv<Y>, Z>.
171 */
172 using batched_derivs_domain_type
173 = ddc::detail::convert_type_seq_to_discrete_domain<ddc::type_seq_replace_t<
174 ddc::detail::TypeSeq<IDimX...>,
175 ddc::detail::TypeSeq<interpolation_mesh_type1, interpolation_mesh_type2>,
176 ddc::detail::TypeSeq<deriv_type1, deriv_type2>>>;

184 /**
185 * @brief Build a SplineBuilder2D acting on batched_interpolation_domain.
186 *
187 * @param batched_interpolation_domain The domain on which the interpolation points are defined.
188 *
189 * @param cols_per_chunk A parameter used by the slicer (internal to the solver) to define the size
190 * of a chunk of right-hand-sides of the linear problem to be computed in parallel (chunks are treated
191 * by the linear solver one-after-the-other).
192 * This value is optional. If no value is provided then the default value is chosen by the requested solver.
193 *
194 * @param preconditionner_max_block_size A parameter used by the slicer (internal to the solver) to
195 * define the size of a block used by the Block-Jacobi preconditioner.
196 * This value is optional. If no value is provided then the default value is chosen by the requested solver.
197 *
198 * @see SplinesLinearProblemSparse
199 */
200 explicit SplineBuilder2D(
201 batched_interpolation_domain_type const& batched_interpolation_domain,
202 std::optional<std::size_t> cols_per_chunk = std::nullopt,
203 std::optional<unsigned int> preconditionner_max_block_size = std::nullopt)
204 : m_spline_builder1(
205 batched_interpolation_domain,
206 cols_per_chunk,
207 preconditionner_max_block_size)
208 , m_spline_builder_deriv1(ddc::replace_dim_of<interpolation_mesh_type2, deriv_type2>(
209 m_spline_builder1.batched_interpolation_domain(),
210 ddc::DiscreteDomain<deriv_type2>(
211 ddc::DiscreteElement<deriv_type2>(1),
212 ddc::DiscreteVector<deriv_type2>(bsplines_type2::degree() / 2))))
213 , m_spline_builder2(
214 m_spline_builder1.batched_spline_domain(),
215 cols_per_chunk,
216 preconditionner_max_block_size)
217 {
218 }

309 spline_domain());

312 /**
313 * @brief Compute a 2D spline approximation of a function.
314 *
315 * Use the values of a function (defined on
316 * SplineBuilder2D::batched_interpolation_domain) and the derivatives of the
317 * function at the boundaries (in the case of BoundCond::HERMITE only)
318 * to calculate a 2D spline approximation of this function.
319 *
320 * The spline approximation is stored as a ChunkSpan of coefficients
321 * associated with B-splines.
322 *
323 * @param[out] spline
324 * The coefficients of the spline computed by this SplineBuilder.
325 * @param[in] vals
326 * The values of the function at the interpolation mesh.
327 * @param[in] derivs_min1
328 * The values of the derivatives at the lower boundary in the first dimension.
329 * @param[in] derivs_max1
330 * The values of the derivatives at the upper boundary in the first dimension.
331 * @param[in] derivs_min2
332 * The values of the derivatives at the lower boundary in the second dimension.
333 * @param[in] derivs_max2
334 * The values of the derivatives at the upper boundary in the second dimension.
335 * @param[in] mixed_derivs_min1_min2
336 * The values of the the cross-derivatives at the lower boundary in the first dimension
337 * and the lower boundary in the second dimension.
338 * @param[in] mixed_derivs_max1_min2
339 * The values of the the cross-derivatives at the upper boundary in the first dimension
340 * and the lower boundary in the second dimension.
341 * @param[in] mixed_derivs_min1_max2
342 * The values of the the cross-derivatives at the lower boundary in the first dimension
343 * and the upper boundary in the second dimension.
344 * @param[in] mixed_derivs_max1_max2
345 * The values of the the cross-derivatives at the upper boundary in the first dimension
346 * and the upper boundary in the second dimension.
347 */
348 template <class Layout>
349 void operator()(
350 ddc::ChunkSpan<double, batched_spline_domain_type, Layout, memory_space> spline,
351 ddc::ChunkSpan<double const, batched_interpolation_domain_type, Layout, memory_space>
352 vals,
353 std::optional<ddc::ChunkSpan<
354 double const,
355 batched_derivs_domain_type1,
356 Layout,
357 memory_space>> const derivs_min1
358 = std::nullopt,
359 std::optional<ddc::ChunkSpan<
360 double const,
361 batched_derivs_domain_type1,
362 Layout,
363 memory_space>> const derivs_max1
364 = std::nullopt,
365 std::optional<ddc::ChunkSpan<
366 double const,
367 batched_derivs_domain_type2,
368 Layout,
369 memory_space>> const derivs_min2
370 = std::nullopt,
371 std::optional<ddc::ChunkSpan<
372 double const,
373 batched_derivs_domain_type2,
374 Layout,
375 memory_space>> const derivs_max2
376 = std::nullopt,
377 std::optional<ddc::ChunkSpan<
378 double const,
379 batched_derivs_domain_type,
380 Layout,
381 memory_space>> const mixed_derivs_min1_min2
382 = std::nullopt,
383 std::optional<ddc::ChunkSpan<
384 double const,
385 batched_derivs_domain_type,
386 Layout,
387 memory_space>> const mixed_derivs_max1_min2
388 = std::nullopt,
389 std::optional<ddc::ChunkSpan<
390 double const,
391 batched_derivs_domain_type,
392 Layout,
393 memory_space>> const mixed_derivs_min1_max2
394 = std::nullopt,
395 std::optional<ddc::ChunkSpan<
396 double const,
397 batched_derivs_domain_type,
398 Layout,
399 memory_space>> const mixed_derivs_max1_max2
400 = std::nullopt) const;
401};

431operator()(
432 ddc::ChunkSpan<double, batched_spline_domain_type, Layout, memory_space> spline,
433 ddc::ChunkSpan<double const, batched_interpolation_domain_type, Layout, memory_space> vals,
434 std::optional<ddc::ChunkSpan<
435 double const,
436 batched_derivs_domain_type1,
437 Layout,
438 memory_space>> const derivs_min1,
439 std::optional<ddc::ChunkSpan<
440 double const,
441 batched_derivs_domain_type1,
442 Layout,
443 memory_space>> const derivs_max1,
444 std::optional<ddc::ChunkSpan<
445 double const,
446 batched_derivs_domain_type2,
447 Layout,
448 memory_space>> const derivs_min2,
449 std::optional<ddc::ChunkSpan<
450 double const,
451 batched_derivs_domain_type2,
452 Layout,
453 memory_space>> const derivs_max2,
454 std::optional<ddc::ChunkSpan<
455 double const,
456 batched_derivs_domain_type,
457 Layout,
458 memory_space>> const mixed_derivs_min1_min2,
459 std::optional<ddc::ChunkSpan<
460 double const,
461 batched_derivs_domain_type,
462 Layout,
463 memory_space>> const mixed_derivs_max1_min2,
464 std::optional<ddc::ChunkSpan<
465 double const,
466 batched_derivs_domain_type,
467 Layout,
468 memory_space>> const mixed_derivs_min1_max2,
469 std::optional<ddc::ChunkSpan<
470 double const,
471 batched_derivs_domain_type,
472 Layout,
473 memory_space>> const mixed_derivs_max1_max2) const
474{
475 // TODO: perform computations along dimension 1 on different streams ?
476 // Spline1-approximate derivs_min2 (to spline1_deriv_min)
477 ddc::Chunk spline1_deriv_min_alloc(
478 m_spline_builder_deriv1.batched_spline_domain(),
479 ddc::KokkosAllocator<double, MemorySpace>());
480 auto spline1_deriv_min = spline1_deriv_min_alloc.span_view();
481 auto spline1_deriv_min_opt = std::optional(spline1_deriv_min.span_cview());
482 if constexpr (BcXmin2 == ddc::BoundCond::HERMITE) {
483 m_spline_builder_deriv1(
484 spline1_deriv_min,
485 *derivs_min2,
486 mixed_derivs_min1_min2,
487 mixed_derivs_max1_min2);
488 } else {
489 spline1_deriv_min_opt = std::nullopt;
490 }
491
492 // Spline1-approximate vals (to spline1)
493 ddc::Chunk spline1_alloc(
494 m_spline_builder1.batched_spline_domain(),
495 ddc::KokkosAllocator<double, MemorySpace>());
496 ddc::ChunkSpan spline1 = spline1_alloc.span_view();
497
498 m_spline_builder1(spline1, vals, derivs_min1, derivs_max1);
499
500 // Spline1-approximate derivs_max2 (to spline1_deriv_max)
501 ddc::Chunk spline1_deriv_max_alloc(
502 m_spline_builder_deriv1.batched_spline_domain(),
503 ddc::KokkosAllocator<double, MemorySpace>());
504 auto spline1_deriv_max = spline1_deriv_max_alloc.span_view();
505 auto spline1_deriv_max_opt = std::optional(spline1_deriv_max.span_cview());
506 if constexpr (BcXmax2 == ddc::BoundCond::HERMITE) {
507 m_spline_builder_deriv1(
508 spline1_deriv_max,
509 *derivs_max2,
510 mixed_derivs_min1_max2,
511 mixed_derivs_max1_max2);
512 } else {
513 spline1_deriv_max_opt = std::nullopt;
514 }
515
516 // Spline2-approximate spline1
517 m_spline_builder2(spline, spline1.span_cview(), spline1_deriv_min_opt, spline1_deriv_max_opt);
518}
