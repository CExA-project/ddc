15/** A CoordinateElement the type of the scalar used to represent elements of coordinates in the
16 * continuous space.
17 */
18using CoordinateElement = Real;
19
20/** A Coordinate represents a coordinate in the continuous space
21 *
22 * It is tagged by its dimensions.
23 */
24
25template <class... CDims>
26using Coordinate = detail::TaggedVector<CoordinateElement, CDims...>;
27
28template <class... DDim, std::enable_if_t<(sizeof...(DDim) > 1), int> = 0>
29KOKKOS_FUNCTION Coordinate<typename DDim::continuous_dimension_type...> coordinate(
30 DiscreteElement<DDim...> const& c)
31{
32 return Coordinate<typename DDim::continuous_dimension_type...>(
33 coordinate(DiscreteElement<DDim>(c))...);
34}
35
36// Gives access to the type of the coordinates of a discrete element
37// Example usage : "using Coords = coordinate_of_t<DElem>;"
38template <class T>
39struct coordinate_of
40{
41 static_assert(is_discrete_element_v<T>, "Parameter T must be of type DiscreteElement");
42 using type = decltype(coordinate(std::declval<T>()));
43};
44
45/// Helper type of \ref ddc::coordinate_of
46template <class T>
47using coordinate_of_t = typename coordinate_of<T>::type;
