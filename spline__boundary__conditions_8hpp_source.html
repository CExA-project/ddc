23static inline std::ostream& operator<<(std::ostream& out, ddc::BoundCond const bc)
24{

39constexpr int n_boundary_equations(ddc::BoundCond const bc, std::size_t const degree)
40{
41 if (bc == ddc::BoundCond::PERIODIC) {
42 return 0;
43 } else if (bc == ddc::BoundCond::HERMITE) {
44 return degree / 2;
45 } else if (bc == ddc::BoundCond::GREVILLE) {
46 return 0;
47 } else if (bc == ddc::BoundCond::NATURAL) {
48 return degree / 2;
49 } else {
50 throw std::runtime_error("ddc::BoundCond not handled");
51 }
52}
