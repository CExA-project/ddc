17template <class T>
18static constexpr PDI_inout_t default_access_v
19 = (std::is_lvalue_reference_v<T> && !std::is_const_v<std::remove_reference_t<T>>)
20 ? PDI_INOUT
21 : PDI_OUT;
22
23template <class T>
24static constexpr PDI_inout_t chunk_default_access_v = is_writable_chunk_v<T> ? PDI_INOUT : PDI_OUT;

26class PdiEvent
27{
28 std::string m_event;
29
30 std::vector<std::string> m_names;
31
32 /// a memory buffer where temporary variables are stored until the class is destroyed
33 std::pmr::monotonic_buffer_resource m_metadata;
34
35public:
36 PdiEvent(std::string const& event_name) : m_event(event_name) {}
37
38 /// @{
39 /// API with access argument
40
41 template <
42 PDI_inout_t access,
43 class BorrowedChunk,
44 std::enable_if_t<is_borrowed_chunk_v<BorrowedChunk>, int> = 0>
45 PdiEvent& with(std::string const& name, BorrowedChunk&& data)
46 {
47 static_assert(
48 !(access & PDI_IN) || (chunk_default_access_v<BorrowedChunk> & PDI_IN),
49 "Invalid access for constant data");
50 auto extents = detail::array(data.domain().extents());
51 size_t& rank = *std::pmr::polymorphic_allocator<size_t>(&m_metadata).allocate(1);
52 rank = extents.size();
53 PDI_share((name + "_rank").c_str(), &rank, PDI_OUT);
54 m_names.push_back(name + "_rank");
55 PDI_share(
56 (name + "_extents").c_str(),
57 std::pmr::vector<size_t>(extents.begin(), extents.end(), &m_metadata).data(),
58 PDI_OUT);
59 m_names.push_back(name + "_extents");
60 PDI_share(
61 name.c_str(),
62 const_cast<chunk_value_t<BorrowedChunk>*>(data.data_handle()),
63 access);
64 m_names.push_back(name);
65 return *this;
66 }
67
68 template <
69 PDI_inout_t access,
70 class Arithmetic,
71 std::enable_if_t<std::is_arithmetic_v<std::remove_reference_t<Arithmetic>>, int> = 0>
72 PdiEvent& with(std::string const& name, Arithmetic&& data)
73 {
74 static_assert(
75 !(access & PDI_IN) || (default_access_v<Arithmetic> & PDI_IN),
76 "Invalid access for constant data");
77 using value_type = std::remove_cv_t<std::remove_reference_t<Arithmetic>>;
78 value_type* data_ptr = const_cast<value_type*>(&data);
79 // for read-only data, we share a copy instead of the data itself in case we received a ref on a temporary,
80 if constexpr (!(access & PDI_IN)) {
81 data_ptr = std::pmr::polymorphic_allocator<value_type>(&m_metadata).allocate(1);
82 *data_ptr = data;
83 }
84 PDI_share(name.c_str(), data_ptr, access);
85 m_names.push_back(name);
86 return *this;
87 }
88
89 template <PDI_inout_t access, class T>
90 PdiEvent& and_with(std::string const& name, T&& t)
91 {
92 return with<access>(name, std::forward<T>(t));
93 }
94
95 /// @}
96 /// API with access deduction
97 /// @{
98
99 /// Borrowed chunk overload (Chunk (const)& or ChunkSpan&& or ChunkSpan (const)&)
100 template <class BorrowedChunk, std::enable_if_t<is_borrowed_chunk_v<BorrowedChunk>, int> = 0>
101 PdiEvent& with(std::string const& name, BorrowedChunk&& data)
102 {
103 return with<chunk_default_access_v<BorrowedChunk>>(name, std::forward<BorrowedChunk>(data));
104 }
105
106 /// Arithmetic overload
107 template <
108 class Arithmetic,
109 std::enable_if_t<std::is_arithmetic_v<std::remove_reference_t<Arithmetic>>, int> = 0>
110 PdiEvent& with(std::string const& name, Arithmetic&& data)
111 {
112 return with<default_access_v<Arithmetic>>(name, std::forward<Arithmetic>(data));
113 }
114
115 /// With synonym
116 template <class T>
117 PdiEvent& and_with(std::string const& name, T&& t)
118 {
119 return with(name, std::forward<T>(t));
120 }
121
122 /// @}
123
124 ~PdiEvent()
125 {
126 PDI_event(m_event.c_str());
127 for (std::string const& one_name : m_names) {
128 PDI_reclaim(one_name.c_str());
129 }
130 }
131};
