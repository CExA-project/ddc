251 /**
252 * @brief Build a 2D spline approximation of a function.
253 *
254 * Use the values of a function at known grid points (as specified by
255 * SplineBuilder2D::interpolation_domain_type) and the derivatives of the
256 * function at the boundaries (if necessary for the chosen boundary
257 * conditions) to calculate a 2D spline approximation of a function.
258 *
259 * The spline approximation is stored as a ChunkSpan of coefficients
260 * associated with basis-splines.
261 *
262 * @param[out] spline
263 * The coefficients of the spline calculated by the function.
264 * @param[in] vals
265 * The values of the function at the grid points.
266 * @param[in] derivs_xmin
267 * The values of the derivatives at the lower boundary in the first dimension.
268 * @param[in] derivs_xmax
269 * The values of the derivatives at the upper boundary in the first dimension.
270 * @param[in] derivs_ymin
271 * The values of the derivatives at the lower boundary in the second dimension.
272 * @param[in] derivs_ymax
273 * The values of the derivatives at the upper boundary in the second dimension.
274 * @param[in] mixed_derivs_xmin_ymin
275 * The values of the the cross-derivatives at the lower boundary in the first dimension
276 * and the lower boundary in the second dimension.
277 * @param[in] mixed_derivs_xmax_ymin
278 * The values of the the cross-derivatives at the upper boundary in the first dimension
279 * and the lower boundary in the second dimension.
280 * @param[in] mixed_derivs_xmin_ymax
281 * The values of the the cross-derivatives at the lower boundary in the first dimension
282 * and the upper boundary in the second dimension.
283 * @param[in] mixed_derivs_xmax_ymax
284 * The values of the the cross-derivatives at the upper boundary in the first dimension
285 * and the upper boundary in the second dimension.
286 */
287 template <class Layout>
288 void operator()(
289 ddc::ChunkSpan<double, spline_domain_type, Layout, memory_space> spline,
290 ddc::ChunkSpan<double const, vals_domain_type, Layout, memory_space> vals,
291 std::optional<
292 ddc::ChunkSpan<double const, derivs_domain_type1, Layout, memory_space>> const
293 derivs_min1
294 = std::nullopt,
295 std::optional<
296 ddc::ChunkSpan<double const, derivs_domain_type1, Layout, memory_space>> const
297 derivs_max1
298 = std::nullopt,
299 std::optional<
300 ddc::ChunkSpan<double const, derivs_domain_type2, Layout, memory_space>> const
301 derivs_min2
302 = std::nullopt,
303 std::optional<
304 ddc::ChunkSpan<double const, derivs_domain_type2, Layout, memory_space>> const
305 derivs_max2
306 = std::nullopt,
307 std::optional<
308 ddc::ChunkSpan<double const, derivs_domain_type, Layout, memory_space>> const
309 mixed_derivs_min1_min2
310 = std::nullopt,
311 std::optional<
312 ddc::ChunkSpan<double const, derivs_domain_type, Layout, memory_space>> const
313 mixed_derivs_max1_min2
314 = std::nullopt,
315 std::optional<
316 ddc::ChunkSpan<double const, derivs_domain_type, Layout, memory_space>> const
317 mixed_derivs_min1_max2
318 = std::nullopt,
319 std::optional<
320 ddc::ChunkSpan<double const, derivs_domain_type, Layout, memory_space>> const
321 mixed_derivs_max1_max2
322 = std::nullopt) const;
323};

339template <class Layout>
340void SplineBuilder2D<
341 ExecSpace,
342 MemorySpace,
343 BSpline1,
344 BSpline2,
345 IDimI1,
346 IDimI2,
347 BcXmin1,
348 BcXmax1,
349 BcXmin2,
350 BcXmax2,
351 Solver,
352 IDimX...>::
353operator()(
354 ddc::ChunkSpan<double, spline_domain_type, Layout, memory_space> spline,
355 ddc::ChunkSpan<double const, vals_domain_type, Layout, memory_space> vals,
356 std::optional<ddc::ChunkSpan<double const, derivs_domain_type1, Layout, memory_space>> const
357 derivs_min1,
358 std::optional<ddc::ChunkSpan<double const, derivs_domain_type1, Layout, memory_space>> const
359 derivs_max1,
360 std::optional<ddc::ChunkSpan<double const, derivs_domain_type2, Layout, memory_space>> const
361 derivs_min2,
362 std::optional<ddc::ChunkSpan<double const, derivs_domain_type2, Layout, memory_space>> const
363 derivs_max2,
364 std::optional<ddc::ChunkSpan<double const, derivs_domain_type, Layout, memory_space>> const
365 mixed_derivs_min1_min2,
366 std::optional<ddc::ChunkSpan<double const, derivs_domain_type, Layout, memory_space>> const
367 mixed_derivs_max1_min2,
368 std::optional<ddc::ChunkSpan<double const, derivs_domain_type, Layout, memory_space>> const
369 mixed_derivs_min1_max2,
370 std::optional<ddc::ChunkSpan<double const, derivs_domain_type, Layout, memory_space>> const
371 mixed_derivs_max1_max2) const
372{
373 // TODO: perform computations along dimension 1 on different streams ?
374 // Spline1-transform derivs_min2 (to spline1_deriv_min)
375 ddc::Chunk spline1_deriv_min_alloc(
376 m_spline_builder_deriv1.spline_domain(),
377 ddc::KokkosAllocator<double, MemorySpace>());
378 auto spline1_deriv_min = spline1_deriv_min_alloc.span_view();
379 auto spline1_deriv_min_opt = std::optional(spline1_deriv_min.span_cview());
380 if constexpr (BcXmin1 == ddc::BoundCond::HERMITE) {
381 m_spline_builder_deriv1(
382 spline1_deriv_min,
383 *derivs_min2,
384 mixed_derivs_min1_min2,
385 mixed_derivs_max1_min2);
386 } else {
387 spline1_deriv_min_opt = std::nullopt;
388 }
389
390 // Spline1-transform vals (to spline1)
391 ddc::Chunk spline1_alloc(
392 m_spline_builder1.spline_domain(),
393 ddc::KokkosAllocator<double, MemorySpace>());
394 ddc::ChunkSpan spline1 = spline1_alloc.span_view();
395
396 m_spline_builder1(spline1, vals, derivs_min1, derivs_max1);
397
398 // Spline1-transform derivs_max2 (to spline1_deriv_max)
399 ddc::Chunk spline1_deriv_max_alloc(
400 m_spline_builder_deriv1.spline_domain(),
401 ddc::KokkosAllocator<double, MemorySpace>());
402 auto spline1_deriv_max = spline1_deriv_max_alloc.span_view();
403 auto spline1_deriv_max_opt = std::optional(spline1_deriv_max.span_cview());
404 if constexpr (BcXmax1 == ddc::BoundCond::HERMITE) {
405 m_spline_builder_deriv1(
406 spline1_deriv_max,
407 *derivs_max2,
408 mixed_derivs_min1_max2,
409 mixed_derivs_max1_max2);
410 } else {
411 spline1_deriv_max_opt = std::nullopt;
412 }
413
414 // Spline2-transform spline1
415 m_spline_builder2(spline, spline1.span_cview(), spline1_deriv_min_opt, spline1_deriv_max_opt);
416}
