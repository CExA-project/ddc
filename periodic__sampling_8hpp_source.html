23/** PeriodicSampling models a periodic discretization of the provided continuous dimension

73 /** @brief Construct a `Impl` from a point and a spacing step.
74 *
75 * @param origin the real coordinate of mesh coordinate 0

114 /// @brief Convert a mesh index into a position in `CDim`
115 KOKKOS_FUNCTION continuous_element_type
116 coordinate(discrete_element_type const& icoord) const noexcept
117 {
118 return m_origin
119 + continuous_element_type(
120 static_cast<int>((icoord.uid() + m_n_period / 2 - 1) % m_n_period)
121 - static_cast<int>(m_n_period / 2 - 1))
122 * m_step;
123 }

126 /** Construct a Impl<Kokkos::HostSpace> and associated discrete_domain_type from a segment
127 * \f$[a, b] \subset [a, +\infty[\f$ and a number of points `n`.
128 *
129 * @param a coordinate of the first point of the domain
130 * @param b coordinate of the last point of the domain
131 * @param n number of points to map on the segment \f$[a, b]\f$ including a & b
132 * @param n_period the number of steps in a period
133 */
134 static std::tuple<Impl<Kokkos::HostSpace>, discrete_domain_type> init(
135 continuous_element_type a,
136 continuous_element_type b,
137 discrete_vector_type n,
138 discrete_vector_type n_period)
139 {
140 assert(a < b);
141 assert(n > 1);
142 assert(n_period > 1);
143 Impl<Kokkos::HostSpace> disc(a, continuous_element_type {(b - a) / (n - 1)}, n_period);
144 discrete_domain_type domain {disc.front(), n};
145 return std::make_tuple(std::move(disc), std::move(domain));
146 }
147
148 /** Construct a periodic `DiscreteDomain` from a segment \f$[a, b] \subset [a, +\infty[\f$ and a
149 * number of points `n`.
150 *
151 * @param a coordinate of the first point of the domain
152 * @param b coordinate of the last point of the domain
153 * @param n the number of points to map the segment \f$[a, b]\f$ including a & b
154 * @param n_period the number of steps in a period
155 * @param n_ghosts_before number of additional "ghost" points before the segment
156 * @param n_ghosts_after number of additional "ghost" points after the segment
157 */
158 static std::tuple<
159 Impl<Kokkos::HostSpace>,
160 discrete_domain_type,
161 discrete_domain_type,
162 discrete_domain_type,
163 discrete_domain_type>
164 init_ghosted(
165 continuous_element_type a,
166 continuous_element_type b,
167 discrete_vector_type n,
168 discrete_vector_type n_period,
169 discrete_vector_type n_ghosts_before,
170 discrete_vector_type n_ghosts_after)
171 {
172 using discrete_domain_type = discrete_domain_type;
173 assert(a < b);
174 assert(n > 1);
175 assert(n_period > 1);
176 Real discretization_step {(b - a) / (n - 1)};
177 Impl<Kokkos::HostSpace>
178 disc(a - n_ghosts_before.value() * discretization_step,
179 discretization_step,
180 n_period);
181 discrete_domain_type ghosted_domain
182 = discrete_domain_type(disc.front(), n + n_ghosts_before + n_ghosts_after);
183 discrete_domain_type pre_ghost
184 = discrete_domain_type(ghosted_domain.front(), n_ghosts_before);
185 discrete_domain_type main_domain
186 = discrete_domain_type(ghosted_domain.front() + n_ghosts_before, n);
187 discrete_domain_type post_ghost
188 = discrete_domain_type(main_domain.back() + 1, n_ghosts_after);
189 return std::make_tuple(
190 std::move(disc),
191 std::move(main_domain),
192 std::move(ghosted_domain),
193 std::move(pre_ghost),
194 std::move(post_ghost));
195 }
196
197 /** Construct a periodic `DiscreteDomain` from a segment \f$[a, b] \subset [a, +\infty[\f$ and a
198 * number of points `n`.
199 *
200 * @param a coordinate of the first point of the domain
201 * @param b coordinate of the last point of the domain
202 * @param n the number of points to map the segment \f$[a, b]\f$ including a & b
203 * @param n_period the number of steps in a period
204 * @param n_ghosts number of additional "ghost" points before and after the segment
205 */
206 static std::tuple<
207 Impl<Kokkos::HostSpace>,
208 discrete_domain_type,
209 discrete_domain_type,
210 discrete_domain_type,
211 discrete_domain_type>
212 init_ghosted(
213 continuous_element_type a,
214 continuous_element_type b,
215 discrete_vector_type n,
216 discrete_vector_type n_period,
217 discrete_vector_type n_ghosts)
218 {
219 return init_ghosted(a, b, n, n_period, n_ghosts, n_ghosts);
220 }
221};
