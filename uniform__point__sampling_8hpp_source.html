120 template <class DDim>
121 static std::tuple<typename DDim::template Impl<DDim, Kokkos::HostSpace>, DiscreteDomain<DDim>>
122 init(Coordinate<CDim> a, Coordinate<CDim> b, DiscreteVector<DDim> n)
123 {
124 assert(a < b);
125 assert(n > 1);
126 typename DDim::template Impl<DDim, Kokkos::HostSpace>
127 disc(a, Coordinate<CDim> {(b - a) / (n - 1)});
128 DiscreteDomain<DDim> domain {disc.front(), n};
129 return std::make_tuple(std::move(disc), std::move(domain));
130 }

141 template <class DDim>
142 static std::tuple<
143 typename DDim::template Impl<DDim, Kokkos::HostSpace>,
144 DiscreteDomain<DDim>,
145 DiscreteDomain<DDim>,
146 DiscreteDomain<DDim>,
147 DiscreteDomain<DDim>>
148 init_ghosted(
149 Coordinate<CDim> a,
150 Coordinate<CDim> b,
151 DiscreteVector<DDim> n,
152 DiscreteVector<DDim> n_ghosts_before,
153 DiscreteVector<DDim> n_ghosts_after)
154 {
155 using discrete_domain_type = DiscreteDomain<DDim>;
156 assert(a < b);
157 assert(n > 1);
158 Real discretization_step {(b - a) / (n - 1)};
159 typename DDim::template Impl<DDim, Kokkos::HostSpace>
160 disc(a - n_ghosts_before.value() * discretization_step, discretization_step);
161 discrete_domain_type ghosted_domain
162 = discrete_domain_type(disc.front(), n + n_ghosts_before + n_ghosts_after);
163 discrete_domain_type pre_ghost
164 = discrete_domain_type(ghosted_domain.front(), n_ghosts_before);
165 discrete_domain_type main_domain
166 = discrete_domain_type(ghosted_domain.front() + n_ghosts_before, n);
167 discrete_domain_type post_ghost
168 = discrete_domain_type(main_domain.back() + 1, n_ghosts_after);
169 return std::make_tuple(
170 std::move(disc),
171 std::move(main_domain),
172 std::move(ghosted_domain),
173 std::move(pre_ghost),
174 std::move(post_ghost));
175 }

185 template <class DDim>
186 static std::tuple<
187 typename DDim::template Impl<DDim, Kokkos::HostSpace>,
188 DiscreteDomain<DDim>,
189 DiscreteDomain<DDim>,
190 DiscreteDomain<DDim>,
191 DiscreteDomain<DDim>>
192 init_ghosted(
193 Coordinate<CDim> a,
194 Coordinate<CDim> b,
195 DiscreteVector<DDim> n,
196 DiscreteVector<DDim> n_ghosts)
197 {
198 return init_ghosted(a, b, n, n_ghosts, n_ghosts);
199 }
