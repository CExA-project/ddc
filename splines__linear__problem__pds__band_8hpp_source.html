34template <class ExecSpace>
35class SplinesLinearProblemPDSBand : public SplinesLinearProblem<ExecSpace>
36{
37public:
38 using typename SplinesLinearProblem<ExecSpace>::MultiRHS;
39 using SplinesLinearProblem<ExecSpace>::size;
40
41protected:
42 Kokkos::View<double**, Kokkos::HostSpace> m_q; // pds band matrix representation
43
44public:
51 explicit SplinesLinearProblemPDSBand(std::size_t const mat_size, std::size_t const kd)
52 : SplinesLinearProblem<ExecSpace>(mat_size)
53 , m_q("q", kd + 1, mat_size)
54 {
55 assert(m_q.extent(0) <= mat_size);
56
57 Kokkos::deep_copy(m_q, 0.);
58 }
59
60 virtual double get_element(std::size_t i, std::size_t j) const override
61 {
62 assert(i < size());
63 assert(j < size());
64
65 // Indices are swapped for an element on subdiagonal
66 if (i > j) {
67 std::swap(i, j);
68 }
69 if (j - i < m_q.extent(0)) {
70 return m_q(j - i, i);
71 } else {
72 return 0.0;
73 }
74 }
75
76 virtual void set_element(std::size_t i, std::size_t j, double const aij) override
77 {
78 assert(i < size());
79 assert(j < size());
80
81 // Indices are swapped for an element on subdiagonal
82 if (i > j) {
83 std::swap(i, j);
84 }
85 if (j - i < m_q.extent(0)) {
86 m_q(j - i, i) = aij;
87 } else {
88 assert(std::fabs(aij) < 1e-20);
89 }
90 }
91
97 void setup_solver() override
98 {
99 int const info = LAPACKE_dpbtrf(
100 LAPACK_ROW_MAJOR,
101 'L',
102 size(),
103 m_q.extent(0) - 1,
104 m_q.data(),
105 m_q.stride(
106 0) // m_q.stride(0) if LAPACK_ROW_MAJOR, m_q.stride(1) if LAPACK_COL_MAJOR
107 );
108 if (info != 0) {
109 throw std::runtime_error(
110 "LAPACKE_dpbtrf failed with error code " + std::to_string(info));
111 }
112 }
113
122 void solve(MultiRHS b, bool const) const override
123 {
124 assert(b.extent(0) == size());
125
126 auto b_host = create_mirror_view(Kokkos::DefaultHostExecutionSpace(), b);
127 Kokkos::deep_copy(b_host, b);
128 int const info = LAPACKE_dpbtrs(
129 LAPACK_ROW_MAJOR,
130 'L',
131 b_host.extent(0),
132 m_q.extent(0) - 1,
133 b_host.extent(1),
134 m_q.data(),
135 m_q.stride(
136 0), // m_q.stride(0) if LAPACK_ROW_MAJOR, m_q.stride(1) if LAPACK_COL_MAJOR
137 b_host.data(),
138 b_host.stride(0));
139 if (info != 0) {
140 throw std::runtime_error(
141 "LAPACKE_dpbtrs failed with error code " + std::to_string(info));
142 }
143 Kokkos::deep_copy(b, b_host);
144 }
145};
146
147} // namespace ddc::detail
