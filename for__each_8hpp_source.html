14 template <class RetType, class Element, std::size_t N, class Functor, class... Is>
15 inline void for_each_serial(
16  std::array<Element, N> const& start,
17  std::array<Element, N> const& end,
18  Functor const& f,
19  Is const&... is) noexcept
20 {
21  static constexpr std::size_t I = sizeof...(Is);
22  if constexpr (I == N) {
23  f(RetType(is...));
24  } else {
25  for (Element ii = start[I]; ii <= end[I]; ++ii) {
26  for_each_serial<RetType>(start, end, f, is..., ii);
27  }
28  }
29 }

31 template <class RetType, class Element, std::size_t N, class Functor>
32 inline void for_each_omp(
33  std::array<Element, N> const& start,
34  std::array<Element, N> const& end,
35  Functor&& f) noexcept
36 {
37  Element const ib = start[0];
38  Element const ie = end[0];
39 #pragma omp parallel for default(none) shared(ib, ie, start, end, f)
40  for (Element ii = ib; ii <= ie; ++ii) {
41  if constexpr (N == 1) {
42  f(RetType(ii));
43  } else {
44  detail::for_each_serial<RetType>(start, end, f, ii);
45  }
46  }
47 }
