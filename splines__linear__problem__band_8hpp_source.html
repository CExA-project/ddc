35template <class ExecSpace>
36class SplinesLinearProblemBand : public SplinesLinearProblem<ExecSpace>
37{
38public:
39 using typename SplinesLinearProblem<ExecSpace>::MultiRHS;
40 using SplinesLinearProblem<ExecSpace>::size;
41
42protected:
43 std::size_t m_kl; // no. of subdiagonals
44 std::size_t m_ku; // no. of superdiagonals
45 Kokkos::View<int*, Kokkos::HostSpace> m_ipiv; // pivot indices
46 Kokkos::View<double**, Kokkos::HostSpace> m_q; // band matrix representation
47
48public:
56 explicit SplinesLinearProblemBand(
57 std::size_t const mat_size,
58 std::size_t const kl,
59 std::size_t const ku)
60 : SplinesLinearProblem<ExecSpace>(mat_size)
61 , m_kl(kl)
62 , m_ku(ku)
63 , m_ipiv("ipiv", mat_size)
64 /*
65 * The matrix itself stored in band format requires a (kl + ku + 1)*mat_size
66 * allocation, but the LU-factorization requires an additional kl*mat_size block
67 */
68 , m_q("q", 2 * kl + ku + 1, mat_size)
69 {
70 assert(m_kl <= mat_size);
71 assert(m_ku <= mat_size);
72
73 Kokkos::deep_copy(m_q, 0.);
74 }
75
76private:
77 std::size_t band_storage_row_index(std::size_t const i, std::size_t const j) const
78 {
79 return m_kl + m_ku + i - j;
80 }
81
82public:
83 double get_element(std::size_t const i, std::size_t const j) const override
84 {
85 assert(i < size());
86 assert(j < size());
87 /*
88 * The "row index" of the band format storage identify the (sub/super)-diagonal
89 * while the column index is actually the column index of the matrix. Two layouts
90 * are supported by LAPACKE. The m_kl first lines are irrelevant for the storage of
91 * the matrix itself but required for the storage of its LU factorization.
92 */
93 if (i >= std::
94 max(static_cast<std::ptrdiff_t>(0),
95 static_cast<std::ptrdiff_t>(j) - static_cast<std::ptrdiff_t>(m_ku))
96 && i < std::min(size(), j + m_kl + 1)) {
97 return m_q(band_storage_row_index(i, j), j);
98 } else {
99 return 0.0;
100 }
101 }
102
103 void set_element(std::size_t const i, std::size_t const j, double const aij) override
104 {
105 assert(i < size());
106 assert(j < size());
107 /*
108 * The "row index" of the band format storage identify the (sub/super)-diagonal
109 * while the column index is actually the column index of the matrix. Two layouts
110 * are supported by LAPACKE. The m_kl first lines are irrelevant for the storage of
111 * the matrix itself but required for the storage of its LU factorization.
112 */
113 if (i >= std::
114 max(static_cast<std::ptrdiff_t>(0),
115 static_cast<std::ptrdiff_t>(j) - static_cast<std::ptrdiff_t>(m_ku))
116 && i < std::min(size(), j + m_kl + 1)) {
117 m_q(band_storage_row_index(i, j), j) = aij;
118 } else {
119 assert(std::fabs(aij) < 1e-20);
120 }
121 }
122
128 void setup_solver() override
129 {
130 int const info = LAPACKE_dgbtrf(
131 LAPACK_ROW_MAJOR,
132 size(),
133 size(),
134 m_kl,
135 m_ku,
136 m_q.data(),
137 m_q.stride(
138 0), // m_q.stride(0) if LAPACK_ROW_MAJOR, m_q.stride(1) if LAPACK_COL_MAJOR
139 m_ipiv.data());
140 if (info != 0) {
141 throw std::runtime_error(
142 "LAPACKE_dgbtrf failed with error code " + std::to_string(info));
143 }
144 }
145
154 void solve(MultiRHS b, bool const transpose) const override
155 {
156 assert(b.extent(0) == size());
157
158 auto b_host = create_mirror_view(Kokkos::DefaultHostExecutionSpace(), b);
159 Kokkos::deep_copy(b_host, b);
160 int const info = LAPACKE_dgbtrs(
161 LAPACK_ROW_MAJOR,
162 transpose ? 'T' : 'N',
163 b_host.extent(0),
164 m_kl,
165 m_ku,
166 b_host.extent(1),
167 m_q.data(),
168 m_q.stride(
169 0), // m_q.stride(0) if LAPACK_ROW_MAJOR, m_q.stride(1) if LAPACK_COL_MAJOR
170 m_ipiv.data(),
171 b_host.data(),
172 b_host.stride(0));
173 if (info != 0) {
174 throw std::runtime_error(
175 "LAPACKE_dgbtrs failed with error code " + std::to_string(info));
176 }
177 Kokkos::deep_copy(b, b_host);
178 }
179};
180
