40 template <typename U = BSplines>
41 static auto get_sampling()
42 {
43 if constexpr (U::is_uniform()) {
44 using Sampling = ddc::UniformPointSampling<tag_type>;
45 return std::get<0>(
46 Sampling::
47 init(ddc::discrete_space<BSplines>().rmin(),
48 ddc::discrete_space<BSplines>().rmax(),
49 ddc::DiscreteVector<ddc::UniformPointSampling<tag_type>>(
50 ddc::discrete_space<BSplines>().ncells() + 1)));
51 } else {
52 using Sampling = ddc::NonUniformPointSampling<tag_type>;
53 using SamplingImpl = typename Sampling::template Impl<Kokkos::HostSpace>;
54 std::vector<double> knots(ddc::discrete_space<BSplines>().npoints());
55 for (int i(0); i < ddc::discrete_space<BSplines>().npoints(); ++i) {
56 knots[i] = ddc::discrete_space<BSplines>().get_knot(i);
57 }
58 return SamplingImpl(knots);
59 }
60 }

74 ddc::DiscreteElement<interpolation_mesh_type>(0),
