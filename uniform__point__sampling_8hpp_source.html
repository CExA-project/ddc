149  continuous_element_type a,

151  discrete_vector_type n)

169  static std::tuple<
170  Impl<Kokkos::HostSpace>,
171  discrete_domain_type,
172  discrete_domain_type,
173  discrete_domain_type,
174  discrete_domain_type>
175  init_ghosted(
176  continuous_element_type a,
177  continuous_element_type b,
178  discrete_vector_type n,
179  discrete_vector_type n_ghosts_before,
180  discrete_vector_type n_ghosts_after)
181  {
182  using continuous_element_type = continuous_element_type;
183  using discrete_domain_type = discrete_domain_type;
184  assert(a < b);
185  assert(n > 1);
186  continuous_element_type discretization_step {(b - a) / (n - 1)};
187  Impl<Kokkos::HostSpace>
188  disc(a - n_ghosts_before.value() * discretization_step, discretization_step);
189  discrete_domain_type ghosted_domain
190  = discrete_domain_type(disc.front(), n + n_ghosts_before + n_ghosts_after);
191  discrete_domain_type pre_ghost
192  = discrete_domain_type(ghosted_domain.front(), n_ghosts_before);
193  discrete_domain_type main_domain
194  = discrete_domain_type(ghosted_domain.front() + n_ghosts_before, n);
195  discrete_domain_type post_ghost
196  = discrete_domain_type(main_domain.back() + 1, n_ghosts_after);
197  return std::make_tuple(
198  std::move(disc),
199  std::move(main_domain),
200  std::move(ghosted_domain),
201  std::move(pre_ghost),
202  std::move(post_ghost));
203  }
