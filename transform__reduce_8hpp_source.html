22 template <
23  class... DDims,
24  class T,
25  class BinaryReductionOp,
26  class UnaryTransformOp,
27  class... DCoords>
28 inline T transform_reduce_serial(
29  DiscreteDomain<DDims...> const& domain,
30  T const neutral,
31  BinaryReductionOp const& reduce,
32  UnaryTransformOp const& transform,
33  DCoords const&... dcoords) noexcept
34 {
35  if constexpr (sizeof...(DCoords) == sizeof...(DDims)) {
36  return transform(DiscreteCoordinate<DDims...>(dcoords...));
37  } else {
38  using CurrentDDim = type_seq_element_t<sizeof...(DCoords), detail::TypeSeq<DDims...>>;
39  T result = neutral;
40  for (DiscreteCoordinate<CurrentDDim> const ii : select<CurrentDDim>(domain)) {
41  result = reduce(
42  result,
43  transform_reduce_serial(domain, neutral, reduce, transform, dcoords..., ii));
44  }
45  return result;
46  }
47 }

60 template <class... DDims, class T, class BinaryReductionOp, class UnaryTransformOp>
61 inline T transform_reduce(
62  [[maybe_unused]] omp_policy policy,
63  DiscreteDomain<DDims...> const& domain,
64  T const neutral,
65  BinaryReductionOp const& reduce,
66  UnaryTransformOp const& transform) noexcept
67 {
68  using FirstDDim = type_seq_element_t<0, detail::TypeSeq<DDims...>>;
69  DiscreteDomainIterator<FirstDDim> const it_b = select<FirstDDim>(domain).begin();
70  DiscreteDomainIterator<FirstDDim> const it_e = select<FirstDDim>(domain).end();
71 
72  T global_result = neutral;
73 #pragma omp parallel default(none) \
74  shared(global_result, neutral, it_b, it_e, domain, reduce, transform)
75  {
76  // Each thread has its private result
77  T thread_result = neutral;
78 
79  // Distribute work among threads
80  // Do not use the global result in this region
81 #pragma omp for
82  for (DiscreteDomainIterator<FirstDDim> it = it_b; it != it_e; ++it) {
83  if constexpr (sizeof...(DDims) == 1) {
84  thread_result = reduce(thread_result, transform(*it));
85  } else {
86  thread_result = reduce(
87  thread_result,
88  transform_reduce_serial(domain, neutral, reduce, transform, *it));
89  }
90  }
91 
92  // Reduce thread private results into the global result
93 #pragma omp critical
94  {
95  global_result = reduce(global_result, thread_result);
96  }
97  }
98  return global_result;
99 }
