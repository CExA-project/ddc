143  static std::tuple<Impl<Kokkos::HostSpace>, discrete_domain_type> init(
144  continuous_element_type a,
145  continuous_element_type b,
146  discrete_vector_type n)
147  {
148  assert(a < b);
149  assert(n > 1);
150  Impl<Kokkos::HostSpace> disc(a, continuous_element_type {(b - a) / (n - 1)});
151  discrete_domain_type domain {disc.front(), n};
152  return std::make_tuple(std::move(disc), std::move(domain));

164  static std::tuple<
165  Impl<Kokkos::HostSpace>,
166  discrete_domain_type,
167  discrete_domain_type,
168  discrete_domain_type,
169  discrete_domain_type>
170  init_ghosted(
171  continuous_element_type a,
172  continuous_element_type b,
173  discrete_vector_type n,
174  discrete_vector_type n_ghosts_before,
175  discrete_vector_type n_ghosts_after)
176  {
177  using continuous_element_type = continuous_element_type;
178  using discrete_domain_type = discrete_domain_type;
179  assert(a < b);
180  assert(n > 1);
181  double discretization_step {(b - a) / (n - 1)};
182  Impl<Kokkos::HostSpace>
183  disc(a - n_ghosts_before.value() * discretization_step, discretization_step);
184  discrete_domain_type ghosted_domain
185  = discrete_domain_type(disc.front(), n + n_ghosts_before + n_ghosts_after);
186  discrete_domain_type pre_ghost
187  = discrete_domain_type(ghosted_domain.front(), n_ghosts_before);
188  discrete_domain_type main_domain
189  = discrete_domain_type(ghosted_domain.front() + n_ghosts_before, n);
190  discrete_domain_type post_ghost
191  = discrete_domain_type(main_domain.back() + 1, n_ghosts_after);
192  return std::make_tuple(
193  std::move(disc),
194  std::move(main_domain),
195  std::move(ghosted_domain),
196  std::move(pre_ghost),
197  std::move(post_ghost));
198  }
