15/** Copy the content of a borrowed chunk into another
16 * @param[out] dst the borrowed chunk in which to copy
17 * @param[in] src the borrowed chunk from which to copy
18 * @return dst as a ChunkSpan
19*/
20template <class ChunkDst, class ChunkSrc>
21auto parallel_deepcopy(ChunkDst&& dst, ChunkSrc&& src)
22{
23 static_assert(is_borrowed_chunk_v<ChunkDst>);
24 static_assert(is_borrowed_chunk_v<ChunkSrc>);
25 static_assert(
26 std::is_assignable_v<chunk_reference_t<ChunkDst>, chunk_reference_t<ChunkSrc>>,
27 "Not assignable");
28 assert(dst.domain().extents() == src.domain().extents());
29 Kokkos::deep_copy(dst.allocation_kokkos_view(), src.allocation_kokkos_view());
30 return dst.span_view();
31}
