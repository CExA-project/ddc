87/** A serial reduction over a nD domain
88 * @param[in] domain the range over which to apply the algorithm
89 * @param[in] neutral the neutral element of the reduction operation
90 * @param[in] reduce a binary FunctionObject that will be applied in unspecified order to the
91 * results of transform, the results of other reduce and neutral.
92 * @param[in] transform a unary FunctionObject that will be applied to each element of the input
93 * range. The return type must be acceptable as input to reduce
94 * @param[in] dcoords discrete elements from dimensions already in a loop
95 */
96template <
97 class... DDims,
98 class T,
99 class BinaryReductionOp,
100 class UnaryTransformOp,
101 class... DCoords>
102inline T transform_reduce_serial(
103 DiscreteDomain<DDims...> const& domain,
104 [[maybe_unused]] T const neutral,
105 BinaryReductionOp const& reduce,
106 UnaryTransformOp const& transform,
107 DCoords const&... dcoords) noexcept
108{
109 DDC_IF_NVCC_THEN_PUSH_AND_SUPPRESS(implicit_return_from_non_void_function)
110 if constexpr (sizeof...(DCoords) == sizeof...(DDims)) {
111 return transform(DiscreteElement<DDims...>(dcoords...));
112 } else {
113 using CurrentDDim = type_seq_element_t<sizeof...(DCoords), detail::TypeSeq<DDims...>>;
114 T result = neutral;
115 for (DiscreteElement<CurrentDDim> const ii : select<CurrentDDim>(domain)) {
116 result = reduce(
117 result,
118 transform_reduce_serial(domain, neutral, reduce, transform, dcoords..., ii));
119 }
120 return result;
121 }
122 DDC_IF_NVCC_THEN_POP
123}

126class TransformReducerKokkosLambdaAdapter

132
133 Functor functor;

135public:

178/** A parallel reduction over a nD domain using the default Kokkos execution space
179 * @param[in] domain the range over which to apply the algorithm
180 * @param[in] neutral the neutral element of the reduction operation
181 * @param[in] reduce a binary FunctionObject that will be applied in unspecified order to the
182 * results of transform, the results of other reduce and neutral.
183 * @param[in] transform a unary FunctionObject that will be applied to each element of the input
184 * range. The return type must be acceptable as input to reduce
185 */
186template <class ExecSpace, class T, class BinaryReductionOp, class UnaryTransformOp>
187inline T transform_reduce_kokkos(
188 DiscreteDomain<> const& domain,
189 T neutral,
190 BinaryReductionOp const& reduce,
191 UnaryTransformOp const& transform) noexcept
192{
193 T result = neutral;
194 if constexpr (need_annotated_operator<ExecSpace>()) {
195 Kokkos::parallel_reduce(
196 Kokkos::RangePolicy<ExecSpace, use_annotated_operator>(0, 1),
197 TransformReducerKokkosLambdaAdapter<
198 BinaryReductionOp,
199 UnaryTransformOp>(reduce, transform),
200 ddc_to_kokkos_reducer_t<BinaryReductionOp>(result));
201 } else {
202 Kokkos::parallel_reduce(
203 Kokkos::RangePolicy<ExecSpace>(0, 1),
204 TransformReducerKokkosLambdaAdapter<
205 BinaryReductionOp,
206 UnaryTransformOp>(reduce, transform),
207 ddc_to_kokkos_reducer_t<BinaryReductionOp>(result));
208 }
209 return result;
210}

212/** A parallel reduction over a nD domain using the default Kokkos execution space
213 * @param[in] domain the range over which to apply the algorithm
214 * @param[in] neutral the neutral element of the reduction operation
215 * @param[in] reduce a binary FunctionObject that will be applied in unspecified order to the
216 * results of transform, the results of other reduce and neutral.
217 * @param[in] transform a unary FunctionObject that will be applied to each element of the input
218 * range. The return type must be acceptable as input to reduce
219 */
220template <class ExecSpace, class DDim0, class T, class BinaryReductionOp, class UnaryTransformOp>
221inline T transform_reduce_kokkos(
222 DiscreteDomain<DDim0> const& domain,
223 T neutral,
224 BinaryReductionOp const& reduce,
225 UnaryTransformOp const& transform) noexcept
226{
227 T result = neutral;
228 if constexpr (need_annotated_operator<ExecSpace>()) {
229 Kokkos::parallel_reduce(
230 Kokkos::RangePolicy<
231 ExecSpace,
232 use_annotated_operator>(domain.front().uid(), domain.back().uid() + 1),
233 TransformReducerKokkosLambdaAdapter<
234 BinaryReductionOp,
235 UnaryTransformOp,
236 DDim0>(reduce, transform),
237 ddc_to_kokkos_reducer_t<BinaryReductionOp>(result));
238 } else {
239 Kokkos::parallel_reduce(
240 Kokkos::RangePolicy<ExecSpace>(domain.front().uid(), domain.back().uid() + 1),
241 TransformReducerKokkosLambdaAdapter<
242 BinaryReductionOp,
243 UnaryTransformOp,
244 DDim0>(reduce, transform),
245 ddc_to_kokkos_reducer_t<BinaryReductionOp>(result));
246 }
247 return result;
248}

250/** A parallel reduction over a nD domain using the default Kokkos execution space
251 * @param[in] domain the range over which to apply the algorithm
252 * @param[in] neutral the neutral element of the reduction operation
253 * @param[in] reduce a binary FunctionObject that will be applied in unspecified order to the
254 * results of transform, the results of other reduce and neutral.
255 * @param[in] transform a unary FunctionObject that will be applied to each element of the input
256 * range. The return type must be acceptable as input to reduce
257 */
258template <
259 class ExecSpace,
260 class DDim0,
261 class DDim1,
262 class... DDims,
263 class T,
264 class BinaryReductionOp,
265 class UnaryTransformOp>
266inline T transform_reduce_kokkos(
267 DiscreteDomain<DDim0, DDim1, DDims...> const& domain,
268 T neutral,
269 BinaryReductionOp const& reduce,
270 UnaryTransformOp const& transform) noexcept
271{
272 T result = neutral;
273 Kokkos::Array<std::size_t, 2 + sizeof...(DDims)> const
274 begin {select<DDim0>(domain).front().uid(),
275 select<DDim1>(domain).front().uid(),
276 select<DDims>(domain).front().uid()...};
277 Kokkos::Array<std::size_t, 2 + sizeof...(DDims)> const
278 end {select<DDim0>(domain).back().uid() + 1,
279 select<DDim1>(domain).back().uid() + 1,
280 (select<DDims>(domain).back().uid() + 1)...};
281 if constexpr (need_annotated_operator<ExecSpace>()) {
282 Kokkos::parallel_reduce(
283 Kokkos::MDRangePolicy<
284 ExecSpace,
285 Kokkos::Rank<2 + sizeof...(DDims)>,
286 use_annotated_operator>(begin, end),
287 TransformReducerKokkosLambdaAdapter<
288 BinaryReductionOp,
289 UnaryTransformOp,
290 DDim0,
291 DDim1,
292 DDims...>(reduce, transform),
293 ddc_to_kokkos_reducer_t<BinaryReductionOp>(result));
294 } else {
295 Kokkos::parallel_reduce(
296 Kokkos::MDRangePolicy<ExecSpace, Kokkos::Rank<2 + sizeof...(DDims)>>(begin, end),
297 TransformReducerKokkosLambdaAdapter<
298 BinaryReductionOp,
299 UnaryTransformOp,
300 DDim0,
301 DDim1,
302 DDims...>(reduce, transform),
303 ddc_to_kokkos_reducer_t<BinaryReductionOp>(result));
304 }
305 return result;
306}
