407template <class CDim, std::size_t D>
408template <class DDim, class MemorySpace>
409template <class RandomIt>
410NonUniformBSplines<CDim, D>::Impl<DDim, MemorySpace>::Impl(
411 RandomIt const breaks_begin,
412 RandomIt const breaks_end)
413 : m_knot_domain(
414 ddc::DiscreteElement<knot_discrete_dimension_type>(0),
415 ddc::DiscreteVector<knot_discrete_dimension_type>(
416 (breaks_end - breaks_begin)
417 + 2 * degree())) // Create a mesh of knots including the eventual periodic point
418 , m_break_point_domain(
419 ddc::DiscreteElement<knot_discrete_dimension_type>(degree()),
420 ddc::DiscreteVector<knot_discrete_dimension_type>(
421 (breaks_end - breaks_begin))) // Create a mesh of break points
422 , m_reference(ddc::create_reference_discrete_element<DDim>())
423{
424 std::vector<ddc::Coordinate<CDim>> knots((breaks_end - breaks_begin) + 2 * degree());
425 // Fill the provided knots
426 int ii = 0;
427 for (RandomIt it = breaks_begin; it < breaks_end; ++it) {
428 knots[degree() + ii] = *it;
429 ++ii;
430 }
431 ddc::Coordinate<CDim> const rmin = knots[degree()];
432 ddc::Coordinate<CDim> const rmax = knots[(breaks_end - breaks_begin) + degree() - 1];
433 assert(rmin < rmax);
434
435 // Fill out the extra knots
436 if constexpr (is_periodic()) {
437 double const period = rmax - rmin;
438 for (std::size_t i = 1; i < degree() + 1; ++i) {
439 knots[degree() + -i] = knots[degree() + ncells() - i] - period;
440 knots[degree() + ncells() + i] = knots[degree() + i] + period;
441 }
442 } else // open
443 {
444 for (std::size_t i = 1; i < degree() + 1; ++i) {
445 knots[degree() + -i] = rmin;
446 knots[degree() + npoints() - 1 + i] = rmax;
447 }
448 }
449 ddc::init_discrete_space<knot_discrete_dimension_type>(knots);
450}

454KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDim> NonUniformBSplines<CDim, D>::
455 Impl<DDim, MemorySpace>::eval_basis(DSpan1D values, ddc::Coordinate<CDim> const& x) const
456{
457 KOKKOS_ASSERT(values.size() == D + 1)
458
459 std::array<double, degree()> left;
460 std::array<double, degree()> right;
461
462 KOKKOS_ASSERT(x - rmin() >= -length() * 1e-14)
463 KOKKOS_ASSERT(rmax() - x >= -length() * 1e-14)
464 KOKKOS_ASSERT(values.size() == degree() + 1)
465
466 // 1. Compute cell index 'icell'
467 ddc::DiscreteElement<knot_discrete_dimension_type> const icell = find_cell_start(x);
468
469 KOKKOS_ASSERT(icell >= m_break_point_domain.front())
470 KOKKOS_ASSERT(icell <= m_break_point_domain.back())
471 KOKKOS_ASSERT(ddc::coordinate(icell) - x <= length() * 1e-14)
472 KOKKOS_ASSERT(x - ddc::coordinate(icell + 1) <= length() * 1e-14)
473
474 // 2. Compute values of B-splines with support over cell 'icell'
475 double temp;
476 values[0] = 1.0;
477 for (std::size_t j = 0; j < degree(); ++j) {
478 left[j] = x - ddc::coordinate(icell - j);
479 right[j] = ddc::coordinate(icell + j + 1) - x;
480 double saved = 0.0;
481 for (std::size_t r = 0; r < j + 1; ++r) {
482 temp = values[r] / (right[r] + left[j - r]);
483 values[r] = saved + right[r] * temp;
484 saved = left[j - r] * temp;
485 }
486 values[j + 1] = saved;
487 }
488
489 return get_first_bspline_in_cell(icell);
490}

494KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDim> NonUniformBSplines<CDim, D>::
495 Impl<DDim, MemorySpace>::eval_deriv(DSpan1D derivs, ddc::Coordinate<CDim> const& x) const
496{
497 std::array<double, degree()> left;
498 std::array<double, degree()> right;
499
500 KOKKOS_ASSERT(x - rmin() >= -length() * 1e-14)
501 KOKKOS_ASSERT(rmax() - x >= -length() * 1e-14)
502 KOKKOS_ASSERT(derivs.size() == degree() + 1)
503
504 // 1. Compute cell index 'icell'
505 ddc::DiscreteElement<knot_discrete_dimension_type> const icell = find_cell_start(x);
506
507 KOKKOS_ASSERT(icell >= m_break_point_domain.front())
508 KOKKOS_ASSERT(icell <= m_break_point_domain.back())
509 KOKKOS_ASSERT(ddc::coordinate(icell) <= x)
510 KOKKOS_ASSERT(ddc::coordinate(icell + 1) >= x)
511
512 // 2. Compute values of derivatives of B-splines with support over cell 'icell'
513
514 /*
515 * Compute nonzero basis functions and knot differences
516 * for splines up to degree degree-1 which are needed to compute derivative
517 * First part of Algorithm A3.2 of NURBS book
518 */
519 double saved;
520 double temp;
521 derivs[0] = 1.0;
522 for (std::size_t j = 0; j < degree() - 1; ++j) {
523 left[j] = x - ddc::coordinate(icell - j);
524 right[j] = ddc::coordinate(icell + j + 1) - x;
525 saved = 0.0;
526 for (std::size_t r = 0; r < j + 1; ++r) {
527 temp = derivs[r] / (right[r] + left[j - r]);
528 derivs[r] = saved + right[r] * temp;
529 saved = left[j - r] * temp;
530 }
531 derivs[j + 1] = saved;
532 }
533
534 /*
535 * Compute derivatives at x using values stored in bsdx and formula
536 * for spline derivative based on difference of splines of degree degree-1
537 */
538 saved = degree() * derivs[0]
539 / (ddc::coordinate(icell + 1) - ddc::coordinate(icell + 1 - degree()));
540 derivs[0] = -saved;
541 for (std::size_t j = 1; j < degree(); ++j) {
542 temp = saved;
543 saved = degree() * derivs[j]
544 / (ddc::coordinate(icell + j + 1) - ddc::coordinate(icell + j + 1 - degree()));
545 derivs[j] = temp - saved;
546 }
547 derivs[degree()] = saved;
548
549 return get_first_bspline_in_cell(icell);
550}

552template <class CDim, std::size_t D>
553template <class DDim, class MemorySpace>
554KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDim> NonUniformBSplines<CDim, D>::
555 Impl<DDim, MemorySpace>::eval_basis_and_n_derivs(
556 ddc::DSpan2D const derivs,
557 ddc::Coordinate<CDim> const& x,
558 std::size_t const n) const
559{
560 std::array<double, degree()> left;
561 std::array<double, degree()> right;
562
563 std::array<double, 2 * (degree() + 1)> a_ptr;
564 Kokkos::mdspan<double, Kokkos::extents<std::size_t, degree() + 1, 2>> const a(a_ptr.data());
565
566 std::array<double, (degree() + 1) * (degree() + 1)> ndu_ptr;
567 Kokkos::mdspan<double, Kokkos::extents<std::size_t, degree() + 1, degree() + 1>> const ndu(
568 ndu_ptr.data());
569
570 KOKKOS_ASSERT(x - rmin() >= -length() * 1e-14)
571 KOKKOS_ASSERT(rmax() - x >= -length() * 1e-14)
572 // KOKKOS_ASSERT(n >= 0) as long as n is unsigned
573 KOKKOS_ASSERT(n <= degree())
574 KOKKOS_ASSERT(derivs.extent(0) == 1 + degree())
575 KOKKOS_ASSERT(derivs.extent(1) == 1 + n)
576
577 // 1. Compute cell index 'icell' and x_offset
578 ddc::DiscreteElement<knot_discrete_dimension_type> const icell = find_cell_start(x);
579
580 KOKKOS_ASSERT(icell >= m_break_point_domain.front())
581 KOKKOS_ASSERT(icell <= m_break_point_domain.back())
582 KOKKOS_ASSERT(ddc::coordinate(icell) <= x)
583 KOKKOS_ASSERT(ddc::coordinate(icell + 1) >= x)
584
585 // 2. Compute nonzero basis functions and knot differences for splines
586 // up to degree (degree-1) which are needed to compute derivative
587 // Algorithm A2.3 of NURBS book
588 //
589 // 21.08.2017: save inverse of knot differences to avoid unnecessary
590 // divisions
591 // [Yaman Gl, Edoardo Zoni]
592
593 double saved;
594 double temp;
595 DDC_MDSPAN_ACCESS_OP(ndu, 0, 0) = 1.0;
596 for (std::size_t j = 0; j < degree(); ++j) {
597 left[j] = x - ddc::coordinate(icell - j);
598 right[j] = ddc::coordinate(icell + j + 1) - x;
599 saved = 0.0;
600 for (std::size_t r = 0; r < j + 1; ++r) {
601 // compute inverse of knot differences and save them into lower
602 // triangular part of ndu
603 DDC_MDSPAN_ACCESS_OP(ndu, r, j + 1) = 1.0 / (right[r] + left[j - r]);
604 // compute basis functions and save them into upper triangular part
605 // of ndu
606 temp = DDC_MDSPAN_ACCESS_OP(ndu, j, r) * DDC_MDSPAN_ACCESS_OP(ndu, r, j + 1);
607 DDC_MDSPAN_ACCESS_OP(ndu, j + 1, r) = saved + right[r] * temp;
608 saved = left[j - r] * temp;
609 }
610 DDC_MDSPAN_ACCESS_OP(ndu, j + 1, j + 1) = saved;
611 }
612 // Save 0-th derivative
613 for (std::size_t j = 0; j < degree() + 1; ++j) {
614 DDC_MDSPAN_ACCESS_OP(derivs, j, 0) = DDC_MDSPAN_ACCESS_OP(ndu, degree(), j);
615 }
616
617 for (int r = 0; r < static_cast<int>(degree() + 1); ++r) {
618 int s1 = 0;
619 int s2 = 1;
620 DDC_MDSPAN_ACCESS_OP(a, 0, 0) = 1.0;
621 for (int k = 1; k < static_cast<int>(n + 1); ++k) {
622 double d = 0.0;
623 int const rk = r - k;
624 int const pk = degree() - k;
625 if (r >= k) {
626 DDC_MDSPAN_ACCESS_OP(a, 0, s2)
627 = DDC_MDSPAN_ACCESS_OP(a, 0, s1) * DDC_MDSPAN_ACCESS_OP(ndu, rk, pk + 1);
628 d = DDC_MDSPAN_ACCESS_OP(a, 0, s2) * DDC_MDSPAN_ACCESS_OP(ndu, pk, rk);
629 }
630 int const j1 = rk > -1 ? 1 : (-rk);
631 int const j2 = (r - 1) <= pk ? k : (degree() - r + 1);
632 for (int j = j1; j < j2; ++j) {
633 DDC_MDSPAN_ACCESS_OP(a, j, s2)
634 = (DDC_MDSPAN_ACCESS_OP(a, j, s1) - DDC_MDSPAN_ACCESS_OP(a, j - 1, s1))
635 * DDC_MDSPAN_ACCESS_OP(ndu, rk + j, pk + 1);
636 d += DDC_MDSPAN_ACCESS_OP(a, j, s2) * DDC_MDSPAN_ACCESS_OP(ndu, pk, rk + j);
637 }
638 if (r <= pk) {
639 DDC_MDSPAN_ACCESS_OP(a, k, s2) = -DDC_MDSPAN_ACCESS_OP(a, k - 1, s1)
640 * DDC_MDSPAN_ACCESS_OP(ndu, r, pk + 1);
641 d += DDC_MDSPAN_ACCESS_OP(a, k, s2) * DDC_MDSPAN_ACCESS_OP(ndu, pk, r);
642 }
643 DDC_MDSPAN_ACCESS_OP(derivs, r, k) = d;
644 Kokkos::kokkos_swap(s1, s2);
645 }
646 }
647
648 int r = degree();
649 for (int k = 1; k < static_cast<int>(n + 1); ++k) {
650 for (std::size_t i = 0; i < derivs.extent(0); ++i) {
651 DDC_MDSPAN_ACCESS_OP(derivs, i, k) *= r;
652 }
653 r *= degree() - k;
654 }
655
656 return get_first_bspline_in_cell(icell);
657}

659template <class CDim, std::size_t D>
660template <class DDim, class MemorySpace>
661KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<NonUniformBsplinesKnots<DDim>> NonUniformBSplines<
662 CDim,
663 D>::Impl<DDim, MemorySpace>::find_cell_start(ddc::Coordinate<CDim> const& x) const
664{
665 KOKKOS_ASSERT(x - rmin() >= -length() * 1e-14)
666 KOKKOS_ASSERT(rmax() - x >= -length() * 1e-14)
667
668 if (x <= rmin()) {
669 return m_break_point_domain.front();
670 }
671 if (x >= rmax()) {
672 return m_break_point_domain.back() - 1;
673 }
674
675 // Binary search
676 ddc::DiscreteElement<knot_discrete_dimension_type> low = m_break_point_domain.front();
677 ddc::DiscreteElement<knot_discrete_dimension_type> high = m_break_point_domain.back();
678 ddc::DiscreteElement<knot_discrete_dimension_type> icell = low + (high - low) / 2;
679 while (x < ddc::coordinate(icell) || x >= ddc::coordinate(icell + 1)) {
680 if (x < ddc::coordinate(icell)) {
681 high = icell;
682 } else {
683 low = icell;
684 }
685 icell = low + (high - low) / 2;
686 }
687 return icell;
688}
