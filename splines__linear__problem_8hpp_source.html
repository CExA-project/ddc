17/**
18 * @brief The parent class for linear problems dedicated to the computation of spline approximations.
19 *
20 * Store a square matrix and provide method to solve a multiple right-hand sides linear problem.
21 * Implementations may have different storage formats, filling methods and multiple right-hand sides linear solvers.
22 */
23template <class ExecSpace>
24class SplinesLinearProblem
25{
26public:
27 /// @brief The type of a Kokkos::View storing multiple right-hand sides.
28 using MultiRHS = Kokkos::View<double**, Kokkos::LayoutRight, ExecSpace>;
29
30private:
31 std::size_t m_size;
32
33protected:
34 explicit SplinesLinearProblem(const std::size_t size) : m_size(size) {}
35
36public:
37 /// @brief Destruct
38 virtual ~SplinesLinearProblem() = default;
39
40 /**
41 * @brief Get an element of the matrix at indexes i, j. It must not be called after `setup_solver`.
42 *
43 * @param i The row index of the desired element.
44 * @param j The column index of the desired element.
45 *
46 * @return The value of the element of the matrix.
47 */
48 virtual double get_element(std::size_t i, std::size_t j) const = 0;
49
50 /**
51 * @brief Set an element of the matrix at indexes i, j. It must not be called after `setup_solver`.
52 *
53 * @param i The row index of the setted element.
54 * @param j The column index of the setted element.
55 * @param aij The value to set in the element of the matrix.
56 */
57 virtual void set_element(std::size_t i, std::size_t j, double aij) = 0;
58
59 /**
60 * @brief Perform a pre-process operation on the solver. Must be called after filling the matrix.
61 */
62 virtual void setup_solver() = 0;
63
64 /**
65 * @brief Solve the multiple right-hand sides linear problem Ax=b or its transposed version A^tx=b inplace.
66 *
67 * @param[in, out] multi_rhs A 2D Kokkos::View storing the multiple right-hand sides of the problem and receiving the corresponding solution.
68 * @param transpose Choose between the direct or transposed version of the linear problem.
69 */
70 virtual void solve(MultiRHS b, bool transpose = false) const = 0;
71
72 /**
73 * @brief Get the size of the square matrix in one of its dimensions.
74 *
75 * @return The size of the matrix in one of its dimensions.
76 */
77 std::size_t size() const
78 {
79 return m_size;
80 }
81
82 /**
83 * @brief Get the required number of rows of the multi-rhs view passed to solve().
84 *
85 * Implementations may require a number of rows larger than what `size` returns for optimization purposes.
86 *
87 * @return The required number of rows of the multi-rhs view. It is guaranteed to be greater or equal to `size`.
88 */
89 std::size_t required_number_of_rhs_rows() const
90 {
91 std::size_t const nrows = impl_required_number_of_rhs_rows();
92 assert(nrows >= size());
93 return nrows;
94 }
95
96private:
97 virtual std::size_t impl_required_number_of_rhs_rows() const
98 {
99 return m_size;
100 }
101};
102
103/**
104 * @brief Prints the matrix of a SplinesLinearProblem in a std::ostream. It must not be called after `setup_solver`.
105 *
106 * @param[out] os The stream in which the matrix is printed.
107 * @param[in] linear_problem The SplinesLinearProblem of the matrix to print.
108 *
109 * @return The stream in which the matrix is printed.
110**/
111template <class ExecSpace>
112std::ostream& operator<<(std::ostream& os, SplinesLinearProblem<ExecSpace> const& linear_problem)
113{
114 std::size_t const n = linear_problem.size();
115 for (std::size_t i = 0; i < n; ++i) {
116 for (std::size_t j = 0; j < n; ++j) {
117 os << std::fixed << std::setprecision(3) << std::setw(10)
118 << linear_problem.get_element(i, j);
119 }
120 os << std::endl;
121 }
122 return os;
123}
