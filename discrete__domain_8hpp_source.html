150 template <class... ODDims>
151 constexpr auto restrict(DiscreteDomain<ODDims...> const& odomain) const
152 {
153 assert(((uid<ODDims>(m_element_begin) <= uid<ODDims>(odomain.m_element_begin)) && ...));
154 assert(((uid<ODDims>(m_element_end) >= uid<ODDims>(odomain.m_element_end)) && ...));
155 const DiscreteVector<DDims...> myextents = extents();
156 const DiscreteVector<ODDims...> oextents = odomain.extents();
157 return DiscreteDomain(
158 DiscreteElement<DDims...>(
159 (uid_or<DDims>(odomain.m_element_begin, uid<DDims>(m_element_begin)))...),
160 DiscreteVector<DDims...>((get_or<DDims>(oextents, get<DDims>(myextents)))...));
161 }

368// Computes the cartesian product of DiscreteDomain types
369// Example usage : "using DDom = cartesian_prod_t<DDom1,DDom2,DDom3>;"
370template <typename... DDom>
371struct cartesian_prod;

373template <typename... DDim1, typename... DDim2>
374struct cartesian_prod<ddc::DiscreteDomain<DDim1...>, ddc::DiscreteDomain<DDim2...>>
375{
376 using type = ddc::DiscreteDomain<DDim1..., DDim2...>;
377};

380struct cartesian_prod<DDom>
381{
382 using type = DDom;

386struct cartesian_prod<DDom1, DDom2, Tail...>
387{
388 using type =

390};

395// Computes the substraction DDom_a - DDom_b in the sense of linear spaces(retained dimensions are those in DDom_a which are not in DDom_b)
396template <class... DDimsA, class... DDimsB>
397constexpr auto remove_dims_of(
398 DiscreteDomain<DDimsA...> const& DDom_a,
399 [[maybe_unused]] DiscreteDomain<DDimsB...> const& DDom_b) noexcept
400{
401 using TagSeqA = detail::TypeSeq<DDimsA...>;
402 using TagSeqB = detail::TypeSeq<DDimsB...>;
403
404 using type_seq_r = type_seq_remove_t<TagSeqA, TagSeqB>;
405 return detail::convert_type_seq_to_discrete_domain<type_seq_r>(DDom_a);
406}

449template <class DDim>
450struct DiscreteDomainIterator
451{
452private:
453 DiscreteElement<DDim> m_value = DiscreteElement<DDim>();

455public:

461

463
464 constexpr explicit DiscreteDomainIterator(DiscreteElement<DDim> value) : m_value(value) {}
465
466 constexpr DiscreteElement<DDim> operator*() const noexcept
467 {
468 return m_value;
469 }
470
471 constexpr DiscreteDomainIterator& operator++()
472 {
473 ++m_value.uid();
474 return *this;
475 }
476
477 constexpr DiscreteDomainIterator operator++(int)
478 {
479 auto tmp = *this;
480 ++*this;
481 return tmp;
482 }
483
484 constexpr DiscreteDomainIterator& operator--()
485 {
486 --m_value.uid();
487 return *this;
488 }
489
490 constexpr DiscreteDomainIterator operator--(int)
491 {
492 auto tmp = *this;
493 --*this;
494 return tmp;
495 }
496
497 constexpr DiscreteDomainIterator& operator+=(difference_type n)
498 {
499 if (n >= difference_type(0))
500 m_value.uid() += static_cast<DiscreteElementType>(n);
501 else
502 m_value.uid() -= static_cast<DiscreteElementType>(-n);
503 return *this;
504 }
505
506 constexpr DiscreteDomainIterator& operator-=(difference_type n)
507 {
508 if (n >= difference_type(0))
509 m_value.uid() -= static_cast<DiscreteElementType>(n);
510 else
511 m_value.uid() += static_cast<DiscreteElementType>(-n);
512 return *this;
513 }
514
515 constexpr DiscreteElement<DDim> operator[](difference_type n) const
516 {
517 return m_value + n;
518 }
