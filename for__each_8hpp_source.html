61template <class ExecSpace, class Functor>
62inline void for_each_kokkos(
63 [[maybe_unused]] DiscreteDomain<> const& domain,
64 Functor const& f) noexcept
65{
66 if constexpr (need_annotated_operator<ExecSpace>()) {
67 Kokkos::parallel_for(
68 Kokkos::RangePolicy<ExecSpace, use_annotated_operator>(0, 1),
69 ForEachKokkosLambdaAdapter<Functor>(f));
70 } else {
71 Kokkos::parallel_for(
72 Kokkos::RangePolicy<ExecSpace>(0, 1),
73 ForEachKokkosLambdaAdapter<Functor>(f));
74 }
75}

77template <class ExecSpace, class Functor, class DDim0>
78inline void for_each_kokkos(DiscreteDomain<DDim0> const& domain, Functor const& f) noexcept
79{
80 DiscreteElement<DDim0> const ddc_begin = domain.front();
81 DiscreteElement<DDim0> const ddc_end = domain.front() + domain.extents();
82 std::size_t const begin = ddc::uid<DDim0>(ddc_begin);
83 std::size_t const end = ddc::uid<DDim0>(ddc_end);
84 if constexpr (need_annotated_operator<ExecSpace>()) {
85 Kokkos::parallel_for(
86 Kokkos::RangePolicy<ExecSpace, use_annotated_operator>(begin, end),
87 ForEachKokkosLambdaAdapter<Functor, DDim0>(f));
88 } else {
89 Kokkos::parallel_for(
90 Kokkos::RangePolicy<ExecSpace>(begin, end),
91 ForEachKokkosLambdaAdapter<Functor, DDim0>(f));
92 }
93}

95template <class ExecSpace, class Functor, class DDim0, class DDim1, class... DDims>
96inline void for_each_kokkos(
97 DiscreteDomain<DDim0, DDim1, DDims...> const& domain,
98 Functor&& f) noexcept
99{
100 DiscreteElement<DDim0, DDim1, DDims...> const ddc_begin = domain.front();
101 DiscreteElement<DDim0, DDim1, DDims...> const ddc_end = domain.front() + domain.extents();
102 Kokkos::Array<std::size_t, 2 + sizeof...(DDims)> const
103 begin {ddc::uid<DDim0>(ddc_begin),
104 ddc::uid<DDim1>(ddc_begin),
105 ddc::uid<DDims>(ddc_begin)...};
106 Kokkos::Array<std::size_t, 2 + sizeof...(DDims)> const
107 end {ddc::uid<DDim0>(ddc_end), ddc::uid<DDim1>(ddc_end), ddc::uid<DDims>(ddc_end)...};
108 if constexpr (need_annotated_operator<ExecSpace>()) {
109 Kokkos::parallel_for(
110 Kokkos::MDRangePolicy<
111 ExecSpace,
112 Kokkos::Rank<
113 2 + sizeof...(DDims),
114 Kokkos::Iterate::Right,
115 Kokkos::Iterate::Right>,
116 use_annotated_operator>(begin, end),
117 ForEachKokkosLambdaAdapter<Functor, DDim0, DDim1, DDims...>(f));
118 } else {
119 Kokkos::parallel_for(
120 Kokkos::MDRangePolicy<
121 ExecSpace,
122 Kokkos::Rank<
123 2 + sizeof...(DDims),
124 Kokkos::Iterate::Right,
125 Kokkos::Iterate::Right>>(begin, end),
126 ForEachKokkosLambdaAdapter<Functor, DDim0, DDim1, DDims...>(f));
127 }
128}

130template <class RetType, class Element, std::size_t N, class Functor, class... Is>
131inline void for_each_serial(
132 std::array<Element, N> const& begin,
133 std::array<Element, N> const& end,
134 Functor const& f,
135 Is const&... is) noexcept
136{
137 static constexpr std::size_t I = sizeof...(Is);
138 if constexpr (I == N) {
139 f(RetType(is...));
140 } else {
141 for (Element ii = begin[I]; ii < end[I]; ++ii) {
142 for_each_serial<RetType>(begin, end, f, is..., ii);
143 }
144 }
145}

154/** iterates over a nD domain using the serial execution policy
155 * @param[in] domain the domain over which to iterate
156 * @param[in] f a functor taking an index as parameter
157 */
158template <class... DDims, class Functor>
159inline void for_each(
160 serial_host_policy,
161 DiscreteDomain<DDims...> const& domain,
162 Functor&& f) noexcept
163{
164 DiscreteElement<DDims...> const ddc_begin = domain.front();
165 DiscreteElement<DDims...> const ddc_end = domain.front() + domain.extents();
166 std::array const begin = detail::array(ddc_begin);
167 std::array const end = detail::array(ddc_end);
168 detail::for_each_serial<DiscreteElement<DDims...>>(begin, end, std::forward<Functor>(f));
169}

232template <typename ExecSpace>
233constexpr auto policy([[maybe_unused]] ExecSpace exec_space)
234{
235 if constexpr (std::is_same_v<ExecSpace, Kokkos::Serial>) {
236 return ddc::policies::serial_host;
237#ifdef KOKKOS_ENABLE_OPENMP
238 } else if constexpr (std::is_same_v<ExecSpace, Kokkos::OpenMP>) {
239 return ddc::policies::parallel_host;
240#endif
241#if defined(KOKKOS_ENABLE_CUDA) || defined(KOKKOS_ENABLE_HIP)
242 } else {
243 return ddc::policies::parallel_device;
244#endif
245 }
