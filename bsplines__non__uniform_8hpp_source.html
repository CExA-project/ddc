211template <class Tag, std::size_t D>
212template <class DDim, class MemorySpace>
213template <class RandomIt>
214NonUniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::Impl(
215 RandomIt const break_begin,
216 RandomIt const break_end)
217 : m_domain(
218 ddc::DiscreteElement<mesh_type>(0),
219 ddc::DiscreteVector<mesh_type>(
220 (break_end - break_begin)
221 + 2 * degree())) // Create a mesh including the eventual periodic point
222 , m_nknots((break_end - break_begin) + 2 * degree())
223{
224 std::vector<ddc::Coordinate<Tag>> knots((break_end - break_begin) + 2 * degree());
225 // Fill the provided knots
226 int ii = 0;
227 for (RandomIt it = break_begin; it < break_end; ++it) {
228 knots[degree() + ii] = *it;
229 ++ii;
230 }
231 ddc::Coordinate<Tag> const rmin = knots[degree()];
232 ddc::Coordinate<Tag> const rmax = knots[(break_end - break_begin) + degree() - 1];
233 assert(rmin < rmax);
234
235 // Fill out the extra knots
236 if constexpr (is_periodic()) {
237 double const period = rmax - rmin;
238 for (std::size_t i = 1; i < degree() + 1; ++i) {
239 knots[degree() + -i] = knots[degree() + ncells() - i] - period;
240 knots[degree() + ncells() + i] = knots[degree() + i] + period;
241 }
242 } else // open
243 {
244 for (std::size_t i = 1; i < degree() + 1; ++i) {
245 knots[degree() + -i] = rmin;
246 knots[degree() + npoints() - 1 + i] = rmax;
247 }
248 }
249 ddc::init_discrete_space<mesh_type>(knots);
250}

252template <class Tag, std::size_t D>
253template <class DDim, class MemorySpace>
254KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDim> NonUniformBSplines<Tag, D>::
255 Impl<DDim, MemorySpace>::eval_basis(DSpan1D values, ddc::Coordinate<Tag> const& x) const
256{
257 assert(values.size() == D + 1);
258
259 std::array<double, degree()> left;
260 std::array<double, degree()> right;
261
262 assert(x >= rmin());
263 assert(x <= rmax());
264 assert(values.size() == degree() + 1);
265
266 // 1. Compute cell index 'icell'
267 int const icell = find_cell(x);
268
269 assert(icell >= 0);
270 assert(icell <= int(ncells() - 1));
271 assert(get_knot(icell) <= x);
272 assert(get_knot(icell + 1) >= x);
273
274 // 2. Compute values of B-splines with support over cell 'icell'
275 double temp;
276 values[0] = 1.0;
277 for (std::size_t j = 0; j < degree(); ++j) {
278 left[j] = x - get_knot(icell - j);
279 right[j] = get_knot(icell + j + 1) - x;
280 double saved = 0.0;
281 for (std::size_t r = 0; r < j + 1; ++r) {
282 temp = values[r] / (right[r] + left[j - r]);
283 values[r] = saved + right[r] * temp;
284 saved = left[j - r] * temp;
285 }
286 values[j + 1] = saved;
287 }
288
289 return discrete_element_type(icell);
290}

293template <class DDim, class MemorySpace>
294KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDim> NonUniformBSplines<Tag, D>::
295 Impl<DDim, MemorySpace>::eval_deriv(DSpan1D derivs, ddc::Coordinate<Tag> const& x) const
296{
297 std::array<double, degree()> left;
298 std::array<double, degree()> right;
299
300 assert(x >= rmin());
301 assert(x <= rmax());
302 assert(derivs.size() == degree() + 1);
303
304 // 1. Compute cell index 'icell'
305 int const icell = find_cell(x);
306
307 assert(icell >= 0);
308 assert(icell <= int(ncells() - 1));
309 assert(get_knot(icell) <= x);
310 assert(get_knot(icell + 1) >= x);
311
312 // 2. Compute values of derivatives of B-splines with support over cell 'icell'
313
314 /*
315 * Compute nonzero basis functions and knot differences
316 * for splines up to degree degree-1 which are needed to compute derivative
317 * First part of Algorithm A3.2 of NURBS book
318 */
319 double saved, temp;
320 derivs[0] = 1.0;
321 for (std::size_t j = 0; j < degree() - 1; ++j) {
322 left[j] = x - get_knot(icell - j);
323 right[j] = get_knot(icell + j + 1) - x;
324 saved = 0.0;
325 for (std::size_t r = 0; r < j + 1; ++r) {
326 temp = derivs[r] / (right[r] + left[j - r]);
327 derivs[r] = saved + right[r] * temp;
328 saved = left[j - r] * temp;
329 }
330 derivs[j + 1] = saved;
331 }
332
333 /*
334 * Compute derivatives at x using values stored in bsdx and formula
335 * for spline derivative based on difference of splines of degree degree-1
336 */
337 saved = degree() * derivs[0] / (get_knot(icell + 1) - get_knot(icell + 1 - degree()));
338 derivs[0] = -saved;
339 for (std::size_t j = 1; j < degree(); ++j) {
340 temp = saved;
341 saved = degree() * derivs[j]
342 / (get_knot(icell + j + 1) - get_knot(icell + j + 1 - degree()));
343 derivs[j] = temp - saved;
344 }
345 derivs[degree()] = saved;
346
347 return discrete_element_type(icell);
348}

351template <class DDim, class MemorySpace>
352KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDim> NonUniformBSplines<Tag, D>::
353 Impl<DDim, MemorySpace>::eval_basis_and_n_derivs(
354 ddc::DSpan2D const derivs,
355 ddc::Coordinate<Tag> const& x,
356 std::size_t const n) const
357{
358 std::array<double, degree()> left;
359 std::array<double, degree()> right;
360
361 std::array<double, 2 * (degree() + 1)> a_ptr;
362 std::experimental::
363 mdspan<double, std::experimental::extents<std::size_t, degree() + 1, 2>> const a(
364 a_ptr.data());
365
366 std::array<double, (degree() + 1) * (degree() + 1)> ndu_ptr;
367 std::experimental::mdspan<
368 double,
369 std::experimental::extents<std::size_t, degree() + 1, degree() + 1>> const
370 ndu(ndu_ptr.data());
371
372 assert(x >= rmin());
373 assert(x <= rmax());
374 // assert(n >= 0); as long as n is unsigned
375 assert(n <= degree());
376 assert(derivs.extent(0) == 1 + degree());
377 assert(derivs.extent(1) == 1 + n);
378
379 // 1. Compute cell index 'icell' and x_offset
380 int const icell = find_cell(x);
381
382 assert(icell >= 0);
383 assert(icell <= int(ncells() - 1));
384 assert(get_knot(icell) <= x);
385 assert(get_knot(icell + 1) >= x);
386
387 // 2. Compute nonzero basis functions and knot differences for splines
388 // up to degree (degree-1) which are needed to compute derivative
389 // Algorithm A2.3 of NURBS book
390 //
391 // 21.08.2017: save inverse of knot differences to avoid unnecessary
392 // divisions
393 // [Yaman Güçlü, Edoardo Zoni]
394
395 double saved, temp;
396 ndu(0, 0) = 1.0;
397 for (std::size_t j = 0; j < degree(); ++j) {
398 left[j] = x - get_knot(icell - j);
399 right[j] = get_knot(icell + j + 1) - x;
400 saved = 0.0;
401 for (std::size_t r = 0; r < j + 1; ++r) {
402 // compute inverse of knot differences and save them into lower
403 // triangular part of ndu
404 ndu(r, j + 1) = 1.0 / (right[r] + left[j - r]);
405 // compute basis functions and save them into upper triangular part
406 // of ndu
407 temp = ndu(j, r) * ndu(r, j + 1);
408 ndu(j + 1, r) = saved + right[r] * temp;
409 saved = left[j - r] * temp;
410 }
411 ndu(j + 1, j + 1) = saved;
412 }
413 // Save 0-th derivative
414 for (std::size_t j = 0; j < degree() + 1; ++j) {
415 derivs(j, 0) = ndu(degree(), j);
416 }
417
418 for (int r = 0; r < int(degree() + 1); ++r) {
419 int s1 = 0;
420 int s2 = 1;
421 a(0, 0) = 1.0;
422 for (int k = 1; k < int(n + 1); ++k) {
423 double d = 0.0;
424 int const rk = r - k;
425 int const pk = degree() - k;
426 if (r >= k) {
427 a(0, s2) = a(0, s1) * ndu(rk, pk + 1);
428 d = a(0, s2) * ndu(pk, rk);
429 }
430 int const j1 = rk > -1 ? 1 : (-rk);
431 int const j2 = (r - 1) <= pk ? k : (degree() - r + 1);
432 for (int j = j1; j < j2; ++j) {
433 a(j, s2) = (a(j, s1) - a(j - 1, s1)) * ndu(rk + j, pk + 1);
434 d += a(j, s2) * ndu(pk, rk + j);
435 }
436 if (r <= pk) {
437 a(k, s2) = -a(k - 1, s1) * ndu(r, pk + 1);
438 d += a(k, s2) * ndu(pk, r);
439 }
440 derivs(r, k) = d;
441 Kokkos::kokkos_swap(s1, s2);
442 }
443 }
444
445 int r = degree();
446 for (int k = 1; k < int(n + 1); ++k) {
447 for (std::size_t i = 0; i < derivs.extent(0); i++) {
448 derivs(i, k) *= r;
449 }
450 r *= degree() - k;
451 }
452
453 return discrete_element_type(icell);
454}

456template <class Tag, std::size_t D>
457template <class DDim, class MemorySpace>
458KOKKOS_INLINE_FUNCTION int NonUniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::find_cell(
459 ddc::Coordinate<Tag> const& x) const
460{
461 if (x > rmax())
462 return -1;
463 if (x < rmin())
464 return -1;
465
466 if (x == rmin())
467 return 0;
468 if (x == rmax())
469 return ncells() - 1;
470
471 // Binary search
472 int low = 0, high = ncells();
473 int icell = (low + high) / 2;
474 while (x < get_knot(icell) || x >= get_knot(icell + 1)) {
475 if (x < get_knot(icell)) {
476 high = icell;
477 } else {
478 low = icell;
479 }
480 icell = (low + high) / 2;
481 }
482 return icell;
483}

485template <class Tag, std::size_t D>
486template <class DDim, class MemorySpace>
487template <class Layout, class MemorySpace2>
488KOKKOS_INLINE_FUNCTION ddc::ChunkSpan<double, ddc::DiscreteDomain<DDim>, Layout, MemorySpace2>
489NonUniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::integrals(
490 ddc::ChunkSpan<double, ddc::DiscreteDomain<DDim>, Layout, MemorySpace2> int_vals) const
491{
492 if constexpr (is_periodic()) {
493 assert(int_vals.size() == nbasis() || int_vals.size() == size());
494 } else {
495 assert(int_vals.size() == nbasis());
496 }
497
498 double const inv_deg = 1.0 / (degree() + 1);
499
500 discrete_domain_type const dom_bsplines(
501 full_domain().take_first(discrete_vector_type {nbasis()}));
502 for (auto ix : dom_bsplines) {
503 int_vals(ix) = (get_last_support_knot(ix) - get_first_support_knot(ix)) * inv_deg;
504 }
505
506 if constexpr (is_periodic()) {
507 if (int_vals.size() == size()) {
508 discrete_domain_type const dom_bsplines_wrap(
509 full_domain().take_last(discrete_vector_type {degree()}));
510 for (auto ix : dom_bsplines_wrap) {
511 int_vals(ix) = 0;
512 }
513 }
514 }
515 return int_vals;
516}
