105// LastSelector: returns a if Dim==Last, else b
106template <typename T, typename Dim, typename Last>
107KOKKOS_FUNCTION constexpr T LastSelector(const T a, const T b)
108{
109 return std::is_same_v<Dim, Last> ? a : b;
110}
111
112template <typename T, typename Dim, typename First, typename Second, typename... Tail>
113KOKKOS_FUNCTION constexpr T LastSelector(const T a, const T b)
114{
115 return LastSelector<T, Dim, Second, Tail...>(a, b);
116}

328// N,a,b from x_mesh
329template <typename DDim, typename... DDimX>
330int N(ddc::DiscreteDomain<DDimX...> x_mesh)
331{
332 static_assert(
333 (is_uniform_point_sampling_v<DDimX> && ...),
334 "DDimX dimensions should derive from UniformPointSampling");
335 return ddc::get<DDim>(x_mesh.extents());
336}
337
338template <typename DDim, typename... DDimX>
339double a(ddc::DiscreteDomain<DDimX...> x_mesh)
340{
341 static_assert(
342 (is_uniform_point_sampling_v<DDimX> && ...),
343 "DDimX dimensions should derive from UniformPointSampling");
344 return ((2 * N<DDim>(x_mesh) - 1) * coordinate(ddc::select<DDim>(x_mesh).front())
345 - coordinate(ddc::select<DDim>(x_mesh).back()))
346 / 2 / (N<DDim>(x_mesh) - 1);
347}
348
349template <typename DDim, typename... DDimX>
350double b(ddc::DiscreteDomain<DDimX...> x_mesh)
351{
352 static_assert(
353 (is_uniform_point_sampling_v<DDimX> && ...),
354 "DDimX dimensions should derive from UniformPointSampling");
355 return ((2 * N<DDim>(x_mesh) - 1) * coordinate(ddc::select<DDim>(x_mesh).back())
356 - coordinate(ddc::select<DDim>(x_mesh).front()))
357 / 2 / (N<DDim>(x_mesh) - 1);
358}

360// core
361template <typename Tin, typename Tout, typename ExecSpace, typename MemorySpace, typename... DDimX>
362void core(
363 ExecSpace const& execSpace,
364 Tout* out_data,
365 Tin* in_data,
366 ddc::DiscreteDomain<DDimX...> mesh,
367 const kwArgs_core& kwargs)
368{
369 static_assert(
370 std::is_same_v<real_type_t<Tin>, float> || std::is_same_v<real_type_t<Tin>, double>,
371 "Base type of Tin (and Tout) must be float or double.");
372 static_assert(
373 std::is_same_v<real_type_t<Tin>, real_type_t<Tout>>,
374 "Types Tin and Tout must be based on same type (float or double)");
375 static_assert(
376 Kokkos::SpaceAccessibility<ExecSpace, MemorySpace>::accessible,
377 "MemorySpace has to be accessible for ExecutionSpace.");
378 static_assert(
379 (is_uniform_point_sampling_v<DDimX> && ...),
380 "DDimX dimensions should derive from UniformPointSampling");
381
382 std::array<int, sizeof...(DDimX)> n = {static_cast<int>(ddc::get<DDimX>(mesh.extents()))...};
383 int idist = 1;
384 int odist = 1;
385 for (std::size_t i = 0; i < sizeof...(DDimX); i++) {
386 idist = transform_type_v<Tin, Tout> == TransformType::C2R && i == sizeof...(DDimX) - 1
387 ? idist * (n[i] / 2 + 1)
388 : idist * n[i];
389 odist = transform_type_v<Tin, Tout> == TransformType::R2C && i == sizeof...(DDimX) - 1
390 ? odist * (n[i] / 2 + 1)
391 : odist * n[i];
392 }
393
394 if constexpr (false) {
395 } // Trick to get only else if
396#if fftw_serial_AVAIL
397 else if constexpr (std::is_same_v<ExecSpace, Kokkos::Serial>) {
398 _fftw_plan<Tin> plan = _fftw_plan_many_dft<Tin, Tout>(
399 kwargs.direction == ddc::FFT_Direction::FORWARD ? FFTW_FORWARD : FFTW_BACKWARD,
400 FFTW_ESTIMATE,
401 static_cast<int>(sizeof...(DDimX)),
402 n.data(),
403 1,
404 reinterpret_cast<typename _fftw_type<Tin>::type*>(in_data),
405 static_cast<int*>(nullptr),
406 1,
407 idist,
408 reinterpret_cast<typename _fftw_type<Tout>::type*>(out_data),
409 static_cast<int*>(nullptr),
410 1,
411 odist);
412 if constexpr (std::is_same_v<real_type_t<Tin>, float>) {
413 fftwf_execute(plan);
414 fftwf_destroy_plan(plan);
415 } else {
416 fftw_execute(plan);
417 fftw_destroy_plan(plan);
418 }
419 }
420#endif
421#if fftw_omp_AVAIL
422 else if constexpr (std::is_same_v<ExecSpace, Kokkos::OpenMP>) {
423 if constexpr (std::is_same_v<real_type_t<Tin>, float>) {
424 fftwf_init_threads();
425 fftwf_plan_with_nthreads(execSpace.concurrency());
426 } else {
427 fftw_init_threads();
428 fftw_plan_with_nthreads(execSpace.concurrency());
429 }
430 _fftw_plan<Tin> plan = _fftw_plan_many_dft<Tin, Tout>(
431 kwargs.direction == ddc::FFT_Direction::FORWARD ? FFTW_FORWARD : FFTW_BACKWARD,
432 FFTW_ESTIMATE,
433 static_cast<int>(sizeof...(DDimX)),
434 n.data(),
435 1,
436 reinterpret_cast<typename _fftw_type<Tin>::type*>(in_data),
437 static_cast<int*>(nullptr),
438 1,
439 idist,
440 reinterpret_cast<typename _fftw_type<Tout>::type*>(out_data),
441 static_cast<int*>(nullptr),
442 1,
443 odist);
444 if constexpr (std::is_same_v<real_type_t<Tin>, float>) {
445 fftwf_execute(plan);
446 fftwf_destroy_plan(plan);
447 } else {
448 fftw_execute(plan);
449 fftw_destroy_plan(plan);
450 }
451 }
452#endif
453#if cufft_AVAIL
454 else if constexpr (std::is_same_v<ExecSpace, Kokkos::Cuda>) {
455 cudaStream_t stream = execSpace.cuda_stream();
456
457 cufftHandle unmanaged_plan = -1;
458 cufftResult cufft_rt = cufftCreate(&unmanaged_plan);
459
460 if (cufft_rt != CUFFT_SUCCESS)
461 throw std::runtime_error("cufftCreate failed");
462
463 std::unique_ptr<cufftHandle, std::function<void(cufftHandle*)>> const
464 managed_plan(&unmanaged_plan, [](cufftHandle* handle) { cufftDestroy(*handle); });
465
466 cufftSetStream(unmanaged_plan, stream);
467 cufft_rt = cufftPlanMany(
468 &unmanaged_plan, // plan handle
469 sizeof...(DDimX),
470 n.data(), // Nx, Ny...
471 nullptr,
472 1,
473 idist,
474 nullptr,
475 1,
476 odist,
477 cufft_transform_type<Tin, Tout>(),
478 1);
479
480 if (cufft_rt != CUFFT_SUCCESS)
481 throw std::runtime_error("cufftPlan failed");
482
483 cufft_rt = _cufftExec<Tin, Tout>(
484 kwargs.direction == ddc::FFT_Direction::FORWARD ? CUFFT_FORWARD : CUFFT_INVERSE,
485 unmanaged_plan,
486 reinterpret_cast<typename _cufft_type<Tin>::type*>(in_data),
487 reinterpret_cast<typename _cufft_type<Tout>::type*>(out_data));
488 if (cufft_rt != CUFFT_SUCCESS)
489 throw std::runtime_error("cufftExec failed");
490 }
491#endif
492#if hipfft_AVAIL
493 else if constexpr (std::is_same_v<ExecSpace, Kokkos::HIP>) {
494 hipStream_t stream = execSpace.hip_stream();
495
496 hipfftHandle unmanaged_plan;
497 hipfftResult hipfft_rt = hipfftCreate(&unmanaged_plan);
498
499 if (hipfft_rt != HIPFFT_SUCCESS)
500 throw std::runtime_error("hipfftCreate failed");
501
502 std::unique_ptr<hipfftHandle, std::function<void(hipfftHandle*)>> const
503 managed_plan(&unmanaged_plan, [](hipfftHandle* handle) { hipfftDestroy(*handle); });
504
505 hipfftSetStream(unmanaged_plan, stream);
506 hipfft_rt = hipfftPlanMany(
507 &unmanaged_plan, // plan handle
508 sizeof...(DDimX),
509 n.data(), // Nx, Ny...
510 nullptr,
511 1,
512 idist,
513 nullptr,
514 1,
515 odist,
516 hipfft_transform_type<Tin, Tout>(),
517 1);
518
519 if (hipfft_rt != HIPFFT_SUCCESS)
520 throw std::runtime_error("hipfftPlan failed");
521
522 hipfft_rt = _hipfftExec<Tin, Tout>(
523 kwargs.direction == ddc::FFT_Direction::FORWARD ? HIPFFT_FORWARD : HIPFFT_BACKWARD,
524 unmanaged_plan,
525 reinterpret_cast<typename _hipfft_type<Tin>::type*>(in_data),
526 reinterpret_cast<typename _hipfft_type<Tout>::type*>(out_data));
527 if (hipfft_rt != HIPFFT_SUCCESS)
528 throw std::runtime_error("hipfftExec failed");
529 }
530#endif
531
532 if (kwargs.normalization != ddc::FFT_Normalization::OFF) {
533 real_type_t<Tout> norm_coef = 1;
534 switch (kwargs.normalization) {
535 case ddc::FFT_Normalization::OFF:
536 break;
537 case ddc::FFT_Normalization::FORWARD:
538 norm_coef = kwargs.direction == ddc::FFT_Direction::FORWARD
539 ? 1. / (ddc::get<DDimX>(mesh.extents()) * ...)
540 : 1.;
541 break;
542 case ddc::FFT_Normalization::BACKWARD:
543 norm_coef = kwargs.direction == ddc::FFT_Direction::BACKWARD
544 ? 1. / (ddc::get<DDimX>(mesh.extents()) * ...)
545 : 1.;
546 break;
547 case ddc::FFT_Normalization::ORTHO:
548 norm_coef = 1. / Kokkos::sqrt((ddc::get<DDimX>(mesh.extents()) * ...));
549 break;
550 case ddc::FFT_Normalization::FULL:
551 norm_coef = kwargs.direction == ddc::FFT_Direction::FORWARD
552 ? (((coordinate(ddc::select<DDimX>(mesh).back())
553 - coordinate(ddc::select<DDimX>(mesh).front()))
554 / (ddc::get<DDimX>(mesh.extents()) - 1)
555 / Kokkos::sqrt(2 * Kokkos::numbers::pi))
556 * ...)
557 : ((Kokkos::sqrt(2 * Kokkos::numbers::pi)
558 / (coordinate(ddc::select<DDimX>(mesh).back())
559 - coordinate(ddc::select<DDimX>(mesh).front()))
560 * (ddc::get<DDimX>(mesh.extents()) - 1)
561 / ddc::get<DDimX>(mesh.extents()))
562 * ...);
563 break;
564 }
565
566 Kokkos::parallel_for(
567 Kokkos::RangePolicy<ExecSpace>(
568 execSpace,
569 0,
570 is_complex_v<Tout> && transform_type_v<Tin, Tout> != TransformType::C2C
571 ? (LastSelector<double, DDimX, DDimX...>(
572 ddc::get<DDimX>(mesh.extents()) / 2 + 1,
573 ddc::get<DDimX>(mesh.extents()))
574 * ...)
575 : (ddc::get<DDimX>(mesh.extents()) * ...)),
576 KOKKOS_LAMBDA(const int& i) { out_data[i] = out_data[i] * norm_coef; });
577 }
578}

588 is_uniform_point_sampling_v<DDimX>,
589 "DDimX dimensions should derive from UniformPointSampling");
590 static_assert(
591 is_periodic_sampling_v<DDimFx>,
592 "DDimFx dimensions should derive from PeriodicPointSampling");

595 ddc::Coordinate<typename DDimFx::continuous_dimension_type>(

599 ddc::DiscreteVector<DDimFx>(ddc::detail::fft::N<DDimX>(x_mesh)),

601 return std::move(impl);
602}
