194  {
195  namespace stdex = std::experimental;
196  extents_type extents_s((front<DDims>(domain) + ::extents<DDims>(domain)).uid()...);
197  std::array<std::size_t, sizeof...(DDims)> strides_s {allocation_mdspan.mapping().stride(
198  type_seq_rank_v<DDims, detail::TypeSeq<DDims...>>)...};
199  stdex::layout_stride::mapping<extents_type> mapping_s(extents_s, strides_s);
200  this->m_internal_mdspan = internal_mdspan_type(
201  allocation_mdspan.data() - mapping_s(front<DDims>(domain).uid()...),

203  this->m_domain = domain;

264  template <class... ODDims>
265  constexpr reference operator()(DiscreteElement<ODDims> const&... delems) const noexcept
266  {
267  static_assert(sizeof...(ODDims) == sizeof...(DDims), "Invalid number of dimensions");
268  assert(((delems >= front<ODDims>(this->m_domain)) && ...));
269  assert(((delems <= back<ODDims>(this->m_domain)) && ...));
270  return this->m_internal_mdspan(uid(take<DDims>(delems...))...);
271  }

281  assert(((select<ODDims>(delems) >= front<ODDims>(this->m_domain)) && ...));
282  assert(((select<ODDims>(delems) <= back<ODDims>(this->m_domain)) && ...));
283  return this->m_internal_mdspan(uid<DDims>(delems)...);
