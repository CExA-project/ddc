380template <class Tag, std::size_t D>
381template <class DDim, class MemorySpace>
382KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDim> UniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::
383 eval_basis(
384 DSpan1D values,
385 ddc::Coordinate<Tag> const& x,
386 [[maybe_unused]] std::size_t const deg) const
387{
388 assert(values.size() == deg + 1);
389
390 double offset;
391 int jmin;
392 // 1. Compute cell index 'icell' and x_offset
393 // 2. Compute index range of B-splines with support over cell 'icell'
394 get_icell_and_offset(jmin, offset, x);
395
396 // 3. Compute values of aforementioned B-splines
397 double xx, temp, saved;
398 values(0) = 1.0;
399 for (std::size_t j = 1; j < values.size(); ++j) {
400 xx = -offset;
401 saved = 0.0;
402 for (std::size_t r = 0; r < j; ++r) {
403 xx += 1;
404 temp = values(r) / j;
405 values(r) = saved + xx * temp;
406 saved = (j - xx) * temp;
407 }
408 values(j) = saved;
409 }
410
411 return discrete_element_type(jmin);
412}

414template <class Tag, std::size_t D>
415template <class DDim, class MemorySpace>
416KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDim> UniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::
417 eval_deriv(DSpan1D derivs, ddc::Coordinate<Tag> const& x) const
418{
419 assert(derivs.size() == degree() + 1);
420
421 double offset;
422 int jmin;
423 // 1. Compute cell index 'icell' and x_offset
424 // 2. Compute index range of B-splines with support over cell 'icell'
425 get_icell_and_offset(jmin, offset, x);
426
427 // 3. Compute derivatives of aforementioned B-splines
428 // Derivatives are normalized, hence they should be divided by dx
429 double xx, temp, saved;
430 derivs(0) = 1.0 / ddc::step<knot_mesh_type>();
431 for (std::size_t j = 1; j < degree(); ++j) {
432 xx = -offset;
433 saved = 0.0;
434 for (std::size_t r = 0; r < j; ++r) {
435 xx += 1.0;
436 temp = derivs(r) / j;
437 derivs(r) = saved + xx * temp;
438 saved = (j - xx) * temp;
439 }
440 derivs(j) = saved;
441 }
442
443 // Compute derivatives
444 double bjm1 = derivs[0];
445 double bj = bjm1;
446 derivs(0) = -bjm1;
447 for (std::size_t j = 1; j < degree(); ++j) {
448 bj = derivs(j);
449 derivs(j) = bjm1 - bj;
450 bjm1 = bj;
451 }
452 derivs(degree()) = bj;
453
454 return discrete_element_type(jmin);
455}

457template <class Tag, std::size_t D>
458template <class DDim, class MemorySpace>
459KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDim> UniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::
460 eval_basis_and_n_derivs(
461 ddc::DSpan2D const derivs,
462 ddc::Coordinate<Tag> const& x,
463 std::size_t const n) const
464{
465 std::array<double, (degree() + 1) * (degree() + 1)> ndu_ptr;
466 std::experimental::mdspan<
467 double,
468 std::experimental::extents<std::size_t, degree() + 1, degree() + 1>> const
469 ndu(ndu_ptr.data());
470 std::array<double, 2 * (degree() + 1)> a_ptr;
471 std::experimental::
472 mdspan<double, std::experimental::extents<std::size_t, degree() + 1, 2>> const a(
473 a_ptr.data());
474 double offset;
475 int jmin;
476
477 assert(x >= rmin());
478 assert(x <= rmax());
479 // assert(n >= 0); as long as n is unsigned
480 assert(n <= degree());
481 assert(derivs.extent(0) == 1 + degree());
482 assert(derivs.extent(1) == 1 + n);
483
484 // 1. Compute cell index 'icell' and x_offset
485 // 2. Compute index range of B-splines with support over cell 'icell'
486 get_icell_and_offset(jmin, offset, x);
487
488 // 3. Recursively evaluate B-splines (eval_basis)
489 // up to self%degree, and store them all in the upper-right triangle of
490 // ndu
491 double xx, temp, saved;
492 ndu(0, 0) = 1.0;
493 for (std::size_t j = 1; j < degree() + 1; ++j) {
494 xx = -offset;
495 saved = 0.0;
496 for (std::size_t r = 0; r < j; ++r) {
497 xx += 1.0;
498 temp = ndu(j - 1, r) / j;
499 ndu(j, r) = saved + xx * temp;
500 saved = (j - xx) * temp;
501 }
502 ndu(j, j) = saved;
503 }
504 for (std::size_t i = 0; i < ndu.extent(1); ++i) {
505 derivs(i, 0) = ndu(degree(), i);
506 }
507
508 for (int r = 0; r < int(degree() + 1); ++r) {
509 int s1 = 0;
510 int s2 = 1;
511 a(0, 0) = 1.0;
512 for (int k = 1; k < int(n + 1); ++k) {
513 double d = 0.0;
514 int const rk = r - k;
515 int const pk = degree() - k;
516 if (r >= k) {
517 a(0, s2) = a(0, s1) / (pk + 1);
518 d = a(0, s2) * ndu(pk, rk);
519 }
520 int const j1 = rk > -1 ? 1 : (-rk);
521 int const j2 = (r - 1) <= pk ? k : (degree() - r + 1);
522 for (int j = j1; j < j2; ++j) {
523 a(j, s2) = (a(j, s1) - a(j - 1, s1)) / (pk + 1);
524 d += a(j, s2) * ndu(pk, rk + j);
525 }
526 if (r <= pk) {
527 a(k, s2) = -a(k - 1, s1) / (pk + 1);
528 d += a(k, s2) * ndu(pk, r);
529 }
530 derivs(r, k) = d;
531 Kokkos::kokkos_swap(s1, s2);
532 }
533 }
534
535 // Multiply result by correct factors:
536 // degree!/(degree-n)! = degree*(degree-1)*...*(degree-n+1)
537 // k-th derivatives are normalized, hence they should be divided by dx^k
538 double const inv_dx = inv_step();
539 double d = degree() * inv_dx;
540 for (int k = 1; k < int(n + 1); ++k) {
541 for (std::size_t i = 0; i < derivs.extent(0); ++i) {
542 derivs(i, k) *= d;
543 }
544 d *= (degree() - k) * inv_dx;
545 }
546
547 return discrete_element_type(jmin);
548}

550template <class Tag, std::size_t D>
551template <class DDim, class MemorySpace>
552KOKKOS_INLINE_FUNCTION void UniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::get_icell_and_offset(
553 int& icell,
554 double& offset,
555 ddc::Coordinate<Tag> const& x) const
556{
557 assert(x >= rmin());
558 assert(x <= rmax());
559
560 double const inv_dx = inv_step();
561 if (x == rmin()) {
562 icell = 0;
563 offset = 0.0;
564 } else if (x == rmax()) {
565 icell = ncells() - 1;
566 offset = 1.0;
567 } else {
568 offset = (x - rmin()) * inv_dx;
569 icell = static_cast<int>(offset);
570 offset = offset - icell;
571
572 // When x is very close to xmax, round-off may cause the wrong answer
573 // icell=ncells and x_offset=0, which we convert to the case x=xmax:
574 if (icell == int(ncells()) && offset == 0.0) {
575 icell = ncells() - 1;
576 offset = 1.0;
577 }
578 }
579}

581template <class Tag, std::size_t D>
582template <class DDim, class MemorySpace>
583template <class Layout, class MemorySpace2>
584KOKKOS_INLINE_FUNCTION ddc::ChunkSpan<double, ddc::DiscreteDomain<DDim>, Layout, MemorySpace2>
585UniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::integrals(
586 ddc::ChunkSpan<double, discrete_domain_type, Layout, MemorySpace2> int_vals) const
587{
588 if constexpr (is_periodic()) {
589 assert(int_vals.size() == nbasis() || int_vals.size() == size());
590 } else {
591 assert(int_vals.size() == nbasis());
592 }
593 discrete_domain_type const full_dom_splines(full_domain());
594
595 if constexpr (is_periodic()) {
596 discrete_domain_type const dom_bsplines(
597 full_dom_splines.take_first(discrete_vector_type {nbasis()}));
598 for (auto ix : dom_bsplines) {
599 int_vals(ix) = ddc::step<knot_mesh_type>();
600 }
601 if (int_vals.size() == size()) {
602 discrete_domain_type const dom_bsplines_repeated(
603 full_dom_splines.take_last(discrete_vector_type {degree()}));
604 for (auto ix : dom_bsplines_repeated) {
605 int_vals(ix) = 0;
606 }
607 }
608 } else {
609 discrete_domain_type const dom_bspline_entirely_in_domain
610 = full_dom_splines
611 .remove(discrete_vector_type(degree()), discrete_vector_type(degree()));
612 for (auto ix : dom_bspline_entirely_in_domain) {
613 int_vals(ix) = ddc::step<knot_mesh_type>();
614 }
615
616 std::array<double, degree() + 2> edge_vals_ptr;
617 std::experimental::
618 mdspan<double, std::experimental::extents<std::size_t, degree() + 2>> const
619 edge_vals(edge_vals_ptr.data());
620
621 eval_basis(edge_vals, rmin(), degree() + 1);
622
623 double const d_eval = ddc::detail::sum(edge_vals);
624
625 for (std::size_t i = 0; i < degree(); ++i) {
626 double const c_eval = ddc::detail::sum(edge_vals, 0, degree() - i);
627
628 double const edge_value = ddc::step<knot_mesh_type>() * (d_eval - c_eval);
629
630 int_vals(discrete_element_type(i)) = edge_value;
631 int_vals(discrete_element_type(nbasis() - 1 - i)) = edge_value;
632 }
633 }
634 return int_vals;
635}
