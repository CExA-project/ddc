14 template <class Element, class... DDims, class Functor, class... DCoords>
15 inline void for_each_serial(
16  detail::TaggedVector<Element, DDims...> const& start,
17  detail::TaggedVector<Element, DDims...> const& end,
18  Functor const& f,
19  DCoords const&... dcs) noexcept
20 {
21  if constexpr (sizeof...(DCoords) == sizeof...(DDims)) {
22  f(detail::TaggedVector<Element, DDims...> {dcs...});
23  } else {
24  using CurrentDDim = type_seq_element_t<sizeof...(DCoords), detail::TypeSeq<DDims...>>;
25  for (Element ii = select<CurrentDDim>(start); ii <= select<CurrentDDim>(end); ++ii) {
26  for_each_serial(start, end, f, dcs..., ii);
27  }
28  }
29 }

31 template <class Element, class... DDims, class Functor>
32 inline void for_each_omp(
33  detail::TaggedVector<Element, DDims...> const& start,
34  detail::TaggedVector<Element, DDims...> const& end,
35  Functor&& f) noexcept
36 {
37  using FirstDDim = type_seq_element_t<0, detail::TypeSeq<DDims...>>;
38  Element const ib = select<FirstDDim>(start);
39  Element const ie = select<FirstDDim>(end);
40 #pragma omp parallel for default(none) shared(ib, ie, start, end, f)
41  for (Element ii = ib; ii <= ie; ++ii) {
42  if constexpr (sizeof...(DDims) == 1) {
43  f(detail::TaggedVector<Element, FirstDDim> {ii});
44  } else {
45  detail::for_each_serial(start, end, f, ii);
46  }
47  }
48 }
