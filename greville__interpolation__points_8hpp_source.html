41 template <class Sampling, typename U = BSplines, class = std::enable_if_t<U::is_uniform()>>
42 static auto uniform_greville_points()
43 {
44 using SamplingImpl = typename Sampling::template Impl<Sampling, Kokkos::HostSpace>;
45
46 double constexpr shift = (BSplines::degree() % 2 == 0) ? 0.5 : 0.0;
47 double dx
48 = (ddc::discrete_space<BSplines>().rmax() - ddc::discrete_space<BSplines>().rmin())
49 / ddc::discrete_space<BSplines>().ncells();
50 return SamplingImpl(
51 ddc::Coordinate<tag_type>(ddc::discrete_space<BSplines>().rmin() + shift * dx),
52 ddc::Coordinate<tag_type>(dx));
53 }

55 template <class Sampling, typename U = BSplines, class = std::enable_if_t<!U::is_uniform()>>
56 static auto non_uniform_greville_points()
57 {
58 using SamplingImpl = typename Sampling::template Impl<Sampling, Kokkos::HostSpace>;
59
60 int n_greville_points = ddc::discrete_space<BSplines>().nbasis();
61 if constexpr (U::is_periodic()) {
62 n_greville_points += 1;
63 }
64
65 std::vector<double> greville_points(n_greville_points);
66 ddc::DiscreteDomain<BSplines> bspline_domain
67 = ddc::discrete_space<BSplines>().full_domain().take_first(
68 ddc::DiscreteVector<BSplines>(ddc::discrete_space<BSplines>().nbasis()));
69
70 ddc::DiscreteVector<NonUniformBsplinesKnots<BSplines>> n_points_in_average(
71 BSplines::degree());
72
73 ddc::DiscreteElement<BSplines> ib0(bspline_domain.front());
74
75 ddc::for_each(bspline_domain, [&](ddc::DiscreteElement<BSplines> ib) {
76 // Define the Greville points from the bspline knots
77 greville_points[ib - ib0] = 0.0;
78 ddc::DiscreteDomain<NonUniformBsplinesKnots<BSplines>> sub_domain(
79 ddc::discrete_space<BSplines>().get_first_support_knot(ib) + 1,
80 n_points_in_average);
81 ddc::for_each(sub_domain, [&](auto ik) {
82 greville_points[ib - ib0] += ddc::coordinate(ik);
83 });
84 greville_points[ib - ib0] /= n_points_in_average.value();
85 });
86
87 std::vector<double> temp_knots(BSplines::degree());
88 // Use periodicity to ensure all points are in the domain
89 if constexpr (U::is_periodic()) {
90 int npoints(0);
91 // Count the number of interpolation points that need shifting to preserve the ordering
92 while (greville_points[npoints] < ddc::discrete_space<BSplines>().rmin()) {
93 temp_knots[npoints]
94 = greville_points[npoints] + ddc::discrete_space<BSplines>().length();
95 npoints++;
96 }
97 // Shift the points
98 for (std::size_t i = 0; i < ddc::discrete_space<BSplines>().nbasis() - npoints; ++i) {
99 greville_points[i] = greville_points[i + npoints];
100 }
101 for (int i = 0; i < npoints; ++i) {
102 greville_points[ddc::discrete_space<BSplines>().nbasis() - npoints + i]
103 = temp_knots[i];
104 }
105
106 // Save a periodic point to initialise the domain size
107 greville_points[n_greville_points - 1]
108 = greville_points[0] + ddc::discrete_space<BSplines>().length();
109 }
110
111 return SamplingImpl(greville_points);
112 }

114 static constexpr int N_BE_MIN = n_boundary_equations(BcXmin, BSplines::degree());
115 static constexpr int N_BE_MAX = n_boundary_equations(BcXmax, BSplines::degree());
116 template <class U>
117 static constexpr bool is_uniform_mesh_v
118 = U::is_uniform() && ((N_BE_MIN != 0 && N_BE_MAX != 0) || U::is_periodic());
119
120public:
130 template <
131 class Sampling,
132 typename U = BSplines,
133 std::enable_if_t<
134 is_uniform_mesh_v<U>,
135 bool> = true> // U must be in condition for SFINAE
136 static auto get_sampling()
137 {
138 return uniform_greville_points<Sampling>();
139 }
140
146 template <
147 class Sampling,
148 typename U = BSplines,
149 std::enable_if_t<
150 !is_uniform_mesh_v<U>,
151 bool> = true> // U must be in condition for SFINAE
152 static auto get_sampling()
153 {
154 using SamplingImpl = typename Sampling::template Impl<Sampling, Kokkos::HostSpace>;
155 if constexpr (U::is_uniform()) {
156 using IntermediateSampling = IntermediateUniformSampling<Sampling>;
157 auto points_wo_bcs = uniform_greville_points<IntermediateSampling>();
158 int const n_break_points = ddc::discrete_space<BSplines>().ncells() + 1;
159 int const npoints = ddc::discrete_space<BSplines>().nbasis() - N_BE_MIN - N_BE_MAX;
160 std::vector<double> points_with_bcs(npoints);
161
162 // Construct Greville-like points at the edge
163 if constexpr (BcXmin == ddc::BoundCond::GREVILLE) {
164 for (std::size_t i(0); i < BSplines::degree() / 2 + 1; ++i) {
165 points_with_bcs[i]
166 = (BSplines::degree() - i) * ddc::discrete_space<BSplines>().rmin();
167 ddc::DiscreteElement<BSplines> spline_idx(i);
168 ddc::DiscreteVector<UniformBsplinesKnots<BSplines>> n_knots_in_domain(i);
169 ddc::DiscreteDomain<UniformBsplinesKnots<BSplines>> sub_domain(
170 ddc::discrete_space<BSplines>().get_last_support_knot(spline_idx)
171 - n_knots_in_domain,
172 n_knots_in_domain);
173 ddc::for_each(
174 sub_domain,
175 [&](DiscreteElement<UniformBsplinesKnots<BSplines>> ik) {
176 points_with_bcs[i] += ddc::coordinate(ik);
177 });
178 points_with_bcs[i] /= BSplines::degree();
179 }
180 } else {
181 points_with_bcs[0]
182 = points_wo_bcs.coordinate(ddc::DiscreteElement<IntermediateSampling>(0));
183 }
184
185 int const n_start
186 = (BcXmin == ddc::BoundCond::GREVILLE) ? BSplines::degree() / 2 + 1 : 1;
187 int const domain_size = n_break_points - 2;
188 ddc::DiscreteElement<IntermediateSampling> domain_start(1);
189 ddc::DiscreteDomain<IntermediateSampling> const
190 domain(domain_start, ddc::DiscreteVector<IntermediateSampling>(domain_size));
191
192 // Copy central points
193 ddc::for_each(domain, [&](auto ip) {
194 points_with_bcs[ip - domain_start + n_start] = points_wo_bcs.coordinate(ip);
195 });
196
197 // Construct Greville-like points at the edge
198 if constexpr (BcXmax == ddc::BoundCond::GREVILLE) {
199 for (std::size_t i(0); i < BSplines::degree() / 2 + 1; ++i) {
200 points_with_bcs[npoints - 1 - i]
201 = (BSplines::degree() - i) * ddc::discrete_space<BSplines>().rmax();
202 ddc::DiscreteElement<BSplines> spline_idx(
203 ddc::discrete_space<BSplines>().nbasis() - 1 - i);
204 ddc::DiscreteVector<UniformBsplinesKnots<BSplines>> n_knots_in_domain(i);
205 ddc::DiscreteDomain<UniformBsplinesKnots<BSplines>> sub_domain(
206 ddc::discrete_space<BSplines>().get_first_support_knot(spline_idx) + 1,
207 n_knots_in_domain);
208 ddc::for_each(
209 sub_domain,
210 [&](DiscreteElement<UniformBsplinesKnots<BSplines>> ik) {
211 points_with_bcs[npoints - 1 - i] += ddc::coordinate(ik);
212 });
213 points_with_bcs[npoints - 1 - i] /= BSplines::degree();
214 }
215 } else {
216 points_with_bcs[npoints - 1]
217 = points_wo_bcs.coordinate(ddc::DiscreteElement<IntermediateSampling>(
218 ddc::discrete_space<BSplines>().ncells() - 1
219 + BSplines::degree() % 2));
220 }
221 return SamplingImpl(points_with_bcs);
222 } else {
223 using IntermediateSampling = IntermediateNonUniformSampling<Sampling>;
224 if constexpr (N_BE_MIN == 0 && N_BE_MAX == 0) {
225 return non_uniform_greville_points<Sampling>();
226 } else {
227 auto points_wo_bcs = non_uniform_greville_points<IntermediateSampling>();
228 // All points are Greville points. Extract unnecessary points near the boundary
229 std::vector<double> points_with_bcs(points_wo_bcs.size() - N_BE_MIN - N_BE_MAX);
230 int constexpr n_start = N_BE_MIN;
231
232 using length = ddc::DiscreteVector<IntermediateSampling>;
233
234 ddc::DiscreteElement<IntermediateSampling> domain_start(n_start);
235 ddc::DiscreteDomain<IntermediateSampling> const
236 domain(domain_start, length(points_with_bcs.size()));
237
238 points_with_bcs[0] = points_wo_bcs.coordinate(domain.front());
239 ddc::for_each(domain.remove(length(1), length(1)), [&](auto ip) {
240 points_with_bcs[ip - domain_start] = points_wo_bcs.coordinate(ip);
241 });
242 points_with_bcs[points_with_bcs.size() - 1]
243 = points_wo_bcs.coordinate(domain.back());
244
245 return SamplingImpl(points_with_bcs);
246 }
247 }
248 }

265 template <class Sampling>
266 static ddc::DiscreteDomain<Sampling> get_domain()
267 {
268 int const npoints = ddc::discrete_space<BSplines>().nbasis() - N_BE_MIN - N_BE_MAX;
269 return ddc::DiscreteDomain<Sampling>(
270 ddc::DiscreteElement<Sampling>(0),
271 ddc::DiscreteVector<Sampling>(npoints));
272 }
