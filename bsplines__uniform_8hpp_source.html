223template <class Tag, std::size_t D>
224template <class DDim, class MemorySpace>
225template <std::size_t Size>
226KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDim> UniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::
227 eval_basis(
228 std::array<double, Size>& values,
229 ddc::Coordinate<Tag> const& x,
230 [[maybe_unused]] std::size_t const deg) const
231{
232 assert(values.size() == deg + 1);
233
234 double offset;
235 int jmin;
236 // 1. Compute cell index 'icell' and x_offset
237 // 2. Compute index range of B-splines with support over cell 'icell'
238 get_icell_and_offset(jmin, offset, x);
239
240 // 3. Compute values of aforementioned B-splines
241 double xx, temp, saved;
242 values[0] = 1.0;
243 for (std::size_t j = 1; j < Size; ++j) {
244 xx = -offset;
245 saved = 0.0;
246 for (std::size_t r = 0; r < j; ++r) {
247 xx += 1;
248 temp = values[r] / j;
249 values[r] = saved + xx * temp;
250 saved = (j - xx) * temp;
251 }
252 values[j] = saved;
253 }
254
255 return discrete_element_type(jmin);
256}

258template <class Tag, std::size_t D>
259template <class DDim, class MemorySpace>
260KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDim> UniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::
261 eval_deriv(std::array<double, D + 1>& derivs, ddc::Coordinate<Tag> const& x) const
262{
263 assert(derivs.size() == degree() + 1);
264
265 double offset;
266 int jmin;
267 // 1. Compute cell index 'icell' and x_offset
268 // 2. Compute index range of B-splines with support over cell 'icell'
269 get_icell_and_offset(jmin, offset, x);
270
271 // 3. Compute derivatives of aforementioned B-splines
272 // Derivatives are normalized, hence they should be divided by dx
273 double xx, temp, saved;
274 derivs[0] = 1.0 / ddc::step<mesh_type>();
275 for (std::size_t j = 1; j < degree(); ++j) {
276 xx = -offset;
277 saved = 0.0;
278 for (std::size_t r = 0; r < j; ++r) {
279 xx += 1.0;
280 temp = derivs[r] / j;
281 derivs[r] = saved + xx * temp;
282 saved = (j - xx) * temp;
283 }
284 derivs[j] = saved;
285 }
286
287 // Compute derivatives
288 double bjm1 = derivs[0];
289 double bj = bjm1;
290 derivs[0] = -bjm1;
291 for (std::size_t j = 1; j < degree(); ++j) {
292 bj = derivs[j];
293 derivs[j] = bjm1 - bj;
294 bjm1 = bj;
295 }
296 derivs[degree()] = bj;
297
298 return discrete_element_type(jmin);
299}

302template <class DDim, class MemorySpace>
303KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDim> UniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::
304 eval_basis_and_n_derivs(
305 ddc::DSpan2D const derivs,
306 ddc::Coordinate<Tag> const& x,
307 std::size_t const n) const
308{
309 std::array<double, (degree() + 1) * (degree() + 1)> ndu_ptr;
310 std::experimental::mdspan<
311 double,
312 std::experimental::extents<std::size_t, degree() + 1, degree() + 1>> const
313 ndu(ndu_ptr.data());
314 std::array<double, 2 * (degree() + 1)> a_ptr;
315 std::experimental::
316 mdspan<double, std::experimental::extents<std::size_t, degree() + 1, 2>> const a(
317 a_ptr.data());
318 double offset;
319 int jmin;
320
321 assert(x >= rmin());
322 assert(x <= rmax());
323 // assert(n >= 0); as long as n is unsigned
324 assert(n <= degree());
325 assert(derivs.extent(0) == 1 + degree());
326 assert(derivs.extent(1) == 1 + n);
327
328 // 1. Compute cell index 'icell' and x_offset
329 // 2. Compute index range of B-splines with support over cell 'icell'
330 get_icell_and_offset(jmin, offset, x);
331
332 // 3. Recursively evaluate B-splines (see
333 // "sll_s_uniform_BSplines_eval_basis")
334 // up to self%degree, and store them all in the upper-right triangle of
335 // ndu
336 double xx, temp, saved;
337 ndu(0, 0) = 1.0;
338 for (std::size_t j = 1; j < degree() + 1; ++j) {
339 xx = -offset;
340 saved = 0.0;
341 for (std::size_t r = 0; r < j; ++r) {
342 xx += 1.0;
343 temp = ndu(j - 1, r) / j;
344 ndu(j, r) = saved + xx * temp;
345 saved = (j - xx) * temp;
346 }
347 ndu(j, j) = saved;
348 }
349 for (std::size_t i = 0; i < ndu.extent(1); ++i) {
350 derivs(i, 0) = ndu(degree(), i);
351 }
352
353 for (int r = 0; r < int(degree() + 1); ++r) {
354 int s1 = 0;
355 int s2 = 1;
356 a(0, 0) = 1.0;
357 for (int k = 1; k < int(n + 1); ++k) {
358 double d = 0.0;
359 int const rk = r - k;
360 int const pk = degree() - k;
361 if (r >= k) {
362 a(0, s2) = a(0, s1) / (pk + 1);
363 d = a(0, s2) * ndu(pk, rk);
364 }
365 int const j1 = rk > -1 ? 1 : (-rk);
366 int const j2 = (r - 1) <= pk ? k : (degree() - r + 1);
367 for (int j = j1; j < j2; ++j) {
368 a(j, s2) = (a(j, s1) - a(j - 1, s1)) / (pk + 1);
369 d += a(j, s2) * ndu(pk, rk + j);
370 }
371 if (r <= pk) {
372 a(k, s2) = -a(k - 1, s1) / (pk + 1);
373 d += a(k, s2) * ndu(pk, r);
374 }
375 derivs(r, k) = d;
376 Kokkos::kokkos_swap(s1, s2);
377 }
378 }
379
380 // Multiply result by correct factors:
381 // degree!/(degree-n)! = degree*(degree-1)*...*(degree-n+1)
382 // k-th derivatives are normalized, hence they should be divided by dx^k
383 double const inv_dx = inv_step();
384 double d = degree() * inv_dx;
385 for (int k = 1; k < int(n + 1); ++k) {
386 for (std::size_t i = 0; i < derivs.extent(0); ++i) {
387 derivs(i, k) *= d;
388 }
389 d *= (degree() - k) * inv_dx;
390 }
391
392 return discrete_element_type(jmin);
393}

395template <class Tag, std::size_t D>
396template <class DDim, class MemorySpace>
397KOKKOS_INLINE_FUNCTION void UniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::get_icell_and_offset(
398 int& icell,
399 double& offset,
400 ddc::Coordinate<Tag> const& x) const
401{
402 assert(x >= rmin());
403 assert(x <= rmax());
404
405 double const inv_dx = inv_step();
406 if (x == rmin()) {
407 icell = 0;
408 offset = 0.0;
409 } else if (x == rmax()) {
410 icell = ncells() - 1;
411 offset = 1.0;
412 } else {
413 offset = (x - rmin()) * inv_dx;
414 icell = static_cast<int>(offset);
415 offset = offset - icell;
416
417 // When x is very close to xmax, round-off may cause the wrong answer
418 // icell=ncells and x_offset=0, which we convert to the case x=xmax:
419 if (icell == int(ncells()) && offset == 0.0) {
420 icell = ncells() - 1;
421 offset = 1.0;
422 }
423 }
424}

426template <class Tag, std::size_t D>
427template <class DDim, class MemorySpace>
428template <class Layout, class MemorySpace2>
429KOKKOS_INLINE_FUNCTION ddc::ChunkSpan<double, ddc::DiscreteDomain<DDim>, Layout, MemorySpace2>
430UniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::integrals(
431 ddc::ChunkSpan<double, ddc::DiscreteDomain<DDim>, Layout, MemorySpace2> int_vals) const
432{
433 if constexpr (is_periodic()) {
434 assert(int_vals.size() == nbasis() || int_vals.size() == size());
435 } else {
436 assert(int_vals.size() == nbasis());
437 }
438 discrete_domain_type const full_dom_splines(full_domain());
439
440 if constexpr (is_periodic()) {
441 discrete_domain_type const dom_bsplines(
442 full_dom_splines.take_first(discrete_vector_type {nbasis()}));
443 for (auto ix : dom_bsplines) {
444 int_vals(ix) = ddc::step<mesh_type>();
445 }
446 if (int_vals.size() == size()) {
447 discrete_domain_type const dom_bsplines_repeated(
448 full_dom_splines.take_last(discrete_vector_type {degree()}));
449 for (auto ix : dom_bsplines_repeated) {
450 int_vals(ix) = 0;
451 }
452 }
453 } else {
454 discrete_domain_type const dom_bspline_entirely_in_domain
455 = full_dom_splines
456 .remove(discrete_vector_type(degree()), discrete_vector_type(degree()));
457 for (auto ix : dom_bspline_entirely_in_domain) {
458 int_vals(ix) = ddc::step<mesh_type>();
459 }
460
461 std::array<double, degree() + 2> edge_vals_ptr;
462 std::experimental::
463 mdspan<double, std::experimental::extents<std::size_t, degree() + 2>> const
464 edge_vals(edge_vals_ptr.data());
465
466 eval_basis(edge_vals_ptr, rmin(), degree() + 1);
467
468 double const d_eval = ddc::detail::sum(edge_vals);
469
470 for (std::size_t i = 0; i < degree(); ++i) {
471 double const c_eval = ddc::detail::sum(edge_vals, 0, degree() - i);
472
473 double const edge_value = ddc::step<mesh_type>() * (d_eval - c_eval);
474
475 int_vals(discrete_element_type(i)) = edge_value;
476 int_vals(discrete_element_type(nbasis() - 1 - i)) = edge_value;
477 }
478 }
479 return int_vals;
480}
