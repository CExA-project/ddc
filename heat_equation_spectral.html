51template <class ChunkType>
52void display(double time, ChunkType temp)
53{
54 double const mean_temp = ddc::transform_reduce(
55 temp.domain(),
56 0.,
57 ddc::reducer::sum<double>(),
58 temp)
59 / temp.domain().size();
60 std::cout << std::fixed << std::setprecision(3);
61 std::cout << "At t = " << time << ",\n";
62 std::cout << " * mean temperature = " << mean_temp << "\n";
63 // take a slice in the middle of the box
64 ddc::ChunkSpan const temp_slice
65 = temp[ddc::get_domain<DDimY>(temp).front()
66 + ddc::get_domain<DDimY>(temp).size() / 2];
67 std::cout << " * temperature[y:"
68 << ddc::get_domain<DDimY>(temp).size() / 2 << "] = {";
69 ddc::for_each(
70 ddc::get_domain<DDimX>(temp),
71 [=](ddc::DiscreteElement<DDimX> const ix) {
72 std::cout << std::setw(6) << temp_slice(ix);
73 });
74 std::cout << " }\n" << std::flush;
75}

77int main(int argc, char** argv)
78{
79 Kokkos::ScopeGuard const kokkos_scope(argc, argv);
80 ddc::ScopeGuard const ddc_scope(argc, argv);
81
82 // some parameters that would typically be read from some form of
83 // configuration file in a more realistic code
84
85 // Start of the domain of interest in the X dimension
86 double const x_start = -1.;
87 // End of the domain of interest in the X dimension
88 double const x_end = 1.;
89 // Number of discretization points in the X dimension
90 std::size_t const nb_x_points = 10;
91 // Thermal diffusion coefficient
92 double const kx = .01;
93 // Start of the domain of interest in the Y dimension
94 double const y_start = -1.;
95 // End of the domain of interest in the Y dimension
96 double const y_end = 1.;
97 // Number of discretization points in the Y dimension
98 std::size_t const nb_y_points = 100;
99 // Thermal diffusion coefficient
100 double const ky = .002;
101 // Simulated time at which to start simulation
102 double const start_time = 0.;
103 // Simulated time to reach as target of the simulation
104 double const end_time = 10.;
105 // Number of time-steps between outputs
106 std::ptrdiff_t const t_output_period = 10;
107
108 // Initialization of the global domain in X including periodic point to have correct step
109 auto const x_domain_with_periodic_point
110 = ddc::init_discrete_space<DDimX>(DDimX::init<DDimX>(
111 ddc::Coordinate<X>(x_start),
112 ddc::Coordinate<X>(x_end),
113 ddc::DiscreteVector<DDimX>(nb_x_points)));
114 ddc::DiscreteDomain<DDimX> const x_domain
115 = x_domain_with_periodic_point.remove_last(
116 ddc::DiscreteVector<DDimX>(1));
117
118 // Initialization of the global domain in Y including periodic point to have correct step
119 auto const y_domain_with_periodic_point
120 = ddc::init_discrete_space<DDimY>(DDimY::init<DDimY>(
121 ddc::Coordinate<Y>(y_start),
122 ddc::Coordinate<Y>(y_end),
123 ddc::DiscreteVector<DDimY>(nb_y_points)));
124 ddc::DiscreteDomain<DDimY> const y_domain
125 = y_domain_with_periodic_point.remove_last(
126 ddc::DiscreteVector<DDimY>(1));
127
128 double const max_rkx = Kokkos::numbers::pi / ddc::step<DDimX>();
129 double const max_rky = Kokkos::numbers::pi / ddc::step<DDimY>();
130 ddc::Coordinate<T> const dt(
131 2. / (kx * max_rkx * max_rkx + ky * max_rky * max_rky));
132
133 // number of time intervals required to reach the end time
134 ddc::DiscreteVector<DDimT> const nb_time_steps(
135 std::ceil((end_time - start_time) / dt) + .2);
136 // Initialization of the global domain in time:
137 // - the number of discrete time-points is equal to the number of
138 // steps + 1
139 ddc::DiscreteDomain<DDimT> const time_domain
140 = ddc::init_discrete_space<DDimT>(DDimT::init<DDimT>(
141 ddc::Coordinate<T>(start_time),
142 ddc::Coordinate<T>(end_time),
143 nb_time_steps + 1));
144
145 ddc::DiscreteDomain<DDimX, DDimY> const
146 xy_domain(x_domain, y_domain);
147
148 // Maps temperature into the full domain (including ghosts) twice:
149 // - once for the last fully computed time-step
150 ddc::Chunk _last_temp(
151 "_last_temp",
152 xy_domain,
153 ddc::DeviceAllocator<double>());
154
155 // - once for time-step being computed
156 ddc::Chunk _next_temp(
157 "_next_temp",
158 xy_domain,
159 ddc::DeviceAllocator<double>());
160
161 ddc::ChunkSpan const initial_temp = _last_temp.span_view();
162 // Initialize the temperature on the main domain
163 ddc::parallel_for_each(
164 xy_domain,
165 KOKKOS_LAMBDA(ddc::DiscreteElement<DDimX, DDimY> const ixy) {
166 double const x = ddc::coordinate(
167 ddc::DiscreteElement<DDimX>(ixy));
168 double const y = ddc::coordinate(
169 ddc::DiscreteElement<DDimY>(ixy));
170 initial_temp(ixy) = 9.999 * ((x * x + y * y) < 0.25);
171 });
172
173 ddc::Chunk _host_temp = ddc::create_mirror(_last_temp.span_cview());
174
175 // display the initial data
176 ddc::parallel_deepcopy(_host_temp, _last_temp);
177 display(ddc::coordinate(time_domain.front()),
178 _host_temp.span_cview());
179 // time of the iteration where the last output happened
180 ddc::DiscreteElement<DDimT> last_output = time_domain.front();
181
182 ddc::init_discrete_space<DDimFx>(
183 ddc::init_fourier_space<DDimFx>(x_domain));
184 ddc::init_discrete_space<DDimFy>(
185 ddc::init_fourier_space<DDimFy>(y_domain));
186
187 ddc::DiscreteDomain<DDimFx, DDimFy> const k_mesh
188 = ddc::fourier_mesh<DDimFx, DDimFy>(xy_domain, false);
189 ddc::Chunk Ff_allocation(
190 "Ff_allocation",
191 k_mesh,
192 ddc::DeviceAllocator<Kokkos::complex<double>>());
193 ddc::ChunkSpan const Ff = Ff_allocation.span_view();
194
195 for (ddc::DiscreteElement<DDimT> const iter :
196 time_domain.remove_first(ddc::DiscreteVector<DDimT>(1))) {
197 // a span excluding ghosts of the temperature at the time-step we
198 // will build
199 ddc::ChunkSpan const next_temp = _next_temp.span_view();
200 // a read-only view of the temperature at the previous time-step
201 ddc::ChunkSpan const last_temp = _last_temp.span_view();
202
203 // Stencil computation on the main domain
204 ddc::kwArgs_fft const kwargs {ddc::FFT_Normalization::BACKWARD};
205 Kokkos::DefaultExecutionSpace const execution_space;
206 ddc::fft(execution_space, Ff, last_temp, kwargs);
207 ddc::parallel_for_each(
208 execution_space,
209 k_mesh,
210 KOKKOS_LAMBDA(ddc::DiscreteElement<DDimFx, DDimFy> const
211 ikxky) {
212 ddc::DiscreteElement<DDimFx> const ikx(ikxky);
213 ddc::DiscreteElement<DDimFy> const iky(ikxky);
214 double const rkx = ddc::coordinate(ikx);
215 double const rky = ddc::coordinate(iky);
216 Ff(ikxky)
217 *= 1
218 - (kx * rkx * rkx + ky * rky * rky) * dt;
219 });
220 ddc::ifft(execution_space, next_temp, Ff, kwargs);
221
222 if (iter - last_output >= t_output_period) {
223 last_output = iter;
224 ddc::parallel_deepcopy(_host_temp, _next_temp);
225 display(ddc::coordinate(iter), _host_temp.span_cview());
226 }
227
228 // Swap our two buffers
229 std::swap(_last_temp, _next_temp);
230 }
231
232 if (last_output < time_domain.back()) {
233 ddc::parallel_deepcopy(_host_temp, _last_temp);
234 display(ddc::coordinate(time_domain.back()),
235 _host_temp.span_cview());
236 }
237}
