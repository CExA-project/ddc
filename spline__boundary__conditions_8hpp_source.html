12/** @brief An enum representing a spline boundary condition. Please refer to
13 * Emily Bourne's thesis (https://www.theses.fr/2022AIXM0412.pdf)
14 */
15enum class BoundCond {
16 PERIODIC, ///< Periodic boundary condition u(1)=u(n)
17 HERMITE, ///< Hermite boundary condition
18 GREVILLE, ///< Use Greville points instead of conditions on derivative for B-Spline interpolation
19};

21/**
22 * @brief Prints a boundary condition in a std::ostream.
23 *
24 * @param out The stream in which the boundary condition is printed.
25 * @param degree The boundary condition.
26 *
27 * @return The stream in which the boundary condition is printed.
28 **/
29static inline std::ostream& operator<<(std::ostream& out, ddc::BoundCond const bc)
30{
31 switch (bc) {
32 case ddc::BoundCond::PERIODIC:
33 return out << "PERIODIC";
34 case ddc::BoundCond::HERMITE:
35 return out << "HERMITE";
36 case ddc::BoundCond::GREVILLE:
37 return out << "GREVILLE";
38 default:
39 throw std::runtime_error("ddc::BoundCond not handled");
40 }
41}

43/**
44 * @brief Return the number of equations needed to describe a given boundary condition.
45 *
46 * @param bc The boundary condition.
47 * @param degree The degree of the spline.
48 *
49 * @return The number of equations.
50 **/
51constexpr int n_boundary_equations(ddc::BoundCond const bc, std::size_t const degree)
52{
53 if (bc == ddc::BoundCond::PERIODIC) {
54 return 0;
55 } else if (bc == ddc::BoundCond::HERMITE) {
56 return degree / 2;
57 } else if (bc == ddc::BoundCond::GREVILLE) {
58 return 0;
59 } else {
60 throw std::runtime_error("ddc::BoundCond not handled");
61 }
62}
