16inline std::shared_ptr<gko::Executor> create_default_host_executor()
17{
18#ifdef KOKKOS_ENABLE_SERIAL
19 if constexpr (std::is_same_v<Kokkos::DefaultHostExecutionSpace, Kokkos::Serial>) {
20 return gko::ReferenceExecutor::create();
21 }
22#endif
23#ifdef KOKKOS_ENABLE_OPENMP
24 if constexpr (std::is_same_v<Kokkos::DefaultHostExecutionSpace, Kokkos::OpenMP>) {
25 return gko::OmpExecutor::create();
26 }
27#endif
28} // Comes from "Basic Kokkos Extension" Ginkgo MR

30template <typename ExecSpace>
31std::shared_ptr<gko::Executor> create_gko_exec()
32{
33#ifdef KOKKOS_ENABLE_SERIAL
34 if constexpr (std::is_same_v<ExecSpace, Kokkos::Serial>) {
35 return gko::ReferenceExecutor::create();
36 }
37#endif
38#ifdef KOKKOS_ENABLE_OPENMP
39 if constexpr (std::is_same_v<ExecSpace, Kokkos::OpenMP>) {
40 return gko::OmpExecutor::create();
41 }
42#endif
43#ifdef KOKKOS_ENABLE_CUDA
44 if constexpr (std::is_same_v<ExecSpace, Kokkos::Cuda>) {
45 ExecSpace exec_space;
46 return gko::CudaExecutor::
47 create(exec_space.cuda_device(),
48 create_default_host_executor(),
49 std::make_shared<gko::CudaAllocator>(),
50 exec_space.cuda_stream());
51 }
52#endif
53#ifdef KOKKOS_ENABLE_HIP
54 if constexpr (std::is_same_v<ExecSpace, Kokkos::HIP>) {
55 ExecSpace exec_space;
56 return gko::HipExecutor::
57 create(exec_space.hip_device(),
58 create_default_host_executor(),
59 std::make_shared<gko::HipAllocator>(),
60 exec_space.hip_stream());
61 }
62#endif
63}

65} // namespace ddc::detail
