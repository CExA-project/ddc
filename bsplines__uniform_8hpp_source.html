215template <class Tag, std::size_t D>
216template <class DDim, class MemorySpace>
217KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDim> UniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::
218 eval_basis(
219 DSpan1D values,
220 ddc::Coordinate<Tag> const& x,
221 [[maybe_unused]] std::size_t const deg) const
222{
223 assert(values.size() == deg + 1);
224
225 double offset;
226 int jmin;
227 // 1. Compute cell index 'icell' and x_offset
228 // 2. Compute index range of B-splines with support over cell 'icell'
229 get_icell_and_offset(jmin, offset, x);
230
231 // 3. Compute values of aforementioned B-splines
232 double xx, temp, saved;
233 values(0) = 1.0;
234 for (std::size_t j = 1; j < values.size(); ++j) {
235 xx = -offset;
236 saved = 0.0;
237 for (std::size_t r = 0; r < j; ++r) {
238 xx += 1;
239 temp = values(r) / j;
240 values(r) = saved + xx * temp;
241 saved = (j - xx) * temp;
242 }
243 values(j) = saved;
244 }
245
246 return discrete_element_type(jmin);
247}

249template <class Tag, std::size_t D>
250template <class DDim, class MemorySpace>
251KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDim> UniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::
252 eval_deriv(DSpan1D derivs, ddc::Coordinate<Tag> const& x) const
253{
254 assert(derivs.size() == degree() + 1);
255
256 double offset;
257 int jmin;
258 // 1. Compute cell index 'icell' and x_offset
259 // 2. Compute index range of B-splines with support over cell 'icell'
260 get_icell_and_offset(jmin, offset, x);
261
262 // 3. Compute derivatives of aforementioned B-splines
263 // Derivatives are normalized, hence they should be divided by dx
264 double xx, temp, saved;
265 derivs(0) = 1.0 / ddc::step<mesh_type>();
266 for (std::size_t j = 1; j < degree(); ++j) {
267 xx = -offset;
268 saved = 0.0;
269 for (std::size_t r = 0; r < j; ++r) {
270 xx += 1.0;
271 temp = derivs(r) / j;
272 derivs(r) = saved + xx * temp;
273 saved = (j - xx) * temp;
274 }
275 derivs(j) = saved;
276 }
277
278 // Compute derivatives
279 double bjm1 = derivs[0];
280 double bj = bjm1;
281 derivs(0) = -bjm1;
282 for (std::size_t j = 1; j < degree(); ++j) {
283 bj = derivs(j);
284 derivs(j) = bjm1 - bj;
285 bjm1 = bj;
286 }
287 derivs(degree()) = bj;
288
289 return discrete_element_type(jmin);
290}

293template <class DDim, class MemorySpace>
294KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDim> UniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::
295 eval_basis_and_n_derivs(
296 ddc::DSpan2D const derivs,
297 ddc::Coordinate<Tag> const& x,
298 std::size_t const n) const
299{
300 std::array<double, (degree() + 1) * (degree() + 1)> ndu_ptr;
301 std::experimental::mdspan<
302 double,
303 std::experimental::extents<std::size_t, degree() + 1, degree() + 1>> const
304 ndu(ndu_ptr.data());
305 std::array<double, 2 * (degree() + 1)> a_ptr;
306 std::experimental::
307 mdspan<double, std::experimental::extents<std::size_t, degree() + 1, 2>> const a(
308 a_ptr.data());
309 double offset;
310 int jmin;
311
312 assert(x >= rmin());
313 assert(x <= rmax());
314 // assert(n >= 0); as long as n is unsigned
315 assert(n <= degree());
316 assert(derivs.extent(0) == 1 + degree());
317 assert(derivs.extent(1) == 1 + n);
318
319 // 1. Compute cell index 'icell' and x_offset
320 // 2. Compute index range of B-splines with support over cell 'icell'
321 get_icell_and_offset(jmin, offset, x);
322
323 // 3. Recursively evaluate B-splines (see
324 // "sll_s_uniform_BSplines_eval_basis")
325 // up to self%degree, and store them all in the upper-right triangle of
326 // ndu
327 double xx, temp, saved;
328 ndu(0, 0) = 1.0;
329 for (std::size_t j = 1; j < degree() + 1; ++j) {
330 xx = -offset;
331 saved = 0.0;
332 for (std::size_t r = 0; r < j; ++r) {
333 xx += 1.0;
334 temp = ndu(j - 1, r) / j;
335 ndu(j, r) = saved + xx * temp;
336 saved = (j - xx) * temp;
337 }
338 ndu(j, j) = saved;
339 }
340 for (std::size_t i = 0; i < ndu.extent(1); ++i) {
341 derivs(i, 0) = ndu(degree(), i);
342 }
343
344 for (int r = 0; r < int(degree() + 1); ++r) {
345 int s1 = 0;
346 int s2 = 1;
347 a(0, 0) = 1.0;
348 for (int k = 1; k < int(n + 1); ++k) {
349 double d = 0.0;
350 int const rk = r - k;
351 int const pk = degree() - k;
352 if (r >= k) {
353 a(0, s2) = a(0, s1) / (pk + 1);
354 d = a(0, s2) * ndu(pk, rk);
355 }
356 int const j1 = rk > -1 ? 1 : (-rk);
357 int const j2 = (r - 1) <= pk ? k : (degree() - r + 1);
358 for (int j = j1; j < j2; ++j) {
359 a(j, s2) = (a(j, s1) - a(j - 1, s1)) / (pk + 1);
360 d += a(j, s2) * ndu(pk, rk + j);
361 }
362 if (r <= pk) {
363 a(k, s2) = -a(k - 1, s1) / (pk + 1);
364 d += a(k, s2) * ndu(pk, r);
365 }
366 derivs(r, k) = d;
367 Kokkos::kokkos_swap(s1, s2);
368 }
369 }
370
371 // Multiply result by correct factors:
372 // degree!/(degree-n)! = degree*(degree-1)*...*(degree-n+1)
373 // k-th derivatives are normalized, hence they should be divided by dx^k
374 double const inv_dx = inv_step();
375 double d = degree() * inv_dx;
376 for (int k = 1; k < int(n + 1); ++k) {
377 for (std::size_t i = 0; i < derivs.extent(0); ++i) {
378 derivs(i, k) *= d;
379 }
380 d *= (degree() - k) * inv_dx;
381 }
382
383 return discrete_element_type(jmin);
384}

386template <class Tag, std::size_t D>
387template <class DDim, class MemorySpace>
388KOKKOS_INLINE_FUNCTION void UniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::get_icell_and_offset(
389 int& icell,
390 double& offset,
391 ddc::Coordinate<Tag> const& x) const
392{
393 assert(x >= rmin());
394 assert(x <= rmax());
395
396 double const inv_dx = inv_step();
397 if (x == rmin()) {
398 icell = 0;
399 offset = 0.0;
400 } else if (x == rmax()) {
401 icell = ncells() - 1;
402 offset = 1.0;
403 } else {
404 offset = (x - rmin()) * inv_dx;
405 icell = static_cast<int>(offset);
406 offset = offset - icell;
407
408 // When x is very close to xmax, round-off may cause the wrong answer
409 // icell=ncells and x_offset=0, which we convert to the case x=xmax:
410 if (icell == int(ncells()) && offset == 0.0) {
411 icell = ncells() - 1;
412 offset = 1.0;
413 }
414 }
415}

417template <class Tag, std::size_t D>
418template <class DDim, class MemorySpace>
419template <class Layout, class MemorySpace2>
420KOKKOS_INLINE_FUNCTION ddc::ChunkSpan<double, ddc::DiscreteDomain<DDim>, Layout, MemorySpace2>
421UniformBSplines<Tag, D>::Impl<DDim, MemorySpace>::integrals(
422 ddc::ChunkSpan<double, ddc::DiscreteDomain<DDim>, Layout, MemorySpace2> int_vals) const
423{
424 if constexpr (is_periodic()) {
425 assert(int_vals.size() == nbasis() || int_vals.size() == size());
426 } else {
427 assert(int_vals.size() == nbasis());
428 }
429 discrete_domain_type const full_dom_splines(full_domain());
430
431 if constexpr (is_periodic()) {
432 discrete_domain_type const dom_bsplines(
433 full_dom_splines.take_first(discrete_vector_type {nbasis()}));
434 for (auto ix : dom_bsplines) {
435 int_vals(ix) = ddc::step<mesh_type>();
436 }
437 if (int_vals.size() == size()) {
438 discrete_domain_type const dom_bsplines_repeated(
439 full_dom_splines.take_last(discrete_vector_type {degree()}));
440 for (auto ix : dom_bsplines_repeated) {
441 int_vals(ix) = 0;
442 }
443 }
444 } else {
445 discrete_domain_type const dom_bspline_entirely_in_domain
446 = full_dom_splines
447 .remove(discrete_vector_type(degree()), discrete_vector_type(degree()));
448 for (auto ix : dom_bspline_entirely_in_domain) {
449 int_vals(ix) = ddc::step<mesh_type>();
450 }
451
452 std::array<double, degree() + 2> edge_vals_ptr;
453 std::experimental::
454 mdspan<double, std::experimental::extents<std::size_t, degree() + 2>> const
455 edge_vals(edge_vals_ptr.data());
456
457 eval_basis(edge_vals, rmin(), degree() + 1);
458
459 double const d_eval = ddc::detail::sum(edge_vals);
460
461 for (std::size_t i = 0; i < degree(); ++i) {
462 double const c_eval = ddc::detail::sum(edge_vals, 0, degree() - i);
463
464 double const edge_value = ddc::step<mesh_type>() * (d_eval - c_eval);
465
466 int_vals(discrete_element_type(i)) = edge_value;
467 int_vals(discrete_element_type(nbasis() - 1 - i)) = edge_value;
468 }
469 }
470 return int_vals;
471}
