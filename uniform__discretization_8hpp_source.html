141  static std::tuple<Impl<Kokkos::HostSpace>, ddom_type> init(
142  rcoord_type a,
143  rcoord_type b,
144  dvect_type n)
145  {
146  assert(a < b);
147  assert(n > 1);
148  Impl<Kokkos::HostSpace> disc(a, rcoord_type {(b - a) / (n - 1)});
149  ddom_type domain {disc.front(), n};
150  return std::make_tuple(std::move(disc), std::move(domain));
151  }

162  static std::tuple<Impl<Kokkos::HostSpace>, ddom_type, ddom_type, ddom_type, ddom_type>
163  init_ghosted(
164  rcoord_type a,
165  rcoord_type b,
166  dvect_type n,
167  dvect_type n_ghosts_before,
168  dvect_type n_ghosts_after)
169  {
170  using rcoord_type = rcoord_type;
171  using ddom_type = ddom_type;
172  assert(a < b);
173  assert(n > 1);
174  rcoord_type discretization_step {(b - a) / (n - 1)};
175  Impl<Kokkos::HostSpace>
176  disc(a - n_ghosts_before.value() * discretization_step, discretization_step);
177  ddom_type ghosted_domain = ddom_type(disc.front(), n + n_ghosts_before + n_ghosts_after);
178  ddom_type pre_ghost = ddom_type(ghosted_domain.front(), n_ghosts_before);
179  ddom_type main_domain = ddom_type(ghosted_domain.front() + n_ghosts_before, n);
180  ddom_type post_ghost = ddom_type(main_domain.back() + 1, n_ghosts_after);
181  return std::make_tuple(
182  std::move(disc),
183  std::move(main_domain),
184  std::move(ghosted_domain),
185  std::move(pre_ghost),
186  std::move(post_ghost));
187  }
188
