143 template <class Layout1, class Layout2, class Layout3, class... CoordsDims>
144 void operator()(
145 ddc::ChunkSpan<double, vals_domain_type, Layout1, memory_space> const spline_eval,
146 ddc::ChunkSpan<
147 ddc::Coordinate<CoordsDims...> const,
148 vals_domain_type,
149 Layout2,
150 memory_space> const coords_eval,
151 ddc::ChunkSpan<double const, spline_domain_type, Layout3, memory_space> const
152 spline_coef) const
153 {
154 interpolation_domain_type const interpolation_domain(spline_eval.domain());
155 batch_domain_type const batch_domain(spline_eval.domain());
156
157 ddc::parallel_for_each(
158 exec_space(),
159 batch_domain,
160 KOKKOS_CLASS_LAMBDA(typename batch_domain_type::discrete_element_type const j) {
161 const auto spline_eval_1D = spline_eval[j];
162 const auto coords_eval_1D = coords_eval[j];
163 const auto spline_coef_1D = spline_coef[j];
164 for (auto const i : interpolation_domain) {
165 spline_eval_1D(i) = eval(coords_eval_1D(i), spline_coef_1D);
166 }
167 });
168 }

207 template <class Layout1, class Layout2>
208 void integrate(
209 ddc::ChunkSpan<double, batch_domain_type, Layout1, memory_space> const integrals,
210 ddc::ChunkSpan<double const, spline_domain_type, Layout2, memory_space> const
211 spline_coef) const
212 {
213 batch_domain_type const batch_domain(integrals.domain());
214 ddc::Chunk values_alloc(
215 ddc::DiscreteDomain<bsplines_type>(spline_coef.domain()),
216 ddc::KokkosAllocator<double, memory_space>());
217 ddc::ChunkSpan values = values_alloc.span_view();
218 Kokkos::parallel_for(
219 Kokkos::RangePolicy<exec_space>(0, 1),
220 KOKKOS_LAMBDA(int) { ddc::discrete_space<bsplines_type>().integrals(values); });
221
222 ddc::parallel_for_each(
223 exec_space(),
224 batch_domain,
225 KOKKOS_LAMBDA(typename batch_domain_type::discrete_element_type const j) {
226 integrals(j) = 0;
227 for (typename bsplines_domain_type::discrete_element_type const i :
228 values.domain()) {
229 integrals(j) += spline_coef(i, j) * values(i);
230 }
231 });
232 }

235 template <class Layout, class... CoordsDims>
236 KOKKOS_INLINE_FUNCTION double eval(
237 ddc::Coordinate<CoordsDims...> const& coord_eval,
238 ddc::ChunkSpan<double const, bsplines_domain_type, Layout, memory_space> const
239 spline_coef) const
240 {
241 ddc::Coordinate<typename interpolation_mesh_type::continuous_dimension_type>
242 coord_eval_interpolation
243 = ddc::select<typename interpolation_mesh_type::continuous_dimension_type>(
244 coord_eval);
245 if constexpr (bsplines_type::is_periodic()) {
246 if (coord_eval_interpolation < ddc::discrete_space<bsplines_type>().rmin()
247 || coord_eval_interpolation > ddc::discrete_space<bsplines_type>().rmax()) {
248 coord_eval_interpolation -= Kokkos::floor(
249 (coord_eval_interpolation
250 - ddc::discrete_space<bsplines_type>().rmin())
251 / ddc::discrete_space<bsplines_type>().length())
252 * ddc::discrete_space<bsplines_type>().length();
253 }
254 } else {
255 if (coord_eval_interpolation < ddc::discrete_space<bsplines_type>().rmin()) {
256 return m_left_extrap_rule(coord_eval_interpolation, spline_coef);
257 }
258 if (coord_eval_interpolation > ddc::discrete_space<bsplines_type>().rmax()) {
259 return m_right_extrap_rule(coord_eval_interpolation, spline_coef);
260 }
261 }
262 return eval_no_bc<eval_type>(coord_eval_interpolation, spline_coef);
263 }

265 template <class EvalType, class Layout, class... CoordsDims>
266 KOKKOS_INLINE_FUNCTION double eval_no_bc(
267 ddc::Coordinate<CoordsDims...> const& coord_eval,
268 ddc::ChunkSpan<double const, bsplines_domain_type, Layout, memory_space> const
269 spline_coef) const
270 {
271 static_assert(
272 std::is_same_v<EvalType, eval_type> || std::is_same_v<EvalType, eval_deriv_type>);
273 ddc::DiscreteElement<bsplines_type> jmin;
274 std::array<double, bsplines_type::degree() + 1> vals_ptr;
275 std::experimental::mdspan<
276 double,
277 std::experimental::extents<std::size_t, bsplines_type::degree() + 1>> const
278 vals(vals_ptr.data());
279 ddc::Coordinate<typename interpolation_mesh_type::continuous_dimension_type>
280 coord_eval_interpolation
281 = ddc::select<typename interpolation_mesh_type::continuous_dimension_type>(
282 coord_eval);
283 if constexpr (std::is_same_v<EvalType, eval_type>) {
284 jmin = ddc::discrete_space<bsplines_type>().eval_basis(vals, coord_eval_interpolation);
285 } else if constexpr (std::is_same_v<EvalType, eval_deriv_type>) {
286 jmin = ddc::discrete_space<bsplines_type>().eval_deriv(vals, coord_eval_interpolation);
287 }
288 double y = 0.0;
289 for (std::size_t i = 0; i < bsplines_type::degree() + 1; ++i) {
290 y += spline_coef(ddc::DiscreteElement<bsplines_type>(jmin + i)) * vals[i];
291 }
292 return y;
293 }
