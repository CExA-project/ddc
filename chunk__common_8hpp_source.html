248  template <
249  class Mapping = mapping_type,
250  std::enable_if_t<std::is_constructible_v<Mapping, extents_type>, int> = 0>
251  constexpr ChunkCommon(ElementType* ptr, mdomain_type const& domain)
252  {
253  namespace stdex = std::experimental;
254  // Handle the case where an allocation of size 0 returns a nullptr.
255  assert((domain.size() == 0) || ((ptr != nullptr) && (domain.size() != 0)));
256 
257  extents_type extents_r(::extents<DDims>(domain).value()...);
258  mapping_type mapping_r(extents_r);
259 
260  extents_type extents_s((front<DDims>(domain) + ::extents<DDims>(domain)).uid()...);
261  std::array<std::size_t, sizeof...(DDims)> strides_s {
262  mapping_r.stride(type_seq_rank_v<DDims, detail::TypeSeq<DDims...>>)...};
263  stdex::layout_stride::mapping<extents_type> mapping_s(extents_s, strides_s);
264 
265  // Pointer offset to handle non-zero indexing
266  ptr -= mapping_s(front<DDims>(domain).uid()...);
267  m_internal_mdspan = internal_mdspan_type(ptr, mapping_s);
268  m_domain = domain;
269  }
