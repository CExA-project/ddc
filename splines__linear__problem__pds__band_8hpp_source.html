26/**
27 * @brief A positive-definite symmetric band linear problem dedicated to the computation of a spline approximation.
28 *
29 * The storage format is dense row-major. Lapack is used to perform every matrix and linear solver-related operations.
30 *
31 * Given the linear system A*x=b, we assume that A is a square (n by n)
32 * with kd sub and superdiagonals.
33 * All non-zero elements of A are stored in the rectangular matrix q, using
34 * the format required by DPBTRF (LAPACK): (super-)diagonals of A are rows of q.
35 * q has 1 row for the diagonal and kd rows for the superdiagonals.
36 *
37 * @tparam ExecSpace The Kokkos::ExecutionSpace on which operations related to the matrix are supposed to be performed.
38 */
39template <class ExecSpace>
40class SplinesLinearProblemPDSBand : public SplinesLinearProblem<ExecSpace>
41{
42public:
43 using typename SplinesLinearProblem<ExecSpace>::MultiRHS;
44 using SplinesLinearProblem<ExecSpace>::size;
45
46protected:
47 Kokkos::DualView<double**, Kokkos::LayoutRight, typename ExecSpace::memory_space>
48 m_q; // pds band matrix representation
49
50public:
51 /**
52 * @brief SplinesLinearProblemPDSBand constructor.
53 *
54 * @param mat_size The size of one of the dimensions of the square matrix.
55 * @param kd The number of sub/superdiagonals of the matrix.
56 */
57 explicit SplinesLinearProblemPDSBand(std::size_t const mat_size, std::size_t const kd)
58 : SplinesLinearProblem<ExecSpace>(mat_size)
59 , m_q("q", kd + 1, mat_size)
60 {
61 assert(m_q.extent(0) <= mat_size);
62
63 Kokkos::deep_copy(m_q.h_view, 0.);
64 }
65
66 double get_element(std::size_t i, std::size_t j) const override
67 {
68 assert(i < size());
69 assert(j < size());
70
71 // Indices are swapped for an element on subdiagonal
72 if (i > j) {
73 std::swap(i, j);
74 }
75 if (j - i < m_q.extent(0)) {
76 return m_q.h_view(j - i, i);
77 } else {
78 return 0.0;
79 }
80 }
81
82 void set_element(std::size_t i, std::size_t j, double const aij) override
83 {
84 assert(i < size());
85 assert(j < size());
86
87 // Indices are swapped for an element on subdiagonal
88 if (i > j) {
89 std::swap(i, j);
90 }
91 if (j - i < m_q.extent(0)) {
92 m_q.h_view(j - i, i) = aij;
93 } else {
94 assert(std::fabs(aij) < 1e-20);
95 }
96 }
97
98 /**
99 * @brief Perform a pre-process operation on the solver. Must be called after filling the matrix.
100 *
101 * LU-factorize the matrix A and store the pivots using the LAPACK dpbtrf() implementation.
102 */
103 void setup_solver() override
104 {
105 int const info = LAPACKE_dpbtrf(
106 LAPACK_ROW_MAJOR,
107 'L',
108 size(),
109 m_q.extent(0) - 1,
110 m_q.h_view.data(),
111 m_q.h_view.stride(
112 0) // m_q.h_view.stride(0) if LAPACK_ROW_MAJOR, m_q.h_view.stride(1) if LAPACK_COL_MAJOR
113 );
114 if (info != 0) {
115 throw std::runtime_error(
116 "LAPACKE_dpbtrf failed with error code " + std::to_string(info));
117 }
118
119 // Push on device
120 m_q.modify_host();
121 m_q.sync_device();
122 }
123
124 /**
125 * @brief Solve the multiple right-hand sides linear problem Ax=b or its transposed version A^tx=b inplace.
126 *
127 * The solver method is band gaussian elimination with partial pivoting using the LU-factorized matrix A. The implementation is LAPACK method dpbtrs.
128 *
129 * @param[in, out] b A 2D Kokkos::View storing the multiple right-hand sides of the problem and receiving the corresponding solution.
130 * @param transpose Choose between the direct or transposed version of the linear problem (unused for a symmetric problem).
131 */
132 void solve(MultiRHS const b, bool const) const override
133 {
134 assert(b.extent(0) == size());
135
136 auto q_device = m_q.d_view;
137 Kokkos::RangePolicy<ExecSpace> const policy(0, b.extent(1));
138 Kokkos::parallel_for(
139 "pbtrs",
140 policy,
141 KOKKOS_LAMBDA(const int i) {
142 auto sub_b = Kokkos::subview(b, Kokkos::ALL, i);
143 KokkosBatched::SerialPbtrs<
144 KokkosBatched::Uplo::Lower,
145 KokkosBatched::Algo::Pbtrs::Unblocked>::invoke(q_device, sub_b);
146 });
147 }
148};
