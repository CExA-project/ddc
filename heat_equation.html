47template <class ChunkType>
48void display(double time, ChunkType temp)
49{
50 // For `Chunk`/`ChunkSpan`, the ()-operator is used to access stored values with a single `DiscreteElement` as
51 // input. So it is used here as a function that maps indices of the temperature domain
52 // to the temperature value at that point
53 double const mean_temp = ddc::transform_reduce(
54 temp.domain(),
55 0.,
56 ddc::reducer::sum<double>(),
57 temp)
58 / temp.domain().size();
59 std::cout << std::fixed << std::setprecision(3);
60 std::cout << "At t = " << time << ",\n";
61 std::cout << " * mean temperature = " << mean_temp << "\n";
62 // take a slice in the middle of the box
63 ddc::ChunkSpan temp_slice
64 = temp[ddc::get_domain<DDimY>(temp).front()
65 + ddc::get_domain<DDimY>(temp).size() / 2];
66 std::cout << " * temperature[y:"
67 << ddc::get_domain<DDimY>(temp).size() / 2 << "] = {";
68 ddc::for_each(
69 ddc::get_domain<DDimX>(temp),
70 [=](ddc::DiscreteElement<DDimX> const ix) {
71 std::cout << std::setw(6) << temp_slice(ix);
72 });
73 std::cout << " }" << std::endl;
74
75#ifdef DDC_BUILD_PDI_WRAPPER
76 ddc::PdiEvent("display")
77 .with("temp", temp)
78 .and_with("mean_temp", mean_temp)
79 .and_with("temp_slice", temp_slice);
80#endif
81}

86int main(int argc, char** argv)
87{
88#ifdef DDC_BUILD_PDI_WRAPPER
89 auto pdi_conf = PC_parse_string("");
90 PDI_init(pdi_conf);
91#endif
92 Kokkos::ScopeGuard const kokkos_scope(argc, argv);
93 ddc::ScopeGuard const ddc_scope(argc, argv);
94
95 // some parameters that would typically be read from some form of
96 // configuration file in a more realistic code
97
99 // Start of the domain of interest in the X dimension
100 double const x_start = -1.;
101 // End of the domain of interest in the X dimension
102 double const x_end = 1.;
103 // Number of discretization points in the X dimension
104 size_t const nb_x_points = 10;
105 // Thermal diffusion coefficient
106 double const kx = .01;
107 // Start of the domain of interest in the Y dimension
108 double const y_start = -1.;
109 // End of the domain of interest in the Y dimension
110 double const y_end = 1.;
111 // Number of discretization points in the Y dimension
112 size_t const nb_y_points = 100;
113 // Thermal diffusion coefficient
114 double const ky = .002;
115 // Simulated time at which to start simulation
116 double const start_time = 0.;
117 // Simulated time to reach as target of the simulation
118 double const end_time = 10.;
119 // Number of time-steps between outputs
120 ptrdiff_t const t_output_period = 10;
122
125 // Number of ghost points to use on each side in X
126 ddc::DiscreteVector<DDimX> static constexpr gwx {1};
128
130 // Initialization of the global domain in X with gwx ghost points on
131 // each side
132 auto const [x_domain, ghosted_x_domain, x_pre_ghost, x_post_ghost]
133 = ddc::init_discrete_space(DDimX::init_ghosted(
134 ddc::Coordinate<X>(x_start),
135 ddc::Coordinate<X>(x_end),
136 ddc::DiscreteVector<DDimX>(nb_x_points),
137 gwx));
139
141 // our zone at the start of the domain that will be mirrored to the
142 // ghost
143 ddc::DiscreteDomain<DDimX> const
144 x_domain_begin(x_domain.front(), x_post_ghost.extents());
145 // our zone at the end of the domain that will be mirrored to the
146 // ghost
147 ddc::DiscreteDomain<DDimX> const x_domain_end(
148 x_domain.back() - x_pre_ghost.extents() + 1,
149 x_pre_ghost.extents());
151
153 // Number of ghost points to use on each side in Y
154 ddc::DiscreteVector<DDimY> static constexpr gwy {1};
155
156 // Initialization of the global domain in Y with gwy ghost points on
157 // each side
158 auto const [y_domain, ghosted_y_domain, y_pre_ghost, y_post_ghost]
159 = ddc::init_discrete_space(DDimY::init_ghosted(
160 ddc::Coordinate<Y>(y_start),
161 ddc::Coordinate<Y>(y_end),
162 ddc::DiscreteVector<DDimY>(nb_y_points),
163 gwy));
164
165 // our zone at the start of the domain that will be mirrored to the
166 // ghost
167 ddc::DiscreteDomain<DDimY> const
168 y_domain_begin(y_domain.front(), y_post_ghost.extents());
169 // our zone at the end of the domain that will be mirrored to the
170 // ghost
171 ddc::DiscreteDomain<DDimY> const y_domain_end(
172 y_domain.back() - y_pre_ghost.extents() + 1,
173 y_pre_ghost.extents());
175
177 // max(1/dx^2)
178 double const invdx2_max = ddc::transform_reduce(
179 x_domain,
180 0.,
181 ddc::reducer::max<double>(),
182 [](ddc::DiscreteElement<DDimX> ix) {
183 return 1.
184 / (ddc::distance_at_left(ix)
185 * ddc::distance_at_right(ix));
186 });
187 // max(1/dy^2)
188 double const invdy2_max = ddc::transform_reduce(
189 y_domain,
190 0.,
191 ddc::reducer::max<double>(),
192 [](ddc::DiscreteElement<DDimY> iy) {
193 return 1.
194 / (ddc::distance_at_left(iy)
195 * ddc::distance_at_right(iy));
196 });
197 ddc::Coordinate<T> const max_dt {
198 .5 / (kx * invdx2_max + ky * invdy2_max)};
199
200 // number of time intervals required to reach the end time
201 // The + .2 is used to make sure it is rounded to the correct number of steps
202 ddc::DiscreteVector<DDimT> const nb_time_steps {
203 std::ceil((end_time - start_time) / max_dt) + .2};
204 // Initialization of the global domain in time:
205 // - the number of discrete time-points is equal to the number of
206 // steps + 1
207 // `init` takes required information to initialize the attributes of the dimension.
208 ddc::DiscreteDomain<DDimT> const time_domain
209 = ddc::init_discrete_space(
210 DDimT::
211 init(ddc::Coordinate<T>(start_time),
212 ddc::Coordinate<T>(end_time),
213 nb_time_steps + 1));
215
217 // Maps temperature into the full domain (including ghosts) twice:
218 // - once for the last fully computed time-step
219 ddc::Chunk ghosted_last_temp(
220 "ghosted_last_temp",
221 ddc::DiscreteDomain<
222 DDimX,
223 DDimY>(ghosted_x_domain, ghosted_y_domain),
224 ddc::DeviceAllocator<double>());
225
226 // - once for time-step being computed
227 // The `DeviceAllocator` is responsible for allocating memory on the default memory space.
228 ddc::Chunk ghosted_next_temp(
229 "ghosted_next_temp",
230 ddc::DiscreteDomain<
231 DDimX,
232 DDimY>(ghosted_x_domain, ghosted_y_domain),
233 ddc::DeviceAllocator<double>());
235
237 // The const qualifier makes it clear that ghosted_initial_temp always references
238 // the same chunk, `ghosted_last_temp` in this case
239 ddc::ChunkSpan const ghosted_initial_temp
240 = ghosted_last_temp.span_view();
241 // Initialize the temperature on the main domain
242 ddc::parallel_for_each(
243 ddc::DiscreteDomain<DDimX, DDimY>(x_domain, y_domain),
244 KOKKOS_LAMBDA(ddc::DiscreteElement<DDimX, DDimY> const ixy) {
245 double const x
246 = ddc::coordinate(ddc::select<DDimX>(ixy));
247 double const y
248 = ddc::coordinate(ddc::select<DDimY>(ixy));
249 ghosted_initial_temp(ixy)
250 = 9.999 * ((x * x + y * y) < 0.25);
251 });
253
254 ddc::Chunk ghosted_temp(
255 "ghost_temp",
256 ddc::DiscreteDomain<
257 DDimX,
258 DDimY>(ghosted_x_domain, ghosted_y_domain),
259 ddc::HostAllocator<double>());
260
261
263 // display the initial data
264 ddc::parallel_deepcopy(ghosted_temp, ghosted_last_temp);
265 display(ddc::coordinate(time_domain.front()),
266 ghosted_temp[x_domain][y_domain]);
267 // time of the iteration where the last output happened
268 ddc::DiscreteElement<DDimT> last_output_iter = time_domain.front();
270
272 for (auto const iter :
273 time_domain.remove_first(ddc::DiscreteVector<DDimT>(1))) {
275
277 // Periodic boundary conditions
278 ddc::parallel_deepcopy(
279 ghosted_last_temp[x_pre_ghost][y_domain],
280 ghosted_last_temp[y_domain][x_domain_end]);
281 ddc::parallel_deepcopy(
282 ghosted_last_temp[y_domain][x_post_ghost],
283 ghosted_last_temp[y_domain][x_domain_begin]);
284 ddc::parallel_deepcopy(
285 ghosted_last_temp[x_domain][y_pre_ghost],
286 ghosted_last_temp[x_domain][y_domain_end]);
287 ddc::parallel_deepcopy(
288 ghosted_last_temp[x_domain][y_post_ghost],
289 ghosted_last_temp[x_domain][y_domain_begin]);
291
293 // a span excluding ghosts of the temperature at the time-step we
294 // will build
295 ddc::ChunkSpan const next_temp {
296 ghosted_next_temp[x_domain][y_domain]};
297 // a read-only view of the temperature at the previous time-step
298 // span_cview returns a read-only ChunkSpan
299 ddc::ChunkSpan const last_temp {ghosted_last_temp.span_cview()};
301
303 // Stencil computation on the main domain
304 ddc::parallel_for_each(
305 next_temp.domain(),
306 KOKKOS_LAMBDA(
307 ddc::DiscreteElement<DDimX, DDimY> const ixy) {
308 ddc::DiscreteElement<DDimX> const ix
309 = ddc::select<DDimX>(ixy);
310 ddc::DiscreteElement<DDimY> const iy
311 = ddc::select<DDimY>(ixy);
312 double const dx_l = ddc::distance_at_left(ix);
313 double const dx_r = ddc::distance_at_right(ix);
314 double const dx_m = 0.5 * (dx_l + dx_r);
315 double const dy_l = ddc::distance_at_left(iy);
316 double const dy_r = ddc::distance_at_right(iy);
317 double const dy_m = 0.5 * (dy_l + dy_r);
318 next_temp(ix, iy) = last_temp(ix, iy);
319 next_temp(ix, iy)
320 += kx * ddc::step<DDimT>()
321 * (dx_l * last_temp(ix + 1, iy)
322 - 2.0 * dx_m * last_temp(ix, iy)
323 + dx_r * last_temp(ix - 1, iy))
324 / (dx_l * dx_m * dx_r);
325 next_temp(ix, iy)
326 += ky * ddc::step<DDimT>()
327 * (dy_l * last_temp(ix, iy + 1)
328 - 2.0 * dy_m * last_temp(ix, iy)
329 + dy_r * last_temp(ix, iy - 1))
330 / (dy_l * dy_m * dy_r);
331 });
333
335 if (iter - last_output_iter >= t_output_period) {
336 last_output_iter = iter;
337 ddc::parallel_deepcopy(ghosted_temp, ghosted_last_temp);
338 display(ddc::coordinate(iter),
339 ghosted_temp[x_domain][y_domain]);
340 }
342
344 // Swap our two buffers
345 std::swap(ghosted_last_temp, ghosted_next_temp);
347 }
348
350 if (last_output_iter < time_domain.back()) {
351 ddc::parallel_deepcopy(ghosted_temp, ghosted_last_temp);
352 display(ddc::coordinate(time_domain.back()),
353 ghosted_temp[x_domain][y_domain]);
354 }
356
357#ifdef DDC_BUILD_PDI_WRAPPER
358 PDI_finalize();
359 PC_tree_destroy(&pdi_conf);
360#endif
361}
