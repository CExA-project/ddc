37 template <class ExecSpace, class Functor, class DDim0>
38 inline void for_each_kokkos(DiscreteDomain<DDim0> const& domain, Functor const& f) noexcept
39 {
40  Kokkos::parallel_for(
41  Kokkos::RangePolicy<ExecSpace>(
42  select<DDim0>(domain).front().uid(),
43  select<DDim0>(domain).back().uid() + 1),
44  ForEachKokkosLambdaAdapter<Functor, DDim0>(f));

47 template <class ExecSpace, class Functor, class DDim0, class DDim1, class... DDims>
48 inline void for_each_kokkos(
49  DiscreteDomain<DDim0, DDim1, DDims...> const& domain,
50  Functor&& f) noexcept
51 {
52  Kokkos::Array<std::size_t, 2 + sizeof...(DDims)> const
53  begin {select<DDim0>(domain).front().uid(),
54  select<DDim1>(domain).front().uid(),
55  select<DDims>(domain).front().uid()...};
56  Kokkos::Array<std::size_t, 2 + sizeof...(DDims)> const
57  end {(select<DDim0>(domain).back().uid() + 1),
58  (select<DDim1>(domain).back().uid() + 1),
59  (select<DDims>(domain).back().uid() + 1)...};
60  Kokkos::parallel_for(
61  Kokkos::MDRangePolicy<
62  ExecSpace,
63  Kokkos::Rank<
64  2 + sizeof...(DDims),
65  Kokkos::Iterate::Right,
66  Kokkos::Iterate::Right>>(begin, end),
67  ForEachKokkosLambdaAdapter<Functor, DDim0, DDim1, DDims...>(f));
68 }

70 template <class RetType, class Element, std::size_t N, class Functor, class... Is>
71 inline void for_each_serial(
72  std::array<Element, N> const& start,
73  std::array<Element, N> const& end,
74  Functor const& f,
75  Is const&... is) noexcept
76 {
77  static constexpr std::size_t I = sizeof...(Is);
78  if constexpr (I == N) {
79  f(RetType(is...));
80  } else {
81  for (Element ii = start[I]; ii <= end[I]; ++ii) {
82  for_each_serial<RetType>(start, end, f, is..., ii);
83  }
84  }
85 }
