23/**
24 * @param gko_exec[in] A Ginkgo executor that has access to the Kokkos::View memory space
25 * @param view[in] A 2-D Kokkos::View with unit stride in the second dimension
26 * @return A Ginkgo Dense matrix view over the Kokkos::View data
27 */
28template <class KokkosViewType>
29auto to_gko_dense(std::shared_ptr<const gko::Executor> const& gko_exec, KokkosViewType const& view)
30{
31 static_assert((Kokkos::is_view_v<KokkosViewType> && KokkosViewType::rank == 2));
32 using value_type = typename KokkosViewType::traits::value_type;
33
34 if (view.stride_1() != 1) {
35 throw std::runtime_error("The view needs to be contiguous in the second dimension");
36 }
37
38 return gko::matrix::Dense<value_type>::
39 create(gko_exec,
40 gko::dim<2>(view.extent(0), view.extent(1)),
41 gko::array<value_type>::view(gko_exec, view.span(), view.data()),
42 view.stride_0());
43}

45template <class ExecSpace>
46int default_cols_per_chunk() noexcept
47{
48#ifdef KOKKOS_ENABLE_SERIAL
49 if (std::is_same_v<ExecSpace, Kokkos::Serial>) {
50 return 8192;
51 }
52#endif
53#ifdef KOKKOS_ENABLE_OPENMP
54 if (std::is_same_v<ExecSpace, Kokkos::OpenMP>) {
55 return 8192;
56 }
57#endif
58#ifdef KOKKOS_ENABLE_CUDA
59 if (std::is_same_v<ExecSpace, Kokkos::Cuda>) {
60 return 65535;
61 }
62#endif
63#ifdef KOKKOS_ENABLE_HIP
64 if (std::is_same_v<ExecSpace, Kokkos::HIP>) {
65 return 65535;
66 }
67#endif
68 return 1;
69}

97// Matrix class for sparse storage and iterative solve
98template <class ExecSpace>
99class Matrix_Sparse : public Matrix
100{
101 using matrix_sparse_type = gko::matrix::Csr<double, int>;
102
103private:
104 std::unique_ptr<gko::matrix::Dense<double>> m_matrix_dense;
105
106 std::shared_ptr<matrix_sparse_type> m_matrix_sparse;
107
108 std::shared_ptr<gko::solver::Bicgstab<double>> m_solver;
109 std::shared_ptr<gko::LinOp> m_solver_tr;
110
111 int m_cols_per_chunk; // Maximum number of columns of B to be passed to a Ginkgo solver
112
113 unsigned int m_preconditionner_max_block_size; // Maximum size of Jacobi-block preconditionner
114
115public:
116 // Constructor
117 explicit Matrix_Sparse(
118 const int mat_size,
119 std::optional<int> cols_per_chunk = std::nullopt,
120 std::optional<unsigned int> preconditionner_max_block_size = std::nullopt)
121 : Matrix(mat_size)
122 , m_cols_per_chunk(cols_per_chunk.value_or(default_cols_per_chunk<ExecSpace>()))
123 , m_preconditionner_max_block_size(preconditionner_max_block_size.value_or(
124 default_preconditionner_max_block_size<ExecSpace>()))
125 {
126 std::shared_ptr const gko_exec = create_gko_exec<ExecSpace>();
127 m_matrix_dense = gko::matrix::Dense<
128 double>::create(gko_exec->get_master(), gko::dim<2>(mat_size, mat_size));
129 m_matrix_dense->fill(0);
130 m_matrix_sparse = matrix_sparse_type::create(gko_exec, gko::dim<2>(mat_size, mat_size));
131 }
132
133 virtual double get_element([[maybe_unused]] int i, [[maybe_unused]] int j) const override
134 {
135 throw std::runtime_error("MatrixSparse::get_element() is not implemented because no API is "
136 "provided by Ginkgo");
137 }
138
139 virtual void set_element(int i, int j, double aij) override
140 {
141 m_matrix_dense->at(i, j) = aij;
142 }
143
144 int factorize_method() override
145 {
146 // Remove zeros
147 gko::matrix_data<double> matrix_data(gko::dim<2>(get_size(), get_size()));
148 m_matrix_dense->write(matrix_data);
149 m_matrix_dense.reset();
150 matrix_data.remove_zeros();
151 m_matrix_sparse->read(matrix_data);
152 std::shared_ptr const gko_exec = m_matrix_sparse->get_executor();
153
154 // Create the solver factory
155 std::shared_ptr const residual_criterion
156 = gko::stop::ResidualNorm<double>::build().with_reduction_factor(1e-19).on(
157 gko_exec);
158
159 std::shared_ptr const iterations_criterion
160 = gko::stop::Iteration::build().with_max_iters(1000u).on(gko_exec);
161
162 std::shared_ptr const preconditioner
163 = gko::preconditioner::Jacobi<double>::build()
164 .with_max_block_size(m_preconditionner_max_block_size)
165 .on(gko_exec);
166
167 std::unique_ptr const solver_factory
168 = gko::solver::Bicgstab<double>::build()
169 .with_preconditioner(preconditioner)
170 .with_criteria(residual_criterion, iterations_criterion)
171 .on(gko_exec);
172
173 m_solver = solver_factory->generate(m_matrix_sparse);
174 m_solver_tr = m_solver->transpose();
175 gko_exec->synchronize();
176
177 return 0;
178 }
179
180 virtual int solve_inplace_method(double* b, char transpose, int n_equations) const override
181 {
182 std::shared_ptr const gko_exec = m_solver->get_executor();
183
184 int const main_chunk_size = std::min(m_cols_per_chunk, n_equations);
185
186 Kokkos::View<double**, Kokkos::LayoutRight, ExecSpace> const
187 b_view(b, get_size(), n_equations);
188 Kokkos::View<double**, Kokkos::LayoutRight, ExecSpace> const
189 x_view("", get_size(), main_chunk_size);
190
191 int const iend = (n_equations + main_chunk_size - 1) / main_chunk_size;
192 for (int i = 0; i < iend; ++i) {
193 int const subview_begin = i * main_chunk_size;
194 int const subview_end
195 = (i + 1 == iend) ? n_equations : (subview_begin + main_chunk_size);
196
197 auto const b_subview = Kokkos::
198 subview(b_view, Kokkos::ALL, Kokkos::pair(subview_begin, subview_end));
199 auto const x_subview = Kokkos::
200 subview(x_view, Kokkos::ALL, Kokkos::pair(0, subview_end - subview_begin));
201
202 Kokkos::deep_copy(x_subview, b_subview);
203
204 if (transpose == 'N') {
205 m_solver
206 ->apply(to_gko_dense(gko_exec, b_subview),
207 to_gko_dense(gko_exec, x_subview));
208 } else if (transpose == 'T') {
209 m_solver_tr
210 ->apply(to_gko_dense(gko_exec, b_subview),
211 to_gko_dense(gko_exec, x_subview));
212 } else {
213 throw std::domain_error("transpose option not recognized");
214 }
215
216
217 Kokkos::deep_copy(b_subview, x_subview);
218 }
219
220 return 1;
221 }
222};
