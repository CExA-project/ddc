36constexpr bool is_spline_interpolation_mesh_uniform(
37 bool const is_uniform,
38 ddc::BoundCond const BcXmin,
39 ddc::BoundCond const BcXmax,
40 int degree)
41{
42 int N_BE_MIN = n_boundary_equations(BcXmin, degree);
43 int N_BE_MAX = n_boundary_equations(BcXmax, degree);
44 bool is_periodic = (BcXmin == ddc::BoundCond::PERIODIC) && (BcXmax == ddc::BoundCond::PERIODIC);
45 return is_uniform && ((N_BE_MIN != 0 && N_BE_MAX != 0) || is_periodic);
46}

433 class ExecSpace,
434 class MemorySpace,
435 class BSplines,
436 class InterpolationMesh,
437 ddc::BoundCond BcXmin,
438 ddc::BoundCond BcXmax,
439 SplineSolver Solver,

441int SplineBuilder<
442 ExecSpace,
443 MemorySpace,
444 BSplines,
445 InterpolationMesh,
446 BcXmin,
447 BcXmax,
448 Solver,
449 IDimX...>::compute_offset(interpolation_domain_type const& interpolation_domain)
450{
451 int offset;
452 if constexpr (bsplines_type::is_periodic()) {
453 // Calculate offset so that the matrix is diagonally dominant
454 std::array<double, bsplines_type::degree() + 1> values_ptr;
455 std::experimental::mdspan<
456 double,
457 std::experimental::extents<std::size_t, bsplines_type::degree() + 1>> const
458 values(values_ptr.data());
459 ddc::DiscreteElement<interpolation_mesh_type> start(interpolation_domain.front());
460 auto jmin = ddc::discrete_space<BSplines>()
461 .eval_basis(values, ddc::coordinate(start + BSplines::degree()));
462 if constexpr (bsplines_type::degree() % 2 == 0) {
463 offset = jmin.uid() - start.uid() + bsplines_type::degree() / 2 - BSplines::degree();
464 } else {
465 int const mid = bsplines_type::degree() / 2;
466 offset = jmin.uid() - start.uid() + (values(mid) > values(mid + 1) ? mid : mid + 1)
467 - BSplines::degree();
468 }
469 } else {
470 offset = 0;
471 }
472 return offset;
473}

476 class ExecSpace,
477 class MemorySpace,
478 class BSplines,
479 class InterpolationMesh,
480 ddc::BoundCond BcXmin,
481 ddc::BoundCond BcXmax,

484void SplineBuilder<
485 ExecSpace,
486 MemorySpace,
487 BSplines,
488 InterpolationMesh,
489 BcXmin,
490 BcXmax,
491 Solver,
492 IDimX...>::compute_block_sizes_uniform(int& lower_block_size, int& upper_block_size) const
493{
494 switch (BcXmin) {
495 case ddc::BoundCond::PERIODIC:
496 upper_block_size = (bsplines_type::degree()) / 2;
497 break;
498 case ddc::BoundCond::HERMITE:
499 upper_block_size = s_nbc_xmin;
500 break;
501 case ddc::BoundCond::GREVILLE:
502 upper_block_size = bsplines_type::degree() - 1;
503 break;
504 default:
505 throw std::runtime_error("ddc::BoundCond not handled");
506 }
507 switch (BcXmax) {
508 case ddc::BoundCond::PERIODIC:
509 lower_block_size = (bsplines_type::degree()) / 2;
510 break;
511 case ddc::BoundCond::HERMITE:
512 lower_block_size = s_nbc_xmax;
513 break;
514 case ddc::BoundCond::GREVILLE:
515 lower_block_size = bsplines_type::degree() - 1;
516 break;
517 default:
518 throw std::runtime_error("ddc::BoundCond not handled");
519 }
520}

594 // Special case: linear spline

596 // (desactivated)
597 /*
598 if constexpr (bsplines_type::degree() == 1)
599 return;

601
602 matrix = ddc::detail::SplinesLinearProblemMaker::make_new_sparse<ExecSpace>(
603 ddc::discrete_space<BSplines>().nbasis(),

605 preconditionner_max_block_size);

607 build_matrix_system();
608
609 matrix->setup_solver();

612template <
613 class ExecSpace,
614 class MemorySpace,
615 class BSplines,
616 class InterpolationMesh,
617 ddc::BoundCond BcXmin,
618 ddc::BoundCond BcXmax,
619 SplineSolver Solver,
620 class... IDimX>
621void SplineBuilder<
622 ExecSpace,
623 MemorySpace,
624 BSplines,
625 InterpolationMesh,
626 BcXmin,
627 BcXmax,
628 Solver,
629 IDimX...>::build_matrix_system()
630{
631 // Hermite boundary conditions at xmin, if any
632 if constexpr (BcXmin == ddc::BoundCond::HERMITE) {
633 std::array<double, (bsplines_type::degree() / 2 + 1) * (bsplines_type::degree() + 1)>
634 derivs_ptr;
635 ddc::DSpan2D
636 derivs(derivs_ptr.data(),
637 bsplines_type::degree() + 1,
638 bsplines_type::degree() / 2 + 1);
639 ddc::discrete_space<BSplines>().eval_basis_and_n_derivs(
640 derivs,
641 ddc::discrete_space<BSplines>().rmin(),
642 s_nbc_xmin);
643
644 // In order to improve the condition number of the matrix, we normalize
645 // all derivatives by multiplying the i-th derivative by dx^i
646 for (std::size_t i = 0; i < bsplines_type::degree() + 1; ++i) {
647 for (std::size_t j = 1; j < bsplines_type::degree() / 2 + 1; ++j) {
648 derivs(i, j) *= ddc::detail::ipow(m_dx, j);
649 }
650 }
651
652 // iterate only to deg as last bspline is 0
653 for (std::size_t i = 0; i < s_nbc_xmin; ++i) {
654 for (std::size_t j = 0; j < bsplines_type::degree(); ++j) {
655 matrix->set_element(i, j, derivs(j, s_nbc_xmin - i - 1 + s_odd));
656 }
657 }
658 }
659
660 // Interpolation points
661 std::array<double, bsplines_type::degree() + 1> values_ptr;
662 std::experimental::mdspan<
663 double,
664 std::experimental::extents<std::size_t, bsplines_type::degree() + 1>> const
665 values(values_ptr.data());
666
667 int start = interpolation_domain().front().uid();
668 ddc::for_each(interpolation_domain(), [&](auto ix) {
669 auto jmin = ddc::discrete_space<BSplines>().eval_basis(
670 values,
671 ddc::coordinate(ddc::DiscreteElement<interpolation_mesh_type>(ix)));
672 for (std::size_t s = 0; s < bsplines_type::degree() + 1; ++s) {
673 int const j = ddc::detail::
674 modulo(int(jmin.uid() - m_offset + s),
675 static_cast<int>(ddc::discrete_space<BSplines>().nbasis()));
676 matrix->set_element(ix.uid() - start + s_nbc_xmin, j, values(s));
677 }
678 });
679
680 // Hermite boundary conditions at xmax, if any
681 if constexpr (BcXmax == ddc::BoundCond::HERMITE) {
682 std::array<double, (bsplines_type::degree() / 2 + 1) * (bsplines_type::degree() + 1)>
683 derivs_ptr;
684 std::experimental::mdspan<
685 double,
686 std::experimental::extents<
687 std::size_t,
688 bsplines_type::degree() + 1,
689 bsplines_type::degree() / 2 + 1>> const derivs(derivs_ptr.data());
690
691 ddc::discrete_space<BSplines>().eval_basis_and_n_derivs(
692 derivs,
693 ddc::discrete_space<BSplines>().rmax(),
694 s_nbc_xmax);
695
696 // In order to improve the condition number of the matrix, we normalize
697 // all derivatives by multiplying the i-th derivative by dx^i
698 for (std::size_t i = 0; i < bsplines_type::degree() + 1; ++i) {
699 for (std::size_t j = 1; j < bsplines_type::degree() / 2 + 1; ++j) {
700 derivs(i, j) *= ddc::detail::ipow(m_dx, j);
701 }
702 }
703
704 int const i0 = ddc::discrete_space<BSplines>().nbasis() - s_nbc_xmax;
705 int const j0 = ddc::discrete_space<BSplines>().nbasis() - bsplines_type::degree();
706 for (std::size_t j = 0; j < bsplines_type::degree(); ++j) {
707 for (std::size_t i = 0; i < s_nbc_xmax; ++i) {
708 matrix->set_element(i0 + i, j0 + j, derivs(j + 1, i + s_odd));
709 }
710 }
711 }
712}

715 class ExecSpace,
716 class MemorySpace,

720 ddc::BoundCond BcXmax,
721 SplineSolver Solver,
722 class... IDimX>
723template <class Layout>
724void SplineBuilder<
725 ExecSpace,
726 MemorySpace,
727 BSplines,
728 InterpolationMesh,
729 BcXmin,
730 BcXmax,
731 Solver,
732 IDimX...>::
733operator()(
734 ddc::ChunkSpan<double, batched_spline_domain_type, Layout, memory_space> spline,
735 ddc::ChunkSpan<double const, batched_interpolation_domain_type, Layout, memory_space> vals,
736 std::optional<ddc::ChunkSpan<
737 double const,
738 batched_derivs_domain_type,
739 Layout,
740 memory_space>> const derivs_xmin,
741 std::optional<ddc::ChunkSpan<
742 double const,
743 batched_derivs_domain_type,
744 Layout,
745 memory_space>> const derivs_xmax) const
746{
747 assert(vals.template extent<interpolation_mesh_type>()
748 == ddc::discrete_space<bsplines_type>().nbasis() - s_nbc_xmin - s_nbc_xmax);
749
750 assert((BcXmin == ddc::BoundCond::HERMITE)
751 != (!derivs_xmin.has_value() || derivs_xmin->template extent<deriv_type>() == 0));
752 assert((BcXmax == ddc::BoundCond::HERMITE)
753 != (!derivs_xmax.has_value() || derivs_xmax->template extent<deriv_type>() == 0));
754 if constexpr (BcXmin == BoundCond::HERMITE) {
755 assert(ddc::DiscreteElement<deriv_type>(derivs_xmin->domain().front()).uid() == 1);
756 }
757 if constexpr (BcXmax == BoundCond::HERMITE) {
758 assert(ddc::DiscreteElement<deriv_type>(derivs_xmax->domain().front()).uid() == 1);
759 }
760
761 // Hermite boundary conditions at xmin, if any
762 // NOTE: For consistency with the linear system, the i-th derivative
763 // provided by the user must be multiplied by dx^i
764 if constexpr (BcXmin == BoundCond::HERMITE) {
765 assert(derivs_xmin->template extent<deriv_type>() == s_nbc_xmin);
766 auto derivs_xmin_values = *derivs_xmin;
767 auto const dx_proxy = m_dx;
768 ddc::parallel_for_each(
769 exec_space(),
770 batch_domain(),
771 KOKKOS_LAMBDA(typename batch_domain_type::discrete_element_type j) {
772 for (int i = s_nbc_xmin; i > 0; --i) {
773 spline(ddc::DiscreteElement<bsplines_type>(s_nbc_xmin - i), j)
774 = derivs_xmin_values(ddc::DiscreteElement<deriv_type>(i), j)
775 * ddc::detail::ipow(dx_proxy, i + s_odd - 1);
776 }
777 });
778 }
779
780 // TODO : Consider optimizing
781 // Fill spline with vals (to work in spline afterward and preserve vals)
782 auto const& offset_proxy = m_offset;
783 auto const& interp_size_proxy = interpolation_domain().extents();
784 auto const& nbasis_proxy = ddc::discrete_space<bsplines_type>().nbasis();
785 ddc::parallel_for_each(
786 exec_space(),
787 batch_domain(),
788 KOKKOS_LAMBDA(typename batch_domain_type::discrete_element_type j) {
789 for (int i = s_nbc_xmin; i < s_nbc_xmin + offset_proxy; ++i) {
790 spline(ddc::DiscreteElement<bsplines_type>(i), j) = 0.0;
791 }
792 for (int i = 0; i < interp_size_proxy; ++i) {
793 spline(ddc::DiscreteElement<bsplines_type>(s_nbc_xmin + i + offset_proxy), j)
794 = vals(ddc::DiscreteElement<interpolation_mesh_type>(i), j);
795 }
796 });
797
798 // Hermite boundary conditions at xmax, if any
799 // NOTE: For consistency with the linear system, the i-th derivative
800 // provided by the user must be multiplied by dx^i
801 if constexpr (BcXmax == BoundCond::HERMITE) {
802 assert(derivs_xmax->template extent<deriv_type>() == s_nbc_xmax);
803 auto derivs_xmax_values = *derivs_xmax;
804 auto const dx_proxy = m_dx;
805 ddc::parallel_for_each(
806 exec_space(),
807 batch_domain(),
808 KOKKOS_LAMBDA(typename batch_domain_type::discrete_element_type j) {
809 for (int i = 0; i < s_nbc_xmax; ++i) {
810 spline(ddc::DiscreteElement<bsplines_type>(nbasis_proxy - s_nbc_xmax - i),
811 j)
812 = derivs_xmax_values(ddc::DiscreteElement<deriv_type>(i + 1), j)
813 * ddc::detail::ipow(dx_proxy, i + s_odd);
814 }
815 });
816 }

833 // Create a 2D Kokkos::View to manage spline_tr as a matrix
834 Kokkos::View<double**, Kokkos::LayoutRight, exec_space> bcoef_section(

837 batch_domain().size());

848 }
849 });
850
851 // Not sure yet of what this part do

854 exec_space(),
855 batch_domain(),
856 KOKKOS_LAMBDA(typename batch_domain_type::discrete_element_type const j) {
857 if (offset_proxy != 0) {

862 }

865 = spline(ddc::DiscreteElement<bsplines_type>(i), j);
866 }
867 }
868 for (std::size_t i(0); i < bsplines_type::degree(); ++i) {
869 const ddc::DiscreteElement<bsplines_type> i_start(i);
870 const ddc::DiscreteElement<bsplines_type> i_end(nbasis_proxy + i);
871
872 spline(i_end, j) = spline(i_start, j);
873 }
874 });
