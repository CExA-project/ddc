131 template <class Mapping = mapping_type>
132 static KOKKOS_FUNCTION constexpr std::
133 enable_if_t<std::is_constructible_v<Mapping, extents_type>, internal_mdspan_type>
134 make_internal_mdspan(ElementType* ptr, mdomain_type const& domain)
135 {
136 if (domain.empty()) {
137 return internal_mdspan_type(
138 ptr,
139 std::experimental::layout_stride::mapping<extents_type>());
140 }
141 extents_type extents_r(::ddc::extents<DDims>(domain).value()...);
142 mapping_type mapping_r(extents_r);
143
144 extents_type extents_s((front<DDims>(domain) + ddc::extents<DDims>(domain)).uid()...);
145 std::array<std::size_t, sizeof...(DDims)> strides_s {
146 mapping_r.stride(type_seq_rank_v<DDims, detail::TypeSeq<DDims...>>)...};
147 std::experimental::layout_stride::mapping<extents_type> mapping_s(extents_s, strides_s);
148 return internal_mdspan_type(ptr - mapping_s(front<DDims>(domain).uid()...), mapping_s);
149 }

277 KOKKOS_FUNCTION constexpr ElementType* data_handle() const
278 {
279 ElementType* ptr = m_internal_mdspan.data_handle();
280 if (!m_domain.empty()) {
281 ptr += m_internal_mdspan.mapping()(front<DDims>(m_domain).uid()...);
282 }
283 return ptr;
284 }
285
289 KOKKOS_FUNCTION constexpr internal_mdspan_type internal_mdspan() const
290 {
291 return m_internal_mdspan;
292 }
293
297 KOKKOS_FUNCTION constexpr allocation_mdspan_type allocation_mdspan() const
298 {
299 DDC_IF_NVCC_THEN_PUSH_AND_SUPPRESS(implicit_return_from_non_void_function)
300 extents_type extents_s(::ddc::extents<DDims>(m_domain).value()...);
301 if constexpr (std::is_same_v<LayoutStridedPolicy, std::experimental::layout_stride>) {
302 mapping_type map(extents_s, m_internal_mdspan.mapping().strides());
303 return allocation_mdspan_type(data_handle(), map);
304 } else {
305 mapping_type map(extents_s);
306 return allocation_mdspan_type(data_handle(), map);
307 }
308 DDC_IF_NVCC_THEN_POP
309 }
