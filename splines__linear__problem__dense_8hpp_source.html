26/**
27 * @brief A dense linear problem dedicated to the computation of a spline approximation.
28 *
29 * The storage format is dense row-major. Lapack is used to perform every matrix and linear solver-related operations.
30 *
31 * @tparam ExecSpace The Kokkos::ExecutionSpace on which operations related to the matrix are supposed to be performed.
32 */
33template <class ExecSpace>
34class SplinesLinearProblemDense : public SplinesLinearProblem<ExecSpace>
35{
36public:
37 using typename SplinesLinearProblem<ExecSpace>::MultiRHS;
38 using SplinesLinearProblem<ExecSpace>::size;
39
40protected:
41 Kokkos::DualView<double**, Kokkos::LayoutRight, typename ExecSpace::memory_space> m_a;
42 Kokkos::DualView<int*, typename ExecSpace::memory_space> m_ipiv;
43
44public:
45 /**
46 * @brief SplinesLinearProblemDense constructor.
47 *
48 * @param mat_size The size of one of the dimensions of the square matrix.
49 */
50 explicit SplinesLinearProblemDense(std::size_t const mat_size)
51 : SplinesLinearProblem<ExecSpace>(mat_size)
52 , m_a("a", mat_size, mat_size)
53 , m_ipiv("ipiv", mat_size)
54 {
55 Kokkos::deep_copy(m_a.h_view, 0.);
56 }
57
58 double get_element(std::size_t const i, std::size_t const j) const override
59 {
60 assert(i < size());
61 assert(j < size());
62 return m_a.h_view(i, j);
63 }
64
65 void set_element(std::size_t const i, std::size_t const j, double const aij) override
66 {
67 assert(i < size());
68 assert(j < size());
69 m_a.h_view(i, j) = aij;
70 }
71
72 /**
73 * @brief Perform a pre-process operation on the solver. Must be called after filling the matrix.
74 *
75 * LU-factorize the matrix A and store the pivots using the LAPACK dgetrf() implementation.
76 */
77 void setup_solver() override
78 {
79 int const info = LAPACKE_dgetrf(
80 LAPACK_ROW_MAJOR,
81 size(),
82 size(),
83 m_a.h_view.data(),
84 size(),
85 m_ipiv.h_view.data());
86 if (info != 0) {
87 throw std::runtime_error(
88 "LAPACKE_dgetrf failed with error code " + std::to_string(info));
89 }
90
91 // Convert 1-based index to 0-based index
92 for (int i = 0; i < size(); ++i) {
93 m_ipiv.h_view(i) -= 1;
94 }
95
96 // Push on device
97 m_a.modify_host();
98 m_a.sync_device();
99 m_ipiv.modify_host();
100 m_ipiv.sync_device();
101 }
102
103 /**
104 * @brief Solve the multiple right-hand sides linear problem Ax=b or its transposed version A^tx=b inplace.
105 *
106 * The solver method is gaussian elimination with partial pivoting using the LU-factorized matrix A. The implementation is LAPACK method dgetrs.
107 *
108 * @param[in, out] b A 2D Kokkos::View storing the multiple right-hand sides of the problem and receiving the corresponding solution.
109 * @param transpose Choose between the direct or transposed version of the linear problem.
110 */
111 void solve(MultiRHS const b, bool const transpose) const override
112 {
113 assert(b.extent(0) == size());
114
115 // For order 1 splines, size() can be 0 then we bypass the solver call.
116 if (size() == 0)
117 return;
118
119 auto a_device = m_a.d_view;
120 auto ipiv_device = m_ipiv.d_view;
121
122 Kokkos::RangePolicy<ExecSpace> policy(0, b.extent(1));
123
124 if (transpose) {
125 Kokkos::parallel_for(
126 "gerts",
127 policy,
128 KOKKOS_LAMBDA(const int i) {
129 auto sub_b = Kokkos::subview(b, Kokkos::ALL, i);
130 KokkosBatched::SerialGetrs<
131 KokkosBatched::Trans::Transpose,
132 KokkosBatched::Algo::Getrs::Unblocked>::
133 invoke(a_device, ipiv_device, sub_b);
134 });
135 } else {
136 Kokkos::parallel_for(
137 "gerts",
138 policy,
139 KOKKOS_LAMBDA(const int i) {
140 auto sub_b = Kokkos::subview(b, Kokkos::ALL, i);
141 KokkosBatched::SerialGetrs<
142 KokkosBatched::Trans::NoTranspose,
143 KokkosBatched::Algo::Getrs::Unblocked>::
144 invoke(a_device, ipiv_device, sub_b);
145 });
146 }
147 }
148};
