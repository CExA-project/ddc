143 template <class Layout1, class Layout2, class Layout3, class... CoordsDims>
144 void operator()(
145 ddc::ChunkSpan<double, batched_interpolation_domain_type, Layout1, memory_space> const
146 spline_eval,
147 ddc::ChunkSpan<
148 ddc::Coordinate<CoordsDims...> const,
149 batched_interpolation_domain_type,
150 Layout2,
151 memory_space> const coords_eval,
152 ddc::ChunkSpan<double const, batched_spline_domain_type, Layout3, memory_space> const
153 spline_coef) const
154 {
155 interpolation_domain_type const interpolation_domain(spline_eval.domain());
156 batch_domain_type const batch_domain(spline_eval.domain());

158 ddc::parallel_for_each(
159 exec_space(),
160 batch_domain,
161 KOKKOS_CLASS_LAMBDA(typename batch_domain_type::discrete_element_type const j) {
162 const auto spline_eval_1D = spline_eval[j];
163 const auto coords_eval_1D = coords_eval[j];
164 const auto spline_coef_1D = spline_coef[j];
165 for (auto const i : interpolation_domain) {
166 spline_eval_1D(i) = eval(coords_eval_1D(i), spline_coef_1D);
167 }
168 });
169 }

209 template <class Layout1, class Layout2>
210 void integrate(
211 ddc::ChunkSpan<double, batch_domain_type, Layout1, memory_space> const integrals,
212 ddc::ChunkSpan<double const, batched_spline_domain_type, Layout2, memory_space> const
213 spline_coef) const
214 {
215 batch_domain_type const batch_domain(integrals.domain());
216 ddc::Chunk values_alloc(
217 ddc::DiscreteDomain<bsplines_type>(spline_coef.domain()),
218 ddc::KokkosAllocator<double, memory_space>());
219 ddc::ChunkSpan values = values_alloc.span_view();
220 Kokkos::parallel_for(
221 Kokkos::RangePolicy<exec_space>(0, 1),
222 KOKKOS_LAMBDA(int) { ddc::discrete_space<bsplines_type>().integrals(values); });
223
224 ddc::parallel_for_each(
225 exec_space(),
226 batch_domain,
227 KOKKOS_LAMBDA(typename batch_domain_type::discrete_element_type const j) {
228 integrals(j) = 0;
229 for (typename spline_domain_type::discrete_element_type const i :
230 values.domain()) {
231 integrals(j) += spline_coef(i, j) * values(i);
232 }
233 });
234 }

237 template <class Layout, class... CoordsDims>
238 KOKKOS_INLINE_FUNCTION double eval(
239 ddc::Coordinate<CoordsDims...> const& coord_eval,
240 ddc::ChunkSpan<double const, spline_domain_type, Layout, memory_space> const
241 spline_coef) const
242 {
243 ddc::Coordinate<typename interpolation_mesh_type::continuous_dimension_type>
244 coord_eval_interpolation
245 = ddc::select<typename interpolation_mesh_type::continuous_dimension_type>(
246 coord_eval);
247 if constexpr (bsplines_type::is_periodic()) {
248 if (coord_eval_interpolation < ddc::discrete_space<bsplines_type>().rmin()
249 || coord_eval_interpolation > ddc::discrete_space<bsplines_type>().rmax()) {
250 coord_eval_interpolation -= Kokkos::floor(
251 (coord_eval_interpolation
252 - ddc::discrete_space<bsplines_type>().rmin())
253 / ddc::discrete_space<bsplines_type>().length())
254 * ddc::discrete_space<bsplines_type>().length();
255 }
256 } else {
257 if (coord_eval_interpolation < ddc::discrete_space<bsplines_type>().rmin()) {
258 return m_left_extrap_rule(coord_eval_interpolation, spline_coef);
259 }
260 if (coord_eval_interpolation > ddc::discrete_space<bsplines_type>().rmax()) {
261 return m_right_extrap_rule(coord_eval_interpolation, spline_coef);
262 }
263 }
264 return eval_no_bc<eval_type>(coord_eval_interpolation, spline_coef);
265 }

267 template <class EvalType, class Layout, class... CoordsDims>
268 KOKKOS_INLINE_FUNCTION double eval_no_bc(
269 ddc::Coordinate<CoordsDims...> const& coord_eval,
270 ddc::ChunkSpan<double const, spline_domain_type, Layout, memory_space> const
271 spline_coef) const
272 {
273 static_assert(
274 std::is_same_v<EvalType, eval_type> || std::is_same_v<EvalType, eval_deriv_type>);
275 ddc::DiscreteElement<bsplines_type> jmin;
276 std::array<double, bsplines_type::degree() + 1> vals_ptr;
277 std::experimental::mdspan<
278 double,
279 std::experimental::extents<std::size_t, bsplines_type::degree() + 1>> const
280 vals(vals_ptr.data());
281 ddc::Coordinate<typename interpolation_mesh_type::continuous_dimension_type>
282 coord_eval_interpolation
283 = ddc::select<typename interpolation_mesh_type::continuous_dimension_type>(
284 coord_eval);
285 if constexpr (std::is_same_v<EvalType, eval_type>) {
286 jmin = ddc::discrete_space<bsplines_type>().eval_basis(vals, coord_eval_interpolation);
287 } else if constexpr (std::is_same_v<EvalType, eval_deriv_type>) {
288 jmin = ddc::discrete_space<bsplines_type>().eval_deriv(vals, coord_eval_interpolation);
289 }
290 double y = 0.0;
291 for (std::size_t i = 0; i < bsplines_type::degree() + 1; ++i) {
292 y += spline_coef(ddc::DiscreteElement<bsplines_type>(jmin + i)) * vals[i];
293 }
294 return y;
295 }
