24template <class RetType, class Element, std::size_t N, class Functor, class... Is>
25void for_each_serial(
26 std::array<Element, N> const& begin,
27 std::array<Element, N> const& end,
28 Functor const& f,
29 Is const&... is) noexcept
30{
31 static constexpr std::size_t I = sizeof...(Is);
32 if constexpr (I == N) {
33 f(RetType(is...));
34 } else {
35 for (Element ii = begin[I]; ii < end[I]; ++ii) {
36 for_each_serial<RetType>(begin, end, f, is..., ii);
37 }
38 }
39}

47template <class... DDims, class Functor>
48void for_each(DiscreteDomain<DDims...> const& domain, Functor&& f) noexcept
49{
50 DiscreteElement<DDims...> const ddc_begin = domain.front();
51 DiscreteElement<DDims...> const ddc_end = domain.front() + domain.extents();
52 std::array const begin = detail::array(ddc_begin);
53 std::array const end = detail::array(ddc_end);
54 detail::for_each_serial<DiscreteElement<DDims...>>(begin, end, std::forward<Functor>(f));
55}
