95  template <class QueryDDim, class... ODDims>
96  auto get_slicer_for(DiscreteElement<ODDims...> const& c) const
97  {
98  DDC_IF_NVCC_THEN_PUSH_AND_SUPPRESS(implicit_return_from_non_void_function)
99  if constexpr (in_tags_v<QueryDDim, detail::TypeSeq<ODDims...>>) {
100  return (uid<QueryDDim>(c) - front<QueryDDim>(this->m_domain).uid());
101  } else {
102  return std::experimental::full_extent;
103  }
104  DDC_IF_NVCC_THEN_POP
105  }
106 
107  template <class QueryDDim, class... ODDims>
108  auto get_slicer_for(DiscreteDomain<ODDims...> const& c) const
109  {
110  DDC_IF_NVCC_THEN_PUSH_AND_SUPPRESS(implicit_return_from_non_void_function)
111  if constexpr (in_tags_v<QueryDDim, detail::TypeSeq<ODDims...>>) {
112  return std::pair<std::size_t, std::size_t>(
113  front<QueryDDim>(c) - front<QueryDDim>(this->m_domain),
114  back<QueryDDim>(c) + 1 - front<QueryDDim>(this->m_domain));
115  } else {
116  return std::experimental::full_extent;
117  }
118  DDC_IF_NVCC_THEN_POP
119  }

183  constexpr ChunkSpan(allocation_mdspan_type allocation_mdspan, mdomain_type const& domain)
184  {
185  namespace stdex = std::experimental;
186  extents_type extents_s((front<DDims>(domain) + extents<DDims>(domain)).uid()...);
187  std::array<std::size_t, sizeof...(DDims)> strides_s {allocation_mdspan.mapping().stride(
188  type_seq_rank_v<DDims, detail::TypeSeq<DDims...>>)...};
189  stdex::layout_stride::mapping<extents_type> mapping_s(extents_s, strides_s);
190  this->m_internal_mdspan = internal_mdspan_type(
191  allocation_mdspan.data() - mapping_s(front<DDims>(domain).uid()...),
192  mapping_s);
193  this->m_domain = domain;
194  }

222  template <class... QueryDDims>
223  constexpr auto operator[](DiscreteElement<QueryDDims...> const& slice_spec) const
224  {
225  auto subview = std::experimental::
226  submdspan(allocation_mdspan(), get_slicer_for<DDims>(slice_spec)...);
227  using detail::TypeSeq;
228  using selected_meshes = type_seq_remove_t<TypeSeq<DDims...>, TypeSeq<QueryDDims...>>;
229  return ChunkSpan<
230  ElementType,
231  decltype(select_by_type_seq<selected_meshes>(this->m_domain)),
232  typename decltype(subview)::layout_type,
233  memory_space>(subview, select_by_type_seq<selected_meshes>(this->m_domain));
234  }

254  template <class... ODDims>
255  constexpr reference operator()(DiscreteElement<ODDims> const&... delems) const noexcept
256  {
257  static_assert(sizeof...(ODDims) == sizeof...(DDims), "Invalid number of dimensions");
258  assert(((delems >= front<ODDims>(this->m_domain)) && ...));
259  assert(((delems <= back<ODDims>(this->m_domain)) && ...));
260  return this->m_internal_mdspan(uid(take<DDims>(delems...))...);
261  }
262 
267  template <class... ODDims, class = std::enable_if_t<sizeof...(ODDims) != 1>>
268  constexpr reference operator()(DiscreteElement<ODDims...> const& delems) const noexcept
269  {
270  static_assert(sizeof...(ODDims) == sizeof...(DDims), "Invalid number of dimensions");
271  assert(((select<ODDims>(delems) >= front<ODDims>(this->m_domain)) && ...));
272  assert(((select<ODDims>(delems) <= back<ODDims>(this->m_domain)) && ...));
273  return this->m_internal_mdspan(uid<DDims>(delems)...);
274  }
